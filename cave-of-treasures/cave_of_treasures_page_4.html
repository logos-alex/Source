עמוד ד של מערת האוצרות

דף א (עמוד 22)

וְקַדְמָיָא דְכֻלְהוֹן אִיתַוְהִי נוּהֲרָא، הַו דְבֵהּ אֶתְעַטַּף אֱלָהָא אַיְכְּ דַאֲמַר דָוִד דַעֲטַף נוּהֲרָא אַיְכְּ כּוּתִינָא. וְהַו נוּהֲרָא שַׁפִּירָא וְטָבָא וּבְהִירָא וְרַגִיגָא וְשַׁבִּיחָא וְחַדְיָא וּפְצִיחָא וְנַצִיחָא וְנַהִירָא וּבְרִירָא וְרַחִימָא וְחַבִּיבָא. וְהַו נוּהֲרָא לָא אִית לֵהּ מְתוּם דְמוּתָא בְּכֻלְהֵין בִּרְיָתָא. וְהַו נוּהֲרָא שְׁבַע זִבְנִין נַהִיר מִן שִׁמְשָׁא. וְהַו נוּהֲרָא לָא מִתְחַבַּל וְלָא מִשְׁתַחְלַף וְלָא מִצְטַלַּם וְלָא מִתְפַּסַּק וְלָא מִתְעַבַּר וְלָא מִתְמַיֵּשׁ וְלָא מִתְטַלַּק וְלָא מִתְחַלַּט וְלָא מִתְטָרַף וְלָא מִשְׁתָּרֵא וְלָא מִתְחַלַּשׁ וְלָא מִתְטָרַד וְלָא מִתְכָּרֵהּ וְלָא מִתְעַמַּם וְלָא מִתְעַצַּר וְלָא מִתְאַמַּץ וְלָא מִתְגַּזַּם וְלָא מִתְפַּלַּג וְלָא מִתְבַּלְבַּל וְלָא מִתְפָּרַס וְלָא מִתְטָוַשׁ וְלָא מִתְכַּסֵּא וְלָא מִתְעַבַּר וְלָא מִתְחַפֵּא וְלָא מִתְגַּלַּז וְלָא מִתְעַלַּם וְלָא מִתְאַטַּר וְלָא מִתְגַּנַּב וְלָא מִתְטַלַּם וְלָא מִתְטָרַף וְלָא מִשְׁתַּגַּשׁ וְלָא מִתְפָּרַע וְלָא מִתְבַּזַּח וְלָא מִתְטַלַּע וְלָא מִתְטָרַף וְלָא מִתְפָּרַס וְלָא מִתְפַּסַּק וְלָא מִתְבַּצַּר וְלָא מִתְעַלַּב וְלָא מִתְכַּתַּם וְלָא מִתְטַלַּשׁ וְלָא מִתְעַפַּן וְלָא מִתְטַנַּף וְלָא מִתְגַּוְשַׁם וְלָא מִתְדָּרַךְ וְלָא מִתְפַּחַם וְלָא מִתְמַלַּל וְלָא מִתְחַשַׁב וְלָא מִתְפַּסַּר וְלָא מִתְבַּיֵּן וְלָא מִתְתַּרְעָא וְלָא מִתְחַזֵּא וְלָא מִתְגַּשַׁם וְלָא מִתְפָּרַס וְלָא מִתְפַּלַּג וְלָא מִתְטָרַף וְלָא מִתְפָּרַע וְלָא מִתְכָּרַךְ וְלָא מִתְפַּרְנַס וְלָא מִתְפַּרְצַף וְלָא מִתְחַבַּר וְלָא מִתְפָּרַשׁ וְלָא מִתְפַּסַּק וְלָא מִתְחַסַּר וְלָא מִתְגַּמַּר וְלָא מִתְכַּלְכַּל וְלָא מִתְתַּזִיעַ וְלָא מִתְנַפַּץ וְלָא מִתְרַפֵּא וְלָא מִתְחַטַּף וְלָא מִתְעַצַּף וְלָא מִתְפַּסַּר וְלָא מִתְבַּטַּל וְלָא מִתְפַּגַּר וְלָא מִתְעַכַּב וְלָא מִתְעַוֵּךְ וְלָא מִתְעַוֵּר וְלָא מִתְעַבַּר וְלָא מִתְקַדַּם וְלָא מִתְאַחַר וְלָא מִתְעַטַּף וְלָא מִתְגַּלֵּא וְלָא מִתְטַשְׁטַשׁ וְלָא מִתְטָרַף וְלָא מִתְחַלַּף וְלָא מִתְטָרַד. אַףלָא מִנֵּהּ נָסְבוּ חַד מִנְהוֹן.

וְהָרִאשׁוֹן שֶׁבְּכֻלָּם הוּא הָאוֹר, הוּא שֶׁבּוֹ הִתְעַטֵּף אֱלֹהִים כְּמוֹ שֶׁאָמַר דָּוִד: "עֹטֶה אוֹר כַּשַּׂלְמָה" (תהלים קד, ב). וְהָאוֹר הַהוּא יָפֶה וְטוֹב וּבָהִיר וְעָדִין וּמְשֻׁבָּח וְשָׂמֵחַ וְצוֹהֵל וְנוֹצֵץ וּמֵאִיר וּבָרוּר וְאָהוּב וְחָבִיב. וְלָאוֹר הַהוּא אֵין לוֹ מֵעוֹלָם דִּמְיוֹן בְּכָל הַבְּרִיּוֹת. וְהָאוֹר הַהוּא פִּי שִׁבְעָה מֵאִיר מִן הַשֶּׁמֶשׁ. וְהָאוֹר הַהוּא אֵינוֹ מִתְקַלְקֵל וְאֵינוֹ מִשְׁתַּנֶּה וְאֵינוֹ מִצְטַלֵּם וְאֵינוֹ נִפְסָק וְאֵינוֹ עוֹבֵר וְאֵינוֹ נָמֵס וְאֵינוֹ נִשְׁלָל וְאֵינוֹ מִתְעַרְבֵּב וְאֵינוֹ נִטְרָף וְאֵינוֹ נִפְתָּר וְאֵינוֹ נֶחְלָשׁ וְאֵינוֹ נִדְחָק וְאֵינוֹ נִכְאב וְאֵינוֹ מִתְעַמֵּם וְאֵינוֹ נֶעֱצָר וְאֵינוֹ מִתְאַמֵּץ וְאֵינוֹ נִגְזָם וְאֵינוֹ נֶחֱלָק וְאֵינוֹ מִתְבַּלְבֵּל וְאֵינוֹ נִפְרָס וְאֵינוֹ מִטַּשְׁטֵשׁ וְאֵינוֹ מִתְכַּסֶּה וְאֵינוֹ עוֹבֵר וְאֵינוֹ מִתְחַפֶּה וְאֵינוֹ נִגְלָז וְאֵינוֹ נֶעֱלָם וְאֵינוֹ מִתְאַטָּר וְאֵינוֹ נִגְנָב וְאֵינוֹ נִפְגָּם וְאֵינוֹ נִטְרָף וְאֵינוֹ מִשְׁתַּגֵּשׁ וְאֵינוֹ נִפְרָע וְאֵינוֹ מִתְבַּזֶּה וְאֵינוֹ מִתְעַלֵּם וְאֵינוֹ נִטְרָף וְאֵינוֹ נִפְרָס וְאֵינוֹ נִפְסָק וְאֵינוֹ נִבְצָר וְאֵינוֹ נֶעֱלָב וְאֵינוֹ נִכְתָּם וְאֵינוֹ נִטְלָשׁ וְאֵינוֹ מִתְעַפֵּן וְאֵינוֹ מִתְטַנֵּף וְאֵינוֹ מִתְגַּשֵּׁם וְאֵינוֹ נִתְפָּס וְאֵינוֹ מִתְפָּחֵם וְאֵינוֹ מִתְמַלֵּל וְאֵינוֹ נֶחְשָׁב וְאֵינוֹ מִתְפָּרֵשׁ וְאֵינוֹ מִתְבַּיֵּן וְאֵינוֹ מִתְרָעֶה וְאֵינוֹ נִרְאֶה וְאֵינוֹ מִתְגַּשֵּׁם וְאֵינוֹ נִפְרָס וְאֵינוֹ נֶחֱלָק וְאֵינוֹ נִטְרָף וְאֵינוֹ נִפְרָע וְאֵינוֹ מִתְכַּרְבֵּל וְאֵינוֹ מִתְפַּרְנֵס וְאֵינוֹ מִתְפַּרְצֵף וְאֵינוֹ מִתְחַבֵּר וְאֵינוֹ נִפְרָשׁ וְאֵינוֹ נִפְסָק וְאֵינוֹ נֶחְסָר וְאֵינוֹ נִגְמָר וְאֵינוֹ מִתְכַּלְכֵּל וְאֵינוֹ מִזְדַּעְזֵעַ וְאֵינוֹ נִנְעָר וְאֵינוֹ מִתְרַפֶּה וְאֵינוֹ נֶחְטָף וְאֵינוֹ מִתְעַצֵּב וְאֵינוֹ מִתְפָּרֵשׁ וְאֵינוֹ מִתְבַּטֵּל וְאֵינוֹ מִתְפַּגֵּר וְאֵינוֹ מִתְעַכֵּב וְאֵינוֹ מִתְעַוֵּךְ וְאֵינוֹ מִתְעַוֵּר וְאֵינוֹ עוֹבֵר וְאֵינוֹ מִתְקַדֵּם וְאֵינוֹ מִתְאַחֵר וְאֵינוֹ מִתְעַטֵּף וְאֵינוֹ מִתְגַּלֶּה וְאֵינוֹ מִטַּשְׁטֵשׁ וְאֵינוֹ נִטְרָף וְאֵינוֹ מִשְׁתַּנֶּה וְאֵינוֹ נִדְחָק. אַף לֹא אֶחָד מֵהֶם לָקְחוּ.

ביאור ופרשנות

הקטע פותח בתיאור נשגב של ה"אור" (נוּהֲרָא) כראשון לכל הברואים, שבו האל "מתעטף", בהתייחסות ישירה לתהלים קד, ב: "עֹטֶה אוֹר כַּשַּׂלְמָה". האור מתואר בשלל תארים חיוביים – יפה, טוב, בהיר, עדין, משובח, שמח, צוהל, נוצץ, מאיר, ברור, אהוב וחביב – המדגישים את קדושתו וייחודו.* הציטוט מתהלים קד, ב ממחיש את השימוש במקורות מקראיים משותפים ליהדות ולנצרות.

האור מוצג כחסר דמיון בבריאה ומאיר פי שבעה מהשמש, מה שמבליט את טרנסצנדנטיותו. עיקר הקטע מוקדש לרשימה ארוכה של שלילות (תיאולוגיה אפופטית), המתארת מה האור האלוהי איננו: הוא אינו מתקלקל, משתנה, נפסק, מתערער, נתפס במחשבה, נראה במלואו, או מושפע מפגמים פיזיים, מנטליים או מוסריים. רשימה זו נועדה להדגיש את היותו בלתי נתפס ובלתי משתנה.* תיאולוגיה אפופטית, המתארת את האל דרך מה שהוא איננו, נפוצה בתאולוגיה המיסטית היהודית והנוצרית.

הסיום, "אַף לֹא אֶחָד מֵהֶם לָקְחוּ", ניתן לפרש כהצהרה שאף מאפיין שלילי אינו חל על האור, או שאף ברוא לא גרע ממנו דבר, מה שמחזק את שלמותו המוחלטת.

דף ב (עמוד 24)

מְטוּל הָדָא לָא מָצֵא דְנִשְׁכַּח לֵהּ חַד מִנְהוֹן מְטוּל דְכֻלְהוֹן בֵּהּ אֲחִידִין וּבֵהּ מִתְקַיְּמִין וּבֵהּ מִתְתַּחְדִין וּבֵהּ מִתְנַטְּרִין וּבֵהּ מִתְעַלִּין וּבֵהּ מִתְנַצְּחִין וּבֵהּ מִתְפַּרְנְסִין וּבֵהּ מִתְרַבִּין וּבֵהּ מִתְקַדְּשִׁין וּבֵהּ מִשְׁתַּבְּחִין וּבֵהּ מִתְפַּצְּחִין וּבֵהּ מִתְבַּסְּמִין וּבֵהּ מִתְטַיְּבִין וּבֵהּ מִתְנַטְּרִין וּבֵהּ מִתְעַדְּרִין וּבֵהּ מִתְרַחֲמִין וּבֵהּ מִתְחַיְּלִין וּבֵהּ מִתְקַיְּמִין וּבֵהּ מִתְנַהֲרִין וּבֵהּ מִתְבָּרְכִין וּבֵהּ מִתְקַדְּשִׁין וּבֵהּ מִתְגַּמְּרִין. וְכֻלְהוֹן חַיְלָוָתָא דִשְׁמַיָּא וּדְאַרְעָא בֵּהּ מִשְׁתַּמְלִין וּבֵהּ מִתְתַּקְּנִין וּבֵהּ מִתְנַטְּרִין. וּבֵהּ תְּלֵין כֻּלְהֵין בִּרְיָתָא וּמִנֵּהּ נָסְבִין כֻּלְהוֹן טָבָתָא. וְהוּאִיוּ שׁוּרָיָא וְשׁוּלָמָא דְכֹל. וְהוּאִיוּ דִלְעֵל מִן כֹּל. וְהוּאִיוּ דִבְכֹל. וְהוּאִיוּ דְמִן כֹּל. וְהוּאִיוּ דִלְוָת כֹּל. וְהוּאִיוּ דְמִן כֹּל לְכֹל. וְהוּאִיוּ דְעַל כֹּל. וְהוּאִיוּ דִבְכֹל כֹּל. וְהוּאִיוּ דִקֳדָם כֹּל. וְהוּאִיוּ דְמִן בָּתַר כֹּל. וְהוּאִיוּ דִבְגַו כֹּל. וְהוּאִיוּ דְמִן לְבַר מִן כֹּל. וְהוּאִיוּ דַחֲדִיר לְכֹל. וְהוּאִיוּ דְלָא סַכָּא לֵהּ. וְהוּאִיוּ דְלָא אִית לֵהּ שׁוּרָיָא וְלָא שׁוּלָמָא. וְהוּאִיוּ דְלָא אִית לֵהּ שׁוּחְלָפָא וְלָא טוּפְסָא וְלָא טוּרְפָא וְלָא גוּשְׁמָא וְלָא חוּשְׁבָּנָא וְלָא תַרְעִיתָא וְלָא חֶזְוָתָא וְלָא גַשְׁמָא וְלָא פוּרְשָׁנָא וְלָא פוּסְקָנָא. וְהוּאִיוּ דְלָא אִית לֵהּ מַוְתָא וְלָא חַבָּלָא וְלָא טוּלְשָׁא וְלָא טוּפָנָא וְלָא טוּרְפָא וְלָא גוּשְׁמָא וְלָא חוּבָּלָא וְלָא עוּפְיָנָא וְלָא טַנְפָא וְלָא חֲטִיתָא. וְהוּאִיוּ מְרַחֲמָנָא וְחַנָּנָא וּנְגִיר רוּחָא וְסַגִּי חִנָּנָא וְקוּשְׁתָּא. וְהוּאִיוּ חַיֵּא וּבֵהּ חָיְיִן כֹּל. וְהוּאִיוּ נוּהֲרָא וּמִנֵּהּ מִתְנַהֲרִין כֹּל. וְהוּאִיוּ טָבָא וּמִנֵּהּ מִתְטַיְּבִין כֹּל. וְהוּאִיוּ חַכִּימָא וּמִנֵּהּ מִתְחַכְּמִין כֹּל. וְהוּאִיוּ מְרַחֲמָנָא וּמִנֵּהּ מִתְרַחֲמִין כֹּל. וְהוּאִיוּ כֹּל וּבֵהּ כֹּל וּמִנֵּהּ כֹּל וּלְוָתֵהּ כֹּל וְעִלָּוֵהִי כֹּל. וְכֻלְהוֹן דְחַלְתָא קַדְמַיְתָא מִן קֳדָם כֻּלְהוֹן אִיתֵיהוֹן.

מִשּׁוּם כָּךְ אֵין אֶחָד מֵהֶם יָכוֹל לְמָצְאוֹ, מִשּׁוּם שֶׁכֻּלָּם בּוֹ אֲחוּזִים וּבוֹ מִתְקַיְּמִים וּבוֹ מִתְאַחֲדִים וּבוֹ נִשְׁמָרִים וּבוֹ מִתְעַלִּים וּבוֹ מִתְנַצְּחִים וּבוֹ מִתְפַּרְנְסִים וּבוֹ מִתְרַבִּים וּבוֹ מִתְקַדְּשִׁים וּבוֹ מִשְׁתַּבְּחִים וּבוֹ מִתְפַּצְּחִים וּבוֹ מִתְבַּשְׂמִים וּבוֹ מִתְטַיְּבִים וּבוֹ נִשְׁמָרִים וּבוֹ נֶעֱזָרִים וּבוֹ מִתְרַחֲמִים וּבוֹ מִתְחַזְּקִים וּבוֹ מִתְקַיְּמִים וּבוֹ מִתְנַהֲרִים וּבוֹ מִתְבָּרְכִים וּבוֹ מִתְקַדְּשִׁים וּבוֹ נִשְׁלָמִים. וְכָל חֵילוֹת הַשָּׁמַיִם וְהָאָרֶץ בּוֹ נִשְׁלָמִים וּבוֹ מִתְתַּקְּנִים וּבוֹ נִשְׁמָרִים. וּבוֹ תְּלוּיִים כָּל הַבְּרִיּוֹת וּמִמֶּנּוּ לוֹקְחִים כָּל הַטּוֹבוֹת. וְהוּא הָרֵאשִׁית וְהַסּוֹף שֶׁל הַכֹּל. וְהוּא שֶׁמֵּעַל לַכֹּל. וְהוּא שֶׁבַּכֹּל. וְהוּא שֶׁמִּן הַכֹּל. וְהוּא שֶׁאֶל הַכֹּל. וְהוּא שֶׁמִּן הַכֹּל אֶל הַכֹּל. וְהוּא שֶׁעַל הַכֹּל. וְהוּא שֶׁבְּכָל כֹּל. וְהוּא שֶׁלִּפְנֵי הַכֹּל. וְהוּא שֶׁמֵּאַחַר הַכֹּל. וְהוּא שֶׁבְּתוֹךְ הַכֹּל. וְהוּא שֶׁמִּחוּץ לַכֹּל. וְהוּא שֶׁמַּקִּיף אֶת הַכֹּל. וְהוּא שֶׁאֵין לוֹ סוֹף. וְהוּא שֶׁאֵין לוֹ רֵאשִׁית וְלֹא סוֹף. וְהוּא שֶׁאֵין לוֹ שִׁנּוּי וְלֹא דְּפוּס וְלֹא פְּגָם וְלֹא גּוּף וְלֹא מַחְשָׁבָה וְלֹא רַעְיוֹן וְלֹא מַרְאֶה וְלֹא גֶּשֶׁם וְלֹא הֶבְדֵּל וְלֹא הֶפְסֵק. וְהוּא שֶׁאֵין לוֹ מָוֶת וְלֹא קִלְקוּל וְלֹא מְחִיקָה וְלֹא מַגֵּפָה וְלֹא פְּגָם וְלֹא גּוּף וְלֹא חֲבָלָה וְלֹא עֲיֵפוּת וְלֹא טִנֹּפֶת וְלֹא חֲטִיאָה. וְהוּא רַחוּם וְחַנּוּן וְאֶרֶךְ אַפַּיִם וְרַב חֶסֶד וֶאֱמֶת. וְהוּא הַחַיִּים וּבוֹ חַיִּים הַכֹּל. וְהוּא הָאוֹר וּמִמֶּנּוּ מְאוּרִים הַכֹּל. וְהוּא הַטּוֹב וּמִמֶּנּוּ מִתְטַיְּבִים הַכֹּל. וְהוּא הֶחָכָם וּמִמֶּנּוּ מִתְחַכְּמִים הַכֹּל. וְהוּא הָרַחוּם וּמִמֶּנּוּ מִתְרַחֲמִים הַכֹּל. וְהוּא הַכֹּל וּבוֹ הַכֹּל וּמִמֶּנּוּ הַכֹּל וְאֵלָיו הַכֹּל וְעָלָיו הַכֹּל. וְכֻלָּם, הַיִּרְאָה הָרִאשׁוֹנָה לִפְנֵי כֻלָּם הִיא.

ביאור ופרשנות

הקטע ממשיך את התיאור של האלוהות (המזוהה עם האור מהקטע הקודם) בשילוב תיאולוגיה אפופטית (שלילית) וקטפטית (חיובית). הוא פותח בהצהרה כי אף ברוא אינו יכול "למצוא" (להשיג) את האל, שכן כל הברואים תלויים בו לחלוטין: "כֻלָּם בּוֹ אֲחוּזִים וּבוֹ מִתְקַיְּמִים...". רשימה ארוכה של פעלים מתארת את תלות הבריאה באל כמקור קיומה, שמירתה, התעלותה, קידושה ושמחתה.* התלות המוחלטת של הבריאה באל מהדהדת רעיונות מפילוסופיה יהודית ונוצרית, כמו אצל הרמב"ם או אבות הכנסייה.

הקטע מציג תפיסה פנאנתאיסטית: האל הוא מעל הכול, בתוך הכול, מחוץ לכול, מקיף הכול, ראשית וסוף הכול, אך אינו זהה לעולם. הוא חסר ראשית, סוף, שינוי, צורה, גוף או פגם, תוך המשך התיאור האפופטי. במקביל, הוא מתואר בתארים חיוביים: רחום, חנון, ארך אפיים, רב חסד ואמת – תכונות המהדהדות את המידות האלוהיות במקרא (שמות לד:6-7).* התיאור "רחום וחנון..." לקוח ישירות משמות לד:6-7.

האל מוצג כמקור החיים, האור, הטוב, החכמה והרחמים, שממנו הכול נובע ובו הכול תלוי. הסיום, "וְכֻלָּם, הַיִּרְאָה הָרִאשׁוֹנָה לִפְנֵי כֻלָּם הִיא", מצביע על יראת האל כתגובה הראויה של הבריאה כלפיו, או כעומדת בבסיס היחסים בין הבריאה לאל.* יראת האל כערך עליון מופיעה גם בספרות חז"ל ובכתבים נוצריים מוקדמים.

שני הקטעים מהווים שיא תאולוגי ופואטי ב"מערת האוצרות", המשלב שפה פיוטית עשירה עם רעיונות עמוקים על טבע האל והיחס שלו לעולם. הרשימות הארוכות של תארים ושלילות נועדו לעורר יראה והתפעלות מגדולת הבורא.
