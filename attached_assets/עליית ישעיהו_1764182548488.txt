חזון ישעיהו (מהדורה פרשנית)
חלק א': ההתגלות בחצר המלך והעלייה לרקיע
(עמודים 1ב, 2, 2ב)
א ...מִימִינוֹ וּמִשְּׂמֹאלוֹ. וּכְשָׁמְעָם אֶת-קוֹל רוּחַ הַקֹּדֶשׁ, כָּרְעוּ עַל-בִּרְכֵּיהֶם וַיְשַׁבְּחוּ אֶת-אֵל עֶלְיוֹן, הַשּׁוֹכֵן בַּקְּדוֹשִׁים וַאֲשֶׁר נָתַן כֹּחַ כָּזֶה בִּדְבָרוֹ בָּעוֹלָם.
ב וּבְעוֹד שֶׁדִּבֵּר בְּרוּחַ הַקֹּדֶשׁ לְאָזְנֵי כֹּל, פִּתְאֹם שָׁתַק וְחָדַל לִרְאוֹת אֶת-הָעוֹמְדִים לְפָנָיו. עֵינָיו הָיוּ פְקוּחוֹת, אַךְ פִּיו הָיָה סָגוּר, כִּי שְׁאִיפַת הָרוּחַ הָיְתָה עִמּוֹ.
ג וְכֻלָּם חָשְׁבוּ כִּי נֶחְטַף יְשַׁעְיָהוּ. אַךְ הַנְּבִיאִים הֵבִינוּ כִּי בְּהִתְגַּלּוּת הָיָה, וְהֶחָזוֹן אֲשֶׁר חָזָה לֹא הָיָה מִן-הָעוֹלָם הַזֶּה, אֶלָּא מִן-הַנִּסְתָּר מִכָּל-בָּשָׂר.
ד וּכְשֶׁעָמַד מֵחֶזְיוֹנוֹ וְשָׁב אֶל-עַצְמוֹ, סִפֵּר לְחִזְקִיָּהוּ וְלִבְנוֹ נָשׂוֹן, וּלְמִיכָה וְלִשְׁאָר הַנְּבִיאִים, לֵאמֹר: "כַּאֲשֶׁר הִתְנַבֵּאתִי לְאָזְנֵיכֶם, אֲשֶׁר לֹא שְׁמַעְתֶּם, רָאִיתִי מַלְאָךְ נִכְבָּד, לֹא כְּכָבוֹד הַמַּלְאָכִים אֲשֶׁר רָאִיתִי תָּמִיד, אֶלָּא כָּבוֹד אַחֵר, גָּדוֹל וְקָדוֹשׁ, אֲשֶׁר לֹא אוּכַל לְסַפֵּר".
ה "וַיֹּאחֶז בְּיָדִי, וַיַּעֲלֵנִי לַמָּרוֹם. וַאֲנִי אָמַרְתִּי: 'מִי אַתָּה? מַה-שִּׁמְךָ? וּלְאָן אַתָּה מַעֲלֶה אוֹתִי?'. כִּי נִתַּן לִי הַכֹּחַ לְדַבֵּר עִמּוֹ. אָז עָנָה וְאָמַר: 'כַּאֲשֶׁר אַעֲלֶךָ לַמַּעֲלָה וְאַרְאֲךָ אֶת-הֶחָזוֹן שֶׁלְּמַעֲנוֹ שֻׁלַּחְתִּי, אָז תָּבִין מִי אָנֹכִי. אַךְ אֶת-שְׁמִי לֹא תֵדַע, כִּי חָפֵץ אַתָּה לָשׁוּב לְגוּפְךָ. וּכְשֶׁאַעֲלֶךָ, תִּרְאֶה'.
ו "וְשָׂמַחְתִּי כִּי בְּעַנְוָה עָנָה לִי. וַיֹּאמֶר לִי: 'הַאִם שָׂמַחְתָּ כִּי בְּעַנְוָה עָנִיתִי לְךָ? עוֹד תִּרְאֶה גָּדוֹל מִמֶּנִּי, עָנָו יוֹתֵר וּמַשְׂכִּיל יוֹתֵר, הֶחָפֵץ לְשׂוֹחֵחַ עִמְּךָ, וְגָדוֹל מִן-הַגָּדוֹל, מֵאִיר יוֹתֵר וְשָׁלֵו יוֹתֵר. כִּי לְכָךְ שֻׁלַּחְתִּי, לְבַשֵּׂר לְךָ אֶת-כָּל-אֵלֶּה'".
ז "וַנַּעַל אֲנִי וָהוּא אֶל-הָרָקִיעַ הַתַּחְתּוֹן (הַ'טְּבֶרְד'). וָאֵרֶא שָׁם מִלְחָמָה גְדוֹלָה – הַשָּׂטָן וְכֹחוֹתָיו, מִתְנַגְּדִים לַחֲסִידוּת וּמְקַנְּאִים זֶה בָּזֶה. כְּמוֹ שֶׁיֵּשׁ בָּאָרֶץ, כָּךְ יֵשׁ בָּרָקִיעַ הַתַּחְתּוֹן. כִּי צַלְמֵי הָרָקִיעַ הַתַּחְתּוֹן נִמְצָאִים כָּאן, בָּאָרֶץ".
ח "וָאֹמַר לַמַּלְאָךְ: 'מָה הַצָּבָא הַזֶּה וְהַקִּנְאָה וְהַמַּעֲרָכָה? עֲנֵה נָא לִי'. וַיֹּאמֶר: 'זוֹהִי מִלְחֶמֶת הַשָּׂטָן, וְאֵין לָהּ מְנוּחָה, עַד אֲשֶׁר יָבוֹא זֶה אֲשֶׁר תַּחְפֹּץ לִרְאוֹת, וְיַהֲרֹג בְּרוּחַ גְּבוּרָתוֹ'".
הערות וביאורים
מצב האקסטזה (פסוקים ב'-ג'): הטקסט מתאר באופן חי את המעבר של ישעיהו ממצב של נבואה רגילה ("דיבור ברוח הקודש") למצב של חזון מיסטי.
"עֵינָיו פְקוּחוֹת, אַךְ פִּיו סָגוּר": זהו תיאור קלאסי של אקסטזה, שבה החושים החיצוניים עדיין פועלים, אך התודעה מנותקת מהסביבה ומופנית פנימה או מעלה. הנביא רואה את העולם הרוחני, לא את האנשים שלפניו.
"כִּי שְׁאִיפַת הָרוּחַ הָיְתָה עִמּוֹ": הביטוי הסלאבי въдх_новенїе дх_а (שאיפת/השראת הרוח) מדגיש את הפסיביות של הנביא. הרוח היא שפועלת בו, לא הוא שפועל.
"כֻּלָּם חָשְׁבוּ כִּי נֶחְטַף": ההמון הפשוט מפרש את היעלמות תודעתו של ישעיהו כחטיפה פיזית (בדומה לחנוך או אליהו). לעומתם, "הַנְּבִיאִים הֵבִינוּ" שהמדובר בחזון רוחני, "מן הנסתר מכל בשר". הטקסט יוצר הבחנה בין הבנה פשוטה ועממית להבנה עמוקה של יודעי-סוד.
המלאך המלווה (פסוקים ד'-ו'):
"כָּבוֹד אַחֵר, גָּדוֹל וְקָדוֹשׁ": ישעיהו, המורגל בחזיונות מלאכים (ראו ישעיהו ו'), מבין מיד שהוא פוגש ישות מדרגה גבוהה במיוחד.
"אֶת-שְׁמִי לֹא תֵדַע, כִּי חָפֵץ אַתָּה לָשׁוּב לְגוּפְךָ": זוהי אמירה אניגמטית. ידיעת שמו האמיתי של המלאך הייתה מעניקה לישעיהו כוח עליו, או קושרת אותו באופן בלתי הפיך לעולם העליון, ומונעת ממנו לשוב לעולם הגשמי. זהו מוטיב נפוץ של טאבו על שמות אלוהיים או מלאכיים.
"עוֹד תִּרְאֶה גָּדוֹל מִמֶּנִּי... מֵאִיר יוֹתֵר וְשָׁלֵו יוֹתֵר": המלאך המלווה מכין את ישעיהו למפגש עם ישויות גבוהות ממנו בהיררכיה השמימית, ובסופו של דבר עם האל עצמו. יש כאן רמז ברור למבנה ההיררכי של הרקיעים שישעיהו עומד לעבור דרכם.
הרקיע התחתון ('טְּבֶרְד') והדואליזם הקוסמי (פסוקים ז'-ח'):
"הָרָקִיעַ הַתַּחְתּוֹן": המילה הסלאבית твердь (טְבֶרְד') היא התרגום המקובל ל"רקיע" מספר בראשית. כאן, הרקיע הזה אינו מקום מושבו של האל, אלא זירת הקרב של השטן. זהו אזור הגבול בין העולם הארצי לעולמות השמימיים הטהורים.
"כְּמוֹ שֶׁיֵּשׁ בָּאָרֶץ, כָּךְ יֵשׁ בָּרָקִיעַ הַתַּחְתּוֹן": זהו עיקרון יסוד בעל גוון גנוסטי. העולם החומרי אינו אלא השתקפות (=брази - צלמים, דמויות) של מאבק רוחני המתרחש מעליו. הרוע, הקנאה והמלחמה על הארץ נובעים ממקורם בממלכתו של השטן ברקיע.
"עַד אֲשֶׁר יָבוֹא זֶה אֲשֶׁר תַּחְפֹּץ לִרְאוֹת": המלאך רומז לדמות הגואל (המשיח), שרק הוא יכול להכריע את המלחמה הקוסמית הזו ולהביא קץ לשלטון השטן. זה מציב את כל החזון בהקשר משיחי מובהק.
סיכום החלק:
החלק הראשון של החזון מבסס את המסגרת: ישעיהו נכנס למצב אקסטטי, פוגש מדריך מלאכי רב-עוצמה, ועולה לרקיע הראשון, שם הוא חוזה במלחמה הקוסמית בין כוחות החושך לכוחות האור. הוא לומד שהעולם הזה הוא רק צל של המאבק הזה, ושהגאולה תלויה בבואו של המשיח. מכאן, הוא מוכן להמשיך במסעו מעלה, אל הרקיעים הטהורים יותר.

חלק א': ההתגלות בחצר המלך
(עמוד 1ב)
א בְּחֹדֶשׁ מַאי. ¤ הֶחָזוֹן אֲשֶׁר חָזָה יְשַׁעְיָהוּ הַנָּבִיא, בֶּן-אָמוֹץ.
ב בִּשְׁנַת הָעֶשְׂרִים לְמַלְכוּת חִזְקִיָּהוּ מֶלֶךְ יְהוּדָה, בָּא יְשַׁעְיָהוּ בֶּן-אָמוֹץ וְיֵשׁוּעַ בֶּן-שָׂאָב אֶל-חִזְקִיָּהוּ בִּירוּשָׁלַיִם.
ג וּכְבוֹאוֹ, יָשַׁב עַל-מִטַּת הַמֶּלֶךְ. וְכָל-שָׂרֵי יְרוּשָׁלַיִם, יוֹעֲצֵי הַמֶּלֶךְ וְהַסָּרִיסִים, עָמְדוּ לְפָנָיו.
ד וּבָאוּ, לֹא מִן-הָעֲיָרוֹת וְלֹא מִן-הַכְּפָרִים, [אֶלָּא] מִן-הָהָר, נְבִיאִים וּבְנֵי-נְבִיאִים, כְּשָׁמְעָם כִּי בָּא יְשַׁעְיָהוּ מִן-הַגִּלְגָּל אֶל-חִזְקִיָּהוּ, וַיְנַשְּׁקוּהוּ יַחְדָּו, לְבַשֵּׂר לוֹ עַל-אֲשֶׁר עָתִיד לִהְיוֹת.
ה וּכְשֶׁהֵחֵל לְדַבֵּר דִּבְרֵי אֱמֶת, יָרְדָה עָלָיו רוּחַ הַקֹּדֶשׁ, וְכֻלָּם רָאוּ, וְשָׁמְעוּ אֶת-דִּבְרֵי הַקֹּדֶשׁ.
ו כִּי קָרָא הַמֶּלֶךְ לְכָל-הַנְּבִיאִים, וַיָּבֹאוּ כֻּלָּם יַחְדָּו, כָּל-אֲשֶׁר נִמְצְאוּ שָׁם. וְהָיָה שָׁם מִיכָה וַחֲנַנְיָה הַזָּקֵן, יוֹאֵל, וְכָל-אֲשֶׁר יָשְׁבוּ מִימִינוֹ וּמִשְּׂמֹאלוֹ.
הערות וביאורים (עמוד 1ב)
קטע הפתיחה של "חזון ישעיהו" מבסס את הרקע ההיסטורי והתיאולוגי לחזון המיסטי שיבוא בהמשך. הוא נועד להעניק לחיבור סמכות ואמינות.
הכותרת והתאריך (פסוק א'-ב'):
"בְּחֹדֶשׁ מַאי": ציון תאריך ספציפי, גם אם הוא אנכרוניסטי (חודש מאי הוא שם רומי), נועד ליצור רושם של כרוניקה היסטורית מדויקת.
"בִּשְׁנַת הָעֶשְׂרִים לְמַלְכוּת חִזְקִיָּהוּ": מיקום האירוע בתקופת שלטונו של חזקיהו, מלך צדיק שנלחם בעבודה זרה (מלכים ב', י"ח), אינו מקרי. הוא מציב את ישעיהו ואת חזונו בהקשר של טוהר דתי ומאבק רוחני, בניגוד לתקופות של מלכים חוטאים כמו מנשה (שעל פי המסורת בחיבור זה, הרג את ישעיהו).
מעמד הנביא (פסוק ג'):
"יָשַׁב עַל-מִטַּת הַמֶּלֶךְ": זוהי הצהרת מעמד דרמטית. הנביא אינו עומד בפני המלך כנתין, אלא יושב על סמל השלטון עצמו. הדבר מדגיש את עליונותה של הסמכות הנבואית-אלוהית על פני הסמכות המלכותית-ארצית. המלך והשרים עומדים בפניו, לא להפך. זהו ביטוי לרעיון שהנביא הוא שגרירו הישיר של האל, ולכן מעמדו גבוה משל מלך בשר ודם.
קהל הנביאים (פסוק ד', ו'):
"נְבִיאִים וּבְנֵי-נְבִיאִים": החיבור מדגיש שהאירוע מתרחש בפורום רחב של אנשי רוח, מה שמעניק לחזון אישור קולקטיבי. הם באים "מן ההר", מקום ההתבודדות וההתגלות, ולא מן העיר הרוחשת.
"מִן-הַגִּלְגָּל": הגלגל הוא מקום בעל משמעות היסטורית ודתית עמוקה (יהושע ד'-ה'), הקשור לכניסה לארץ, ברית מילה, והתגלות. הגעתו של ישעיהו משם מוסיפה הילה של קדושה למעמד.
נוכחות נביאים נוספים: אזכורם של נביאים מוכרים כמו מיכה ויוֹאֵל לצד חנניה (אולי חנניה בן עזור, יריבו של ירמיהו, המוצג כאן באור חיובי יותר, או דמות אחרת) נועד לחזק את האמינות ההיסטורית של המעמד.
ההתגלות (פסוק ה'):
"יָרְדָה עָלָיו רוּחַ הַקֹּדֶשׁ": זהו תיאור קלאסי של התחלת הנבואה. הרוח "יורדת" על הנביא, והופכת אותו לכלי להעברת המסר האלוהי.
"וְכֻלָּם רָאוּ, וְשָׁמְעוּ": החוויה אינה פרטית של ישעיהו בלבד. כל הנוכחים הם עדים להתגלות, מה שמעניק לה תוקף ציבורי ומונע כל טענה שמדובר בהזיה פרטית.
סיכום החלק:
הפתיחה בונה בקפידה את הבמה לחזון הגדול. היא מציבה אותו בזמן ובמקום קדושים (תקופת חזקיהו, ירושלים), במעמד רב-רושם (הנביא יושב על כס המלך), ובנוכחות עדים רבים וחשובים (המלך, השרים, ונביאים אחרים). כל הפרטים הללו נועדו לבסס את סמכותו של ישעיהו ואת אמינותו של החזון שהוא עומד לחוות, ולהציג אותו כאירוע מרכזי בתולדות הנבואה.


חלק ב': האקסטזה והמפגש עם המלאך המלווה
(עמודים 2, 2ב)
ז וּכְשָׁמְעָם אֶת-קוֹל רוּחַ הַקֹּדֶשׁ, כָּרְעוּ עַל-בִּרְכֵּיהֶם וַיְשַׁבְּחוּ אֶת-אֵל עֶלְיוֹן, הַשּׁוֹכֵן בַּקְּדוֹשִׁים וַאֲשֶׁר נָתַן כֹּחַ כָּזֶה בִּדְבָרוֹ בָּעוֹלָם.
ח וּבְעוֹד שֶׁדִּבֵּר בְּרוּחַ הַקֹּדֶשׁ לְאָזְנֵי כֹּל, פִּתְאֹם שָׁתַק וְחָדַל לִרְאוֹת אֶת-הָעוֹמְדִים לְפָנָיו. עֵינָיו הָיוּ פְקוּחוֹת, אַךְ פִּיו הָיָה סָגוּר, כִּי שְׁאִיפַת הָרוּחַ הָיְתָה עִמּוֹ.
ט וְכֻלָּם חָשְׁבוּ כִּי נֶחְטַף יְשַׁעְיָהוּ. אַךְ הַנְּבִיאִים הֵבִינוּ כִּי בְּהִתְגַּלּוּת הָיָה, וְהֶחָזוֹן אֲשֶׁר חָזָה לֹא הָיָה מִן-הָעוֹלָם הַזֶּה, אֶלָּא מִן-הַנִּסְתָּר מִכָּל-בָּשָׂר.
י וּכְשֶׁעָמַד מֵחֶזְיוֹנוֹ וְשָׁב אֶל-עַצְמוֹ, סִפֵּר לְחִזְקִיָּהוּ וְלִבְנוֹ נָשׂוֹן, וּלְמִיכָה וְלִשְׁאָר הַנְּבִיאִים, לֵאמֹר: "כַּאֲשֶׁר הִתְנַבֵּאתִי לְאָזְנֵיכֶם, אֲשֶׁר לֹא שְׁמַעְתֶּם, רָאִיתִי מַלְאָךְ נִכְבָּד, לֹא כְּכָבוֹד הַמַּלְאָכִים אֲשֶׁר רָאִיתִי תָּמִיד, אֶלָּא כָּבוֹד אַחֵר, גָּדוֹל וְקָדוֹשׁ, אֲשֶׁר לֹא אוּכַל לְסַפֵּר".
יא "וַיֹּאחֶז בְּיָדִי, וַיַּעֲלֵנִי לַמָּרוֹם. וַאֲנִי אָמַרְתִּי: 'מִי אַתָּה? מַה-שִּׁמְךָ? וּלְאָן אַתָּה מַעֲלֶה אוֹתִי?'. כִּי נִתַּן לִי הַכֹּחַ לְדַבֵּר עִמּוֹ. אָז עָנָה וְאָמַר: 'כַּאֲשֶׁר אַעֲלֶךָ לַמַּעֲלָה וְאַרְאֲךָ אֶת-הֶחָזוֹן שֶׁלְּמַעֲנוֹ שֻׁלַּחְתִּי, אָז תָּבִין מִי אָנֹכִי. אַךְ אֶת-שְׁמִי לֹא תֵדַע, כִּי חָפֵץ אַתָּה לָשׁוּב לְגוּפְךָ. וּכְשֶׁאַעֲלֶךָ, תִּרְאֶה'.
יב "וְשָׂמַחְתִּי כִּי בְּעַנְוָה עָנָה לִי. וַיֹּאמֶר לִי: 'הַאִם שָׂמַחְתָּ כִּי בְּעַנְוָה עָנִיתִי לְךָ? עוֹד תִּרְאֶה גָּדוֹל מִמֶּנִּי, עָנָו יוֹתֵר וּמַשְׂכִּיל יוֹתֵר, הֶחָפֵץ לְשׂוֹחֵחַ עִמְּךָ, וְגָדוֹל מִן-הַגָּדוֹל, מֵאִיר יוֹתֵר וְשָׁלֵו יוֹתֵר. כִּי לְכָךְ שֻׁלַּחְתִּי, לְבַשֵּׂר לְךָ אֶת-כָּל-אֵלֶּה'".
הערות וביאורים (עמודים 2, 2ב)
חלק זה מתאר את "נקודת המפנה" של החיבור: המעבר מדיבור נבואי פומבי לחוויה מיסטית אישית של עלייה לשמיים.
תגובת הקהל (פסוק ז'): הנוכחים, בהבינם כי רוח הקודש שורה במקום, כורעים ומשבחים את האל. תגובה זו מדגישה את קדושת המעמד ואת ההכרה בכך שהם עדים להתגלות אלוהית.
תיאור האקסטזה (פסוק ח'):
"פִּתְאֹם שָׁתַק... עֵינָיו פְקוּחוֹת, אַךְ פִּיו סָגוּר": זהו תיאור פנומנולוגי מדויק של כניסה למצב טראנס או אקסטזה. החושים החיצוניים (ראייה) עדיין פועלים, אך התודעה מנותקת מהסביבה המיידית ("חדל לראות את העומדים לפניו") והיכולת לדבר נלקחת ממנו. הנביא הופך מכלי שמדבר את דבר ה' לכלי שרואה את עולמו של ה'.
"שְׁאִיפַת הָרוּחַ": הביטוי הסלאבי въдх_новенїе дх_а (השראת/שאיפת הרוח) מדגיש את הפסיביות של הנביא. הרוח היא שפועלת בו, לא הוא שפועל.
פרשנות האקסטזה (פסוק ט'):
"כֻּלָּם חָשְׁבוּ כִּי נֶחְטַף": ההמון הפשוט מפרש את היעלמות תודעתו של ישעיהו כחטיפה פיזית של הגוף, בדומה למסורות על חנוך (בראשית ה', כ"ד) ואליהו (מלכים ב', ב').
"אַךְ הַנְּבִיאִים הֵבִינוּ": לעומתם, הנביאים, "יודעי הח"ן", מבינים את ההבחנה הדקה בין עלייה גופנית לעלייה רוחנית ("בהתגלות היה"). הם מבינים שהחזון הוא מעולם אחר, "הנסתר מכל בשר". הטקסט יוצר כאן היררכיה של ידע: ישנה הבנה עממית-פשטנית והבנה עמוקה של יודעי-סוד.
המפגש עם המלאך המלווה (פסוקים י'-י"ב):
"כָּבוֹד אַחֵר, גָּדוֹל וְקָדוֹשׁ": ישעיהו, המורגל בחזיונות מלאכים (כמו במעשה המרכבה בישעיהו ו'), מבין מיד שהוא פוגש ישות מדרגה גבוהה במיוחד, שונה מהמלאכים ה"רגילים".
"אֶת-שְׁמִי לֹא תֵדַע, כִּי חָפֵץ אַתָּה לָשׁוּב לְגוּפְךָ": זוהי אמירה אניגמטית ורבת משמעות. בעולם המיסטי הקדום, ידיעת שמו האמיתי של יצור על-טבעי מעניקה כוח ושליטה עליו. יתרה מכך, המלאך רומז שידיעת שמו המלא הייתה קושרת את ישעיהו באופן בלתי הפיך לעולם העליון, ומונעת ממנו לשוב לעולם הגשמי ולמסור את נבואתו. זהו טאבו עתיק על שמות אלוהיים ומלאכיים (השוו: יעקב והמלאך בבראשית ל"ב, ל'; מנוח והמלאך בשופטים י"ג, י"ח).
"עוֹד תִּרְאֶה גָּדוֹל מִמֶּנִּי... מֵאִיר יוֹתֵר וְשָׁלֵו יוֹתֵר": המלאך המלווה (בלטינית: Angelus Interpres, "המלאך המפרש") מכין את ישעיהו למסע היררכי. הוא מבהיר שהוא רק המדריך, ושישעיהו עתיד לפגוש ישויות נעלות ממנו בהרבה, ובסופו של דבר את האל עצמו. יש כאן רמז ברור למבנה המדורג של הרקיעים שישעיהו עומד לעבור דרכם.
סיכום החלק:
חלק זה מהווה את שער הכניסה לעולם המיסטי של החיבור. הוא מתאר את המעבר הפיזי והתודעתי של ישעיהו, מבסס את ההבחנה בין הבנה פשוטה להבנה אזוטרית, ומציג את המדריך המלאכי שיוליך את הנביא במסעו המסוכן והמפואר דרך הרקיעים.

חלק ג': הרקיע התחתון והשמיים הראשונים והשניים
(עמודים 3, 3ב)
יג "וַנַּעַל אֲנִי וָהוּא אֶל-הָרָקִיעַ הַתַּחְתּוֹן (הַ'טְּבֶרְד'). וָאֵרֶא שָׁם מִלְחָמָה גְדוֹלָה – הַשָּׂטָן וְכֹחוֹתָיו, מִתְנַגְּדִים לַחֲסִידוּת וּמְקַנְּאִים זֶה בָּזֶה. כְּמוֹ שֶׁיֵּשׁ בָּאָרֶץ, כָּךְ יֵשׁ בָּרָקִיעַ הַתַּחְתּוֹן. כִּי צַלְמֵי הָרָקִיעַ הַתַּחְתּוֹן נִמְצָאִים כָּאן, בָּאָרֶץ".
יד "וָאֹמַר לַמַּלְאָךְ: 'מָה הַצָּבָא הַזֶּה וְהַקִּנְאָה וְהַמַּעֲרָכָה? עֲנֵה נָא לִי'. וַיֹּאמֶר: 'זוֹהִי מִלְחֶמֶת הַשָּׂטָן, וְאֵין לָהּ מְנוּחָה, עַד אֲשֶׁר יָבוֹא זֶה אֲשֶׁר תַּחְפֹּץ לִרְאוֹת, וְיַהֲרֹג בְּרוּחַ גְּבוּרָתוֹ'".
טו "אַחַר-כָּךְ הֶעֱלַנִי אֶל-הַמָּקוֹם שֶׁמֵּעַל הָרָקִיעַ הַתַּחְתּוֹן, הוּא הַשָּׁמַיִם הָרִאשׁוֹנִים. וָאֵרֶא שָׁם בָּאֶמְצַע כִּסֵּא, וְעָלָיו יוֹשֵׁב מַלְאָךְ בְּכָבוֹד רָב. וּמִימִינוֹ וּמִשְּׂמֹאלוֹ יוֹשְׁבִים מַלְאָכִים, וְכָבוֹד אַחֵר לָאֵלֶּה שֶׁמִּימִין. וְהֵם שָׁרִים בְּקוֹל אֶחָד, וְאֵלֶּה שֶׁמִּשְּׂמֹאל שָׁרִים אַחֲרֵיהֶם, אַךְ שִׁירָתָם אֵינָהּ כְּשִׁירַת הַיְמָנִים".
טז "וָאֶשְׁאַל אֶת-הַמַּלְאָךְ הַמּוֹלִיךְ אוֹתִי: 'לְמִי נִשְׁלַחַת הַשִּׁירָה הַזֹּאת?'. וַיַּעַן הַמַּלְאָךְ: 'לַכָּבוֹד הַגָּדוֹל שֶׁל אֱלֹהִים אֲשֶׁר בַּשָּׁמַיִם הַשְּׁבִיעִיִים, וְלִבְנוֹ אֲהוּבוֹ. מִשָּׁם שֻׁלַּחְתִּי אֵלֶיךָ'".
יז "וְשׁוּב הֶעֱלַנִי אֶל-הַשָּׁמַיִם הַשֵּׁנִיִּים. וְגֹבַהּ הַשָּׁמַיִם הָיָה כְּגֹבַהּ הַשָּׁמַיִם הָרִאשׁוֹנִים מִן-הָאָרֶץ. וָאֵרֶא שָׁם כְּמוֹ בַּשָּׁמַיִם הָרִאשׁוֹנִים, מַלְאָכִים מִימִין וּמִשְּׂמֹאל, וְשִׁירָתָם גְּדוֹלָה מִשֶּׁל הָרִאשׁוֹנִים. וְנָפַלְתִּי עַל-פָּנַי וְהִשְׁתַּחֲוֵיתִי לָהֶם".
יח "וְלֹא עֲזָבַנִי הַמַּלְאָךְ הַמַּדְרִיךְ אוֹתִי, [וְאָמַר]: 'אַל תִּשְׁתַּחֲוֶה לֹא לַכִּסֵּא וְלֹא לְמַלְאָךְ מִן-הַשָּׁמַיִם. כִּי לְשֵׁם כָּךְ שֻׁלַּחְתִּי, לְהַדְרִיכְךָ. רַק לַאֲשֶׁר אֹמַר לְךָ, אֲשֶׁר הוּא מֵעַל כָּל-הַכִּסְאוֹת וְהַבְּגָדִים וְהַכְּתָרִים, אֲשֶׁר אוֹתוֹ תִּרְאֶה מִכָּאן וָהָלְאָה'".
יט "וְשָׂמַחְתִּי שִׂמְחָה גְדוֹלָה מְאֹד, כִּי כָּזֶה הוּא סוֹפָם שֶׁל הַיּוֹדְעִים אֶת-הָעֶלְיוֹן וְהַנִּצְחִי וְאֶת-בְּנוֹ הָאָהוּב, וְכִי עַל-יְדֵי מַלְאַךְ רוּחַ הַקֹּדֶשׁ כָּךְ הֵם יוֹצְאִים".
הערות וביאורים (עמודים 3, 3ב)
הרקיע התחתון ('טְּבֶרְד') והדואליזם הקוסמי (פסוקים י"ג-י"ד):
"הָרָקִיעַ הַתַּחְתּוֹן": המילה הסלאבית твердь (טְבֶרְד') היא התרגום המקובל ל"רקיע" מספר בראשית. בחיבור זה, הרקיע הזה אינו מקום מושבו של האל, אלא זירת הקרב של השטן. זהו אזור הגבול בין העולם הארצי לעולמות השמימיים הטהורים, והוא מלא בקנאה, מאבק ורוע.
"כְּמוֹ שֶׁיֵּשׁ בָּאָרֶץ, כָּךְ יֵשׁ בָּרָקִיעַ הַתַּחְתּוֹן": זהו עיקרון יסוד בעל גוון גנוסטי. העולם החומרי אינו אלא השתקפות (=брази - צלמים, דמויות) של מאבק רוחני המתרחש מעליו. הרוע, הקנאה והמלחמה על הארץ נובעים ממקורם בממלכתו של השטן ברקיע. תפיסה זו מסבירה את מקור הרוע בעולם.
"עַד אֲשֶׁר יָבוֹא זֶה אֲשֶׁר תַּחְפֹּץ לִרְאוֹת": המלאך רומז לדמות הגואל (המשיח), שרק הוא יכול להכריע את המלחמה הקוסמית הזו ולהביא קץ לשלטון השטן. זה מציב את כל החזון בהקשר משיחי מובהק.
ההיררכיה השמימית (פסוק ט"ו):
"הַשָּׁמַיִם הָרִאשׁוֹנִים": לאחר שעבר את אזור הקרב של השטן, ישעיהו נכנס לממלכת השמיים ה"אמיתית".
"מַלְאָכִים מִימִין וּמִשְּׂמֹאל": כאן מתחילה להתגלות ההיררכיה השמימית. בכל רקיע ישנו "מלך" (מלאך היושב על כיסא) וסביבו מלאכים בדרגות שונות. המלאכים שמימין נעלים יותר ("כבוד אחר להם"), ושירתם איכותית יותר. זהו ביטוי לתפיסה היררכית נוקשה של המציאות, המושפעת מהפילוסופיה הניאופלטונית, שבה כל שלב במציאות הוא השתקפות פחותה של השלב שמעליו.
יעד התפילה והשליחות (פסוק ט"ז):
"לַכָּבוֹד הַגָּדוֹל... בַּשָּׁמַיִם הַשְּׁבִיעִיִים, וְלִבְנוֹ אֲהוּבוֹ": המלאך מבהיר שהשירה אינה מיועדת למלאכים המקומיים, אלא מכוונת כלפי מעלה, אל הישות העליונה ברקיע השביעי – האל האב ובנו. זוהי תוספת נוצרית מובהקת, המשלבת את התיאולוגיה של השילוש הקדוש בתוך המבנה הקוסמולוגי היהודי של שבעת הרקיעים.
"מִשָּׁם שֻׁלַּחְתִּי אֵלֶיךָ": המלאך המלווה חושף את מקור סמכותו – הוא שליח ישיר מהרקיע הגבוה ביותר.
איסור ההשתחוויה (פסוקים י"ז-י"ח):
"וְנָפַלְתִּי עַל-פָּנַי וְהִשְׁתַּחֲוֵיתִי לָהֶם": ישעיהו, המום מהכבוד וההדר של הרקיע השני, מגיב באופן אינסטינקטיבי ומשתחווה למלאכים.
"אַל תִּשְׁתַּחֲוֶה לֹא לַכִּסֵּא וְלֹא לְמַלְאָךְ": המלאך המלווה נוזף בו ומלמד אותו עיקרון תיאולוגי חשוב: עבודת האל וההשתחוויה מיועדות אך ורק לאל העליון, ולא לשום יצור נברא, נכבד ככל שיהיה. זהו מוטיב שחוזר באפוקליפטיקה (ראו חזון יוחנן י"ט, י'; כ"ב, ח'-ט'), ומטרתו להילחם בנטייה להעריץ מלאכים.
שמחת היודעים (פסוק י"ט):
"כִּי כָּזֶה הוּא סוֹפָם שֶׁל הַיּוֹדְעִים": ישעיהו שמח לא רק על החזון, אלא על ההבנה שהגורל הנעלה הזה (עלייה לרקיעים) שמור ל"יודעים" – מונח בעל גוון גנוסטי, המתייחס לאלה שזכו לידיעה רוחנית מיוחדת על האל ובנו.
סיכום החלק:
חלק זה מעמיק את ההבנה הקוסמולוגית והתיאולוגית של החיבור. ישעיהו לומד להבחין בין ממלכת השטן שברקיע התחתון לבין השמיים הטהורים, הוא נחשף למבנה ההיררכי של עולם המלאכים, ומקבל שיעור חשוב על עבודת האל הראויה, המכוונת אך ורק כלפי מעלה, אל האב והבן.


חלק ד': הרקיעים השלישי, הרביעי והחמישי
(עמודים 3ב, 4, 4ב)
כ "וְהֶעֱלַנִי אֶל-הַשָּׁמַיִם הַשְּׁלִישִׁיִּים. וְשָׁם כְּמוֹ-כֵן רָאִיתִי כִּסֵּא קָטָן, וּמַלְאָכִים מִימִין וּמִשְּׂמֹאל. אַךְ זֵכֶר הָעוֹלָם הַזֶּה אֵינוֹ נִזְכָּר שָׁם. וְנִקְרָא שָׁם כְּבוֹד רוּחִי. כַּאֲשֶׁר עָלִיתִי לַשָּׁמַיִם, אָמַרְתִּי: 'דָּבָר מֵאוֹתוֹ עוֹלָם אֵינוֹ נִקְרָא כָּאן בְּשֵׁם'. וַיַּעַן לִי הַמַּלְאָךְ וַיֹּאמֶר: 'דָּבָר אֵינוֹ נֶחְשָׁב בִּגְלַל חֻלְשָׁתוֹ, וְדָבָר מִן-הַנַּעֲשֶׂה שָׁם אֵינוֹ נִסְתָּר כָּאן'".
כא "וְהַשִּׁירָה אֲשֶׁר שָׁרוּ, וְהַשֶּׁבַח לַיּוֹשֵׁב [עַל-הַכִּסֵּא], וְהַמַּלְאָךְ, גְּדוֹלִים הָיוּ מִן-הַשֵּׁנִי".
כב "וְשׁוּב הֶעֱלַנִי אֶל-הַשָּׁמַיִם הָרְבִיעִיִּים, גְּדוֹלִים הָיוּ בְּמִדָּה רַבָּה. וְשָׁם שׁוּב רָאִיתִי כִּסֵּא, וּמַלְאָכִים מִימִין וּמִשְּׂמֹאל. וְגַם אֵלֶּה שָׁרוּ. וְהַכָּבוֹד וְהַשִּׁירָה שֶׁל הַמַּלְאָכִים גְּדוֹלִים הָיוּ מִשֶּׁל אֵלֶּה שֶׁמִּשְּׂמֹאל. וְכָבוֹד הַיּוֹשֵׁב [עַל-הַכִּסֵּא] גָּדוֹל הָיָה מִכְּבוֹד הַמַּלְאָכִים שֶׁמִּימִינוֹ. וְכָבוֹד כֻּלָּם גָּדוֹל וְעוֹלֶה עַל-זֶה שֶׁל הַתַּחְתּוֹנִים".
כג "וַנַּעַל אֶל-הַשָּׁמַיִם הַחֲמִישִׁיִּים. וְשָׁם שׁוּב רָאִיתִי לִגְיוֹנוֹת אֵין-מִסְפָּר, וּכְבוֹדָם וְשִׁירָתָם נִכְבָּדִים יוֹתֵר מִשֶּׁל הַשָּׁמַיִם הָרְבִיעִיִּים. וַאֲנִי הִתְדַּהַמְתִּי לִרְאוֹת הָמוֹן רַב כָּל-כָּךְ שֶׁל מַלְאָכִים, מְקֻשָּׁטִים בְּטוּב מִדּוֹת שׁוֹנוֹת, וּלְכָל-אֶחָד כָּבוֹד מִשֶּׁלּוֹ. וְהֵם שִׁבְּחוּ אֶת-הַשּׁוֹכֵן בַּמְּרוֹמִים, אֲשֶׁר שְׁמוֹ לֹא נִגְלָה לְכָל-בָּשָׂר, כִּי כָּבוֹד כָּזֶה נוֹתֵן לַמַּלְאָכִים בְּכָל-שָׁמַיִם וְשָׁמַיִм".
כד "וַיַּעַן הַמַּלְאָךְ וַיֹּאמֶר לִי: 'מַדּוּעַ הִתְפַּלֵּאתָ עַל-כָּךְ? הַאִם רָאִיתָ רַק אֶת-הַפָּנִים הַלָּלוּ? הֲלֹא עוֹד לֹא רָאִיתָ אֶת-כֹּחוֹת הַחֹשֶׁךְ הַבִּלְתִּי-נִתָּנִים לְמַעֲבָר, אֶת-צִבְאוֹת הַמַּלְאָכִים?'".
הערות וביאורים (עמודים 3ב, 4, 4ב)
הרקיע השלישי – הניתוק מהעולם (פסוק כ'):
"זֵכֶר הָעוֹלָם הַזֶּה אֵינוֹ נִזְכָּר שָׁם": זהו שלב חשוב במסע המיסטי. הרקיע השלישי הוא נקודת המפנה שבה הקשר לעולם הגשמי מתחיל להתנתק. המושגים, השמות והדאגות של העולם התחתון מאבדים את משמעותם. זה מזכיר את תיאורו של שאול השליח על עלייתו ל"גן עדן" (שמזוהה לעיתים עם הרקיע השלישי), שם שמע "דברים נסתרים אשר נבצר מאיש לדברם" (השנייה לקורינתים י"ב, ד').
"דָּבָר מִן-הַנַּעֲשֶׂה שָׁם אֵינוֹ נִסְתָּר כָּאן": למרות הניתוק, אין זה אומר שהעולם העליון אינו מודע למתרחש למטה. להפך, הכל גלוי וידוע. הניתוק הוא חד-כיווני: העולם התחתון אינו רלוונטי לעליון, אך העליון צופה ויודע הכל על התחתון.
העלייה ההדרגתית בכבוד (פסוקים כ"א-כ"ג):
"גְּדוֹלִים הָיוּ מִן-הַשֵּׁנִי... נִכְבָּדִים יוֹתֵר מִשֶּׁל הָרְבִיעִיִּים": המחבר מדגיש שוב ושוב את העיקרון ההיררכי. כל רקיע עולה על קודמו בכמה פרמטרים:
גודל פיזי: "גְּדוֹלִים הָיוּ בְּמִדָּה רַבָּה".
כמות המלאכים: "לִגְיוֹנוֹת אֵין-מִסְפָּר".
עוצמת הכבוד (ה"סְלַאבָה"): הכבוד של המלאכים ושל היושב על הכיסא גובר.
איכות השירה: השירה הופכת "נכבדת יותר".
השפעה ניאופלטונית: מבנה זה משקף בבירור את תפיסת ה"אצילה" (אמנציה) הניאופלטונית, שבה כל שלב במציאות נובע מהשלב שמעליו והוא השתקפות פחותה שלו. ככל שמתקרבים למקור האחד (האל), כך גוברים האור, הכבוד והטוהר.
הרקיע החמישי – ריבוי וגיוון (פסוק כ"ג):
"מְקֻשָּׁטִים בְּטוּב מִדּוֹת שׁוֹנוֹת, וּלְכָל-אֶחָד כָּבוֹד מִשֶּׁלּוֹ": בניגוד לרקיעים הנמוכים יותר, שבהם יש חלוקה ברורה לימין ושמאל, ברקיע החמישי יש ריבוי עצום של סוגי מלאכים, שלכל אחד "טוב מידות" (סגולות) ו"כבוד" ייחודיים לו. זה מרמז על התקרבות למקור האלוהי, שהוא מקור כל הגיוון והשפע.
"הַשּׁוֹכֵן בַּמְּרוֹמִים, אֲשֶׁר שְׁמוֹ לֹא נִגְלָה לְכָל-בָּשָׂר": השירה של כל המלאכים הללו מכוונת לאל העליון, שמהותו נסתרת ובלתי ניתנת להשגה על ידי יצורים גשמיים.
נזיפת המלאך (פסוק כ"ד):
"מַדּוּעַ הִתְפַּלֵּאתָ עַל-כָּךְ?": ישעיהו, המום מהמראות, זוכה לנזיפה קלה מהמלאך המלווה.
"הֲלֹא עוֹד לֹא רָאִיתָ אֶת-כֹּחוֹת הַחֹשֶׁךְ הַבִּלְתִּי-נִתָּנִים לְמַעֲבָר": זוהי אמירה קשה ומפתיעה. המלאך רומז לישעיהו שהתפעלותו מהצד החיובי של ההיררכיה היא חד-צדדית. הוא מזכיר לו שכשם שיש היררכיה של אור, ישנה גם היררכיה נגדית של חושך, "כוחות בלתי ניתנים למעבר" של מלאכי חבלה. זהו תזכורת לעיקרון הדואליסטי שהוצג ברקיע התחתון, ומכין את הקורא להבין שהדרמה הקוסמית מורכבת הרבה יותר ממה שנראה לעין.
סיכום החלק:
המסע דרך הרקיעים 3-5 מדגים את העיקרון ההיררכי של הבריאה ואת הניתוק ההדרגתי מהעולם הגשמי. כל שלב מקרב את ישעיהו יותר אל המקור האלוהי, אך גם חושף בפניו את המורכבות של העולם הרוחני, הכולל לא רק היררכיות של קדושה אלא גם היררכיות נגדיות של רוע.


חלק ה': הרקיע השישי – אחדות ושירה
(עמודים 4ב, 5, 5ב)
כה "וְשׁוּב הֶעֱלַנִי אֶל-מְרוֹם הַשָּׁמַיִם הַשִּׁשִּׁיִּים. וָאֵרֶא כָּבוֹד גָּדוֹל בָּהֶם, אֲשֶׁר לֹא רָאִיתִי בַּשָּׁמַיִם הַחֲמִישִׁיִּים. כִּי רָאִיתִי מַלְאָכִים בְּכָבוֹד רַב, סִדְרֵי כֹּחוֹת נִצָּבִים וְעוֹבְרִים. וְשִׁירָתָם קְדוֹשָׁה וְנִפְלָאָה הָיְתָה".
כו "וָאֹמַר לַמַּלְאָךְ הַמּוֹלִיךְ אוֹתִי: 'מָה אֲנִי רוֹאֶה, אֲדֹנִי?'. וַיֹּאמֶר לִי: 'אֵינֶנִּי אֲדוֹנֶךָ, אֶלָּא יוֹעֲצֶךָ'. וְשׁוּב אָמַר עַל-אוֹדוֹת הַשָּׁמַיִם הַשִּׁשִּׁיִּים: 'אֵין כָּאן לֹא כִּסֵּא וְלֹא מַלְאָכִים מִשְּׂמֹאל, אֶלָּא הֵם מֻנְהָגִים מִכֹּחַ הַשָּׁמַיִם הַשְּׁבִיעִיִּים, שָׁם נִמְצָא הַנִּקְרָא בְּשֵׁם וּבְנוֹ יְחִידוֹ. וְכָל-הַשָּׁמַיִם וּמַלְאָכָיו שׁוֹמְעִים לוֹ. וַאֲנִי שֻׁלַּחְתִּי לְהַעֲלוֹתְךָ הֵנָּה, לְמַעַן תִּרְאֶה אֶת-הַכָּבוֹד הַזֶּה שֶׁל אֲדוֹן כָּל-הַשָּׁמַיִם וְהַמַּלְאָכִים וְהַכֹּחוֹת'".
כז "'אוֹמֵר אֲנִי לְךָ, יְשַׁעְיָהוּ, אֵין אָדָם בְּגוּף בָּשָׂר הֶחָפֵץ לָשׁוּב לָעוֹלָם, אֲשֶׁר רָאָה אֶת-הֶחָזוֹן שֶׁאַתָּה רוֹאֶה, וְלֹא יוּכַל לִרְאוֹת אֶת-אֲשֶׁר אַתָּה חָפֵץ לִרְאוֹת. כִּי לְךָ נָפַל בְּגוֹרַל ה' לָבוֹא הֵנָּה'. וְהִתְרוֹמַמְתִּי בַּאֲדֹנִי, כִּי בְּגוֹרָלוֹ אֲנִי הוֹלֵךְ".
כח "וַיֹּאמֶר לִי: 'כַּאֲשֶׁר תַּשְׁלִים וְתָשׁוּב הֵנָּה בִּרְצוֹן אָבִיךָ, אָז תְּקַבֵּל אֶת-בִּגְדְּךָ, וְאָז תִּהְיֶה שָׁוֶה לַמַּלְאָכִים הַשּׁוֹכְנִים בַּשָּׁמַיִם הַשְּׁבִיעִיִּים'".
כט "וְהֶעֱלַנִי אֶל-הַשָּׁמַיִם הַשִּׁשִּׁיִּים. וְכִסֵּא כְּבָר לֹא הָיָה שָׁם, לֹא מַלְאָכִים מִימִין וְלֹא מִשְּׂמֹאל, אֶלָּא לְכֻלָּם הָיָה מַרְאֶה אֶחָד וְשִׁירָתָם שָׁוָה. וְנִתַּן לִי לָשִׁיר עִמָּהֶם. וְהַמַּלְאָךְ אֲשֶׁר עִמִּי, הָיִינוּ כְּמוֹהֶם, כִּי כְבוֹדָם הָיָה אֶחָד. וְהֵם שִׁבְּחוּ אֶת-הָאָב שֶׁל הַכֹּל, וְאֶת-הַבֵּן הָאָהוּב, וְאֶת-רוּחַ הַקֹּדֶשׁ, כֻּלָּם בְּפֶה אֶחָד שָׁרוּ. אַךְ לֹא כְּמוֹ שֶׁהָיָה לָהֶם בַּשָּׁמַיִם הַחֲמִישִׁיִּים, אֶלָּא בְּקוֹל אַחֵר. וְאוֹר גָּדוֹל וְרַב הָיָה שָׁם".
ל "וּכְשֶׁהָיִיתִי בַּשָּׁמַיִם הַשִּׁשִּׁיִּים, חָשַׁבְתִּי שֶׁהָאוֹר בַּשָּׁמַיִם הַחֲמִישִׁיִּים הוּא חֹשֶׁךְ. וְשָׂמַחְתִּי מְאֹד, וְשַׁרְתִּי לְנוֹתֵן דְּבָרִים כָּאֵלֶּה לַמְקַבְּלִים אֶת-רַחֲמָיו. וְהִתְפַּלַּלְתִּי לַמַּלְאָךְ הַמַּדְרִיךְ אוֹתִי, שֶׁלֹּא אָשׁוּב עוֹד לָעוֹלָם עִם בְּנֵי-הַבָּשָׂר. אוֹמֵר אֲנִי לָכֶם, כִּי חֹשֶׁךְ רַב יֵשׁ כָּאן".
לא "וְהַמַּלְאָךְ הַמַּדְרִיךְ אוֹתִי אָמַר לִי: 'מַדּוּעַ שָׂמַחְתָּ עַל-הָאוֹר הַזֶּה? כַּמָּה יוֹתֵר תִּשְׂמַח וְתִתְעַלֵּס כַּאֲשֶׁר תִּרְאֶה אֶת-אוֹר הַשָּׁמַיִם הָאֲמִתִּיִּים, שָׁם יוֹשֵׁב הָאָב הַשְּׁמֵימִי וּבְנוֹ יְחִידוֹ, וְשָׁם מַחֲנוֹת, כִּסְאוֹת וּכְתָרִים מוּנָחִים לַצַּדִּיקִים. וְעַל-זֶה שֶׁלֹּא תָּשׁוּב לְגוּפְךָ – לֹא בָּא הַזְּמַן שֶׁתַּשְׁלִים וְתָבוֹא הֵנָּה'. כְּשָׁמְעִי זֹאת, הִתְעַצַּבְתִּי מְאֹד. וַיֹּאמֶר לִי: 'אַל תִּתְעַצֵּב וְאַל תִּהְיֶה רָפֶה'".
הערות וביאורים (עמודים 4ב, 5, 5ב)
ביטול ההיררכיה (פסוק כ"ט):
"אֵין כָּאן לֹא כִּסֵּא וְלֹא מַלְאָכִים מִשְּׂמֹאל... לְכֻלָּם הָיָה מַרְאֶה אֶחָד וְשִׁירָתָם שָׁוָה": זהו השינוי הדרמטי ביותר עד כה. בעוד שהרקיעים 1-5 התאפיינו בהיררכיה נוקשה (כיסא, ימין ושמאל), הרקיע השישי הוא מרחב של אחדות. אין בו מדרגות, כולם שווים במראם ובשירתם. זה מסמל התקרבות משמעותית למקור האלוהי, שבו כל ההבדלים מתבטלים ומתאחדים.
השתתפות פעילה (פסוק כ"ט):
"וְנִתַּן לִי לָשִׁיר עִמָּהֶם": לראשונה, ישעיהו אינו רק צופה פסיבי, אלא משתתף פעיל בפולחן השמימי. הוא, יחד עם המלאך המלווה, הופכים לחלק מהקהילה השמימית ("הָיִינוּ כְּמוֹהֶם, כִּי כְבוֹדָם הָיָה אֶחָד"). זהו שלב גבוה של חניכה מיסטית, שבו הנשמה מתחילה להתאחד עם העולם העליון.
שירת השילוש (פסוק כ"ט):
"שִׁבְּחוּ אֶת-הָאָב... וְאֶת-הַבֵּן... וְאֶת-רוּחַ הַקֹּדֶשׁ, כֻּלָּם בְּפֶה אֶחָד": השירה ברקיע השישי מכוונת במפורש לשלוש ישויות השילוש הקדוש. זוהי תוספת נוצרית מובהקת, המפרשת את החזון היהודי המקורי במונחים של התיאולוגיה הנוצרית המפותחת.
חווית האור והרצון להישאר (פסוק ל'):
"חָשַׁבְתִּי שֶׁהָאוֹר בַּשָּׁמַיִם הַחֲמִישִׁיִּים הוּא חֹשֶׁךְ": חווית האור כה עזה, עד שהיא גורמת לכל מה שקדם לה להיראות כחושך. זהו דימוי קלאסי בספרות המיסטית לתיאור התקדמות רוחנית והתקרבות לאלוהות.
"שֶׁלֹּא אָשׁוּב עוֹד לָעוֹלָם עִם בְּנֵי-הַבָּשָׂר": הפער בין החוויה השמימית למציאות הארצית ("חֹשֶׁךְ רַב יֵשׁ כָּאן") כה גדול, עד שישעיהו מבקש להישאר בעולם העליון ולא לשוב לגופו. זהו רגע של משבר עבור המיסטיקן, שבו הפיתוי להישאר בעולם הרוח מתנגש עם חובתו לשוב ולפעול בעולם החומר.
הבטחת הגמול העתידי (פסוקים כ"ז-כ"ח, ל"א):
"כִּי לְךָ נָפַל בְּגוֹרַל ה' לָבוֹא הֵנָּה": המלאך מסביר לישעיהו שעלייתו אינה מקרית, אלא חלק מגורלו האלוהי.
"תְּקַבֵּל אֶת-בִּגְדְּךָ, וְאָז תִּהְיֶה שָׁוֶה לַמַּלְאָכִים": המלאך מבטיח לישעיהו שבעתיד, לאחר שישלים את חייו הארציים, הוא יקבל את "בגדו השמימי" (גוף רוחני) ויזכה למעמד שווה לזה של המלאכים ברקיע הגבוה ביותר. זוהי הבטחה לגאולה אישית, המעודדת אותו לשוב ולמלא את שליחותו.
"לֹא בָּא הַזְּמַן... אַל תִּתְעַצֵּב": המלאך מנחם את ישעיהו המאוכזב ומבהיר לו שתפקידו בעולם טרם הסתיים.
סיכום החלק:
הרקיע השישי מהווה שלב של טרנספורמציה. ההיררכיה מפנה את מקומה לאחדות, והצפייה הפסיבית הופכת להשתתפות פעילה. ישעיהו חווה את האור האלוהי בעוצמה כה גדולה, עד שהוא מבקש להתנתק מהעולם הגשמי. המלאך מבהיר לו שזוהי רק טעימה מהעתיד לבוא, ושתפקידו הוא לשוב לעולם עם הידע שרכש. זהו שלב קריטי בהכנתו של הנביא לקראת המפגש עם השיא – הרקיע השביעי.


חלק ו': הכניסה לרקיע השביעי ועולם הצדיקים
(עמודים 6, 6ב)
לב "וַיַּעֲלֵנִי אֶל-אֲוִיר הַשָּׁמַיִם הַשְּׁבִיעִיִּים. וָאֶשְׁמַע קוֹל אוֹמֵר לִי: 'עַד אָן עוֹלֶה זֶה הֶחָפֵץ לִחְיוֹת בְּבָשָׂר?'. וָאִירָא מְאֹד וְרָעַדְתִּי".
לג "וְשׁוּב קוֹל אַחֵר שָׁמַעְתִּי אוֹמֵר: 'אַל תַּפְרִיעוּ, יַעֲלֶה נָא הָרָאוּי לֵאלֹהִים. כִּי כָּאן הוּא בִּגְדוֹ'. וָאֶשְׁאַל אֶת-הַמַּלְאָךְ אֲשֶׁר עִמִּי: 'מִי הוּא הַמּוֹנֵעַ מִמֶּנִּי, וּמִי הוּא שׁוּב הַמְצַוֶּה עָלַי לַעֲלוֹת?'. וַיֹּאמֶר לִי: 'הַמּוֹנֵעַ הוּא זֶה שֶׁעַל-גַּבֵּי הַשָּׁרִים מִן-הַשָּׁמַיִם, הַמַּלְאָךְ. וְהַמְצַוֶּה עָלֶיךָ הוּא בֶּן-הָאֱלֹהִים, וְאֶת-שְׁמוֹ אֵינְךָ יָכוֹל לִשְׁמֹעַ עַד אֲשֶׁר תֵּצֵא מִגּוּפְךָ'".
לד "וְכַאֲשֶׁר עָלִינוּ אֶל-הַשָּׁמַיִם הַשְּׁבִיעִיִּים, רָאִיתִי שָׁם אוֹר נִפְלָא וּבִלְתִּי-נִתָּן לְסִפּוּר, וּמַלְאָכִים אֵין-מִסְפָּר וְצַדִּיקִים".
לה "וָאֵרֶא אֲחֵרִים אֲשֶׁר יָצְאוּ מִבִּגְדֵי-הַבָּשָׂר וְהָיוּ בְּבִגְדֵי-מָרוֹם, וְהֵם עוֹמְדִים בְּכָבוֹד גָּדוֹל. אַךְ עַל-כִּסְאוֹתֵיהֶם לֹא יָשָׁבוּ, וּכְתָרֵי כְבוֹדָם הָיוּ עֲלֵיהֶם. וָאֶשְׁאַל אֶת-הַמַּלְאָךְ וָאֹמַר: 'מַדּוּעַ קִבְּלוּ אֶת-הַבְּגָדִים, וְאֶת-כִּסְאוֹת הַכָּבוֹד וְהַכֶּתֶר אֵינָם מְקַבְּלִים?'".
לו "וַיֹּאמֶר לִי: 'אֵינָם מְקַבְּלִים אוֹתָם עַתָּה, עַד אֲשֶׁר יֵרֵד בֶּן-הָאֱלֹהִים. זֹאת מִקֶּדֶם הֵם יוֹדְעִים, שֶׁלְּמִי שַׁיָּכִים כִּסְאוֹתֵיהֶם וְכִתְרֵיהֶם. כַּאֲשֶׁר יֵרֵד וְלֹא יִהְיֶה בִּדְמוּתְכֶם, וְנֹגֵשׂ הָעוֹלָם הַזֶּה יִפְרֹשׂ אֶת-יָדָיו עָלָיו, וְיִתְלֶה אוֹתוֹ עַל-הָעֵץ, וְיַהֲרֹג מִבְּלִי דַּעַת מִי הוּא'".
לז "'וְהוּא יֵרֵד לִשְׁאוֹל, וְאֶת-צֵל-הַמָּוֶת יָשִׂים לְרֵיק. וְאֶת-נְסִיךְ-הַמָּוֶת יִקַּח בַּשֶּׁבִי, וְיַשְׁחִית אֶת-כָּל-כֹּחוֹ. וְיָקוּם בַּיּוֹם הַשְּׁלִישִׁי, וְעִמּוֹ אֲחֵרִים וְצַדִּיקִים בְּכָל-הָעוֹלָם. וְיַעֲלֶה שׁוּב לַשָּׁמַיִם. אָז יְקַבְּלוּ אֵלֶּה אֶת-כִּסְאוֹתֵיהֶם וְאֶת-כִּתְרֵיהֶם'".
הערות וביאורים (עמודים 6, 6ב)
הכניסה הדרמטית לרקיע השביעי (פסוקים ל"ב-ל"ג):
"עַד אָן עוֹלֶה זֶה הֶחָפֵץ לִחְיוֹת בְּבָשָׂר?": הכניסה לרקיע השביעי אינה חלקה. ישעיהו נתקל ב"שומר סף" מלאכי, המביע התנגדות לכניסתו של יצור גשמי למקום הקדוש ביותר. זהו מוטיב נפוץ של מבחן או מכשול בכניסה לעולם האלוהי.
"יַעֲלֶה נָא הָרָאוּי לֵאלֹהִים. כִּי כָּאן הוּא בִּגְדוֹ": קול שני, גבוה יותר, מתערב ומתיר את כניסתו. הנימוק כפול: ישעיהו "ראוי", וחשוב מכך, "בגדו השמימי" (הגוף הרוחני העתידי שלו) כבר שמור לו כאן. גורלו נקבע מראש.
"הַמּוֹנֵעַ... הַמְצַוֶּה": המלאך המלווה מבהיר את ההיררכיה. המונע הוא מלאך בכיר, אך המצווה הוא "בן-האלוהים", ישות גבוהה עוד יותר, שאת שמה המפורש לא ניתן לשמוע בעוד הנביא בגופו.
עולם הצדיקים (פסוקים ל"ד-ל"ה):
"אוֹר נִפְלָא וּבִלְתִּי-נִתָּן לְסִפּוּר": תיאור האור העליון הוא טופוס קבוע בספרות המיסטית, המדגיש את מגבלות השפה האנושית לתאר את החוויה האלוהית.
"יָצְאוּ מִבִּגְדֵי-הַבָּשָׂר וְהָיוּ בְּבִגְדֵי-מָרוֹם": ישעיהו רואה את נשמות הצדיקים שכבר מתו. הם פשטו את "בגדי הבשר" (הגוף הגשמי) ולבשו "בגדי מרום" (גוף רוחני של אור). מוטיב "בגדי האור" של הנשמה נפוץ בספרות הגנוסטית ובמקורות יהודיים ונוצריים קדומים.
"עַל-כִּסְאוֹתֵיהֶם לֹא יָשָׁבוּ... וְהַכֶּתֶר אֵינָם מְקַבְּלִים": ישעיהו מבחין בפרט תיאולוגי חשוב. הצדיקים כבר נמצאים ברקיע השביעי ולובשים את בגדי הכבוד, אך גמולם עדיין אינו שלם. הם עומדים, אך לא יושבים על כסאותיהם, וכתריהם אינם על ראשיהם. הם נמצאים במצב של המתנה.
הסוד המשיחי (פסוקים ל"ו-ל"ז):
"עַד אֲשֶׁר יֵרֵד בֶּן-הָאֱלֹהִים": המלאך חושף בפני ישעיהו את הסיבה להמתנה. גמולם השלם של הצדיקים תלוי בהשלמת המשימה של "בן-האלוהים" על הארץ.
"יִתְלֶה אוֹתוֹ עַל-הָעֵץ, וְיַהֲרֹג מִבְּלִי דַּעַת מִי הוּא": זוהי נבואה נוצרית מובהקת על צליבתו של ישוע. "נסיך העולם הזה" (השטן) יהרוג אותו, אך יעשה זאת מתוך בורות, מבלי להבין שהוא הורג את בן-האלוהים. בורות זו של השטן היא חלק מהתוכנית האלוהית, והיא שתוביל למפלתו.
"יֵרֵד לִשְׁאוֹל... וְיָקוּם בַּיּוֹם הַשְּׁלִישִׁי": תיאור מדויק של הקרדו (עיקרי האמונה) הנוצרי: ירידתו של ישוע לשאול כדי לשחרר את נשמות הצדיקים (Harrowing of Hell), ותחייתו ביום השלישי.
"אָז יְקַבְּלוּ אֵלֶּה אֶת-כִּסְאוֹתֵיהֶם וְאֶת-כִּתְרֵיהֶם": רק לאחר שישוע ישלים את משימתו ויעלה לשמיים, יוכלו הצדיקים לקבל את גמולם המושלם ולשבת על כסאותיהם. הגאולה האישית של כל צדיק כרוכה באופן בלתי נפרד בדרמה הקוסמית של המשיח.
סיכום החלק:
העלייה לרקיע השביעי חושפת בפני ישעיהו את שיא ההתגלות. הוא לומד שהכניסה לעולם האלוהי מותנית בגורל שנקבע מראש, ורואה את מצבן של נשמות הצדיקים, הממתינות לגמולן השלם. החשוב מכל, הוא מקבל נבואה מפורטת על חייו, מותו ותחייתו של המשיח, ומבין כיצד אירועים אלו, שיתרחשו בעתיד על הארץ, הם המפתח להשלמת הגאולה בעולמות העליונים.


חלק ז': ספרי המעשים, מיכאל, וחזון השילוש
(עמודים 7, 7ב)
לח "וְאַחַר הַדְּבָרִים הָאֵלֶּה, אָמַרְתִּי לוֹ: 'עַל-אֲשֶׁר שְׁאִלְתִּיךָ בַּשָּׁמַיִם הָרִאשׁוֹנִים, סַפֵּר-לִי. כִּי כָּאן הִבְטַחְתָּ לִי לְסַפֵּר, כֵּיצַד הַנַּעֲשֶׂה בָּעוֹלָם הַהוּא יָדוּעַ כָּאן'.
לט "עוֹדֶנִּי מְדַבֵּר עִמּוֹ, וְהִנֵּה מַלְאָךְ אַחֵר מִן-הָעוֹמְדִים, נִכְבָּד וְגָדוֹל מְאֹד מִזֶּה שֶׁהֶעֱלַנִי וּמִכָּל-הַמַּלְאָכִים, וְהֶרְאָה לִי סְפָרִים. וַיִּפְתַּח וַיִּתֵּן לִי, וָאֶרְאֶה אֶת-הַכְּתָב, וְלֹא הָיָה כְּכִתְבֵי הָעוֹלָם הַזֶּה. וָאֶקְרָא, וְהִנֵּה מַעֲשֵׂי יְרוּשָׁלַיִם הָיוּ כְּתוּבִים שָׁם. וְאִישׁ אַחֵר רָאִיתִי, וּמַעֲשָׂיו אֲשֶׁר לֹא יָדַעְתִּי. וְיָדַעְתִּי אָז בֶּאֱמֶת, כִּי דָּבָר אֵינוֹ נִסְתָּר בַּשָּׁמַיִם הַשְּׁבִיעִיִּים מִן-הַנַּעֲשֶׂה בָּעוֹלָם".
מ "וָאֶשְׁאַל אֶת-הַמַּלְאָךְ: 'מִי הוּא זֶה הָעוֹלֶה עַל-הַמַּלְאָכִים בִּכְבוֹדוֹ?'. וַיַּעַן וַיֹּאמֶר לִי: 'הָעוֹלֶה עַל-הַמַּלְאָכִים, זֶהוּ הַמַּלְאָךְ הַגָּדוֹל מִיכָאֵל, הַמִּתְפַּלֵּל תָּמִיד עַל-הָאֱנוֹשׁוּת'".
מא "וָאֵרֶא בְּגָדִים רַבִּים וְכִסְאוֹת וּכְתָרִים רַבִּים מוּנָחִים, [וְהֵם] שֶׁל מַלְאָכִים. הַבְּגָדִים וְהַכִּסְאוֹת וְהַכְּתָרִים הָאֵלֶּה, לָהֶם הֵם שְׁמוּרִים. וַיֹּאמֶר לִי: 'מִן-הַבְּגָדִים הָאֵלֶּה רַבִּים מִתְנַכְּרִים לִי מֵאוֹתוֹ עוֹלָם, [אַף-עַל-פִּי] שֶׁהֶאֱמִינוּ בַּדָּבָר הַהוּא אֲשֶׁר נֶאֱמַר לְךָ'".
מב "וּכְשֶׁפָּנִיתִי, אֶת-הָאָדוֹן רָאִיתִי בְּכָבוֹד גָּדוֹל, וְנִבְעַתִּי מְאֹד. וַיִּגְּשׁוּ אֵלָיו כָּל-הַצַּדִּיקִים אֲשֶׁר רָאִיתִי, וַיִּשְׁתַּחֲווּ לוֹ, וַיָּשִׁירוּ בְּקוֹל אֶחָד. וְהוּא הָיָה דּוֹמֶה לָהֶם, וּכְבוֹדוֹ הָיָה כִּכְבוֹדָם. וְאָז נִגַּשׁ מִיכָאֵל וַיִּשְׁתַּחֲוֶה, וְעִמּוֹ כָּל-הַמַּלְאָכִים הִשְׁתַּחֲווּ וַיְשַׁבְּחוּהוּ. וְהוּא שִׁנָּה אֶת-צוּרָתוֹ וְהָיָה כְּמַלְאָךְ. אָז אָמַר לִי הַמַּלְאָךְ הַמּוֹלִיךְ אוֹתִי: 'הִשְׁתַּחֲוֵה לוֹ וְשַׁבְּחֵהוּ'. וְהִשְׁתַּחֲוֵיתִי וְשִׁבַּחְתִּי אוֹתוֹ. וַיֹּאמֶר לִי הַמַּלְאָךְ: 'זֶהוּ אֲדוֹן כָּל-הַכָּבוֹד אֲשֶׁר רָאִיתָ'".
מג "וְעוֹדוֹ מְדַבֵּר, רָאִיתִי אַחֵר, נִכְבָּד מְאֹד, דּוֹמֶה לוֹ בַּכֹּל".
הערות וביאורים (עמודים 7, 7ב)
ספרי המעשים (פסוקים ל"ח-ל"ט):
"כֵּיצַד הַנַּעֲשֶׂה בָּעוֹלָם הַהוּא יָדוּעַ כָּאן": ישעיהו חוזר לשאלה הפילוסופית שהטרידה אותו ברקיע השלישי. התשובה ניתנת לו לא בדיבור, אלא בחוויה ישירה.
"הֶרְאָה לִי סְפָרִים... וְהִנֵּה מַעֲשֵׂי יְרוּשָׁלַיִם הָיוּ כְּתוּבִים שָׁם": זהו מוטיב "ספרי השמיים" שבהם נכתבים כל מעשי בני האדם. מוטיב זה מוכר היטב מהמקרא (תהלים קל"ט, ט"ז; דניאל ז', י') ומהספרות החיצונית (ספר היובלים, ספר חנוך). כאן, ישעיהו רואה לא רק את מעשי ירושלים, אלא גם מעשים של "איש אחר" שלא הכיר, מה שמדגיש את הידיעה האוניברסלית והמוחלטת של האל.
המלאך מיכאל (פסוק מ'):
"הַמַּלְאָךְ הַגָּדוֹל מִיכָאֵל, הַמִּתְפַּלֵּל תָּמִיד עַל-הָאֱנוֹשׁוּת": המלאך הנכבד שהראה לישעיהו את הספרים מזוהה כמיכאל. תפקידו כ"סניגור" של האנושות ושל עם ישראל מוכר היטב מהמקרא (דניאל י', י"ג, כ"א; י"ב, א') ומהספרות הבתר-מקראית. הוא מוצג כאן כדמות מרכזית בהיררכיה השמימית, שני רק לישויות האלוהיות עצמן.
הבגדים האבודים (פסוק מ"א):
"מִן-הַבְּגָדִים הָאֵלֶּה רַבִּים מִתְנַכְּרִים לִי": זוהי אמירה טראגית. המלאך מראה לישעיהו בגדים, כסאות וכתרים ששמורים לצדיקים, אך רומז שרבים, למרות ש"האמינו בדבר" (כלומר, קיבלו את המסר המשיחי), בסופו של דבר "מתנכרים" לו (כלומר, חוטאים ונופלים) ומאבדים את גמולם השמימי. זהו מסר של אזהרה לקוראי החיבור.
חזון השילוש מתחיל (פסוקים מ"ב-מ"ג):
"אֶת-הָאָדוֹן רָאִיתִי... וְהוּא הָיָה דּוֹמֶה לָהֶם": ישעיהו רואה את הישות הראשונה, המכונה "האדון". באופן מפתיע, הוא דומה במראהו לצדיקים, אם כי כבודו גדול יותר.
"וְהוּא שִׁנָּה אֶת-צוּרָתוֹ וְהָיָה כְּמַלְאָךְ": לאחר שהצדיקים והמלאכים משתחווים לו, "האדון" משנה את צורתו ונעשה דומה למלאך. יכולת שינוי הצורה (טרנספורמציה) היא מאפיין אלוהי.
"זֶהוּ אֲדוֹן כָּל-הַכָּבוֹד אֲשֶׁר רָאִיתָ": המלאך המלווה מזהה את הדמות הזו כ"אדון כל הכבוד", כלומר, הבן (ישוע). הוא מוצג כאן כישות הראשונה בשילוש שישעיהו רואה.
"רָאִיתִי אַחֵר, נִכְבָּד מְאֹד, דּוֹמֶה לוֹ בַּכֹּל": מיד לאחר מכן, מופיעה ישות שנייה, הדומה לבן בכל. זוהי, כפי שיתברר בהמשך, רוח הקודש. החיבור מציג את השילוש לא כמושג פילוסופי, אלא כחזון ויזואלי של שלוש דמויות נפרדות אך קשורות.
סיכום החלק:
בחלק זה, החזון מגיע לשיאו התיאולוגי. ישעיהו מקבל תשובה לשאלתו על ידיעת האל באמצעות חזון ספרי המעשים. הוא פוגש את מיכאל, הסניגור השמימי, ולומד על הסכנה של אובדן הגמול. והחשוב מכל, הוא מתחיל לחזות בדרמה המורכבת של השילוש הקדוש, כאשר שתי הישויות הראשונות, הבן ורוח הקודש, מתגלות בפניו בזו אחר זו.


חזון ישעיהו (מהדורה פרשנית)
חלק ח': שיא חזון השילוש והציווי על הבן
(עמודים 8, 8ב)
מד "וְהַצַּדִּיקִים נִגְּשׁוּ וְהִשְׁתַּחֲווּ וְשִׁבְּחוּ אֶת-זֶה, וְהוּא שִׁבַּח עִמָּהֶם. וְהוּא לֹא שִׁנָּה אֶת-צוּרָתוֹ כְּמוֹ הַנִּרְאִים. וְעִמּוֹ בָּאוּ מַלְאָכִים וְהִשְׁתַּחֲווּ לוֹ. וַיֹּאמֶר לִי הַמַּלְאָךְ: 'הִשְׁתַּחֲוֵה וְשַׁבַּח לוֹ'. וְהִשְׁתַּחֲוֵיתִי וְשִׁבַּחְתִּי לוֹ".
מה "וְשׁוּב, אַחֵר בְּכָבוֹד גָּדוֹל רָאִיתִי. וּכְשֶׁבָּא, שָׁאַלְתִּי אֶת-הַמַּלְאָךְ: 'מִי הוּא זֶה?'. וַיֹּאמֶר לִי: 'הִשְׁתַּחֲוֵה לוֹ, כִּי זֶהוּ הַמַּלְאָךְ שֶׁל הָרוּחַ, הַמְדַבֵּר בְּךָ וּבְכָל-הַצַּדִּיקִים'".
מו "וְאַחֲרֵי-כֵן, כָּבוֹד אַחֵר, שׁוֹנֶה וּבִלְתִּי-נִתָּן לְדִבּוּר, נִגְלָה, אֲשֶׁר בְּעֵינַי הַפְּקוּחוֹת שֶׁל רוּחִי לֹא יָכֹלְתִּי לִרְאוֹת. וְלֹא אֲנִי, וְלֹא הַמַּלְאָךְ שֶׁהֶעֱלַנִי, וְלֹא כָּל-הַמַּלְאָכִים אֲשֶׁר רָאִיתִי מִשְׁתַּחֲוִים לָאָדוֹן. רַק אֶת-הַצַּדִּיקִים בִּלְבַד רָאִיתִי, אֲשֶׁר בְּכֹחַ גָּדוֹל הִבִּיטוּ בִּכְבוֹדוֹ".
מז "וְנִגַּשׁ אֲלֵיהֶם תְּחִלָּה אֲדֹנִי, וְהַמַּלְאָךְ הָרוּחָנִי. וְהִשְׁתַּחֲווּ וְשִׁבְּחוּ, שְׁנֵיהֶם בִּלְבַד. וְאָז כָּל-הַצַּדִּיקִים הִשְׁתַּחֲווּ, וְעִמָּהֶם מִיכָאֵל וְכָל-הַמַּלְאָכִים הִשְׁתַּחֲווּ וְשִׁבְּחוּ".
מח "אַחַר-כָּךְ שָׁמַעְתִּי אֶת-הַקּוֹל הַהוּא, וְאֶת-הַשִּׁירוֹת שָׁמַעְתִּי עוֹלוֹת בַּשָּׁמַיִם עַד לַשָּׁמַיִם הַשְּׁבִיעִיִּים. וְכֻלָּם שִׁבְּחוּ אֶת-זֶה אֲשֶׁר אֶת-כְּבוֹדוֹ לֹא יָכֹלְתִּי לִרְאוֹת. וְכָל-שִׁירַת שִׁבְעַת הָרְקִיעִים לֹא רַק נִשְׁמְעָה, אֶלָּא גַּם נִרְאֲתָה".
מט "וַיֹּאמֶר לִי הַמַּלְאָךְ: 'זֶהוּ הָאֶחָד הַנִּצְחִי, הַחַי בָּעוֹלָם הָעֶלְיוֹן וְשׁוֹכֵן בַּקְּדוֹשִׁים. אֲשֶׁר אֲפִלּוּ אֶת-שְׁמוֹ אֵין אָנוּ יְכוֹלִים לָשֵׂאת. הוּא זֶה אֲשֶׁר מְשֻׁבָּח עַל-יְדֵי רוּחַ הַקֹּדֶשׁ בְּפִי הַצַּדִּיקִים'".
נ "וְאַחַר-זֶה שָׁמַעְתִּי אֶת-קוֹל הַנִּצְחִי אוֹמֵר לָאָדוֹן הַבֵּן: 'צֵא וּרֵד מִכָּל-הַשָּׁמַיִם, וְהֱיֵה בָּעוֹלָם, וְעַד לַמַּלְאָךְ אֲשֶׁר בַּשְּׁאוֹל. הִשְׁתַּנֵּה לְפִי דְּמוּתָם, וְלֹא יַכִּירוּךָ לֹא הַמַּלְאָכִים וְלֹא נְסִיכֵי הָעוֹלָם הַהוּא. וְתִשְׁפֹּט אֶת-הַנָּסִיךְ וְאֶת-מַלְאָכָיו וְאֶת-הָעוֹלָם אֲשֶׁר הוּא מוֹשֵׁל בּוֹ, כִּי מָאֲסוּ בִּי וְאָמְרוּ: "אֲנַחְנוּ יֵשׁ, וּמִלְּבַדֵּנוּ אֵין אַחֵר". וּכְשֶׁתַּעֲלֶה מִן-הָאָרֶץ, לֹא תִּשְׁתַּנֶּה עוֹד בַּשָּׁמַיִם, אֶלָּא בְּכָבוֹד גָּדוֹל תַּעֲלֶה וְתֵשֵׁב לִימִינִי. וְאָז יִשְׁתַּחֲווּ לְךָ נְסִיכִים וְכֹחוֹת וּמַלְאָכִים וְכָל-רֹאשׁ שֶׁבַּשָּׁמַיִם וּבָאָרֶץ וּבַתַּחְתִּיּוֹת'".
הערות וביאורים (עמודים 8, 8ב)
הופעת רוח הקודש (פסוקים מ"ד-מ"ה):
"רָאִיתִי אַחֵר, נִכְבָּד מְאֹד, דּוֹמֶה לוֹ בַּכֹּל": לאחר שראה את הבן, ישעיהו רואה דמות שנייה, דומה לו לחלוטין.
"זֶהוּ הַמַּלְאָךְ שֶׁל הָרוּחַ, הַמְדַבֵּר בְּךָ וּבְכָל-הַצַּדִּיקִים": המלאך המלווה מזהה את הדמות השנייה כ"מלאך הרוח" או רוח הקודש. תפקידה מוגדר כמי שמדבר בתוך הנביאים והצדיקים, כלומר, היא מקור ההשראה הנבואית. החיבור מציג כאן את רוח הקודש כישות נפרדת, בעלת צורה נראית, ולא רק ככוח או השפעה.
הופעת האב והגבול המיסטי (פסוקים מ"ו, מ"ט):
"כָּבוֹד... אֲשֶׁר... לֹא יָכֹלְתִּי לִרְאוֹת": כעת מופיעה הישות השלישית, האב. אך בניגוד לבן ולרוח, את כבודו של האב ישעיהו אינו יכול לראות. זוהי נקודת הגבול של החוויה המיסטית. אפילו במצב הנעלה ביותר, מהותו של האל האב נותרת בלתי נתפסת ובלתי נראית.
"רַק אֶת-הַצַּדִּיקִים בִּלְבַד רָאִיתִי, אֲשֶׁר בְּכֹחַ גָּדוֹל הִבִּיטוּ בִּכְבוֹדוֹ": רק הצדיקים, שכבר פשטו את גופם הגשמי, מסוגלים לעמוד במראה הכבוד הזה. זה מדגיש את המגבלה של הנביא, שעדיין קשור לגופו.
"אֲפִלּוּ אֶת-שְׁמוֹ אֵין אָנוּ יְכוֹלִים לָשֵׂאת": המלאך מאשר את הטרנסצנדנטיות המוחלטת של האב. אפילו המלאכים העליונים אינם יכולים לבטא את שמו.
הפולחן השמימי (פסוקים מ"ז-מ"ח):
"נִגַּשׁ... תְּחִלָּה אֲדֹנִי, וְהַמַּלְאָךְ הָרוּחָנִי... שְׁנֵיהֶם בִּלְבַד": סדר ההשתחוויה חושף את ההיררכיה הפנימית של השילוש. הבן ורוח הקודש ניגשים ראשונים ומשתחווים לאב. הם מתפקדים כמתווכים בין הבריאה לבין האב הבלתי נראה. רק אחריהם ניגשים כל שאר הנבראים (צדיקים, מיכאל, מלאכים).
"שִׁירַת שִׁבְעַת הָרְקִיעִים לֹא רַק נִשְׁמְעָה, אֶלָּא גַּם נִרְאֲתָה": תיאור סינסתטי מובהק. בשיא החוויה המיסטית, החושים מתאחדים. הקול הופך למראה, והשירה הופכת לתופעה ויזואלית.
הציווי על הבן (פסוק נ'):
"צֵא וּרֵד מִכָּל-הַשָּׁמַיִם... הִשְׁתַּנֵּה לְפִי דְּמוּתָם": כאן נחשפת התוכנית האלוהית הגדולה (ה"קנוסיס" – Kenosis). האב מצווה על הבן לרדת דרך כל הרקיעים, ובכל רקיע לשנות את צורתו ולהידמות למלאכים שבו, כדי שלא יכירוהו. זוהי ירידה של הסתרה והתחפשות.
"וְלֹא יַכִּירוּךָ לֹא הַמַּלְאָכִים וְלֹא נְסִיכֵי הָעוֹלָם הַהוּא": ההסתרה נועדה להטעות לא רק את המלאכים הטובים, אלא בעיקר את "נסיכי העולם הזה" (השטן וכוחותיו), כדי שיוציאו לפועל את תוכניתם להרוג אותו, ובכך יביאו על עצמם את מפלתם. זוהי תפיסה גנוסטית מובהקת של גאולה באמצעות תחבולה אלוהית.
"כִּי מָאֲסוּ בִּי וְאָמְרוּ: 'אֲנַחְנוּ יֵשׁ, וּמִלְּבַדֵּנוּ אֵין אַחֵר'": הסיבה למשפט על נסיכי העולם היא חטא הגאווה – הם ראו בעצמם אלים ומרכז המציאות.
"בְּכָבוֹד גָּדוֹל תַּעֲלֶה וְתֵשֵׁב לִימִינִי": בניגוד לירידה, שתהיה בהסתר, העלייה חזרה (ההתעלות) תהיה בכבוד גלוי, ותסתיים בישיבתו של הבן לימין האב, ובקבלת השלטון על כל הבריאה.
סיכום החלק:
חלק זה הוא לב לבה של התיאולוגיה של החיבור. הוא מציג חזון מורכב של השילוש, שבו האב הוא טרנסצנדנטי ובלתי נראה, והבן ורוח הקודש הם המתווכים. החלק חושף את תוכנית הגאולה כירידה מכוונת של הבן בהסתר, במטרה להטעות את כוחות הרשע ולהביסם, ולאחר מכן לעלות בכבוד גלוי ולקבל את מלכותו.


חלק ט': ירידת הבן דרך הרקיעים
(עמודים 9, 9ב)
נא "אֶת-הַכָּבוֹד הַגָּדוֹל הַזֶּה שָׁמַעְתִּי, הַמְּצַוֶּה עַל אֲדֹנִי. וְאָז יָרַד הָאָדוֹן מִן-הַשָּׁמַיִם הַשְּׁבִיעִיִּים וַיֵּרֶד אֶל-הַשָּׁמַיִם הַשִּׁשִּׁיִּים. וְהַמַּלְאָךְ הַמַּדְרִיךְ אוֹתִי אָמַר: 'הָבֵן, יְשַׁעְיָהוּ, וּרְאֵה אֶת-שִׁנּוּי-הַצּוּרָה הֶעָתִיד וְאֶת-יְרִידָתוֹ. כִּי רָאוּהוּ הַמַּלְאָכִים וַיְשַׁבְּחוּהוּ וַיְהַלְלוּהוּ, כִּי לֹא שִׁנָּה אֶת-צוּרָתוֹ כְּמוֹ הַנִּרְאִים. וַאֲנִי שַׁבַּחְתִּי עִמָּהֶם'".
נב "וּכְשֶׁיָּרַד אֶל-הַשָּׁמַיִם הַחֲמִישִׁיִּים, שִׁנָּה אֶת-צוּרָתוֹ לִדְמוּתָם, וְהֵם לֹא שִׁבְּחוּהוּ וְלֹא הִשְׁתַּחֲווּ לוֹ, כִּי דְּמוּתוֹ הָיְתָה כִּדְמוּתָם".
נג "וְיָרַד אֶל-הַשָּׁמַיִם הָרְבִיעִיִּים, וַיֵּרָאֶה לָהֶם כְּמַרְאֵה פְּנֵיהֶם, וְהֵם לֹא שִׁבְּחוּהוּ, כִּי הָיָה כְּמַרְאֵה פְּנֵיהֶם. וַיָּבֹא אֶל-הַשָּׁמַיִם הַשְּׁלִישִׁיִּים, וְהַשֵּׁנִיִּים, וְהָרִאשׁוֹנִים, וַיִּשְׁתַּנֶּה בְּכָל-אֶחָד מֵהֶם. עַל-כֵּן לֹא שָׁרוּ וְלֹא הִשְׁתַּחֲווּ לוֹ, כִּי נִרְאָה דּוֹמֶה לָהֶם".
נד "וְהוּא הֶרְאָה אוֹתוֹת בְּכָל-שָׁמַיִם וְשָׁמַיִם לְשׁוֹמְרֵי הַשְּׁעָרִים. וְיָרַד אֶל-הָרָקִיעַ הַתַּחְתּוֹן (הַ'טְּבֶרְד'), שָׁם נְסִיךְ הָעוֹלָם הַזֶּה יוֹשֵׁב. וְשָׁם נָתַן אוֹת, אַךְ דְּמוּתוֹ הָיְתָה כִּדְמוּתָם, וְהֵם לֹא הִלְּלוּ וְלֹא שָׁרוּ לוֹ".
נה "וְיָרַד אֶל-מַלְאֲכֵי הָאֲוִיר הַזֶּה, וְהָיָה כְּאֶחָד מֵהֶם. וְלֹא נָתַן לָהֶם אוֹת, כִּי לֹא שְׁאֵלוּהוּ".
הערות וביאורים (עמודים 9, 9ב)
חלק זה הוא המימוש של הציווי האלוהי, והוא מתאר את מסע הירידה של הבן, המכונה "קֶנוֹסִיס" (Kenosis) – התרוקנות עצמית. זהו אחד הרעיונות התיאולוגיים המורכבים והנועזים ביותר בחיבור.
הירידה לרקיע השישי (פסוק נ"א):
"כִּי לֹא שִׁנָּה אֶת-צוּרָתוֹ כְּמוֹ הַנִּרְאִים": ברקיע השישי, הבן עדיין שומר על צורתו האלוהית. לכן, המלאכים שם, שכבר מתאפיינים באחדות וברמה רוחנית גבוהה, מזהים אותו ומשבחים אותו. זהו הרקיע האחרון שבו זהותו האמיתית עדיין גלויה.
הירידה לרקיעים 5 עד 1 (פסוקים נ"ב-נ"ג):
"שִׁנָּה אֶת-צוּרָתוֹ לִדְמוּתָם, וְהֵם לֹא שִׁבְּחוּהוּ": החל מהרקיע החמישי ומטה, מתחילה ההסתתרות. בכל רקיע, הבן מאמץ את "דמותם" של המלאכים המקומיים. כתוצאה מכך, הם אינם מזהים את עליונותו ואינם עובדים אותו. הוא חולף דרכם כ"אחד משלהם".
תפיסה גנוסטית: רעיון זה של ירידה דרך רקיעים עוינים או בורים תוך הסוואה הוא לב לבה של המיתולוגיה הגנוסטית. הגואל הגנוסטי יורד מהעולם העליון, חומק דרך ה"ארכונטים" (השליטים) של כל רקיע על ידי שימוש בסיסמאות או שינוי צורה, כדי להגיע לעולם התחתון ולשחרר את ניצוצות האור הכלואים בו. "חזון ישעיהו" מאמץ את המבנה הזה, אך "מכשיר" אותו לתיאולוגיה נוצרית: הרקיעים אינם עוינים, אלא פשוט אינם מזהים את הבן בגלל ההסוואה.
הירידה לרקיע התחתון ולאוויר (פסוקים נ"ד-נ"ה):
"הָרָקִיעַ הַתַּחְתּוֹן (הַ'טְּבֶרְד'), שָׁם נְסִיךְ הָעוֹלָם הַזֶּה יוֹשֵׁב": זוהי נקודת המפגש עם האויב. הבן יורד לממלכתו של השטן עצמו. גם כאן הוא מאמץ את דמותם של מלאכי החושך, והם אינם מזהים אותו. התחבולה מצליחה.
"מַלְאֲכֵי הָאֲוִיר הַזֶּה": השכבה האחרונה לפני העולם הגשמי. גם כאן, הוא חולף מבלי להתגלות.
"נָתַן אוֹת... לֹא נָתַן לָהֶם אוֹת": יש כאן הבחנה מעניינת. לשומרי השערים של הרקיעים העליונים הוא נותן "אות" (סימן זיהוי, סיסמה), אולי כדי לעבור. אך למלאכי האוויר, הדרגה הנמוכה ביותר, הוא אינו נותן אות, "כי לא שאלוהו". ייתכן שהם כה נחותים עד שאינם מודעים כלל למתרחש ואינם דורשים זיהוי.
סיכום החלק:
חלק זה מתאר באופן ויזואלי ודרמטי את תהליך ה"קנוסיס" – ההתרוקנות וההסתתרות של הבן. הוא יורד שלב אחר שלב בסולם הקוסמי, פושט את כבודו האלוהי ולובש את דמותם של הנבראים בכל רקיע, עד שהוא מגיע לעולם התחתון כשהוא מוסווה לחלוטין. הצלחתה של תחבולה זו היא תנאי הכרחי להצלחת משימת הגאולה כולה, שכן היא תגרום לכוחות הרשע להביא על עצמם את מפלתם מבלי דעת.


חלק י': סיכום החזון, העלייה מחדש, והחזרה לעולם
(עמודים 10, 10ב)
נו "וְאַחֲרֵי-כֵן אָמַר לִי הַמַּלְאָךְ: 'הָבֵן, יְשַׁעְיָהוּ בֶּן-אָמוֹץ. כִּי לְשֵׁם כָּךְ שֻׁלַּחְתִּי מֵאֵת ה' לְהַרְאוֹת לְךָ הַכֹּל. כִּי לֹא רָאָה אִישׁ לְפָנֶיךָ, וְגַם אַחֲרֶיךָ לֹא יוּכַל לִרְאוֹת כְּמוֹ שֶׁאַתָּה רוֹאֶה וְשָׁמַעְתָּ. וְהִנֵּה רָאִיתִי דְּמוּת כְּבֶן-אָדָם, וְהוּא שָׁכַן עִם בְּנֵי-אָדָם בָּעוֹלָם, וְלֹא הִכִּירוּהוּ'".
נז "וְרָאִיתִי אוֹתוֹ עוֹלֶה אֶל-הָרָקִיעַ הַתַּחְתּוֹן (הַ'טְּבֶרְד'), וְכָאן כְּבָר לֹא הָיָה מִשְׁתַּנֶּה לְפִי דְּמוּתָם. וּכְשֶׁרָאוּהוּ כָּל-הַמַּלְאָכִים, נִבְעֲתוּ, וַיִּשְׁתַּחֲווּ וַיֹּאמְרוּ: 'אֵיךְ הִסְתַּתַּרְתָּ מֵאִתָּנוּ, אֲדֹנֵנוּ, בְּתוֹכֵנוּ, וְלֹא הִכַּרְנוּ אֶת-מֶלֶךְ הַכָּבוֹד?'. וּמִן-הַשָּׁמַיִם הָרִאשׁוֹנִים הָיָה עוֹלֶה נִכְבָּד יוֹתֵר, וְלֹא הִשְׁתַּנָּה. וְכָל-הַמַּלְאָכִים, הַיְמָנִיִּים וְהַשְּׂמָאלִיִּים, וְהַיּוֹשֵׁב עַל-הַכִּסֵּא בָּאֶמְצַע, הִשְׁתַּחֲווּ לוֹ וַיְשַׁבְּחוּהוּ, אוֹמְרִים: 'אֵיךְ עָבַרְתָּ דַּרְכֵּנוּ, אֲדֹנֵנוּ, וְלֹא רָאִינוּ וְלֹא הִשְׁתַּחֲוֵינוּ לְךָ?'".
נח "וְכָךְ עָלָה אֶל-הַשֵּׁנִי, וְאֶל-הַשְּׁלִישִׁי, וְאֶל-הָרְבִיעִי, וְאֶל-הַחֲמִישִׁי, וְאֶל-הַשִּׁשִּׁי. אַךְ בְּכָל-הַשָּׁמַיִם כְּבוֹדוֹ הָלַךְ וְגָדַל. וּכְשֶׁעָלָה אֶל-הַשָּׁמַיִם הַשְּׁבִיעִיִּים, שִׁבְּחוּהוּ כָּל-הַצַּדִּיקִים וְכָל-הַמַּלְאָכִים וְכָל-הַכֹּחוֹת. וְאָז רָאִיתִי אוֹתוֹ לִימִין הַכָּבוֹד הַגָּדוֹל, אֲשֶׁר לֹא יָכֹלְתִּי לִרְאוֹת. וְאֶת-הַמַּלְאָךְ הָרוּחָנִי רָאִיתִי יוֹשֵׁב מִשְּׂמֹאל".
נט "וַיֹּאמֶר לִי: 'דַּי לְךָ, יְשַׁעְיָהוּ, שֶׁרָאִיתָ. בֶּן-בָּשָׂר, עַיִן לֹא-רָאֲתָה וְאֹזֶן לֹא-שָׁמְעָה וְעַל לֵב-אֱנוֹשׁ לֹא עָלָה, אֶת-אֲשֶׁר הֵכִין אֱלֹהִים לְאֹהֲבָיו'. וַיֹּאמֶר לִי: 'שׁוּב לְבִגְדְּךָ, עַד אֲשֶׁר יִמָּלֵא זְמַן יָמֶיךָ, וְאָז תָּבוֹא הֵנָּה'".
ס "זֹאת רָאִיתִי, אֲנִי יְשַׁעְיָהוּ, וְאָמַרְתִּי לָעוֹמְדִים סְבִיבִי. וּכְשָׁמְעָם אֶת-הַנִּפְלָאוֹת הָאֵלֶּה, הִלְּלוּ וְשִׁבְּחוּ אֶת-הָאֱלֹהִים, אֲשֶׁר נָתַן חֵן כָּזֶה לִבְנֵי-אָדָם".
סא "וַיֹּאמֶר לְחִזְקִיָּהוּ הַמֶּלֶךְ: 'קֵץ הָעוֹלָם הַזֶּה וְכָל-הֶחָזוֹן הַזֶּה יִתְקַיֵּם בַּדּוֹר הָאַחֲרוֹן'. וַיַּשְׁבִּיעֵם לֹא לְגַלּוֹת לִבְנֵי יִשְׂרָאֵל, וְלֹא לָתֵת אֶת-הַדְּבָרִים הָאֵלֶּה לְכָתוּב לְכָל-אָדָם. 'אַךְ כָּל-אֲשֶׁר תָּבִינוּ מִן-הַמֶּלֶךְ, הַנֶּאֱמָרִים בַּנְּבִיאִים, דְּבָרִים כָּאֵלֶּה תָּבִינוּ כֻּלְּכֶם. וְאַתֶּם, הֱיוּ בְּרוּחַ הַקֹּדֶשׁ, לְמַעַן תְּקַבְּלוּ אֶת-בִּגְדֵיכֶם, וְאֶת-הַכִּסְאוֹת וְהַכְּתָרִים שֶׁל כָּבוֹד, הַמּוּנָחִים בַּשָּׁמַיִם'".
סב "וַיַּעֲמֹד מִלְּדַבֵּר, וַיֵּצֵא מֵאֵת חִזְקִיָּהוּ הַמֶּלֶךְ. וְלֵאלֹהֵינוּ הַכָּבוֹד, גַּם-עַתָּה וְתָמִיד וּלְעוֹלְמֵי עוֹלָמִים, אָמֵן. +".
הערות וביאורים (עמודים 10, 10ב)
סיכום וייחודיות החזון (פסוק נ"ו):
"לֹא רָאָה אִישׁ לְפָנֶיךָ, וְגַם אַחֲרֶיךָ לֹא יוּכַל לִרְאוֹת": המלאך מדגיש את ייחודיותו של חזון ישעיהו. הוא מוצג כהתגלות חד-פעמית ונעלה ביותר, מה שמעניק לחיבור כולו מעמד מיוחד.
"שָׁכַן עִם בְּנֵי-אָדָם בָּעוֹלָם, וְלֹא הִכִּירוּהוּ": סיכום תמציתי של חיי ישוע על הארץ – התגלמות בהסתר.
העלייה מחדש (פסוקים נ"ז-נ"ח):
"אֵיךְ הִסְתַּתַּרְתָּ מֵאִתָּנוּ... וְלֹא הִכַּרְנוּ": בניגוד לירידה השקטה, העלייה היא אירוע דרמטי של התגלות. בכל רקיע, המלאכים נדהמים לגלות שהישות הנעלה הזו שהתה בקרבם מבלי שהם ידעו. תדהמתם מדגישה את הצלחת התחבולה האלוהית.
"בְּכָל-הַשָּׁמַיִם כְּבוֹדוֹ הָלַךְ וְגָדַל": בעוד שבירידה הוא "התרוקן" מכבודו, בעלייה הוא "מתמלא" בו מחדש, ואף יותר מכך. בכל שלב הוא פושט את ההסוואה וחוזר לכבודו האמיתי, שהולך ומתעצם ככל שהוא מתקרב למקור.
"יָשַׁב לִימִין הַכָּבוֹד הַגָּדוֹל... וְהַמַּלְאָךְ הָרוּחָנִי... מִשְּׂמֹאל": החזון מסתיים בתמונת יציבות של השילוש. הבן יושב לימין האב, ורוח הקודש לשמאלו. זהו המצב הסופי והנצחי של ההנהגה האלוהית.
סיום החוויה המיסטית (פסוק נ"ט):
"עַיִן לֹא-רָאֲתָה וְאֹזֶן לֹא-שָׁמְעָה...": המלאך מצטט (או מהווה מקור לציטוט) פסוק מפורסם, המופיע גם בישעיהו ס"ד, ג' ובאיגרת הראשונה אל הקורינתים ב', ט'. הציטוט מדגיש את הפער הבלתי נתפס בין העולם הזה לגמול השמור לצדיקים.
"שׁוּב לְבִגְדְּךָ, עַד אֲשֶׁר יִמָּלֵא זְמַן יָמֶיךָ": הפקודה לחזור לגוף ("בגד הבשר") ניתנת יחד עם ההבטחה לחזור לשמיים באופן סופי לאחר מותו.
החזרה לעולם וההוראות (פסוקים ס'-ס"ב):
"וְאָמַרְתִּי לָעוֹמְדִים סְבִיבִי": ישעיהו חוזר להכרה ומספר את חזונו לקהל הנביאים, ובכך סוגר את סיפור המסגרת שהחל בתחילת החיבור.
"קֵץ הָעוֹלָם הַזֶּה... יִתְקַיֵּם בַּדּוֹר הָאַחֲרוֹן": החזון מקבל משמעות אסכטולוגית. הוא אינו רק תיאור של המציאות השמימית, אלא נבואה על אחרית הימים.
"וַיַּשְׁבִּיעֵם לֹא לְגַלּוֹת... וְלֹא לָתֵת... לְכָתוּב": ישעיהו מצווה לשמור את החזון בסוד. זהו מאפיין קלאסי של ספרות אזוטרית (סודית). הסוד נועד רק לקהילה מצומצמת של "יודעים", ולא להמון הרחב. הציווי שלא לכתוב נועד לשמר את המסורת בעל-פה, בתוך חוג התלמידים.
"הֱיוּ בְּרוּחַ הַקֹּדֶשׁ, לְמַעַן תְּקַבְּלוּ אֶת-בִּגְדֵיכֶם": המסר המעשי הסופי הוא קריאה לחיים של קדושה, המהווים תנאי לקבלת הגמול השמימי.
"וְלֵאלֹהֵינוּ הַכָּבוֹד... אָמֵן": החיבור מסתיים ב"דוקסולוגיה" (נוסחת שבח) נוצרית טיפוסית, החותמת את הטקסט ומשייכת אותו באופן מובהק למסורת הכנסייתית.
סיכום החלק והחיבור כולו:
החלק האחרון סוגר את כל הקצוות. הוא מתאר את חזרתו של הבן למקומו לימין האב, את חזרתו של ישעיהו לגופו, ואת ההוראות לשמירת החזון בסוד. החיבור, שהחל כסיפור על נביא בחצר המלך, הפך למסע קוסמי מורכב, חשף תיאולוגיה נוצרית מפותחת על השילוש והגאולה, והסתיים כטקסט אזוטרי המיועד לקהילת מאמינים מצומצמת, הקורא להם לחיות חיי קדושה כדי לזכות בגמול השמימי שישעיהו זכה לראות.



סיכום הממצאים המרכזיים מ"חזון ישעיהו"
"חזון ישעיהו" הוא חיבור אפוקליפטי יוצא דופן, המשלב מסורות יהודיות על עלייה לשמיים עם תיאולוגיה נוצרית מפותחת. ניתוח הטקסט חושף מספר מאפיינים ותמות מרכזיות:
ז'אנר "העלייה" (Ascension):
החיבור שייך באופן מובהק לסוגת ה"עליות לשמיים", שהייתה נפוצה בספרות בית שני (כמו בספרי חנוך). גיבור אנושי (ישעיהו) זוכה לעלות בהדרכת מלאך דרך הרקיעים, לחזות בסודות הקוסמוס, ולקבל התגלות אלוהית.
העלייה מתוארת כחוויה אקסטטית, שבה הנביא יוצא מחושי גופו ("עיניו פקוחות, פיו סגור") ומועבר לעולם הרוח.
קוסמולוגיה היררכית:
העולם מתואר כמבנה היררכי של שבעה רקיעים, בתוספת "רקיע תחתון" (טברד') המשמש כממלכת השטן.
כל רקיע (מ-1 עד 5) עולה על קודמו בכבוד, באור ובאיכות השירה, ומציג היררכיה פנימית של מלאכים (יושב על כיסא, מלאכי ימין ושמאל). מבנה זה מושפע מאוד מתפיסות ניאופלטוניות.
הרקיע השישי מהווה נקודת מפנה, שבה ההיררכיה מתבטלת ומוחלפת באחדות מוחלטת, והרקיע השביעי הוא משכן האל הבלתי נתפס.
תיאולוגיה נוצרית מובהקת:
השילוש הקדוש: החיבור מציג חזון ויזואלי של השילוש: האב (ה"כבוד הגדול" הבלתי נראה), הבן ("האדון"), ורוח הקודש ("המלאך של הרוח"). הם מתוארים כישויות נפרדות אך קשורות, עם היררכיה פנימית ברורה (הבן והרוח משתחווים לאב).
תוכנית הגאולה (קֶנוֹסִיס): לב החיבור הוא תיאור תוכנית הגאולה כפקודה של האב לבן לרדת לעולם. הירידה מתוארת כמסע של הסתתרות מכוונת, שבו הבן משנה את צורתו בכל רקיע כדי להטעות את המלאכים ואת "נסיך העולם הזה" (השטן).
הדרמה המשיחית: החיבור כולל נבואה מפורטת על חייו, צליבתו ("יתלה אותו על העץ"), מותו, ירידתו לשאול ותחייתו של ישוע. גאולת הצדיקים מוצגת כתלויה בהשלמת משימתו של הבן.
דואליזם מרוכך:
החיבור מציג מאבק קוסמי בין כוחות האור לכוחות החושך, המתרחש ב"רקיע התחתון" ומשתקף בעולם הארצי.
עם זאת, זהו אינו דואליזם מוחלט (כמו במניכאיזם), מכיוון שהשטן וכוחותיו מוגבלים לרקיע התחתון בלבד, וכל המערכת כפופה בסופו של דבר לשלטונו של האל האחד.
אופי אזוטרי (סודי):
החיבור מסתיים בציווי מפורש של ישעיהו לשמור את תוכן החזון בסוד, לא להעלות אותו על הכתב, ולמסור אותו רק לחוג מצומצם של "יודעים". זהו מאפיין של קהילות סגורות שראו בעצמן בעלות ידע נסתר ונעלה על פני המון המאמינים.
לסיכום, "חזון ישעיהו" הוא דוגמה מרתקת לאופן שבו קהילה נוצרית קדומה (בעלת גוון גנוסטי) אימצה ז'אנר ומבנים קוסמולוגיים יהודיים, ו"הטעינה" אותם בתוכן תיאולוגי נוצרי מובהק. התוצאה היא יצירה מורכבת ורבת רבדים, המשלבת מיסטיקה, קוסמולוגיה, ונבואה משיחית לכדי דרמה קוסמית אחת גדולה.
