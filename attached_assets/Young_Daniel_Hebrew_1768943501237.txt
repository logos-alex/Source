
חזון דניאל הצעיר

פרק א

תּוּב דָּנִיֵּאל נִרְאָה כִּי גָּדוֹל הוּא מִכֻּלָּם, וֵאלֹהִים גִּדְּלוֹ מִן הַיּוֹם הַהוּא.
בּוֹ בִזְבָרַיִם הוּא, הֶחֱוָה וְגִלָּה אֶת כָּל הַסּוֹד, וּפֵשַׁר אֶת הַדִּין.
וְהֶחֱוָה אֶת הַחַיִּים וְלֹא הִסְתִּיר עַל הַמָּוֶת.

וַיֹּאמֶר דָּנִיֵּאל: מִפִּיכֶם תִּשָּׁפְטוּ, וּבְחַטֹּאתֵיכֶם תָּמוּתוּ, וּמַעֲשֵׂיכֶם יְאַבְּדוּ אֶתְכֶם.
עֵינֵיכֶם תִּכְהֶינָה, וְאָזְנֵיכֶם תֵּחָרַשְׁנָה, וְלִבּוֹתֵיכֶם לֹא יֵדְעוּ.
וּתְשׁוּבוּ בְּחַיֵּי אֱלֹהִים כְּמוֹ הוֹלְכִים אַחֲרֵי הָרוֹעֶה, וְלֹא תֵּדְעוּ.
מִפְּנֵי שֶׁכְּמוֹ צֶלֶם אָדָם תֵּלְכוּ אַחֲרֵי הָרוֹעֶה.

וַיָּבֹא אֶל שֶׁלּוֹ, וְשֶׁלּוֹ לֹא הִכִּירוּהוּ.
וְיִקְרָא לִבְכוֹרֵי עֶדְרוֹ, וְלֹא יַכִּירוּ בְּקוֹלוֹ.
לְאוֹהֲבָיו יִהְיֶה כְּאוֹיֵב, וְאֵלֶּה שֶׁבָּא לְהַחֲיוֹת – יִהְיוּ לוֹ לְאוֹיְבִים.
וְיִסְבֹּל בַּעֲדָם, וְיִקְרָא לָהֶם, וְלֹא יַאֲזִינוּ לְקוֹלוֹ.

וְאֵלֶּה אֲשֶׁר יַבְדִּילוּ אֶת קוֹלוֹ – יִהְיוּ עִמּוֹ, וְלֹא יִפָּרְדוּ מִמֶּנּוּ.
תִּהְיֶה תּוֹרָה לָרַבִּים, וְחַיִּים לָעַמִּים יִנָּתֵן.
תֹּאבַד חָכְמַת חֲכָמִים, וְתֵחָכֵם עָרְמַת עֲרוּמִים.
יֶאֱסֹף אֶת הָאֹבְדִים, וְיָבִיא אֶת הָעַמִּים.

וְיֵצֵא שְׁמוֹ בָּעַמִּים, וּבִמְקוֹם אֲשֶׁר יֵלֵךְ בּוֹ יִשָּׁמַע קוֹלוֹ.
וְהָיָה לְחוֹמוֹת וּלְאוֹר לָאֵלֶּה הַשּׁוֹמְעִים לוֹ.
וְהָיָה לְעֶזְרָה לַדַּלִּים, וּלְרוֹפֵא לַחוֹלִים.
לְמוֹדָע וְדִמְיוֹן הוּא לָעַמִּים.

לֹא מָצָאתִי לְפָרֵשׁ אֶת אֲבוֹתָיו, אֶלָּא בְּרוּחַ בְּסוֹד אֲנִי מְדַבֵּר וּמְנַבֵּא עֲלֵיהֶם.
לֹא יָדַע עֶבֶד לִנְצֹחַ אֶת אֲדוֹנָיו, אֶלָּא אוֹתוֹת אֲנִי נוֹתֵן וּמְנַבֵּא אֲנִי עָלָיו.
וַאֲשֶׁר יִרְאוּ אוֹתוֹ, לֹא בָּעוֹלָם הַזֶּה, אֶלָּא בָּעוֹלָם הַבָּא.
הוּא אֲשֶׁר שֶׁקֶר אֵין בּוֹ, וְלֹא רְדִיפַת עָם.
גַּם נְהָרוֹת אֲשֶׁר יָבְשׁוּ יִזְרְמוּ מַיִם בִּגְלָלוֹ,
וַאֲנָשִׁים אֲבוּדִים יִחְיוּ בִּגְלָלוֹ.

וְיָצְאוּ עַמִּים רַבִּים, מְחָרְפִים וּמְנָאֲצִים מֵאַחֲרֵי הָאֱלִילִים פִּתְאֹם,
וְיִהְיוּ יוֹרְשִׁים בְּמַלְכוּת הַשָּׁמַיִם.

פרק ב

תּוּב נִבָּא דָּנִיֵּאל וַיַּעַן וַיֹּאמַר:
שִׁמְעוּ שָׁמַיִם, וַאֲדַבֵּרָה, וְתִשְׁמַע הָאָרֶץ אִמְרֵי פִי.
שִׁמְעוּ חֵיל הַשָּׁמַיִם, מָה תַּעֲשׂוּ?
קָרְבָּן מֻקְרָב לַאדֹנָי – לֹא זָהָב וְלֹא כֶסֶף,
וְלֹא אֵיל וְלֹא צְפִיר, וְלֹא אֲתוֹן מִן הַחֲמוֹרִים.
כִּי פַת לֶחֶם עַל כַּף יַד אָדָם מֻקְרָב קָרְבָּן לַאדֹנָי.
אָז הֵם יִהְיוּ כֹּהֲנֵי אֲדֹנָי, אָז כַּבְּדוּם לְכַהֵן לַאדֹנָי.
וְהִשָּׁמְרוּ בַּעֲבוֹדַת אֲדֹנָי.
שְׁפָטוּ בֵּינֵיהֶם, כִּי הֵם יִשְׁפְּטוּ אֶתְכֶם.
אֲהֵבוּם בְּאַהֲבָה וְהִשְׁתַּחֲווּ לָהֶם בְּתָמִים.

אֲבָל בָּעֵת הַזֹּאת דְּעוּ וְהִשְׁתַּחֲווּ לַחַיִּים וּלְכָבוֹד אֱלֹהִים.

פרק ג

וְאַחֲרֵי הַדְּבָרִים הָאֵלֶּה, כַּאֲשֶׁר הָיִיתִי אֲנִי דָּנִיֵּאל בְּאֶרֶץ פָּרַס וֶעֵילָם, בִּימֵי דָּרְיָוֶשׁ הַמֶּלֶךְ, נִגְלָה אֵלַי בְּרוּחַ הַקֹּדֶשׁ אֲשֶׁר יִהְיֶה בְּאַחֲרִית הַיָּמִים.
וָאֶכְתֹּב בְּמַרְאוֹת וָאֶחְתֹּם אֵת אֲשֶׁר יִהְיֶה לְדוֹר אַחֲרוֹן,
לְמַעַן יָבִינוּ הַנְּבוֹנִים, כִּי תִּהְיֶינָה צָרוֹת וְרַעַשׁ עַל הָאָרֶץ, וּכְלָיוֹן רְצוֹן אֲדֹנָי.

וְיִפְנֶה אֲדֹנָי אֶת עֵינָיו מִן הָאָרֶץ, וְיַשְׁלֵךְ בָּהּ מְהוּמָה,
וְיָסִית אֶת הָאָדָם בְּאָחִיו,
וְיָקִים רַבִּים, בְּהַכּוֹתוֹ בַּשֵּׁבֶט.
וְיַחֲלֹק בּוֹ אֶת הָעוֹלָם,
וְיַכֶּה בּוֹ עַמִּים רַבִּים,
וְיַצְמִיחוּ שְׂעָרָם כְּנָשִׁים,
וּכְבִגְדֵי שַׂק יִלְבְּשׁוּ.
וְיִוָּסֵר בָּהֶם הָעוֹלָם,
וְיוֹצִיאֵם מִמְּקוֹמָם,
וְיוֹלִיכֵם אֶל מְקוֹמוֹת אֲחֵרִים.
וְיִכָּתֵב עֲלֵיהֶם: לְאַרְבַּע רוּחוֹת הַשָּׁמַיִם.
וְיִשְׂמְחוּ מִמִּזְרָח לְמַעֲרָב, וּמִצָּפוֹן לְתֵימָן.
וְיִתֹּצוּ חוֹמוֹת גְּבוֹהוֹת, וְעַד אָרֶץ יַשְׁפִּילוּם.
וּבֵית הַפֶּסֶל יַהֲרֹסוּ,
וּבֵית הַמֶּלֶךְ לֹא יַחְמֹלוּ,
וְרֹאשׁ הַמֶּלֶךְ בְּמַחֲזִיתוֹ יִתְקַעֲעוּ –
וְלֹא יַחְמֹלוּ.

וְתֵצֵא חֶרֶב בָּאָרֶץ,
וְיִכְרְתוּ זְקֵנִים עִם בַּחוּרִים,
אָבוֹת עִם בָּנִים –
וְלֹא יַחְמֹלוּ.
יַהֲרֹג בְּתוּלוֹת,
וְיִזְרְקוּ דָּמָן עַל גִּלּוּלֵיהֶן,
יְקַרְקְרוּ זָהָב מִפֶּסֶל,
וּבָאֵשׁ יִשְׂרְפוּם.
וְיִקְּחוּ אֶת מִשְׁפָּטָם מִלִּפְנֵי אֱלֹהִים,
עַל רִשְׁעַת מַעַלְלֵיהֶם.
יִקְּחוּ אָזְנַיִם עִם אַף,
וְעֵינֵי אָדָם יְעַוֵּרוּ,
וְיָדַיִם וְרַגְלַיִם יְנַתְּחוּ –
וְלֹא יַחְמֹלוּ.
וְיַשְׁחִיתוּ בְּקֶצֶף עָרִים רַבּוֹת,
וְיִסְגְּרוּ נְפָשׁוֹת.

וְאֵלֶּה הֵם רְצוֹן אֲדֹנָי,
כִּי הֵרֵעוּ אֶת מַעַלְלֵיהֶם.

יָקִימוּ עַם עַל עַם, וְחֶרֶב עַל חֶרֶב – וְלֹא יַחְמֹלוּ.
וְצָבָא אָדָם יִצְעַק אֶל אֱלֹהִים,
וְלֹא יַעֲנֵם עַל רִשְׁעַת מַעַלְלֵיהֶם.
יַהֲפֹכוּ עַצְמוֹת מֵתִים הַשּׁוֹכְבִים,
מִנֹּחַ וְעַד הַדּוֹר הַזֶּה,
וְעַל הַחַיִּים לֹא יַחְמֹלוּ.
יִדְקְרוּ בְרֹמַח אֶת הַלֵּב,
וְיַעֲלוּ זֶרַע אִישׁ מִן הָאֲדָמָה.
יָנִיעוּ עִיר בַּיָּם, וְאֳנִיּוֹת יַטְבִּיעוּ.
יֹאכְלוּ אֶת הָרַךְ, וְאֶת הַיָּבֵשׁ בְּרַגְלֵיהֶם יִרְמֹסוּ.
אִישׁ לֹא יַעֲמֹד בִּפְנֵיהֶם.
יִבְנוּ גֶשֶׁר עַל הַנָּהָר,
וְאֹרַח בַּמִּדְבָּר יִסְלָלוּ.
יָשִׁימוּ שָׁמֵם מָקוֹם נוֹשָׁב,
וְיִבְנוּ מָקוֹם שָׁמֵם.
כָּל אֵלֶּה – רְצוֹן אֲדֹנָי הִתְגַּשֵּׁם.
וְיִתְהַלְלוּ בִּשְׂעָרָם וּבְאַדַּרְתְּ שַׂקָּם.
יִדָּמוּ לַיָּם – אֵלָיו כָּל הַנְּהָרוֹת יִשָּׂאוּ,
וּמִכָּל עִיר אֵלָיו יָבִיאוּ.

יִגְדַּל הָרָשָׁע,
וְיוֹצִיא אֶת רוּחוֹת הָאָדָם כְּאֵיל הָעֵדֶר.
אֵלֶּה הַחוֹגְרוֹת יְחַלְּצוּ אוֹתָן מִמַּתְעִים,
וְאֵלֶּה הַמַּתְעִים יַחְגְּרוּ אוֹתָן.
יִתְאַכְזָר בֵּן בְּאָבִיו וְיֹאמַר: לֹא אָבִי אַתָּה.
וְעֶבֶד יִשְׁוֶה לַאדֹנָיו.
שִׁפְחָה תֵּשֵׁב וּגְבִרְתָּהּ תַּעֲמֹל.
וְיֶלֶד עַל זָקֵן יִשְׁכַּב,
וְלִפְנֵיו יִשְׁתֶּה.
אָז יִתְאַוּ אֲנָשִׁים לְמַאֲכַל כֶּלֶב.
יָקוּם אָדוֹן לְמִשְׁפָּט עִם עַבְדּוֹ,
וְהָעֶבֶד יִנְצַח וַאדֹנָיו יִכָּנַע.

פרק ד

אֲבָל בָּעֵת הַזֹּאת יִרְגְּזוּ חֵילוֹת הַשָּׁמַיִם.
כַּאֲשֶׁר מַלְכֻיּוֹת יִלָּחֲמוּ אֵלֶּה בָּאֵלֶּה,
יָקוּם מֶלֶךְ מִמִּזְרָח,
זֶהוּ הַשֵּׁבֶט אֲשֶׁר יִקְבֹּץ בּוֹ אֲדֹנָי אֶת הָעוֹלָם.
וְיִרְגְּזוּ עַמִּים רַבִּים,
וְתֹאבַד אֲגֻדָּתָם בֶּחָרֶב.

וְעַם עִיר הַפָּסֶל יֻכֶּה וְיִרְגַּז
עַל רִשְׁעַת מַעַלְלֵיהֶם.
וְיִהְיֶה עַד עֵת,
וְעֵת תַּעֲבֹר אֶל עֵת.
וְתִהְיֶה לוֹ צְפִיָּה בָּאָרֶץ.
וְעַמִּים רַבִּים יַגִּידוּ אִישׁ לְאָחִיו
כִּי נִסְּתָה חֶרֶב מִן הָאָרֶץ,
וְתִהְיֶה מְנוּחָה לָאָדָם.
וְלֹא יֵדְעוּ בְּנֵי אָדָם כִּי חֶרֶב נֶעֱצָרָה עַל פְּנֵי הָאָרֶץ.
וְתִהְיֶה צָרָה וּרְעִידָה הַסּוֹחֶפֶת נְפָשׁוֹת אָדָם,
וַאדֹנָי יִשְׁפֹּךְ דָּם עַל פְּנֵי הָאָרֶץ.
וְתִסְבֹּל חַיַּת הַשָּׂדֶה וְעוֹף הַשָּׁמַיִם וְתִשָּׁאֵר.

אֲבָל בָּעֵת הַזֹּאת יִתְנַגְּשׁוּ בְנֵי אָדָם אִישׁ בְּאָחִיו,
וְיֹאמַר אִישׁ לְרֵעֵהוּ וְאִשָּׁה לְרְעוּתָהּ:
מֵאַיִן בָּאת, וְלְאָן תֵּלֵךְ?
וְיֵרָאוּ כְּלַהֲבֵי אֵשׁ וּכְמַרְאֵה חֶרֶב.
וְיַעַל מֶלֶךְ מִמִּזְרָח וְיֵלֵךְ עַל פְּנֵי כָל הָאָרֶץ וְהַיָּם,
וְיִלְכֹּד עַמִּים וְיִרְמֹס עָרִים גְּדוֹלוֹת,
וְיֵלֵךְ מִיָּם לְיָם.
וּבָעֵת הַהִיא יִמְרְדוּ עַמִּים מִצָּפוֹן,
וְרַעַשׁ גָּדוֹל יִהְיֶה עַל פְּנֵי הָאָרֶץ.
וְיִהְיֶה קוֹל כְּקוֹל מַלְאָךְ,
וּכְהֲמוֹן מַחֲנוֹת מִן הַשָּׁמַיִם יִשָּׁמַע,
וְפַחַד גָּדוֹל יִהְיֶה מִן הַשָּׁמַיִם.
עַד הָרִים גְּבוֹהִים בְּתוֹךְ הַשָּׁדֶה יַחְשִׁיכוּ.
כִּי בְסוֹף הַדְּבָרִים הָאֵלֶּה
יִרְגְּזוּ חֵילוֹת הַשָּׁמַיִם,
וּמַלְכֵי הָאָרֶץ יִלָּחֲמוּ אֵלֶּה בָּאֵלֶּה.
וְיִרְגְּזוּ עִמָּהֶם חֵיל מִזְרָח וּמַעֲרָב וְתֵימָן וְצָפוֹן.
בָּעֵת הַזֹּאת תִּהְיֶה רַחֲמִים וְתַחֲנוּנִים מֵאֱלֹהִים עַל הָאָרֶץ.
וְיִפָּתְחוּ דְּלָתַיִם,
וְקוֹל יִשָּׁמַע,
וְיִוָּשְׁרוּ...
יִשָּׁבֵר,
כִּי מִשְׁפַּט אֲדֹנָי יִוָּשַׁר,
וְחַיֵּי אֱלֹהִים יֵרָאוּ.
וְיִקַּח כָּל אִישׁ אֶת כְּבוֹד אֲדֹנָי,
וְכָל הַדּוֹרוֹת יָבִינוּ.

בָּעֵת הַזֹּאת, הַנִּסְתָּר יִגָּלֶה:
שֶׁיִּפָּרֵד עַם מֵעַם,
וְשֶׁהַחַלָּשׁ לֹא יִבָּלַע בַּעֲצוּמִים.
אֲבָל כַּאֲשֶׁר רְאִיתֶם שֶׁנַּעֲשׂוּ הַדְּבָרִים הָאֵלֶּה,
שְׂאוּ נַפְשְׁכֶם לַצֶּדֶק, וְהַאֲמִינוּ בְּחַיֵּי אֱלֹהִים.
כִּי קָרוֹב רְצוֹן אֲדֹנָי,
כִּי קָרוֹב רְצוֹן אֲדֹנָי אֲשֶׁר יִתְגַּשֵּׁם.

פרק ה

תּוּב: קָרוֹב הַמֶּלֶךְ אֲשֶׁר יָרִים נַפְשׁוֹ,
וְעַד הַשּׁוֹר אֲשֶׁר קַרְנֵי רֹאשׁוֹ גְּבוֹהוֹת
וְפַרְסוֹת רַגְלָיו יְשָׁרוֹת.
וְלֹא יִגְדַּל בְּעֵינָיו אוֹצַר הַמְּלָכִים,
וְלֹא יְכֻבְּדוּ עֲטָרוֹתֵיהֶם.
נִשְׁתַּגַּע מַחְשְׁבוֹת רַבִּים בְּנֶפֶשׁ אֱלֹהִים.
אִם לֹא יָרִים נַפְשׁוֹ וְיֵרָאֶה – לֹא יַכִּירוּהוּ.
יֵעוֹר מִמִּזְרָח,
וְאָחוֹר לֹא יִפְנֶה
מִפְּנֵי הַמִּרְמָה.
וְגַם לַמֶּמְשָׁלָה יַעֲבֹר בַּמַּעֲרָב,
וְרַבִּים יֵעוֹרוּ מִמֶּנּוּ.
וְאַחֲרֵי כֵן יִנְצְחֶנּוּ אֲדֹנָי אֶת גְּדִי הָעִזִּים,
וְיִרְמֹס אֶת הַשּׁוֹר לְמַרְאֵה עֵינַיִם,
וְגַם לִבְחֹן אוֹתוֹ.
וְלִפְנֵי פָּנָיו לֹא יָקוּם,
כִּי הֵבִיא עָלָיו קֶצֶף גָּדוֹל,
וְיוֹם נָקָם מִלִּפְנֵי אַפּוֹ שֶׁל אֲדֹנָי.

אֲבָל כַּאֲשֶׁר רְאִיתֶם שֶׁנִּשְׁלְמוּ הַדְּבָרִים הָאֵלֶּה,
דְּעוּ כִּי הָיְתָה שִׂמְחָה עַל הָאָרֶץ,
וּמַמְלָכָה אַחַת עִם אַחַת הִתְרַצְּתָה,
וְיַחְדָּו אָכְלוּ וְשָׁתוּ,
וְקָמוּ וְהֶחֱרִיבוּ אֶת הַחַדָּה,
וְקָמוּ – הֶחֱרִיבוּ אֶת הַחַדָּה.

פרק ו

אֲבָל בָּעֵת הַזֹּאת,
כַּאֲשֶׁר רְאִיתֶם גּוּר הָאַרְיֵה שֶׁנֶּאֱסַף לְמָקוֹם אֶחָד,
וְיֵעוֹרוּ חֵיל צָפוֹן –
דְּעוּ כִּי הִגִּיעָה לוֹ הַקֵּץ, הוּא צְפִיַּת הָעוֹלָם.

וּכְאֲשֶׁר רְאִיתֶם זִמָּה וְאָסוֹן וָזַעַם שֶׁל מְלָכִים,
שֶׁהִתְאַכְזְרוּ אֵלֶּה בָּאֵלֶּה וְהֵם מִמָּקוֹם אֶחָד,
דְּעוּ כִּי בִּגְלָלָם הֶחְשִׁיכָה הָאָרֶץ.
וְאַחֲרֵי כֵן, בְּהֵרָאוֹת חֵילוֹת הַשָּׁמַיִם,
תִּהְיֶה חֵן עַל הָאָדָם.

בָּעֵת הַהִיא,
תִּהְיֶה מִלְחָמָה עַל מִלְחָמָה,
וְשֹׁד עַל שֹׁד,
וְיִמָּסְרוּ בְנֵי הָאָדָם לְפִי חֶרֶב,
וַאדֹנָי לֹא יְכַלֶּה אֶת הָעוֹלָם.
וְתִהְיֶה קְלָלָה עַל הָאָרֶץ.

אִלּוּ הָיְתָה בִּימֵי אָדָם,
לֹא הָיָה מְקֻלָּל וְלֹא בָּרַח מִן הַגַּן.
וְאִלּוּ הָיְתָה בִּימֵי נֹחַ,
לֹא הָיָה הַמַּבּוּל עַל הָאָרֶץ,
וְלֹא הָיְתָה כָּל בָּשָׂר נֶעֱנֶשֶׁת.
וְאִלּוּ הָיְתָה בִּימֵי לוֹט,
לֹא הָיָה יוֹרֵד אֵשׁ וְגָפְרִית מִן הַשָּׁמַיִם,
וְלֹא הָיְתָה שׂוֹרֶפֶת אֶת יוֹשְׁבֵי סְדוֹם.
כִּי תִּהְיֶה תְּפִלָּה עַל הָאָרֶץ בָּעֵת הַזֹּאת,
אֲשֶׁר לֹא הָיְתָה מֵרֵאשִׁית וְעַד הַיּוֹם הַזֶּה.

פרק ז

וַאֲנִי דָּנִיֵּאל – כָּבַד עָלַי בָּעֵת הַהִיא,
אֲשֶׁר בֵּית אֱלֹהִים וּבֵית קְהַל הָעַמִּים הִתְעוֹרְרוּ מִן הַכְּסִילִים.
יִבְכּוּ בָנִים,
וְתִשָּׁמַע קוֹל אוֹהֲבִים,
וְיֻדָּחוּ אֶל מְקוֹמוֹת שֹׁנִים,
כִּי מְדַבְּרִים עַל הַקֵּץ
וּמַרְאִים אֶת הַכָּבוֹד לָרַבִּים.

כִּי אֵלֶּה אֲשֶׁר יִשְׁמְעוּ אֶת הַדְּבָרִים וְיַאֲמִינוּ בָּהֶם –
יִרְאוּ אֲשֶׁר עַיִן לֹא רָאֲתָה
וְאֹזֶן לֹא שָׁמְעָה,
וְעַל לֵב אָדָם לֹא עָלָה.
כִּי מְלָכִים וַאֲדִירִים נִקְבְּצוּ לִרְאוֹת וְלֹא רָאוּ.
כִּי הוּא אֵצֶל אֵלֶּה שֶׁרָאוּהוּ וְעוֹד יִרְאוּהוּ.
לֹא בִּמְקוֹם מֶלֶךְ וְשַׂר הוּא,
אִם לֹא שְׁנֵיהֶם – אָב וָבֵן.

וְהִנֵּה רַבִּים מִתְעוֹרְרִים – אֲנִי אֲשֶׁר אֲדַבֵּר,
וְאָנֹכִי זָקֵן אֲשֶׁר חִפַּשְׂתִּי לְפָרֵשׁ עֲלֵיהֶם.
וְאַעֲזֹב לִי שֵׁם בָּעַמִּים,
שֶׁבִּגְלָלִי רַבִּים יַאֲמִינוּ וְלֹא יֵבוֹשׁוּ.

בָּעֵת הַזֹּאת
תִּסָּתֵר הַחָכְמָה מֵרַבִּים,
גַּם מִבָּנִים,
וְהַכָּבוֹד יֻקַּח מֵאֲדוֹנֶיהָ.
וְאֵלֶּה הַמְשָׁרְתִים אֶת הָעַמּוּד יִוָּדְעוּ מִמֶּנּוּ,
וְהִיא עֵדוּת עֲלֵיהֶם.
וַאֲנִי עֵד אֲנִי לַדְּבָרִים הָאֵלֶּה אֲשֶׁר אֲנִי מְדַבֵּר.

אֲבָל הָאֲנָשִׁים אֲשֶׁר הָיוּ,
וְהַדּוֹרוֹת אֲשֶׁר הָיוּ,
וְהָעָם אֲשֶׁר יֵאָמֵר עָלָיו "לַאדֹנָי" –
הֵם יִקְּחוּ אֶת כְּבוֹד אֱלֹהִים.

כִּי כַּאֲשֶׁר רְאִיתֶם שֶׁנַּעֲשׂוּ כָּל הַדְּבָרִים הָאֵלֶּה,
אֲשֶׁר מִלְמַעְלָה הֵם וְהָיוּ בָּאָרֶץ –
וַאֲנִי לֹא יָכֹלְתִּי לְדַבֵּר וּלְפָרֵשׁ עֲלֵיהֶם.
אֶת אֲשֶׁר לֹא פֵּרַשׁ אַחֵר – אֲנִי פֵּרַשְׁתִּי.

פרק ח

תּוּב: יֵרָאוּ הָאוֹתוֹת הָאֵלֶּה בָּאָרֶץ,
וְפַחַד בָּאִיִּים,
וְאֵשׁ תִּבְעַר בָּהֶן יוֹמָם וָלָיְלָה.
בָּעֵת הַהִיא,
הַשֶּׁמֶשׁ כַּשַּׂק תִּתְעַטֵּף,
וְהַיָּרֵחַ דָּם תִּלְבַּשׁ.
וְתֵרָעַשׁ הָאָרֶץ וְהַיָּם,
וְרַבִּים מִבְּנֵי אָדָם בְּזַעַם יִפֹּלוּ.

בָּעֵת הַהִיא יִהְיֶה מִרְמָה בָּאָרֶץ:
יִמְאַס בֵּן בְּאָבִיו,
וְאָח בְּאָחִיו,
וְרֵעַ בְּרֵעֵהוּ יַעְשִׁיק.
וְיִוָּסֵר אֱלֹהִים אֶת הָאָרֶץ בַּיָּמִים הָהֵם,
וְתִהְיֶה רָעָב גָּדוֹל וָדֶבֶר,
וְגַלְגַּל חָזָק,
וְחֹם וְיֵרָקוֹן וְאַרְבֶּה וְחָסִיל.
וְיֹאכְלוּ אֶת עֵשֶׂב הַשָּׂדֶה כָּל הָאָרֶץ בַּיָּמִים הָהֵם.
חֹשֶׁךְ יְכַסֶּה אֶת הָאָרֶץ,
וַאֲפֵלָה לַדּוֹרוֹת,
וְדִין הָאָרֶץ – עַוְלָה.
וְתַהַו וְתֹהוּ תַּהַר, וְתֵלֵד אָוֶן.
וְיִכָּלֵא הַטַּל מִן הֶעָבִים,
וְתִפֹּל אֵשׁ מִן הַשָּׁמַיִם עַל הָאָרֶץ,
וְתִבְעַר עַל פְּנֵי הַצָּפוֹן,
וְתִבְעַר יוֹמָם וָלָיְלָה,
וְתֹאכַל אֶת הֶעָפָר וְאֶת הָעֵץ וְאֶת שָׁרְשֵׁיהֶם.

בַּיָּמִים הָהֵם תֵּרָעַשׁ הָאָרֶץ,
וְיָקוּם עַם עַל עַם וְעִיר עַל עִיר,
וְעִיר וּמְבוּצָה אַחַת עַל אַחַת תִּמְרֹד.
בַּיָּמִים הָהֵם יֵצְאוּ מַלְאֲכֵי זַעַם
לְאַרְבַּע רוּחוֹת הַשָּׁמַיִם
לַעֲשׂוֹת נִקְמַת זַעַם מִן הָאָרֶץ.
יִתְוַעֲדוּ לְהַשְׁמִידָם וּלְהַכּוֹתָם בַּחֶרֶב
וּבַדֶּבֶר וּבְנִסָּיוֹן רַב אֲשֶׁר כָּל דִּין.

וְאַחֲרֵיהֶן תִּהְיֶה מְנוּחָה בָּאָרֶץ,
וְיִרְבֶּה עוֹלָם,
וְיִכָּרְתוּ עַמִּים.
וְתֵרָעַשׁ הָאָרֶץ מִיּוֹשְׁבֶיהָ,
וְיִתְמַלְּאוּ יָמִים וּגְזֵרָה, עֲמוֹרָה וְדוֹר.
תִּבָּנֶה עִיר, וְיִרְבּוּ מָחוֹזִים.
וְתִתְקַשֵּׁט הָאָרֶץ וְהַיָּם
בַּמְּצָדִים וּבֶעָרִים וּבְהֵיכָלוֹת וּבִבְנִינִים.
תִּבָּנֶה עִיר עַל הַר,
וְחוֹמָה וּמִגְדָּל בַּבִּקְעָה.

וּמִן הַמְּנוּחָה יֵרָגְזוּ רוּחוֹת הַשָּׁמַיִם,
וּמַלְאַךְ הַשָּׁמַיִם יֵלֵךְ עָלֶיהָ,
וְתֵרָעַשׁ הָאָרֶץ מִיּוֹשְׁבֶיהָ,
וְיָם וּגְזֵרָה מִיּוֹשְׁבֵיהֶם.

וּבְכָל מָקוֹם יִרְבּוּ:
תֹּהוּ הוֹן אֶרֶץ,
וְרוּחַ תִּנָּשֵׂא.
גַּם הֵיכָלוֹת לְמִרְמָס יִהְיוּ,
וּבְנִינִים לְחָרְבָּה.

בָּעֵת הַהִיא
יִתְהַפְּכוּ הָרוּחַ וְהַיָּם,
וְעָפָר מִן הַשָּׁמַיִם עַל הָאָרֶץ יֵרֵד,
וְהָרִים יִהְיוּ לְאֵפֶר,
וּנְהָרוֹת רַבִּים יִיבָשׁוּ.
וְיִקָּצְרוּ הַיָּמִים,
וִימֵי הַחֹדֶשׁ יִתְחַלְּפוּ,
וְיִתְחַלְּפוּ עִתֵּי הַשָּׁנָה,
וְמַחֲזוֹר הַשֶּׁמֶשׁ וְהַיָּרֵחַ יֵרָאֶה כָּזָב.
בָּעֵת הַהִיא
יֵאָסְרוּ הָרוּחוֹת וְלֹא יִנָּשְׁבוּ,
וְיִתְאַוּוּ הֶעָבִים אֶל הָרָקִיעַ וְלֹא יֵלֵכוּ,
וּמָטָר הַשָּׁמַיִם לֹא יֵרֵד.
תֶּחֱשַׁךְ נֹגַהּ הַשָּׁמַיִם,
וְנֹגַהּ הַיָּרֵחַ לֹא תֵּרָאֶה,
וְגַם כּוֹכָב לֹא יָאִיר,
וְחֹשֶׁךְ וַאֲפֵלָה עַל פְּנֵי הָאָרֶץ
יִהְיֶה חָזָק.
וְתִרְבֶּה הָרָעָה עַל הָאָרֶץ וְעַל יוֹשְׁבֶיהָ,
וְתִגְרַע הָאָרֶץ אֶת יְבוּלָהּ,
וְהֶהָרִים אֶת יְסוֹדָם.

וְקוֹל יִשָּׁמַע מִן הַשָּׁמַיִם,
וְקוֹלוֹ וּפַחַד בֶּהָר יִהְיוּ,
וְאֵימָה וָרַעַד בְּכָל מָקוֹם,
וְקוֹלוֹת וּרְעָמִים וּסְעָרוֹת.
וּמַלְאֲכֵי הַשָּׁמַיִם כִּבְנֵי אָדָם יֵרָאוּ עַל הָאָרֶץ,
וְחֹשֶׁךְ עַל הָאָרֶץ יִהְיֶה.

וְיָמִים שְׁלֹשָׁה
הַשֶּׁמֶשׁ לֹא תֵּרָאֶה,
וְהַיָּרֵחַ לֹא יָאִיר,
וְכוֹכָב לֹא יֵרָאֶה.
וְיִזְעוּ הָרֵי הָאָרֶץ,
וּמַעְיְנוֹת הָאָרֶץ יִיבָשׁוּ.
וּנְהָרוֹת רַבִּים לְמִסְתָּרִים יִפְנוּ.

בָּעֵת הַהִיא
יִכָּסֶה מְצָד בַּיָּם,
וְעִיר בַּיָּם תִּגָּרֵף,
וּמְקוֹמוֹת רַבִּים יֵחָרְבוּ מִנְּחָשִׁים.
וְעַמִּים רַבִּים בֶּחָרֶב יָסֹבּוּ,
וְגַם עִיר רַבָּה בַּדֶּבֶר תֹּאבַד.
כְּרָכִים וּכְפָרִים בָּאֵשׁ יִצַּתּוּ.
בָּעֵת הַהִיא
תִּרְבֶּה הַחֵמָה עַל הָאָרֶץ,
וְטוֹבָה בָּעוֹלָם תִּסָּגֵר.
וְתִרְבֶּה הַחַטָּאת בָּאָרֶץ,
וְהָרָשָׁע רֹאשׁוֹ יָרִים.
וְיִמְעֲטוּ וְיִדַלְּלוּ,
וְיִרְבּוּ רוּחוֹת וּפְגָעִים,
וְיֵצְאוּ לְעוֹרֵר תְּבוּסָה בָּאָרֶץ.
וְהַנָּחָשׁ אֲשֶׁר בָּאָרֶץ יִמְשֹׁל בִּבְנֵי אָדָם,
כִּי יִשָּׁחֵתוּ דִּבְרֵי בְנֵי אָדָם.
וְתִקָּרַע הָאָרֶץ לִרְסִיסִים רַבִּים,
כְּמוֹ שִׂמְלָה תִּקָּרַע,
וְתָבֹא עֲדֵי תְהוֹם.

וּבְנֵי אָדָם רַבִּים חַיִּים יִבָּלְעוּ בְּתוֹךְ הָאָרֶץ,
וְהַר יִזַּע מִמְּקוֹמוֹ,
וְגִבְעָה תָּמוּשׁ מִמּוֹסָדָהּ.

פרק ט

וְיֵרָאוּ עַמּוּדֵי אֵשׁ בַּמִּזְרָח,
וְתַנּוּר לֶהָבָה מִתּוֹךְ עָנָן.
וְיֵרָאוּ בָּרָקִים בַּשָּׁמַיִם
כְּסוּסֵי אֵשׁ וּכְמֶרְכָּבוֹת קְרָב
וְכָל צָבָא בְּתַנּוּר אֵשׁ.
כַּאֲשֶׁר יֹאחֲזוּ חֶרֶב בַּרְזֶל וְקֶשֶׁת קְרָב.

מֶלֶךְ מִמִּזְרָח יָבוֹא,
וְיָסֹף אֶת כָּל רוּחוֹת הַשָּׁמַיִם.
וּקְרָב גָּדוֹל וְעָצוּם יִהְיֶה,
כְּמַשָּׂא יִכָּבְסוּ,
וְיָבוֹא עַל הָאָרֶץ.
וְתִשָּׁבַר קַרְנָהּ,
וְתִּלָּכֵד מַלְכוּת הָאָרֶץ,
וְיֻשְׁלַךְ בָּהּ מְהוּמָה.
וּמִכָּל צִדָּיו כָּל מְשַׂנְאָיו יְבַקְשׁוּ אֶת נַפְשׁוֹ,
לִפְנֵי מְאַהֲבָיו לֹא יַעֲמֹד.
וּמִדָּם אֲשֶׁר לָוָה מִמִּזְרָח תִּפְרַע,
וּמִדָּם אֲשֶׁר בָּלַע תָּקִיא.
וְיִתְחַזֵּק קֶרֶן מַעֲרָב,
וְיִתְפַּזְּרוּ רוּחוֹת הַשָּׁמַיִם,
וְהִיא תִּמְשֹׁל עַד קְצֵה הָאָרֶץ.

כַּאֲשֶׁר רְאִיתֶם כִּי נִשְׁלַם הַדָּבָר הַזֶּה,
דְּעוּ כִּי הִגִּיעָה לוֹ הַקֵּץ.

פרק י

הָאַנְטִיכְרִיסְטוֹס יִהְיֶה אַחֲרֵי הַיָּמִים הָאֵלֶּה:
אִשָּׁה תֵּלֵד זָכָר מִשֵּׁבֶט לֵוִי.
וְיֵרָאוּ בּוֹ הָאוֹתוֹת הָאֵלֶּה:
יִצֻּיְּרוּ עַל בְּשָׂרוֹ כִּכְלֵי מִלְחָמָה –
חֲנִית וְקֶשֶׁת וְסַיִף וְרֹמַח,
וְחֶרֶב בַּרְזֶל,
וְגַם מֶרְכָּבַת קְרָב.
פָּנָיו כְּפָנֵי תַנּוּר בּוֹעֵר,
וְעֵינָיו כְּגֶחָלֵי לֶהָבָה.
וְקֶרֶן אַחַת יֵשׁ לוֹ בֵּין עֵינָיו,
וְרֹאשָׁהּ נִשְׁבָּר.
וְיֵצֵא מִמֶּנּוּ כְּמַרְאֵה נָחָשׁ.

כַּאֲשֶׁר רְאִיתֶם שֶׁיִּהְיוּ הַדְּבָרִים הָאֵלֶּה,
דְּעוּ כִּי קְרוֹבָה בִּאָתוֹ
שֶׁל הָאֱוִיל לָבוֹא מִכַּנְפֵי מִזְרָח.

וְיִגָּלֶה בָּעֵת הַזֹּאת בֶּן הָאֲבַדּוֹן עַל הָאָרֶץ,
לְהַתְעוֹת אֶת יוֹשְׁבֵי תֵבֵל.
וְיֹאמַר עַל נַפְשׁוֹ: אֲנִי הַמָּשִׁיחַ.
וְיֵצֵא מִבֶּטֶן צִפְעוֹנִי וּמִמֵּעֵי נָחָשׁ.
וְיָבוֹאוּ עַמִּים וּמְלָכִים גְּדוֹלִים,
וְמִשְׁמָרוֹת רַבּוֹת.

וְאֵלֶּה הֵן אוֹתוֹתָיו:
מַרְאֵה קוֹמָתוֹ וְרֹאשׁוֹ גָּדוֹל,
שְׂעָרוֹתָיו כְּרֹאשׁ תַּרְנְגוֹל,
עֵינָיו זוֹהֲרוֹת,
גַּבּוֹתָיו רָמוֹת.
וּשְׁתֵּי קְרָנַיִם יֵשׁ לוֹ בְּאָזְנָיו,
וּבָשָׂר יָתֵר יֵשׁ לוֹ בְּתוֹךְ אָזְנוֹ.
וְעָרְפּוֹ חָזָק וַחֲזֵהוּ רָחָב,
זְרוֹעוֹתָיו אֲרֻכּוֹת,
וְאֶצְבְּעוֹתָיו קְצָרוֹת.
מַרְאֵהוּ זָעוּף,
מַפְחִיד וְחַם אַף,
וּמַרְאֵה תִמְהָה קוֹמָתוֹ,
וְכִלְפִיד אֵשׁ בְּתוֹךְ מַחֲנוֹת קְרָב.
קַלִּים מִנָּמֵר סוּסָיו,
וְחַרִיפִים מִזְּאֵב עָרְבִּית רִכְבּוֹ.
קוֹמָתוֹ מְרוֹמָמֶת לְמַעְלָה מֵהַר,
וְעִם עָנָן בַּשָּׁמַיִם מִשְׁתָּוָה.
וְעִמּוֹ צְבָא נְחָשִׁים.
