חלק ו' – חזון הבריאה, החטא הקדמון וגורל האנושות (פרקים כ"א-ל"ב)
מה וָאַבִּיט מִן-הָהָר אֲשֶׁר עָמַדְתִּי עָלָיו, מַטָּה אֶל-הָרָקִיעַ הַחֲמִישִׁי, וָאֵרֶא שָׁם נַפְשׁוֹת מַלְאָכִים בְּלִי גוּף, וְהָמוֹן הַמְּבַצְּעִים אֶת-צִוּוּיֵי מַלְאֲכֵי הָאֵשׁ אֲשֶׁר בָּרָקִיעַ הַשְּׁמִינִי.
מו וָאֵרֶא שָׁם, בָּרָקִיעַ הַשִּׁשִּׁי, כֹּחוֹת שֶׁל כּוֹכָבִים, הַמְבַצְּעִים אֶת-צִוּוּיֵיהֶם, וְהֵם נִשְׁמָעִים לְצִיוּוּיֵי הָאָרֶץ.
מז וַיֹּאמֶר אֵלַי הַקַּדְמוֹן הֶחָזָק: "אַבְרָם, אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמֶר: "הַבֵּט לְמַעְלָה בַּכּוֹכָבִים אֲשֶׁר מִתַּחְתֶּיךָ, וּסְפֹר אוֹתָם, וְהַגֵּד לִי אֶת-מִסְפָּרָם". וָאֹמַр: "אֵיךְ אוּכַל, כִּי אֲנִי עָפָר וָאֵפֶר?". וַיֹּאמֶר לִי: "כְּמִסְפַּר הַכּוֹכָבִים וּכְכֹחָם אָשִׂים לְזַרְעֲךָ עַמִּים וְגוֹיִם, הַנִּבְדָּלִים לִי בְּגוֹרָלִי, עִם עֲזָאזֵל".
מח וָאֵרֶא שָׁם אֶת-גַּן-עֵדֶן וְאֶת-פִּרְיוֹ, וְאֶת-הַמַּעְיָן הַמּוֹצִיא מִמֶּנּוּ נָהָר, וְאֶת-עֲצֵי הַגָּן וְאֶת-פִּרְחֵיהֶם, וְאֶת-עוֹשֵׂי הַצֶּדֶק. וָאֵרֶא בּוֹ אֶת-מְזוֹנָם וְאֶת-מְנוּחָתָם. וָאֵרֶא שָׁם עַם רַב, גְּבָרִים וְנָשִׁים וִילָדִים, חֶצְיָם מִצַּד יָמִין שֶׁל הַתְּמוּנָה, וְחֶצְיָם מִצַּד שְׂמֹאל.
מט וָאַבִּיט בַּתְּמוּנָה, וַתִּזְלַגְנָה עֵינַי אֶל-צִדֵּי גַּן-עֵדֶן. וָאֵרֶא שָׁם אִישׁ גָּבֹהַּ מְאֹד בְּקוֹמָתוֹ וְנוֹרָא בְּרָחְבּוֹ, בְּלִי דִמְיוֹן בְּמַרְאֵהוּ, שָׁזוּר עִם אִשָּׁה, אֲשֶׁר גַּם-הִיא הִשְׁתַּוְּתָה לוֹ בְּמַרְאֶה וּבְקוֹמָה. וְהֵם עוֹמְדִים תַּחַת עֵץ אֶחָד שֶׁל גַּן-עֵדֶן. וּפְרִי הָעֵץ הָיָה כְּמַרְאֵה אֶשְׁכּוֹל תְּמָרִים.
נ וּמֵאֲחוֹרֵי הָעֵץ הָיָה עוֹמֵד נָחָשׁ, יָדָיו וְרַגְלָיו כְּשֶׁל אָדָם, וּכְנָפַיִם עַל-כְּתֵפָיו, שֵׁשׁ מִימִינוֹ וְשֵׁשׁ מִשְּׂמֹאלוֹ. וְהוּא אוֹחֵז בְּיָדוֹ אֶשְׁכּוֹל מִפְּרִי הָעֵץ, וְהַשְּׁנַיִם אֲשֶׁר רָאִיתִי שְׁזוּרִים יַחַד אָכְלוּ.
נא וָאֹמַр: "מִי הֵם אֵלֶּה הַשְּׁזוּרִים זֶה בָּזֶה, אוֹ מִי הוּא זֶה שֶׁבֵּינֵיהֶם, אוֹ מַהוּ הַפְּרִי שֶׁהֵם אוֹכְלִים, חָזָק וְנִצְחִי?". וַיֹּאמַр: "זֶהוּ מַחֲשֶׁבֶת בְּנֵי-הָאָדָם, זֶהוּ אָדָם, וְזֹאת חַוָּה. וַאֲשֶׁר בֵּינֵיהֶם – זֶהוּ חֹסֶר-הַכָּבוֹד, תְּחִלַּת דַּרְכָּם לָאֲבַדּוֹן, הוּא עֲזָאזֵל".
נב וָאֵרֶא שָׁם אֶת-הַתַּאֲוָה, וּבְיָדָהּ רֹאשׁ כָּל-חֹסֶר-חֹק, וְאֶת-הַמּוֹלִיכִים שׁוֹלָל, וְאֶת-גְּמוּלָם – הַשְׁלָכָה לָאֲבַדּוֹן.
נג וָאֹמַр: "מָהוּ הַפֶּסֶל הַזֶּה... וּמַהוּ הַהֵיכָל הַיָּפֶה... תַּחַת כִּסֵּא כְבוֹדֶךָ הַמְחֻלָּל?". וַיֹּאמֶр: "שְׁמַע, אַבְרָם. זֶה אֲשֶׁר רָאִיתָ – הַהֵיכָל וְהַמִּזְבֵּחַ וְהַיֹּפִי – זֶהוּ רַעְיוֹן הַכֹּהֲנִים שֶׁל שֵׁם כְּבוֹדִי... וְהַפֶּסֶל אֲשֶׁר רָאִיתָ – הוּא קִנְאָתִי, שֶׁבָּהּ יַקְנִיטוּ אוֹתִי הָעָם הַבָּא מִמְּךָ".
נד וָאֶעֱנֶה וָאֹמַр: "חָזָק וְקָדוֹשׁ... הַאִם יִהְיֶה זְמַן רַב לְכָל-אֲשֶׁר רָאִיתִי?". וַיֹּאמֶр: "בִּגְלַל אַרְבַּע הַמְּבוֹאוֹת שֶׁרָאִיתָ, יַכְעִיסוּ אוֹתִי, וּבָהֶם יִהְיֶה גְּמוּל מַעֲשֵׂיהֶם מִמֶּנִּי. בְּאַרְבַּע מְבוֹאוֹת, מֵאָה שָׁנָה, וְשָׁעָה אַחַת שֶׁל הָעוֹלָם, שֶׁהִיא מֵאָה שָׁנָה, תִּהְיֶה בָּרָעָה בְּקֶרֶב הַגּוֹיִם. וְשָׁעָה שֶׁל טוֹבָה וְשָׁעָה שֶׁל חֶרְפָּה בְּקֶרֶב הַגּוֹיִם".
נה וָאֹמַр: "נִצְחִי! וְכַמָּה זְמַן הִיא שְׁעַת הָעוֹלָם?". וַיֹּאמֶр: "שְׁתֵּים-עֶשְׂרֵה שָׁנָה קָבַעְתִּי לָעוֹלָם הָרָשָׁע הַזֶּה לִמְשֹׁל בַּגּוֹיִים וּבְזַרְעֲךָ. וְעַד קֵץ הַזְּמַן יִהְיֶה כָּל-אֲשֶׁר רָאִיתָ".
נו וָאֵרֶא אִישׁ יוֹצֵא מִצַּד שְׂמֹאל... וַיִּשְׁתַּחֲווּ לוֹ... וְעוֹדֶנִּי מִסְתַּכֵּל, וַיֵּצְאוּ מִצַּד יָמִין, וּקְצָתָם בִּיְּשׁוּ אֶת-הָאִישׁ, וּקְצָתָם הִכּוּהוּ, וּקְצָתָם הִשְׁתַּחֲווּ לוֹ. וָאֵרֶא כַּאֲשֶׁר הִשְׁתַּחֲווּ לוֹ, וַיָּרָץ עֲזָאזֵל וַיִּשְׁתַּחֲוֶה, וַיְּנַשֵּׁק אֶת-פָּנָיו, וַיִּסּוֹב וַיַּעֲמֹד מֵאֲחוֹרָיו.
נז וַיַּעַן וַיֹּאמֶр: "שְׁמַע, אַבְרָם. הָאִישׁ אֲשֶׁר רָאִיתָ מְבֻזֶּה וּמֻכֶּה וְאַחַр-כָּךְ נִסְגָד, הוּא הַגְּאוּלָה מִן-הַגּוֹיִים לָעָם הַבָּא מִמְּךָ בְּאַחֲרִית הַיָּמִים, בַּשָּׁעָה הַשְּׁתֵּים-עֶשְׂרֵה שֶׁל עוֹלָמִי הַמִּסְתַּיֵּם. אֶת-הָאִישׁ הַזֶּה אָקִים מִשִּׁבְטְךָ...".
נח וְאָז אֶתְקַע בַּשּׁוֹפָר מִן-הָאֲוִיר, וְאֶשְׁלַח אֶת-בְּחִירִי, וּבוֹ כָּל-כֹּחוֹתַי, בְּמִדָּה אַחַת. וְהוּא יִקְרָא אֶת-עַמִּי הַמְבֻזֶּה מִבֵּין הַגּוֹיִים. וַאֲנִי אֶשְׂרֹף בָּאֵשׁ אֶת-אֵלֶּה שֶׁחֵרְפוּ אוֹתָם וְשָׁלְטוּ בָּהֶם בָּעוֹלָם הַזֶּה. וְאֶתֵּן אֶת-אֵלֶּה שֶׁכִּסּוּ אוֹתִי בְּחֶרְפָּה לְחֶרְפַּת הָעוֹלָם הַבָּא. כִּי הֲכִינוֹתִי אוֹתָם לְמִשְׁתֵּה אֵשׁ הַגֵּיהִנּוֹם וּלְשִׁיטָפוֹן בִּלְתִּי-פּוֹסֵק בַּאֲוִיר, בִּתְהוֹמוֹת הָאָרֶץ, לְמַלֵּא בֶּטֶן שֶׁל תּוֹלַעַת רָעָה... וְיִרְאוּ בָּם עוֹשֵׂי הַצֶּדֶק... וְיִשְׂמְחוּ בְּשִׂמְחָה עַל-אַבְדַן הָאֲנָשִׁים הַנּוֹתָרִים... כִּי שָׁכְחוּ אוֹתִי וְהִלְּלוּ אֶת-הַזָּר.
נט "שְׁמַע, אַבְרָם, וּרְאֵה, זֶהוּ דּוֹרְךָ הַשְּׁבִיעִי. הֵם יֵלְכוּ עִמְּךָ, וְיֵצְאוּ לְאֶרֶץ נֵכָר, וְיַעַבְדוּ בָּם וִיעַנּוּ אוֹתָם, כְּשָׁעָה אַחַת שֶׁל עוֹלָם הָרְשָׁעָה. אַךְ אֶת-הַגּוֹי אֲשֶׁר יַעֲבֹדוּ, אָנֹכִי אָדוּן".
הערות וביאורים (ניתוח השוואתי)
השוואה נקודתית:
פסוק מ"ה-מ"ו: בלבול בסדר הרקיעים – גרסה ב' מציגה בלבול ברור בסדר הרקיעים. היא מתארת את הרקיע החמישי, ואז את השישי, אך מציינת שהמלאכים ברקיע החמישי מצייתים לאלה שברקיע השמיני (שכלל לא הוזכר). זהו שיבוש טקסטואלי מובהק, המעיד על העתקה פחות זהירה בהשוואה לגרסה א', שהציגה סדר הגיוני (שביעי > שישי > חמישי).
פסוק מ"ז: "אֲנִי עָפָר וָאֵפֶר" – ציטוט מקראי מדויק. בגרסה א', אברהם פשוט אומר "אני אדם". כאן, הוא משתמש בביטוי המדויק מבראשית י"ח, כ"ז. זה מראה על היכרות עמוקה יותר עם לשון המקרא אצל עורך גרסה ב'.
פסוק מ"ט: "אֶשְׁכּוֹל תְּמָרִים" – שינוי בתיאור פרי עץ הדעת. בגרסה א', הפרי היה "כמראה אשכול ענבים". כאן, הוא "כמראה אשכול תמרים". זהו שינוי קטן, אך הוא מעיד על מסורות שונות לגבי זהותו של הפרי האסור.
פסוק נ"ד: "וְשָׁעָה שֶׁל טוֹבָה וְשָׁעָה שֶׁל חֶרְפָּה" – תוספת המדגישה את הדואליות. גרסה א' דיברה רק על "שעה של רעה". גרסה ב' מוסיפה "שעה של טובה ושעה של חרפה", מה שמחזק את התפיסה שההיסטוריה מורכבת מתקופות מתחלפות של טוב ורע, אור וחושך.
פסוק נ"ז: "אֶת-הָאִישׁ הַזֶּה אָקִים מִשִּׁבְטְךָ" – שינוי קטן אך משמעותי. בגרסה א' נאמר "מזרעך". כאן נאמר "משבטך". המילה "שבט" עשויה לרמוז באופן ספציפי יותר לשבט יהודה, מה שמחזק עוד יותר את הקישור לדמותו של ישוע.
פסוק נ"ח: "וּבוֹ כָּל-כֹּחוֹתַי, בְּמִדָּה אַחַת" – ניסוח תיאולוגי מורכב יותר. בגרסה א' נאמר "מידה אחת מכל כוחותיי". כאן, נאמר "כל כוחותיי, במידה אחת". הניסוח בגרסה ב' הוא פרדוקסלי ומתוחכם יותר: הגואל מכיל את כל כוחות האל, אך הם דחוסים לתוך "מידה אחת". זהו ניסיון להסביר כיצד ההתגלמות יכולה להיות גם מלאה וגם חלקית בו-זמנית.
פסוק נ"ח: תיאור גורל הרשעים – הרחבה גרפית ומפורטת. גרסה א' מסתפקת בכך שהרשעים יידונו לאש הגיהינום. גרסה ב' מוסיפה תיאורים איומים: "למלא בטן של תולעת רעה", "יִרְאוּ בָּם עוֹשֵׂי הַצֶּדֶק... וְיִשְׂמְחוּ בְּשִׂמְחָה עַל-אַבְדַן הָאֲנָשִׁים". תיאור זה, שבו הצדיקים שמחים במפלת הרשעים, משקף תפיסה נוקשה ונקמנית יותר של צדק אלוהי, המוכרת מספרות קומראן ומחיבורים אפוקליפטיים אחרים.
פסוק נ"ט: "זֶהוּ דּוֹרְךָ הַשְּׁבִיעִי... וְיֵצְאוּ לְאֶרֶץ נֵכָר" – זוהי התוספת החשובה והמפתיעה ביותר בטקסט כולו. לפתע, בסוף החזון, המחבר קושר את כל הנבואה האוניברסלית בחזרה לסיפור המקראי הספציפי של ברית בין הבתרים (בראשית ט"ו, י"ג): "יָדֹעַ תֵּדַע כִּי-גֵר יִהְיֶה זַרְעֲךָ בְּאֶרֶץ לֹא לָהֶם, וַעֲבָדוּם וְעִנּוּ אֹתָם...".
הזכרת "דור שביעי" קושרת את אברהם למסורת של חנוך, שהיה הדור השביעי מאדם.
הקישור הישיר לפסוק מבראשית מראה שהעורך רצה "לעגן" את החזון האפוקליפטי הפרוע בחזרה בתוך הסיפור המקראי המוכר. זהו ניסיון להראות שכל החזון הגדול על אחרית הימים, הגואל והמלחמה הקוסמית, היה בעצם פירוש והרחבה של ההבטחה המקורית שניתנה לאברהם בברית בין הבתרים.
תוספת זו חסרה לחלוטין בגרסה א', מה שהופך את סיומה של גרסה ב' למעגלי ומסורתי יותר.
