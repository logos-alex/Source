חלק ד' – התגלות המלאך יָהוֹאֵל (פרקים י'-י"ב)
כז וַיָּבֹא הַמַּלְאָךְ אֲשֶׁר שֻׁלַּח אֵלַי בִּדְמוּת אִישׁ, וַיֹּאחֶז בִּיד יְמִינִי וַיְקִימֵנִי עַל-רַגְלָי. וַיֹּאמֶר אֵלַי: "קוּם, יְדִיד הָאֵל אֲשֶׁר אֲהֵבְךָ, פֶּן-יֶאֱחָזְךָ רַעַד בְּנֵי-אָדָם. הִנֵּה שָׁלוּחַ אֲנִי אֵלֶיךָ לְחַזֶּקְךָ וּלְבָרֶכְךָ בְּשֵׁם הָאֵל אֲשֶׁר אֲהֵבְךָ, יוֹצֵר הַשָּׁמַיִם וְהָאָרֶץ. חֲזַק וְהִתְאַזֵּר לִקְרָאתוֹ.
כח אָנֹכִי הוּא יָהוֹאֵל, הַנִּקְרָא בְשֵׁם הַמַּרְעִיד אֶת-אֵלֶּה אֲשֶׁר עִמִּי בָּרָקִיעַ הַשְּׁבִיעִי, עַל-הַמִּשְׁטָח. כֹּחַ בְּתוֹךְ הַמְּתוֹכְכִים לַשֵּׁם הַנִּשְׂגָּב, הַמְשָׁרְתִים אֲשֶׁר לֹא לִי.
כט אָנֹכִי הוּא הַנִּתָּן, עַל-פִּי צִוּוּיוֹ, לְהַשְׁפִּיל אֶת-הַכְּרוּבִים, הַחַיּוֹת הַמִּתְנַצְּחוֹת זוֹ עִם זוֹ, וּלְלַמֵּד אֶת-שִׁירַת הַקֹּדֶשׁ בְּאֶמְצַע לֵיל הָאָדָם בַּשָּׁעָה הַשְּׁבִיעִית.
ל אָנֹכִי הוּא הַמְמֻנֶּה לְהַחְזִיק בַּלִּוְיְתָנִים, כִּי בִי הֻכְנְעוּ. וְעַל כָּל-הַתַּקָּפוֹת וְהַאיוּמִים שֶׁל כָּל-רֶמֶשׂ.
לא אָנֹכִי הוּא הַמְצֻוֶּה לְהַצִּית אֶת-בֵּית אָבִיךָ וְאוֹתוֹ עִמּוֹ, כִּי עָבַד לַמֵּתִים.
לב אָנֹכִי שָׁלוּחַ אֵלֶיךָ עַתָּה לְבָרֶכְךָ וְאֶת-הָאָרֶץ אֲשֶׁר הֵכִין לְךָ הַקַּדְמוֹן, הַנִּקְרָא בְשִׁמְךָ. וּלְמַעַנְךָ הִכְשַׁרְתִּי אֶת-דֶּרֶךְ הָאָרֶץ. קוּם, אַבְרָם, לֵךְ בְּבִטְחָה, שְׂמַח מְאֹד וַעֲלֹז, וְאָנֹכִי עִמָּךְ, כִּי הוּכַן לְךָ כְּבוֹד עוֹלָמִים מֵאֵת הַקַּדְמוֹן. לֵךְ וְהַשְׁלֵם אֶת-הַקָּרְבָּן שֶׁצֻּוֵּיתָ. הִנֵּה אָנֹכִי מְמֻנֶּה לִהְיוֹת עִמְּךָ וְעִם הַזֶּרַע שֶׁיֻּכַן מִמְּךָ. וְעִמִּי מִיכָאֵל, לְבָרֶכְךָ עַד-עוֹלָם. חֲזַק וְלֵךְ".
לג וָאָקָם וָאֵרֶא אֶת-אֲשֶׁר אָחַז בִּימִינִי וְהֵקִים אוֹתִי עַל-רַגְלָי. מַרְאֵה גוּפוֹ כְּאֶבֶן סַפִּיר, וּמַרְאֵה פָנָיו כְּתַרְשִׁישׁ, וּשְׂעַר רֹאשׁוֹ כַּשֶּׁלֶג. וּמִצְנֶפֶת עַל-רֹאשׁוֹ כְּמַרְאֵה הַקֶּשֶׁת בֶּעָנָן. וּלְבוּשׁוֹ אַרְגָּמָן, וּמַטֵּה זָהָב בְּיַד יְמִינוֹ.
לד וַיֹּאמֶר אֵלַי: "אַבְרָם!". וָאֹמַר: "הִנֵּה עַבְדְּךָ". וַיֹּאמֶר: "אַל-יַפְחִידְךָ מַרְאִי וְאַל-יַבְהִילְךָ דִּבּוּרִי, וְאַל-תִּסָּעֵר נַפְשֶׁךָ. לֵךְ עִמִּי, וְאֵלֵךְ עִמְּךָ עַד-הַקָּרְבָּן, נִרְאֶה. וּלְאַחַר הַקָּרְבָּן, בִּלְתִּי-נִרְאֶה עַד-עוֹלָם. חֲזַק וּלֵךְ".
הערות וביאורים
פסוק כ"ח
אָנֹכִי הוּא יָהוֹאֵל: המלאך מציג את עצמו בשמו המלא. כפי שצוין, זהו שם המזוהה עם מטטרון, "שר העולם".
בָּרָקִיעַ הַשְּׁבִיעִי: על פי הקוסמולוגיה של בית שני, השמיים מחולקים לשבעה רקיעים. הרקיע השביעי הוא הגבוה והקדוש מכולם, מקום מושבו של כיסא הכבוד. מיקומו של יָהוֹאֵל ברקיע השביעי מעיד על מעמדו הרם.
כֹּחַ בְּתוֹךְ הַמְּתוֹכְכִים לַשֵּׁם הַנִּשְׂגָּב: תיאור עצמי מורכב. הוא כוח הפועל בתוך קבוצת מלאכים מתווכים אחרים, המשרתים את השם האלוהי.
פסוק כ"ט
לְהַשְׁפִּיל אֶת-הַכְּרוּבִים, הַחַיּוֹת הַמִּתְנַצְּחוֹת: אחד מתפקידיו הוא להשליט סדר בקרב הדרגות הגבוהות ביותר של המלאכים, ה"חיות" (הכרובים) של המרכבה, המתוארות כאן כ"מתנצחות". זהו רעיון מפתיע, המרמז על מתח ודינמיקה אפילו בעולם המלאכים.
לְלַמֵּד אֶת-שִׁירַת הַקֹּדֶשׁ... בַּשָּׁעָה הַשְּׁבִיעִית: יָהוֹאֵל הוא המנצח על מקהלת המלאכים, המלמד אותם את שירת הקדושה ("קדוש, קדוש, קדוש"). השירה נאמרת בלילה, כשהעולם האנושי שקט.
פסוק ל'
לְהַחְזִיק בַּלִּוְיְתָנִים: תפקיד קוסמי נוסף. הוא הכוח השולט במפלצות הכאוס הקדמוניות, הלוויתנים. הוא שומר על גבולות הבריאה ומונע מהכאוס להשתלט.
פסוק ל"א
הַמְצֻוֶּה לְהַצִּית אֶת-בֵּית אָבִיךָ: המלאך לוקח אחריות אישית על חורבן בית תרח. הוא פועל כשליח הדין האלוהי.
פסוק ל"ב
וְעִמִּי מִיכָאֵל: יָהוֹאֵל אינו פועל לבד. הוא מזכיר את מיכאל, השר הממונה על ישראל, כשותף במשימת הברכה. הדבר מעיד על היררכיה ושיתוף פעולה בעולם המלאכים.
פסוק ל"ג
מַרְאֵה גוּפוֹ כְּאֶבֶן סַפִּיר...: תיאור מראהו של המלאך שואב ישירות מתיאורי המרכבה ביחזקאל ודניאל.
סַפִּיר: מזכיר את "לִבְנַת הַסַּפִּיר" תחת רגלי אלוהים (שמות כ"ד) ואת מראה הכיסא ביחזקאל א'.
תַרְשִׁישׁ: אבן חן זוהרת, מתארת את גופו של האיש הלבוש בבדים בדניאל י', ו'.
שְׂעַר... כַּשֶׁלֶג: מזכיר את תיאור "עתיק יומין" בדניאל ז', ט'.
הַקֶּשֶׁת בֶּעָנָן: דימוי הלקוח ישירות מחזון המרכבה ביחזקאל א', כ"ח.
לְבוּשׁוֹ אַרְגָּמָן, וּמַטֵּה זָהָב: סמלי מלכות ושלטון. המחבר מעניק למלאך יָהוֹאֵל את כל התכונות החיצוניות של הכבוד האלוהי, ובכך מדגיש את מעמדו כנציגו הישיר של האל.
פסוק ל"ד
עַד-הַקָּרְבָּן, נִרְאֶה. וּלְאַחַר הַקָּרְבָּן, בִּלְתִּי-נִרְאֶה: המלאך מסביר את טבע נוכחותו. הוא ילווה את אברהם באופן גלוי עד לרגע הקורבן. לאחר מכן, הקשר יהפוך לרוחני ובלתי נראה.
הגענו לנקודה שבה אברהם והמלאך יוצאים למסע אל ההר.

חלק ה' – המסע, הקורבן והעימות עם עזאזל (פרקים י"ב-י"ד)
לה וָנֵלֵךְ שְׁנֵינוּ יַחְדָּו, אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה. וְלֶחֶם לֹא אָכַלְתִּי וּמַיִם לֹא שָׁתִיתִי, כִּי מַאֲכָלִי הָיָה לִרְאוֹת אֶת-הַמַּלְאָךְ אֲשֶׁר עִמִּי, וּמִשְׁתִּי – דְּבָרָיו אֲשֶׁר דִּבֵּר אֵלָי.
לו וָאָבֹא אֶל-הַר חוֹרֵב, הַר הָאֱלֹהִים. וָאֹמַר אֶל-הַמַּלְאָךְ, שַׂר הַקַּדְמוֹן: "הִנֵּה אֵין עִמִּי קָרְבָּן, וְאֶת-מְקוֹם הַמִּזְבֵּחַ עַל-הָהָר לֹא יָדַעְתִּי. אֵיךְ אֶעֱשֶׂה אֶת-הַקָּרְבָּן?". וַיֹּאמַר: "הַבֵּט אֲחוֹרֶיךָ". וָאַבִּיט, וְהִנֵּה הוֹלְכִים אַחֲרֵינוּ כָּל-הַקֳּרְבָּנוֹת שֶׁצֻּוּוּ: הָעֶגְלָה, וְהָעֵז, וְהָאַיִל, וְהַתּוֹר, וְהַגּוֹזָל.
לז וַיֹּאמֶר אֵלַי הַמַּלְאָךְ: "אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמֶר לִי: "שְׁחַט אֶת-כֻּלָּם וְנַתַּח אוֹתָם. אֶת-הַנְּתָחִים שִׂים זֶה מוּל זֶה, אַךְ אֶת-הָעוֹף לֹא תְבַתֵּר. וְתֵן אוֹתָם לָאֲנָשִׁים אֲשֶׁר אַרְאֶךָּ עוֹמְדִים לְצִדְּךָ, כִּי הֵם הַמִּזְבֵּחַ עַל-הָהָר לְהַקְרִיב קָרְבָּן לַקַּדְמוֹן. וְאֶת-הַתּוֹר וְאֶת-הַגּוֹזָל תֵּן לִי, וַאֲנִי אֶעֱלֶה עַל-כַּנְפֵי הָעוֹפוֹת לְהַרְאוֹתְךָ אֶת-אֲשֶׁר בַּשָּׁמַיִם וּבָאָרֶץ וּבַיָּם וּבַתְּהוֹמוֹת, וּבְגַן-עֵדֶן וּבִנְהָרוֹתָיו, וְאֶת-מִלְאוֹ הַתֵּבֵל כֻּלָּהּ. וְאֶת-סוֹבֵב הָעוֹלָם תִּרְאֶה בְּכֻלּוֹ".
לח וָאֶעֱשׂ כְּכֹל אֲשֶׁר צִוָּה הַמַּלְאָךְ. וָאֶתֵּן לַמַּלְאָכִים אֲשֶׁר בָּאוּ אֵלֵינוּ אֶת-הַבְּהֵמוֹת הַמְנֻתָּחוֹת. וְאֶת-הָעוֹפוֹת לָקַח הַמַּלְאָךְ. וָאֲחַכֶּה לְמִנְחַת עָרֶב.
לט וַיֵּרֶד עַיִט טָמֵא עַל-הַפְּגָרִים, וָאַבְרֵחַ אוֹתוֹ. וַיִּקְרָא אֵלַי הָעַיִט הַטָּמֵא וַיֹּאמַר: "מַה-לְּךָ, אַבְרָם, בַּמְּרוֹמִים הַקְּדוֹשִׁים, אֲשֶׁר שָׁם לֹא אוֹכְלִים וְלֹא שׁוֹתִים, וְאֵין בָּהֶם מַאֲכַל אָדָם, אֶלָּא הַכֹּל בָּאֵשׁ נֶאֱכָלִים וְאוֹתְךָ יְכַלּוּ? עֲזֹב אֶת-הָאִישׁ אֲשֶׁר עִמָּךְ וּבְרַח, כִּי אִם תַּעֲלֶה לַמָּרוֹם, יַשְׁחִיתוּךָ!".
מ וַיְהִי כִּרְאוֹתִי אֶת-הָעוֹף הַמְדַבֵּר, וָאֹמַר אֶל-הַמַּלְאָךְ: "מֶה-הוּא זֶה, אֲדֹנִי?". וַיֹּאמַר: "זֶהוּ חֹסֶר-הַצֶּדֶק, זֶהוּ עֲזָאזֵל. כִּי בָּחַר לוֹ לָגוּר עַל-הָאָרֶץ, וְחֶלְקוֹ עַל-הָאָרֶץ, וְאַתָּה חֶלְקְךָ בַּשָּׁמָיִם". וַיֹּאמֶר לוֹ: "חֶרְפָּה לְךָ, עֲזָאזֵל! כִּי בָּחַרְתָּ וְאָהַבְתָּ אֶת-הַמָּקוֹם הַמְטֻנָּף. עַל-כֵּן נְתָנְךָ הָאָדוֹן הַקַּדְמוֹן, הֶחָזָק, לִהְיוֹת שׁוֹכֵן עַל-הָאָרֶץ. וּבְךָ רוּחַ הַשֶּׁקֶר הָרָעָה, וּבְךָ חֵמָה וּמַכְאוֹבִים עַל-דּוֹרוֹת הָרְשָׁעָה. כִּי לֹא נָתַן הָאָדוֹן הַקַּדְמוֹן, הֶחָזָק, אֶת-גוּפוֹת הַצַּדִּיקִים בְּיָדְךָ, לְמַעַן יִתְחַזְּקוּ בָּהֶם חַיֵּי הַצַּדִּיק וְאַבְדַן הָרָשָׁע. שְׁמַע, יְדִידִי, וְהִכָּלֵם מִמֶּנִּי! כִּי לֹא נִתַּן לְךָ לְפַתּוֹת אֶת-כָּל-הַצַּדִּיקִים. סוּר מֵאִישׁ זֶה! אֵינְךָ יָכוֹל לְהַטְעוֹתוֹ, כִּי אוֹיֵב הוּא לְךָ, וְגַם לַהוֹלְכִים בְּעִקְּבוֹתֶיךָ וְלָאוֹהֲבִים אֶת-אֲשֶׁר תִּרְצֶה. הִנֵּה הַלְּבוּשׁ אֲשֶׁר הָיָה לְךָ בַּשָּׁמַיִם נִטַּל מִמְּךָ מִקֶּדֶם, וְהַכָּבוֹד אֲשֶׁר הָיָה עַל-גּוּפְךָ, עָבַר אֵלָיו".
מא וַיֹּאמֶר הַמַּלְאָךְ לְאַבְרָם: "אֱמֹר לוֹ: 'הֱיֵה גֶּחָלִים לַאֲדָמָה! סוּר, עֲזָאזֵל, אֶל-מְקוֹמוֹת הַשְּׁאוֹל הַבִּלְתִּי-נִכְנָסִים!'". וַיֹּאמֶר לִי הַמַּלְאָךְ: "אַל-תַּעֲנֶה לוֹ, כִּי נִתְּנָה לוֹ הָרְשׁוּת לְפַתּוֹת אֶת-אֵלֶּה הָעוֹנִים לוֹ". וְכַאֲשֶׁר דִּבֵּר אֵלַי, לֹא עָנִיתִי לוֹ.
מב וַיְהִי כַּאֲשֶׁר הַשֶּׁמֶשׁ בָּאָה, וְהִנֵּה עָשָׁן כְּעֶשַׁן כִּבְשָׁן. וַיַּעֲלוּ הַמַּלְאָכִים מֵעַל הַנְּתָחִים, מֵרֹאשׁ הַכִּבְשָׁן הַמַּעֲלֶה עָשָׁן.
הערות וביאורים
פסוק ל"ה
מַאֲכָלִי הָיָה לִרְאוֹת אֶת-הַמַּלְאָךְ... וּמִשְׁתִּי – דְּבָרָיו: תיאור פיוטי של הצום המיסטי. אברהם אינו ניזון ממזון גשמי, אלא מהנוכחות והדיבור של המלאך. זהו מעבר ממצב אנושי למצב רוחני, שבו הגוף אינו זקוק עוד למזון רגיל.
פסוק ל"ו
הַר חוֹרֵב, הַר הָאֱלֹהִים: זיהוי מפתיע. בבראשית, מקום ברית בין הבתרים אינו מזוהה. כאן, המחבר קושר אותו להר סיני/חורב, הר ההתגלות המרכזי בתורה. בכך הוא מעניק לברית עם אברהם מעמד שווה למעמד הר סיני.
פסוק ל"ז
הֵם הַמִּזְבֵּחַ עַל-הָהָר: רעיון מהפכני. המזבח אינו עשוי אבן או אדמה, אלא המלאכים עצמם הם המזבח החי. הקורבן אינו מוקרב על חומר דומם, אלא נמסר לישויות שמימיות.
וַאֲנִי אֶעֱלֶה עַל-כַּנְפֵי הָעוֹפוֹת: המלאך והוא עצמו ישתמשו בעופות שלא בותרו (התור והגוזל) כמעין "מרכבה" לעלות לשמיים. זהו דימוי ייחודי המשלב את טקס הקורבן עם מסע שמימי.
פסוק ל"ט
וַיֵּרֶד עַיִט טָמֵא עַל-הַפְּגָרִים: המחבר משלב כאן את הפסוק מבראשית ט"ו, י"א: "וַיֵּרֶד הָעַיִט עַל-הַפְּגָרִים וַיַּשֵּׁב אֹתָם אַבְרָם". אך בעוד שבבראשית זהו רק "עיט", כאן הוא הופך לדמות מדברת, למפתה, ליריבו של אברהם.
"מַה-לְּךָ, אַבְרָם, בַּמְּרוֹמִים הַקְּדוֹשִׁים...": הפיתוי הראשון. העיט (עזאזל) מנסה להפחיד את אברהם, וטוען שהעולם השמימי אינו מיועד לבני אדם והוא מסוכן עבורם.
פסוק מ'
זֶהוּ עֲזָאזֵל: המלאך מזהה את העיט הטמא עם דמותו של עזאזל. עזאזל מוכר לנו מטקס יום הכיפורים (ויקרא ט"ז), שם נשלח אליו "השעיר לעזאזל". בספרות בית שני (במיוחד בספר חנוך), עזאזל הוא אחד מראשי המלאכים שנפלו, לימד את בני האדם חטאים, ומקומו במדבר או במקומות חורבן. כאן הוא מתואר כיריב קוסמי לאברהם ולאל.
חֶלְקוֹ עַל-הָאָרֶץ, וְאַתָּה חֶלְקְךָ בַּשָּׁמָיִם: המלאך מציג דואליזם ברור. העולם מחולק לשני תחומים: תחום השמיים והצדקה השייך לאברהם ולזרעו, ותחום הארץ והטומאה השייך לעזאזל.
הַלְּבוּשׁ אֲשֶׁר הָיָה לְךָ בַּשָּׁמַיִם נִטַּל מִמְּךָ... וְעָבַר אֵלָיו: רעיון דרמטי. כבודו השמימי של עזאזל, שאבד לו בנפילתו, הועבר לאברהם. אברהם יורש את מקומו של המלאך הסורר.
פסוק מ"א
"הֱיֵה גֶּחָלִים לַאֲדָמָה!": זוהי קללה שנועדה לגרש את עזאזל.
"אַל-תַּעֲנֶה לוֹ, כִּי נִתְּנָה לוֹ הָרְשׁוֹת לְפַתּוֹת אֶת-אֵלֶּה הָעוֹנִים לוֹ": עצה טקטית חשובה. אסור להיכנס לדיאלוג עם המקטרג. עצם המענה נותן לו פתחון פה וכוח. השתיקה היא ההגנה הטובה ביותר.
פסוק מ"ב
עָשָׁן כְּעֶשַׁן כִּבְשָׁן: ציטוט מדויק מבראשית ט"ו, י"ז ("וְהִנֵּה תַנּוּר עָשָׁן וְלַפִּיד אֵשׁ"). המחבר חוזר לתיאור המקראי כדי לסמן את תחילת החזון השמימי. העשן והאש הם השער שדרכו אברהם יעלה לשמיים.
