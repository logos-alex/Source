חזון אברהם (מהדורה פרשנית)
חלק ח' – חזון החטא הקדמון וגורל האנושות (פרקים כ"ג-כ"ט)
סה "הַבֵּט עוֹד בַּתְּמוּנָה, מִי הוּא אֲשֶׁר פִּתָּה אֶת-חַוָּה, וּמַהוּ פְּרִי הָעֵץ, וּמֶה יִהְיֶה, וְכֵיצַד יִהְיֶה. כִּי בְּשִׁמְךָ וּבְעַמְּךָ יִהְיֶה הַדָּבָר בְּאַחֲרִית הַיָּמִים וּבְקֵץ הָעוֹלָם. לֵךְ וְהָבֵן, כִּי אֲנִי אַגִּיד לְךָ אֶת-אֲשֶׁר הָיָה טוֹב בְּעֵינַי, וְאֶת-אֲשֶׁר צָפַנְתִּי בְּלִבִּי אֲגַלֶּה לְךָ".
סו וָאַבִּיט בַּתְּמוּנָה, וַתִּזְלַגְנָה עֵינַי אֶל-צִדֵּי גַּן-עֵדֶן. וָאֵרֶא שָׁם אִישׁ גָּבֹהַּ מְאֹד בְּקוֹמָתוֹ וְנוֹרָא בְּרָחְבּוֹ, בְּלִי דִמְיוֹן בְּמַרְאֵהוּ, שָׁזוּר עִם אִשָּׁה, אֲשֶׁר גַּם-הִיא הִשְׁתַּוְּתָה לוֹ בְּמַרְאֶה וּבְקוֹמָה. וְהֵם עוֹמְדִים תַּחַת עֵץ אֶחָד שֶׁל גַּן-עֵדֶן. וּפְרִי הָעֵץ הָיָה כְּמַרְאֵה אֶשְׁכּוֹל עֲנָבִים.
סז וּמֵאֲחוֹרֵי הָעֵץ הָיָה עוֹמֵד נָחָשׁ, יָדָיו וְרַגְלָיו כְּשֶׁל אָדָם, וּכְנָפַיִם עַל-כְּתֵפָיו, שֵׁשׁ מִימִינוֹ וְשֵׁשׁ מִשְּׂמֹאלוֹ. וְהוּא אוֹחֵז בְּיָדוֹ אֶשְׁכּוֹל מִפְּרִי הָעֵץ, וּמְדַבֵּר עִם הַשְּׁנַיִם אֲשֶׁר רָאִיתִי שְׁזוּרִים יַחַד.
סח וָאֹמַר: "מִי הֵם אֵלֶּה הַשְּׁזוּרִים זֶה בָּזֶה, אוֹ מִי הוּא זֶה שֶׁבֵּינֵיהֶם, אוֹ מַהוּ הַפְּרִי שֶׁהֵם אוֹכְלִים, חָזָק וְנִצְחִי?". וַיֹּאמַר: "זֶהוּ מַחֲשֶׁבֶת בְּנֵי-הָאָדָם, זֶהוּ אָדָם, וְזֹאת חַוָּה. וַאֲשֶׁר בֵּינֵיהֶם – זֶהוּ חֹסֶר-הַצֶּדֶק, תְּחִלַּת דַּרְכָּם לָאֲבַדּוֹן, הוּא עֲזָאזֵל".
סט וָאֹמַר: "נִצְחִי וְחָזָק! מַדּוּעַ נָתַתָּ לוֹ שִׁלְטוֹן כָּזֶה, לְהַשְׁחִית אֶת-דּוֹר בְּנֵי-הָאָדָם בְּמַעֲשֵׂיהֶם עַל-הָאָרֶץ?". וַיֹּאמֶר אֵלַי: "שְׁמַע, אַבְרָם. אֶת-אֵלֶּה הַחוֹפְצִים בָּרַע, וְאֶת-כָּל-אֲשֶׁר שָׂנֵאתִי בְּעוֹשֵׂי הָרַע, עֲלֵיהֶם נָתַתִּי לוֹ שִׁלְטוֹן, וְלִהְיוֹת אָהוּב עֲלֵיהֶם".
ע וָאֶעֱנֶה וָאֹמַר: "נִצְחִי וְחָזָק! מַדּוּעַ רָצִיתָ שֶׁיִּהְיֶה רָצוֹן לָרַע בְּלֵב הָאָדָם? הֲרֵי אַתָּה כּוֹעֵס עַל-מַה שֶּׁנִּבְחַר בְּךָ, וְעַל-הָעוֹשֶׂה דָּבָר שֶׁאֵינוֹ הָגוּן בְּאוֹרְךָ".
עא וַיֹּאמֶר לִי: "בְּשׂוֹרָה קְרוֹבָה לְךָ וּלְאַחֲרֶיךָ לָעָם הַנִּבְדָּל. וּמִזַּרְעֲךָ, כַּאֲשֶׁר תִּרְאֶה בַּתְּמוּנָה, יִהְיוּ נִטְעָנִים עֲלֵיהֶם. וַאֲנִי אַגִּיד לְךָ מֶה יִהְיֶה וְכֵיצַד יִהְיֶה בְּאַחֲרִית הַיָּמִים. הַבֵּט-נָא בַּתְּמוּנָה".
עב וָאַבִּיט וָאֵרֶא שָׁם אֶת-אֲשֶׁר הָיָה לְפָנַי בַּבְּרִיאָה. רָאִיתִי אֶת-אָדָם וְאֶת-חַוָּה אֲשֶׁר עִמּוֹ, וְאֶת-הָאוֹיֵב הָרַע עִמָּם. וְאֶת-קַיִן, שֶׁבִּגְלַל הָאוֹיֵב חָטָא וְהָרַג אֶת-הֶבֶל. וְאֶת-הָאֲבַדּוֹן שֶׁהוּבָא עָלָיו בִּגְלַל חֹסֶר-הַחֹק.
עג וָאֵרֶא שָׁם אֶת-הַנִּאוּף וְאֶת-הַחוֹפְצִים בּוֹ, וְאֶת-קִנְאָתָם, וְאֶת-אֵשׁ הַשְׁחָתָם בִּתְהוֹמוֹת הָאָרֶץ. וָאֵרֶא שָׁם אֶת-הַגְּנֵבָה וְאֶת-הַחוֹמְדִים לָהּ, וְאֶת-גְּמוּלָם. וָאֵרֶא שָׁם אִישׁ וְאִשָּׁה זֶה כְּנֶגֶד זֶה, וְאֶת-בָּשְׁתָּם, וְאֶת-הַנֵּזֶק שֶׁעוֹשִׂים זֶה לָזֶה, וְאֶת-גְּמוּלָם.
עד וָאֵרֶא שָׁם אֶת-הַתַּאֲוָה, וּבְיָדָהּ רֹאשׁ כָּל-חֹסֶר-חֹק. וָאֵרֶא שָׁם אֶת-פֶּסֶל הַקִּנְאָה, כִּדְמוּת מְלֶאכֶת-עֵץ כְּמוֹ שֶׁעָשָׂה אָבִי. וְלִפְנָיו אִישׁ מִשְׁתַּחֲוֶה לוֹ, וּמִזְבֵּחַ כְּנֶגְדּוֹ. וִילָדִים נִשְׁחָטִים עָלָיו, לִפְנֵי הַפֶּסֶל.
עה וָאֹמַר: "מָהוּ הַפֶּסֶל הַזֶּה, אוֹ מָהוּ הַמִּזְבֵּחַ? מִי הֵם הַנִּזְבָּחִים, אוֹ מִי הוּא הַזּוֹבֵחַ? וּמַהוּ הַהֵיכָל הַיָּפֶה הַזֶּה שֶׁאֲנִי רוֹאֶה, מְלֶאכֶת מַחְשֶׁבֶת וְיֹפִי, תַּחַת כִּסֵּא כְבוֹדֶךָ?".
עו וַיֹּאמֶר: "שְׁמַע, אַבְרָם. זֶה אֲשֶׁר רָאִיתָ – הַהֵיכָל וְהַמִּזְבֵּחַ וְהַיֹּפִי – זֶהוּ רַעְיוֹן הַכֹּהֲנִים שֶׁל שֵׁם כְּבוֹדִי, שֶׁבּוֹ יִשְׁכְּנוּ כָּל-תְּפִלּוֹת הָאָדָם, וְעֲלִיַּת הַמְּלָכִים וְהַנְּבִיאִים, וְכָל-קָרְבָּן שֶׁאָשִׂים לַעֲשׂוֹת לִי בְּקֶרֶב עַמִּי הַבָּא מִמְּךָ.
עז וְהַפֶּסֶל אֲשֶׁר רָאִיתָ – הוּא קִנְאָתִי, שֶׁבָּהּ יַקְנִיטוּ אוֹתִי הָעָם הַבָּא מִמְּךָ. וְהָאִישׁ אֲשֶׁר רָאִיתָ זוֹבֵחַ – הוּא הַמַּקְנִיא בְּקָרְבָּן רֶצַח. וְאֵלֶּה הַנִּשְׁחָטִים – הֵם עֵדוּת לַדִּין בְּקֵץ הַבְּרִיאָה".
הערות וביאורים
פסוק ס"ו-ס"ח
אִישׁ גָּבֹהַּ מְאֹד... שָׁזוּר עִם אִשָּׁה: אדם וחוה מתוארים כיצורים ענקיים, כמעט אלוהיים, השזורים זה בזה. ייתכן שזהו רמז למדרש על כך שאדם הראשון נברא דו-פרצופין (זכר ונקבה יחד).
נָחָשׁ, יָדָיו וְרַגְלָיו כְּשֶׁל אָדָם, וּכְנָפַיִם: זהו תיאור ייחודי של הנחש. הוא אינו זוחל, אלא יצור דמוי-מלאך, בעל גפיים וכנפיים. זה מתחבר למסורות (בספר חנוך ובמקורות אחרים) המזהות את הנחש עם אחד המלאכים שנפלו.
הוּא עֲזָאזֵל: הטקסט מזהה במפורש את הנחש המפתה בגן עדן עם עזאזל. בכך, הוא קושר את חטא גן עדן ישירות למרד המלאכים. עזאזל אינו רק יריב, אלא מקור החטא הראשון.
פסוק ס"ט-ע'
מַדּוּעַ נָתַתָּ לוֹ שִׁלְטוֹן...?: אברהם חוזר על שאלתו התיאולוגית הנוקבת. הוא אינו מקבל את קיומו של הרוע כמובן מאליו, אלא מאתגר את האל להסביר את הצדק שבמתן שלטון לעזאזל.
"עֲלֵיהֶם נָתַתִּי לוֹ שִׁלְטוֹן": התשובה האלוהית מטילה את האחריות על בני האדם. עזאזל שולט רק על אלה שבוחרים ברע מרצונם. זהו ניסיון לפתור את בעיית הגזירה הקדומה (פרה-דסטינציה) באמצעות רעיון הבחירה החופשית.
פסוק ע"ב-ע"ד
וָאַבִּיט וָאֵרֶא: אברהם רואה כעת את תוצאות החטא מתגלגלות בהיסטוריה. הוא רואה את רצח הבל, את חטאי המין, הגניבה, הקנאה, ושיא הרשעות – עבודת האלילים.
פֶּסֶל הַקִּנְאָה... וִילָדִים נִשְׁחָטִים עָלָיו: זהו חזון של עבודת המולך, שבה הקריבו ילדים. המחבר קושר את עבודת האלילים של אביו (מלאכת עץ) עם הצורה הקיצונית והאכזרית ביותר של עבודה זרה.
פסוק ע"ה-ע"ז
הַהֵיכָל הַיָּפֶה... תַּחַת כִּסֵּא כְבוֹדֶךָ: אברהם רואה את בית המקדש בירושלים, אך הוא רואה אותו במיקומו השמימי, "תחת כיסא הכבוד". זהו רעיון של "ירושלim של מעלה".
זֶהוּ רַעְיוֹן הַכֹּהֲנִים: המקדש הארצי הוא רק השתקפות של רעיון או תבנית שמימית.
הַפֶּסֶל... הוּא קִנְאָתִי, שֶׁבָּהּ יַקְnִיטוּ אוֹתִי: האל מסביר שהפסל שאברהם ראה בתוך ההיכאל הוא דימוי לחטא עבודה זרה שיבוצע על ידי זרעו של אברהם עצמו בתוך המקדש. זהו רמז לחטאי המלכים (כמו מנשה) שהכניסו עבודה זרה למקדש, והובילו לחורבנו.
הַזּוֹבֵחַ – הוּא הַמַּקְנִיא בְּקָרְבָּן רֶצַח: הקרבת הקורבנות לאלילים נחשבת לרצח, והיא המעוררת את קנאת האל ומובילה לדין.
הגענו לנקודה שבה אברהם מבין שההיסטוריה כולה, כולל חטאי זרעו וחורבן המקדש, היא תוצאה של אותו חטא קדמון.


חלק ט' – חורבן המקדש, עשר המכות ובוא הגואל (פרקים כ"ט-ל"ב)
עח וָאֹמַר: "נִצְחִי וְחָזָק! מַדּוּעַ הֲקִימוֹתָ שֶׁיִּהְיֶה כָּךְ? הָשֵׁב נָא אֶת-הָרָעוֹת הַלָּלוּ!". וַיֹּאמֶר אֵלַי: "שְׁמַע, אַבְרָם, וְהָבֵן אֶת-אֲשֶׁר אֲנִי שׁוֹאֵל אוֹתְךָ. אָבִיךָ תֶּרַח, מַדּוּעַ לֹא שָׁמַע לְקוֹלְךָ וְלֹא עָזַב אֶת-עֲבוֹדַת הָאֱלִילִים, עַד-אֲשֶׁר אָבַד עִם כָּל-בֵּיתוֹ?".
עט וָאֹמַר: "נִצְחִי! כִּי לֹא רָצָה לִשְׁמֹעַ לִי, וְגַם אֲנִי לֹא הָלַכְתִּי אַחֲרֵי מַעֲשָׂיו". וַיֹּאמֶר: "שְׁמַע, אַבְרָם. כְּמוֹ שֶׁרְצוֹנוֹ שֶׁל אָבִיךָ בּוֹ, וּכְמוֹ שֶׁרְצוֹנְךָ בְּךָ, כָּךְ גַּם רְצוֹנִי בִּי, מוּכָן בַּיָּמִים הַבָּאִים. אַךְ אֶת-אֲשֶׁר יִהְיֶה, לֹא תִרְאֶה בְּעֵינֶיךָ, כִּי מִזַּרְעֲךָ הֵם. הַבֵּט בַּתְּמוּנָה".
פ וָאַבִּיט, וָאֵרֶא וְהִנֵּה הַתְּמוּנָה מִתְנוֹעַעַת. וּמִצַּד שְׂמֹאל יָצָא עַם שֶׁל גּוֹיִים, וַיִּשְׁדְּדוּ אֶת-אֵלֶּה שֶׁמִּצַּד יָמִין – גְּבָרִים, נָשִׁים וִילָדִים. וְאֶת-קְצָתָם הָרְגוּ וְאֶת-קְצָתָם הֶחֱזִיקוּ בְּשִׁבְיָה. וָאֵרֶא אוֹתָם רָצִים אֲלֵיהֶם בְּאַרְבַּע מְבוֹאוֹת, וַיִּשְׂרְפוּ אֶת-הַהֵיכָל בָּאֵשׁ, וְאֶת-כָּל-כְּלֵי הַקֹּדֶשׁ שֶׁבּוֹ בָּזְזוּ.
פא וָאֹמַר: "נִצְחִי! הִנֵּה הָעָם הַבָּא מִמֶּנִּי נִשְׁדָּד עַל-יְדֵי הַגּוֹיִים, וְקְצָתָם נֶהֱרָגִים וּקְצָתָם נִלְקָחִים בַּשֶּׁבִי. וְאֶת-הַהֵיכָל שָׂרְפוּ בָּאֵשׁ, וְאֶת-כְּלֵי הַחֶמְדָּה שֶׁבּוֹ הֵפִיצוּ. נִצְחִי וְחָזָק! אִם-כָּכָה הוּא, מַדּוּעַ הִכְאַבְתָּ אֶת-לִבִּי, וּמַדּוּעַ זֶה יִהְיֶה?".
פב וַיֹּאמֶר אֵלַי: "שְׁמַע, אַבְרָם. כָּל-אֲשֶׁר רָאִיתָ יִהְיֶה בִּגְלַל זַרְעֲךָ, שֶׁיַּכְעִיסוּ אוֹתִי בִּגְלַל הַפֶּסֶל שֶׁרָאִיתָ וְהָרֶצַח בְּתוֹךְ הֵיכַל הַקִּנְאָה. וְכָל-אֲשֶׁר רָאִיתָ – כָּךְ יִהְיֶה".
פג וָאֹמַר: "נִצְחִי וְחָזָק! יַעַבְרוּ-נָא הָרָעוֹת בְּמַעֲשֵׂי הָרְשָׁעָה, וְיִצְלְחוּ מַעֲשֵׂי הַצַּדִּיקִים. הֲלֹא יָכוֹל אַתָּה לַעֲשׂוֹת זֹאת". וַיֹּאמֶר אֵלַי: "זְמַן הַצַּדִּיקִים יָבוֹא אַחֲרֵי הַמְּלָכִים הָרִאשׁוֹנִים, וּבַצֶּדֶק אֲשֶׁר בָּרָאתִי לִשְׁפֹּט אוֹתָם, מֵהֶם יֵצְאוּ הָאֲנָשִׁים אֲשֶׁר הִגַּדְתִּי לְךָ וְרָאִיתָ".
פד וָאֶעֱנֶה וָאֹמַר: "חָזָק וְקָדוֹשׁ! בְּכֹחֲךָ, הֱיֵה-נָא קָשּׁוּב לִשְׁאֵלָתִי, כִּי לָזֹאת הִגַּדְתָּ לִי, לְאוֹהַבְךָ, אֶת-כָּל-אֲשֶׁר אֶשְׁאַל. הַאִם יִהְיֶה זְמַן רַב לְכָל-אֲשֶׁר רָאִיתִי?". וַיַּרְאֵנִי אֶת-מִסְפַּר עַמּוֹ. וַיֹּאמֶר אֵלַי: "בִּגְלַל אַרְבַּע הַמְּבוֹאוֹת שֶׁרָאִיתָ, יַכְעִיסוּ אוֹתִי, וּבָהֶם יִהְיֶה גְּמוּל מַעֲשֵׂיהֶם מִמֶּנִּי. בְּאַרְבַּע מְבוֹאוֹת, מֵאָה שָׁנָה, וְשָׁעָה אַחַת שֶׁל הָעוֹלָם, שֶׁהִיא מֵאָה שָׁנָה, תִּהְיֶה בַּשְּׁאוֹל בְּקֶרֶב הַגּוֹיִם".
פה וָאֹמַר: "נִצְחִי! וְכַמָּה זְמַן הִיא שְׁעַת הָעוֹלָם?". וַיֹּאמֶר: "שְׁתֵּים-עֶשְׂרֵה שָׁנָה קָבַעְתִּי לָעוֹלָם הָרָשָׁע הַזֶּה לִמְשֹׁל בַּגּוֹיִים וּבְזַרְעֲךָ. וְעַד קֵץ הַזְּמַן יִהְיֶה כָּל-אֲשֶׁר רָאִיתָ. סְפֹר וְדַע וְהַבֵּט בַּתְּמוּנָה".
פו וָאֵרֶא אִישׁ יוֹצֵא מִצַּד שְׂמֹאל, מִצַּד הַגּוֹיִים. וַיָּבוֹאוּ גְּבָרִים וְנָשִׁים וִילָדִים מִצַּד הַגּוֹיִים, הֲמוֹנִים רַבִּים, וַיִּשְׁתַּחֲווּ לוֹ. וְעוֹדֶנִּי מִסְתַּכֵּל, וַיֵּצְאוּ מִצַּד יָמִין, וּקְצָתָם בִּיְּשׁוּ אֶת-הָאִישׁ הַהוּא, וּקְצָתָם הִכּוּהוּ, וּקְצָתָם הִשְׁתַּחֲווּ לוֹ. וָאֵרֶא כַּאֲשֶׁר הִשְׁתַּחֲווּ לוֹ, וַיָּרָץ עֲזָאזֵל וַיִּשְׁתַּחֲוֶה, וַיְּנַשֵּׁק אֶת-פָּנָיו, וַיִּסּוֹב וַיַּעֲמֹד מֵאֲחוֹרָיו.
פז וָאֹמַר: "נִצְחִי וְחָזָק! מִי הוּא הָאִישׁ הַמְבֻזֶּה וְהַמֻּכֶּה עַל-יְדֵי הַגּוֹיִים, וְעִם-זֹאת נִסְגָד עַל-יְדֵי עֲזָאזֵל?". וַיַּעַן וַיֹּאמֶר: "שְׁמַע, אַבְרָם. הָאִישׁ אֲשֶׁר רָאִיתָ מְבֻזֶּה וּמֻכֶּה וְאַחַר-כָּךְ נִסְגָד, הוּא הַגְּאוּלָה מִן-הַגּוֹיִים לָעָם הַבָּא מִמְּךָ בְּאַחֲרִית הַיָּמִים, בַּשָּׁעָה הַשְּׁתֵּים-עֶשְׂרֵה שֶׁל עוֹלָם הָרְשָׁעָה, בַּשָּׁנָה הַשְּׁתֵּים-עֶשְׂרֵה שֶׁל עוֹלָמִי הַמִּסְתַּיֵּם. אֶת-הָאִישׁ הַזֶּה אָקִים מִזַּרְעֲךָ, זֶה אֲשֶׁר רָאִיתָ.
פח וְרַבִּים מֵעַמִּי יִדְמוּ לוֹ, וְיִקָּרְאוּ כְּאִלּוּ נִקְרְאוּ בִשְׁמִי, אַךְ יִשְׁתַּנּוּ בִּרְצוֹנוֹתֵיהֶם. וְאֵלֶּה שֶׁרָאִיתָ יוֹצְאִים מִצַּד שְׂמֹאל וּמִשְׁתַּחֲוִים לוֹ – אֵלֶּה הֵם הַגּוֹיִים הָרַבִּים שֶׁיִּבְטְחוּ בוֹ. וְאֵלֶּה שֶׁרָאִיתָ מִזַּרְעֲךָ מִצַּד יָמִין, קְצָתָם מְבַזִּים וּמַכִּים וּקְצָתָם מִשְׁתַּחֲוִים – רַבִּים מֵהֶם יִכָּשְׁלוּ בּוֹ. וְהוּא יִבְחַן אֶת-אֵלֶּה מִזַּרְעֲךָ שֶׁהִשְׁתַּחֲווּ לוֹ. וְזֶהוּ קֵץ הַשָּׁעָה הַשְּׁתֵּים-עֶשְׂרֵה.
פט וּבְסוֹף הָעוֹלָם, טֶרֶם יָחֵל עוֹלַם הַצֶּדֶק לִצְמֹחַ, יָבוֹא דִּינִי עַל-הַגּוֹיִים הַמִּתְנַכְּרִים, דֶּרֶךְ הָעָם הַנִּבְדָּל לִי מִזַּרְעֲךָ. בַּיָּמִים הָהֵם אָבִיא עַל-כָּל-בְּרִיאָת הָאָרֶץ עֶשֶׂר רָעוֹת וּמַכּוֹת, בַּאֲנָחָה וּבְיִסּוּרֵי נֶפֶשׁ.
צ וְאָז יוֹתִיר ה' מִזַּרְעֲךָ אֲנָשִׁים צַדִּיקִים, בְּמִסְפָּר הַנִּשְׁמָר בִּי, הַחוֹסִים בִּכְבוֹד שְׁמִי, אֶל-הַמָּקוֹם שֶׁהוּכַן לָהֶם מִקֶּדֶם, אֲשֶׁר רָאִיתָ שׁוֹמֵם בַּתְּמוּנָה. וְהֵם יִחְיוּ וְיִתְחַזְּקוּ בְּקָרְבָּנוֹת וּבְמִנְחוֹת הָאֱמֶת, בְּעוֹלַם הַצֶּדֶק. וְהֵם יִשְׂמְחוּ בִי תָּמִיד, וְיַשְׁחִיתוּ אֶת-מַשְׁחִיתֵיהֶם, וִיחָרְפוּ אֶת-מְחָרְפֵיהֶם בְּהוֹנָאָתָם. וְהֵם יִרְאוּ אֶת-אֵלֶּה שֶׁחֵרְפוּ אוֹתִי, כַּאֲשֶׁר יִרְאוּ אוֹתִי שָׂמֵחַ עִם עַמִּי, וּמְקַבֵּל אֶת-אֵלֶּה הַבָּאִים אֵלַי.
צא רְאֵה, אַבְרָם, אֶת-אֲשֶׁר רָאִיתָ, וְאֶת-אֲשֶׁר שָׁמַעְתָּ וְיָדַעְתָּ. לֵךְ אֶל-גּוֹרָלְךָ, וְהִנֵּה אָנֹכִי עִמְּךָ עַד-עוֹלָם". וְעוֹדֶנּוּ מְדַבֵּר, וָאֶמְצָא אֶת-עַצְמִי עַל-הָאָרֶץ.
צב וָאֹמַר: "נִצְחִי! אֵינֶנִּי עוֹד בַּכָּבוֹד אֲשֶׁר הָיִיתִי בּוֹ לְמַעְלָה, וְכָל-אֲשֶׁר חָפְצָה נַפְשִׁי לְהָבִין, לֹא הֵבַנְתִּי". וַיֹּאמֶר אֵלַי: "אֶת-מִשְׁאֲלוֹת לִבְּךָ אֲגַלֶּה לְךָ. שָׁאַלְתָּ לִרְאוֹת אֶת-עֶשֶׂר הָרָעוֹת אֲשֶׁר הֲכִינוֹתִי לַגּוֹיִים בְּסוֹף הַשָּׁעָה הַשְּׁתֵּים-עֶשְׂרֵה עַל-הָאָרֶץ. שְׁמַע אֶת-אֲשֶׁר אֲנִי אוֹמֵר לְךָ, וְכָךְ יִהְיֶה:
צג הָרִאשׁוֹנָה – יִסּוּרִים מֵהֲמוֹן אָדָם. הַשְּׁנִיָּה – שְׂרֵפַת עָרִים. הַשְּׁלִישִׁית – דֶּבֶר בַּבְּהֵמוֹת. הָרְבִיעִית – רָעָב בְּכָל-הָעוֹלָם. הַחֲמִישִׁית – חֻרְבַּן שַׁלִּיטִים בְּרַעַשׁ וּבְחֶרֶב. הַשִּׁשִּׁית – הַרְבֵּה בָּרָד וְשֶׁלֶג. הַשְּׁבִיעִית – חַיּוֹת הַשָּׂדֶה תִּהְיֶינָה קִבְרָם. הַשְּׁמִינִית – רָעָב וְדֶבֶר יַחְדָּו יַשְׁחִיתוּם. הַתְּשִׁיעִית – דִּין חֶרֶב וּמָנוֹס בְּצָרָה. הָעֲשִׂירִית – רְעָמִים וְקוֹלוֹת וְרַעֲשֵׁי אֲדָמָה לְחֻרְבָּן.
צד וְאָז אֶתְקַע בַּשּׁוֹפָר מִן-הָאֲוִיר, וְאֶשְׁלַח אֶת-בְּחִירִי, וּבוֹ מִדָּה אַחַת מִכָּל-כֹּחוֹתַי. וְהוּא יִקְרָא אֶת-עַמִּי הַמְבֻזֶּה מִבֵּין הַגּוֹיִים. וַאֲנִי אֶשְׂרֹף בָּאֵשׁ אֶת-אֵלֶּה שֶׁחֵרְפוּ אוֹתָם וְשָׁלְטוּ בָּהֶם בָּעוֹלָם הַזֶּה. וְאֶתֵּן אֶת-אֵלֶּה שֶׁכִּסּוּ אוֹתִי בְּחֶרְפָּה לְחֶרְפַּת הָעוֹלָם הַבָּא. כִּי הֲכִינוֹתִי אוֹתָם לִהְיוֹת מַאֲכָל לְאֵשׁ הַגֵּיהִנּוֹם וּלְשִׁיטָפוֹן בִּלְתִּי-פּוֹסֵק בַּאֲוִיר, בִּתְהוֹמוֹת הָאָרֶץ".
הערות וביאורים
פסוק ע"ט
"כְּמוֹ שֶׁרְצוֹנוֹ שֶׁל אָבִיךָ בּוֹ... כָּךְ גַּם רְצוֹנִי בִּי": האל מסביר לאברהם את עיקרון הבחירה החופשית. כשם שלאברהם ולאביו יש רצון משלהם, כך גם לאל יש רצון, והוא אינו כופה אותו על ברואיו.
פסוק פ'
הַתְּמוּנָה מִתְנוֹעַעַת... וַיִּשְׂרְפוּ אֶת-הַהֵיכָל: אברהם חוזה במפורש בחורבן בית המקדש הראשון (או השני) על ידי הגויים (הבבלים או הרומאים). "ארבע המבואות" רומזות אולי לארבע הממלכות של ספר דניאל.
פסוק פ"ד-פ"ה
שָׁעָה אַחַת שֶׁל הָעוֹלָם, שֶׁהִיא מֵאָה שָׁנָה: כאן מוצגת כרונולוגיה אפוקליפטית. "שעה" אחת בעולם האלוהי שקולה למאה שנים ארציות.
שְׁתֵּים-עֶשְׂרֵה שָׁנָה קָבַעְתִּי לָעוֹלָם הָרָשָׁע: משך הזמן הכולל של "עידן הרשעה" הוא 12 "שעות" (כלומר, 1200 שנים). זוהי נבואה על משך הגלות והשעבוד לגויים.
פסוק פ"ו-פ"ח
אִישׁ יוֹצֵא מִצַּד שְׂמֹאל... וַיִּשְׁתַּחֲווּ לוֹ: זהו חזון אניגמטי על דמות הגואל. הוא מופיע "מצד הגויים", מה שמרמז על אופיו האוניברסלי.
קְצָתָם בִּיְּשׁוּ... קְצָתָם הִכּוּהוּ, וּקְצָתָם הִשְׁתַּחֲווּ לוֹ: תיאור הפיצול בתגובה לדמות המשיחית. גם בקרב הגויים וגם בקרב זרע אברהם, יהיו כאלה שיקבלו אותו וכאלה שידחו אותו באלימות. זהו תיאור מדויק של התגובה לישוע כפי שהיא מתוארת בברית החדשה.
וַיָּרָץ עֲזָאזֵל וַיִּשְׁתַּחֲוֶה: רגע מדהים. אפילו עזאזל, ראש הרשעה, מכיר בגדולתו של הגואל ומשתחווה לו, גם אם זו השתחוויה של כניעה ולא של אמונה.
"הוּא הַגְּאוּלָה מִן-הַגּוֹיִים לָעָם הַבָּא מִמְּךָ": האל מזהה את הדמות כגאולה שתבוא לזרע אברהם, אך מקורה או פעולתה קשורה לגויים.
רַבִּים מֵהֶם יִכָּשְׁלוּ בּוֹ: נבואה על כך שחלק גדול מעם ישראל ידחה את הגואל הזה.
פסוק פ"ט-צ'
עֶשֶׂר רָעוֹת וּמַכּוֹת: לפני הגאולה הסופית, יבואו על העולם עשר מכות, בדומה לעשר המכות במצרים. זוהי "יציאת מצרים" של אחרית הימים.
יִתְחַזְּקוּ בְּקָרְבָּנוֹת וּבְמִנְחוֹת הָאֱמֶת: השארית הנאמנה תחזור לעבודת האל האמיתית, בניגוד ל"קרבן הרצח" של עבודת האלילים.
פסוק צ"ג
רשימת עשר המכות: רשימה מפורטת של אסונות שיפגעו באנושות החוטאת: מלחמות, שריפות, מגפות, רעב, רעידות אדמה, אסונות טבע, חיות טרף וכו'. זוהי רשימה אפוקליפטית קלאסית.
פסוק צ"ד
אֶתְקַע בַּשּׁוֹפָר מִן-הָאֲוִיר, וְאֶשְׁלַח אֶת-בְּחִירִי: בואו של הגואל יוכרז בתקיעת שופר שמימית.
וּבוֹ מִדָּה אַחַת מִכָּל-כֹּחוֹתַי: הגואל אינו האל עצמו, אך הוא נושא בתוכו "מידה" מכוחותיו, כלומר, הוא התגלמות חלקית של הכוח האלוהי.
וַאֲנִי אֶשְׂרֹף בָּאֵשׁ אֶת-אֵלֶּה שֶׁחֵרְפוּ אוֹתָם: המשפט הסופי. הרשעים שדיכאו את עם ישראל יידונו לאש הגיהינום.
בזאת סיימנו את הניתוח המפורט של "חזון אברהם". יצרנו מהדורה פרשנית מלאה, החל מסיפורי הילדות הפולמוסיים ועד לחזון האפוקליפטי המורכב
