חלק ג' – ההתגלות, חורבן בית תרח והציווי על הקורבן (פרקים ח'-ט')
יט וַיְהִי כַּאֲשֶׁר דִּבַּרְתִּי אֶל-תֶּרַח אָבִי בַּחֲצַר בֵּיתִי, וְקוֹל חָזָק נָפַל מִן-הַשָּׁמַיִם בְּשֶׁטֶף שֶׁל אֵשׁ, קוֹרֵא וְאוֹמֵר: "אַבְרָם, אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמַר: "אֶת-הָאֵל הַנֶּעְלָם וְהַיּוֹצֵר אַתָּה מְחַפֵּשׂ בְּדַעַת לִבֶּךָ. אָנֹכִי הוּא. צֵא מִתֶּרַח אָבִיךָ וּמִבֵּיתוֹ צֵא, פֶּן-תִּסָּפֶה גַּם-אַתָּה בַּחֲטָאֵי בֵּית אָבִיךָ".
כ וָאֵצֵא. וַיְהִי כְּצֵאתִי, וְטֶרֶם הִסְפַּקְתִּי לַעֲבֹר אֶת-שַׁעַר הֶחָצֵר, וְקוֹל רַעַם בָּא וַיִּשְׂרֹף אוֹתוֹ וְאֶת-בֵּיתוֹ וְאֶת-כָּל-אֲשֶׁר בְּבֵיתוֹ, עַד לָאָרֶץ, אַרְבָּעִים אַמָּה.
כא אָז בָּא הַקּוֹל אֵלַי שְׁנֵי פְעָמִים: "אַבְרָם, אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמַר: "הִנְנִי, אַל-תִּירָא, כִּי אָנֹכִי הָאֵל הַקַּדְמוֹן וְהֶחָזָק, אֲשֶׁר בָּרָא רִאשׁוֹנָה אֶת-אוֹר הָעוֹלָם. וְאָנֹכִי מָגֵן לָךְ וְעוֹזֵר לָךְ.
כב לֵךְ וְקַח-לִי עֶגְלָה מְשֻׁלֶּשֶׁת וְעֵז מְשֻׁלֶּשֶׁת וְאַיִל מְשֻׁלָּשׁ, וְתֹר וְגוֹזָל, וְהַעֲלֵה לִי קָרְבָּן טָהוֹר. וּבַקָּרְבָּן הַזֶּה אָשִׂים אֶת-הָעוֹלָמוֹת, וְאַגִּיד לְךָ אֶת-אֲשֶׁר צָפַנְתִּי, וְדִבְרֵי גְדוֹלוֹת תִּרְאֶה אֲשֶׁר לֹא רָאִיתָ, כִּי אַתָּה אָהַבְתָּ לְחַפֵּשׂ אוֹתִי, וְאוֹתְךָ קָרָאתִי 'אוֹהֲבִי'.
כג אַךְ הִשָּׁמֶר לְךָ מִכָּל-מַאֲכָל הַיּוֹצֵא מִן-הָאֵשׁ, וּמִכָּל-מִשְׁתֵּה יַיִן, וּמִסִּיכַת שֶׁמֶן, אַרְבָּעִים יוֹם. וְאָז תַּקְרִיב לִי אֶת-הַקָּרְבָּן אֲשֶׁר צִוִּיתִיךָ, בַּמָּקוֹם אֲשֶׁר אַרְאֶךָּ עַל-הַר גָּבֹהַּ.
כד וְשָׁם אַרְאֶךָּ אֶת-הָעוֹלָמוֹת הַנִּקְרָאִים בִּשְׁמִי – הַנִּבְרָאִים וְהַנֶּעֱזָבִים, הַנִּבְנִים וְהַנֶּחֱרָבִים. וְאַגִּיד לְךָ אֶת-אֲשֶׁר יִהְיֶה בָּהֶם, הָרָעוֹת וְהַצְדָקוֹת בְּדוֹר בְּנֵי-הָאָדָם".
כה וַיְהִי כְּשָׁמְעִי אֶת-הַקּוֹל הַמַּכְרִיז לִי דְּבָרִים כָּאֵלֶּה, וָאַבִּיט הֵנָּה וָהֵנָּה, וְהִנֵּה אֵין רוּחַ אָדָם. וְכִמְעַט כָּלְתָה רוּחִי וְנִשְׁמָתִי יָצְאָה מִמֶּנִּי, וָהְיֶה כְּאֶבֶן וָאֶפֹּל אַפַּיִם אַרְצָה, כִּי לֹא-הָיָה בִי עוֹד כֹּחַ לַעֲמֹד עַל-הָאָרֶץ.
כו וְעוֹדֶנִּי נָפוּל עַל-פָּנַי אַרְצָה, וָאֶשְׁמַע קוֹל אוֹמֵר: "לֵךְ, יָהוֹאֵל, אַתָּה הַמְּתוֹלֵל בְּשֵׁם שְׁמִי הַבִּלְתִּי-נִשְׁאָל, וְהָקֵם אֶת-הָאִישׁ הַזֶּה וְחַזְּקֵהוּ מֵרַעֲדוֹ".
הערות וביאורים
פסוק י"ט
קוֹל חָזָק נָפַל מִן-הַשָּׁמַיִם בְּשֶׁטֶף שֶׁל אֵשׁ: ההתגלות הראשונה אינה חוויה עדינה, אלא אירוע קוסמי, מפחיד ואלים. האש, שקודם לכן הייתה רק כלי לבחינת האלילים, מתגלה כעת כביטוי לנוכחות האלוהית עצמה.
"צֵא מִתֶּרַח אָבִיךָ... פֶּן-תִּסָּפֶה": זוהי גרסה דרמטית של הציווי "לך לך" (בראשית י"ב). בעוד שבבראשית הציווי הוא למען העתיד ("ואעשך לגוי גדול"), כאן הוא ציווי הצלה מיידי. אברהם צריך לברוח כדי לא למות עם משפחתו החוטאת. זה מזכיר את בריחת לוט מסדום.
פסוק כ'
וַיִּשְׂרֹף אוֹתוֹ וְאֶת-בֵּיתוֹ: חורבן בית תרח הוא אקט של משפט אלוהי מיידי. המסר חד: עבודת אלילים דינה כליה. המדרש המפורסם על השלכת אברהם לכבשן האש על ידי נמרוד מקבל כאן היפוך: לא אברהם נזרק לאש, אלא בית אביו נשרף באש מן השמיים.
פסוק כ"ב
עֶגְלָה מְשֻׁלֶּשֶׁת וְעֵז מְשֻׁלֶּשֶׁת וְאַיִל מְשֻׁלָּשׁ: הציווי לקחת בעלי חיים "משולשים" (בני שלוש שנים) לקוח ישירות מתיאור ברית בין הבתרים בבראשית ט"ו, ט'. המחבר משתמש במסגרת המוכרת של ברית בין הבתרים, אך ממלא אותה בתוכן חדש.
וּבַקָּרְבָּן הַזֶּה אָשִׂים אֶת-הָעוֹלָמוֹת: משפט מיסטי ויוצא דופן. הקורבן אינו רק ברית על הארץ, אלא אקט קוסמולוגי שבו האל יגלה לאברהם את סודות הבריאה כולה, את העולמות העליונים והתחתונים.
"אוֹהֲבִי": האל מעניק לאברהם את התואר "אוהבי", תואר המופיע בישעיהו מ"א, ח' ("אַבְרָהָם אֹהֲבִי").
פסוק כ"ג
הִשָּׁמֶר לְךָ... אַרְבָּעִים יוֹם: לפני הקרבת הקורבן, אברהם מצווה על תקופת התקדשות של 40 יום. הצום וההינזרות (ממאכל מבושל, יין ושמן) הם הכנה רוחנית לקראת החזון השמימי. 40 יום הוא מספר טיפולוגי להתבודדות והכנה רוחנית, כמו 40 יום של משה על הר סיני.
פסוק כ"ה
וָהְיֶה כְּאֶבֶן וָאֶפֹּל אַפַּיִם אַרְצָה: תיאור התגובה הפיזית והנפשית להתגלות האלוהית. עוצמת החוויה מרוקנת את האדם מכוחו והופכת אותו לחומר דומם. זוהי תגובה אופיינית לנביאים בחזיונותיהם (השווה דניאל י', ח'-ט').
פסוק כ"ו
"לֵךְ, יָהוֹאֵל": כאן מוצגת לראשונה דמותו של המלאך המתווך, שילווה את אברהם במסעו השמימי.
יָהוֹאֵל (Yahoel): זהו שם תיאופורי מובהק, המכיל בתוכו את השם המפורש (יהו + אל). בספרות המיסטית של בית שני, יָהוֹאֵל הוא שם הקוד של המלאך מטטרון, המלאך הגבוה ביותר, "שר הפנים", ששמו כשם רבו. הוא המדריך של אברהם, בדומה למלאך המלווה את חנוך במסעותיו.
הַמְּתוֹלֵל בְּשֵׁם שְׁמִי הַבִּלְתִּי-נִשְׁאָל: "המתולל" (בסלאבית: посредъствка, מתווך) בשם ה'. יָהוֹאֵל הוא המתווך המורשה להשתמש בשם האלוהי. "השם הבלתי-נשאל" הוא כינוי לשם המפורש, שאין שואלים או הוגים אותו.
הגענו לנקודה שבה המלאך יָהוֹאֵל מתגלה לאברהם כדי להכין אותו לעלייה השמימית.
האם נמשיך לחלק הבא, שבו המלאך יָהוֹאֵל מציג את עצמו ואת תפקידיו הקוסמיים?

חלק ד' – התגלות המלאך יָהוֹאֵל (פרקים י'-י"ב)
כז וַיָּבֹא הַמַּלְאָךְ אֲשֶׁר שֻׁלַּח אֵלַי בִּדְמוּת אִישׁ, וַיֹּאחֶז בִּיד יְמִינִי וַיְקִימֵנִי עַל-רַגְלָי. וַיֹּאמֶר אֵלַי: "קוּם, יְדִיד הָאֵל אֲשֶׁר אֲהֵבְךָ, פֶּן-יֶאֱחָזְךָ רַעַד בְּנֵי-אָדָם. הִנֵּה שָׁלוּחַ אֲנִי אֵלֶיךָ לְחַזֶּקְךָ וּלְבָרֶכְךָ בְּשֵׁם הָאֵל אֲשֶׁר אֲהֵבְךָ, יוֹצֵר הַשָּׁמַיִם וְהָאָרֶץ. חֲזַק וְהִתְאַזֵּר לִקְרָאתוֹ.
כח אָנֹכִי הוּא יָהוֹאֵל, הַנִּקְרָא בְשֵׁם הַמַּרְעִיד אֶת-אֵלֶּה אֲשֶׁר עִמִּי בָּרָקִיעַ הַשְּׁבִיעִי, עַל-הַמִּשְׁטָח. כֹּחַ בְּתוֹךְ הַמְּתוֹכְכִים לַשֵּׁם הַנִּשְׂגָּב, הַמְשָׁרְתִים אֲשֶׁר לֹא לִי.
כט אָנֹכִי הוּא הַנִּתָּן, עַל-פִּי צִוּוּיוֹ, לְהַשְׁפִּיל אֶת-הַכְּרוּבִים, הַחַיּוֹת הַמִּתְנַצְּחוֹת זוֹ עִם זוֹ, וּלְלַמֵּד אֶת-שִׁירַת הַקֹּדֶשׁ בְּאֶמְצַע לֵיל הָאָדָם בַּשָּׁעָה הַשְּׁבִיעִית.
ל אָנֹכִי הוּא הַמְמֻנֶּה לְהַחְזִיק בַּלִּוְיְתָנִים, כִּי בִי הֻכְנְעוּ. וְעַל כָּל-הַתַּקָּפוֹת וְהַאיוּמִים שֶׁל כָּל-רֶמֶשׂ.
לא אָנֹכִי הוּא הַמְצֻוֶּה לְהַצִּית אֶת-בֵּית אָבִיךָ וְאוֹתוֹ עִמּוֹ, כִּי עָבַד לַמֵּתִים.
לב אָנֹכִי שָׁלוּחַ אֵלֶיךָ עַתָּה לְבָרֶכְךָ וְאֶת-הָאָרֶץ אֲשֶׁר הֵכִין לְךָ הַקַּדְמוֹן, הַנִּקְרָא בְשִׁמְךָ. וּלְמַעַנְךָ הִכְשַׁרְתִּי אֶת-דֶּרֶךְ הָאָרֶץ. קוּם, אַבְרָם, לֵךְ בְּבִטְחָה, שְׂמַח מְאֹד וַעֲלֹז, וְאָנֹכִי עִמָּךְ, כִּי הוּכַן לְךָ כְּבוֹד עוֹלָמִים מֵאֵת הַקַּדְמוֹן. לֵךְ וְהַשְׁלֵם אֶת-הַקָּרְבָּן שֶׁצֻּוֵּיתָ. הִנֵּה אָנֹכִי מְמֻנֶּה לִהְיוֹת עִמְּךָ וְעִם הַזֶּרַע שֶׁיֻּכַן מִמְּךָ. וְעִמִּי מִיכָאֵל, לְבָרֶכְךָ עַד-עוֹלָם. חֲזַק וְלֵךְ".
לג וָאָקָם וָאֵרֶא אֶת-אֲשֶׁר אָחַז בִּימִינִי וְהֵקִים אוֹתִי עַל-רַגְלָי. מַרְאֵה גוּפוֹ כְּאֶבֶן סַפִּיר, וּמַרְאֵה פָנָיו כְּתַרְשִׁישׁ, וּשְׂעַר רֹאשׁוֹ כַּשֶּׁלֶג. וּמִצְנֶפֶת עַל-רֹאשׁוֹ כְּמַרְאֵה הַקֶּשֶׁת בֶּעָנָן. וּלְבוּשׁוֹ אַרְגָּמָן, וּמַטֵּה זָהָב בְּיַד יְמִינוֹ.
לד וַיֹּאמֶר אֵלַי: "אַבְרָם!". וָאֹמַר: "הִנֵּה עַבְדְּךָ". וַיֹּאמֶר: "אַל-יַפְחִידְךָ מַרְאִי וְאַל-יַבְהִילְךָ דִּבּוּרִי, וְאַל-תִּסָּעֵר נַפְשֶׁךָ. לֵךְ עִמִּי, וְאֵלֵךְ עִמְּךָ עַד-הַקָּרְבָּן, נִרְאֶה. וּלְאַחַר הַקָּרְבָּן, בִּלְתִּי-נִרְאֶה עַד-עוֹלָם. חֲזַק וּלֵךְ".
הערות וביאורים
פסוק כ"ח
אָנֹכִי הוּא יָהוֹאֵל: המלאך מציג את עצמו בשמו המלא. כפי שצוין, זהו שם המזוהה עם מטטרון, "שר העולם".
בָּרָקִיעַ הַשְּׁבִיעִי: על פי הקוסמולוגיה של בית שני, השמיים מחולקים לשבעה רקיעים. הרקיע השביעי הוא הגבוה והקדוש מכולם, מקום מושבו של כיסא הכבוד. מיקומו של יָהוֹאֵל ברקיע השביעי מעיד על מעמדו הרם.
כֹּחַ בְּתוֹךְ הַמְּתוֹכְכִים לַשֵּׁם הַנִּשְׂגָּב: תיאור עצמי מורכב. הוא כוח הפועל בתוך קבוצת מלאכים מתווכים אחרים, המשרתים את השם האלוהי.
פסוק כ"ט
לְהַשְׁפִּיל אֶת-הַכְּרוּבִים, הַחַיּוֹת הַמִּתְנַצְּחוֹת: אחד מתפקידיו הוא להשליט סדר בקרב הדרגות הגבוהות ביותר של המלאכים, ה"חיות" (הכרובים) של המרכבה, המתוארות כאן כ"מתנצחות". זהו רעיון מפתיע, המרמז על מתח ודינמיקה אפילו בעולם המלאכים.
לְלַמֵּד אֶת-שִׁירַת הַקֹּדֶשׁ... בַּשָּׁעָה הַשְּׁבִיעִית: יָהוֹאֵל הוא המנצח על מקהלת המלאכים, המלמד אותם את שירת הקדושה ("קדוש, קדוש, קדוש"). השירה נאמרת בלילה, כשהעולם האנושי שקט.
פסוק ל'
לְהַחְזִיק בַּלִּוְיְתָנִים: תפקיד קוסמי נוסף. הוא הכוח השולט במפלצות הכאוס הקדמוניות, הלוויתנים. הוא שומר על גבולות הבריאה ומונע מהכאוס להשתלט.
פסוק ל"א
הַמְצֻוֶּה לְהַצִּית אֶת-בֵּית אָבִיךָ: המלאך לוקח אחריות אישית על חורבן בית תרח. הוא פועל כשליח הדין האלוהי.
פסוק ל"ב
וְעִמִּי מִיכָאֵל: יָהוֹאֵל אינו פועל לבד. הוא מזכיר את מיכאל, השר הממונה על ישראל, כשותף במשימת הברכה. הדבר מעיד על היררכיה ושיתוף פעולה בעולם המלאכים.
פסוק ל"ג
מַרְאֵה גוּפוֹ כְּאֶבֶן סַפִּיר...: תיאור מראהו של המלאך שואב ישירות מתיאורי המרכבה ביחזקאל ודניאל.
סַפִּיר: מזכיר את "לִבְנַת הַסַּפִּיר" תחת רגלי אלוהים (שמות כ"ד) ואת מראה הכיסא ביחזקאל א'.
תַרְשִׁישׁ: אבן חן זוהרת, מתארת את גופו של האיש הלבוש בבדים בדניאל י', ו'.
שְׂעַר... כַּשֶׁלֶג: מזכיר את תיאור "עתיק יומין" בדניאל ז', ט'.
הַקֶּשֶׁת בֶּעָנָן: דימוי הלקוח ישירות מחזון המרכבה ביחזקאל א', כ"ח.
לְבוּשׁוֹ אַרְגָּמָן, וּמַטֵּה זָהָב: סמלי מלכות ושלטון. המחבר מעניק למלאך יָהוֹאֵל את כל התכונות החיצוניות של הכבוד האלוהי, ובכך מדגיש את מעמדו כנציגו הישיר של האל.
פסוק ל"ד
עַד-הַקָּרְבָּן, נִרְאֶה. וּלְאַחַר הַקָּרְבָּן, בִּלְתִּי-נִרְאֶה: המלאך מסביר את טבע נוכחותו. הוא ילווה את אברהם באופן גלוי עד לרגע הקורבן. לאחר מכן, הקשר יהפוך לרוחני ובלתי נראה.
הגענו לנקודה שבה אברהם והמלאך יוצאים למסע אל ההר.
