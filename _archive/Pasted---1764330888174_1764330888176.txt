חזון אברהם (מהדורה פרשנית)
חלק א' – אברהם ואלילי אביו (פרקים א'-ב' על פי החלוקה המקובלת)
א וַיְהִי בַּיּוֹם אֲשֶׁר הִנְחַנִי לִבְדֹּק אֶת-אֱלֹהֵי אָבִי תֶּרַח וְאֶת-אֱלֹהֵי אָחִי נָחוֹר, וָאֶתְבּוֹנֵן מִי הוּא בֶּאֱמֶת אֵל גִּבּוֹר. אֲנִי, אַבְרָם, בְּעֵת שִׁמּוּשִׁי בְּקָרְבַּן אָבִי, כַּאֲשֶׁר כִּלִּיתִי אֶת-עֲבוֹדַת הַזֶּבַח לֵאלֹהֵי אָבִי תֶּרַח – אֱלֹהֵי עֵץ וָאֶבֶן, זָהָב וָכֶסֶף, נְחֹשֶׁת וּבַרְזֶל – וּבְבוֹאִי אֶל-הֵיכָלָם לַעֲבֹד, מָצָאתִי אֶת הָאֵל מָרוּמָת שְׁמוֹ, פֶּסֶל אֶבֶן, נֹפֵל אַפַּיִם אַרְצָה לְרַגְלֵי הָאֵל נָחוֹר, פֶּסֶל בַּרְזֶל.
ב וַיְהִי כִּרְאוֹתִי, וַיִּסָּעֵר לִבִּי, וָאֹמַר בְּדַעְתִּי כִּי לֹא אוּכַל לַהֲשִׁיבוֹ לִמְקוֹמוֹ לְבַדִּי, אֲנִי אַבְרָם, כִּי כָבֵד וְגָדוֹל הָאֶבֶן מְאֹד. וָאֵלֵךְ וָאַגֵּד לְאָבִי. וַיָּבֹא עִמִּי, וּכְשֶׁהֵרַמְנוּ אוֹתוֹ לַהֲשִׁיבוֹ לִמְקוֹמוֹ, נָפַל רֹאשׁוֹ מֵעָלָיו. וְעוֹדֶנִּי אוֹחֵז בְּרֹאשׁוֹ.
ג וַיְהִי כִּרְאוֹת אָבִי כִּי-נָפַל רֹאשׁ מָרוּמָת מֵעָלָיו, וַיֹּאמֶר אֵלַי: "אַבְרָם!" וָאֹמַר: "הִנֵּנִי". וַיֹּאמֶר לִי: "הָבֵא לִי מַעֲצָד וְכֵלִים מִן הַבַּיִת". וָאָבֵא לוֹ. וַיַּחְצֹב 'מָרוּמָת' אַחֵר מֵאֶבֶן אַחֶרֶת, בְּלֹא רֹאשׁ. וְאֶת-הָרֹאשׁ שֶׁנִּתַּק מִמָּרוּמָת וְאֶת שְׁאָר הַפְּסִילִים רִצֵּץ, וַיַּעַשׂ מֵהֶם חֲמִשָּׁה אֱלִילִים קְטַנִּים, וַיִּתֵּן לִי וַיְצַוֵּנִי לְמָכְרָם בַּחוּץ, בִּרְחוֹב הָעִיר.
ד וָאֶחֱבֹשׁ אֶת-חֲמוֹר אָבִי וָאֶשָּׂאֵם עָלָיו, וָאֵצֵא אֶל-הַדֶּרֶךְ לְמָכְרָם. וְהִנֵּה סוֹחֲרִים מִפַּדַּן-אֲרָם בָּאִים עִם גְּמַלֵּיהֶם, הוֹלְכִים מִצְרַיְמָה לִסְחֹר. וָאֶסַע עִמָּהֶם. וַיְהִי כִּי-נָהַק אֶחָד מִן הַגְּמַלִּים, וַיִּבָּעֵת הַחֲמוֹר, וַיִּנְעַר וַיַּשְׁלֵךְ אֶת-הָאֱלִילִים, וַיִּשָּׁבְרוּ שְׁלֹשָׁה מֵהֶם וּשְׁנַיִם נוֹתָרוּ.
ה וַיְהִי כִּרְאוֹת הָאֲרַמִּים כִּי אֱלִילִים עִמִּי, וַיֹּאמְרוּ לִי: "מַדּוּעַ לֹא אָמַרְתָּ לָנוּ? כִּי אִלּוּ יָדַעְנוּ, לֹא הָיָה שׁוֹמֵעַ הַחֲמוֹר אֶת-קוֹל הַגָּמָל וְלֹא הָיָה לְךָ נֵזֶק. כָּעֵת, תֵּן לָנוּ אֶת-הָאֱלִילִים שֶׁלֹּא נִשְׁבְּרוּ, וְנִתֵּן לְךָ מְחִיר רָאוּי עֲבוּר הַשְּׁבוּרִים". וְעַל-הַנּוֹתָרִים לָקַחְתִּי אֶת-הַכֶּסֶף, כִּי בְלִבִּי דָּאַבְתִּי אֵיךְ אָבִיא אֶת-הַמִּמְכָּר לְאָבִי. וְאֶת-שְׁבָרֵי הַשְּׁלֹשָׁה הִשְׁלַכְתִּי אֶל-מֵימֵי נְהַר גּוּר אֲשֶׁר הָיָה בַּמָּקוֹם הַהוּא, וַיִּטְבְּעוּ בַּמְּצוּלוֹת וְלֹא נוֹדַע מְקוֹמָם.
ו וְעוֹדֶנִּי הוֹלֵךְ בַּדֶּרֶךְ וַיִּסָּעֵר לִבִּי, וּמַחְשְׁבוֹתַי בִּי מִתְרוֹצְצוֹת. וָאֹמַר בְּלִבִּי: הִנֵּה מָרוּמָת נָפַל וְלֹא יָכֹל לָקוּם בְּהֵיכָלוֹ, וְגַם אֲנִי לְבַדִּי לֹא יָכֹלְתִּי לַהֲקִימוֹ עַד בּוֹא אָבִי. וּכְשֶׁלֹּא יָכֹלְנוּ לוֹ, נָפַל רֹאשׁוֹ מֵעָלָיו. וַיָּשֶׂם לוֹ אֶבֶן אַחֶרֶת, אֵל אַחֵר שֶׁעָשָׂה בְּלִי רֹאשׁ. וְהַחֲמִשָּׁה הָאֲחֵרִים נִשְׁבְּרוּ עַל-יְדֵי הַחֲמוֹר, וְלֹא יָכְלוּ לְהַצִּיל אֶת-עַצְמָם, וְאַף לֹא לַעֲשׂוֹת רָעָה לַחֲמוֹר עַל-אֲשֶׁר שְׁבָרָם. וְגַם שִׁבְרֵיהֶם לֹא עָלוּ מִן-הַנָּהָר.
ז וָאֹמַר אֶל-לִבִּי: אִם-כֵּן הַדָּבָר, אֵיךְ יָכֹל מָרוּמָת, אֱלוֹהַּ אָבִי, שֶׁרֹאשׁוֹ מֵאֶבֶן אַחַת וְגוּפוֹ מֵאֶבֶן אַחֶרֶת, לְהוֹשִׁיעַ אָדָם, אוֹ לִשְׁמֹעַ תְּפִלַּת אָדָם וּלְהֵעָנוֹת לוֹ?
ח וּבְהַגּוֹתִי כָּזֹאת, בָּאתִי אֶל-בֵּית אָבִי. וָאַשְׁקֶה אֶת-הַחֲמוֹר וָאָשֶׂם לוֹ מִסְפּוֹא. וָאֶקַּח אֶת-הַכֶּסֶף וָאֶתֵּן בְּיַד אָבִי תֶּרַח. וַיַּרְא וַיִּשְׂמַח, וַיֹּאמַר: "בָּרוּךְ אַתָּה, אַבְרָם, לֵאלֹהַי, כִּי הֵבֵאתָ אֶת-מְחִיר הָאֱלִילִים וְלֹא הָיָה עֲמָלִי לָרִיק".
ט וָאֶעֱנֶה וָאֹמַר לוֹ: "שְׁמַע, אָבִי תֶּרַח. בְּרוּכִים הָאֵלִים שֶׁאַתָּה עוֹשֶׂה, כִּי בִרְכָתָם – אָבְדָן, וְכֹחָם – רִיק. הֵם עַצְמָם לֹא יוֹשִׁיעוּ, אֵיךְ לְךָ יוֹשִׁיעוּ אוֹ אוֹתִי יְבָרֵכוּ? אֲנִי בֵּרַכְתִּי אוֹתְךָ, כִּי בְּעָרְמָתִי הֵבֵאתִי לְךָ אֶת-כֶּסֶף הַשְּׁבוּרִים". וַיִּשְׁמַע אֶת-דְּבָרַי וַיִּחַר אַפּוֹ עָלַי מְאֹד, עַל-אֲשֶׁר דִּבַּרְתִּי קָשׁוֹת עַל-אֱלֹהָיו.
הערות וביאורים
פסוק א'
אֱלֹהֵי עֵץ וָאֶבֶן...: רשימת החומרים שמתוכה עשויים האלילים היא טכניקה פולמוסית קלאסית בנבואת ישראל, שנועדה להדגיש את אפסותם. השווה: דברים ד', כ"ח: "וַעֲבַדְתֶּם-שָׁם אֱלֹהִים מַעֲשֵׂה יְדֵי אָדָם, עֵץ וָאֶבֶן...".
מָרוּמָת... נֹפֵל... לְרַגְלֵי הָאֵל נָחוֹר: הסיפור מזכיר באופן מכוון את סיפור נפילת דגון לפני ארון ה' בספר שמואל א' (פרק ה'). שם, דגון נופל לפני ארון ה'; כאן, אל אחד נופל לפני אל אחר. זוהי אירוניה דקה: אפילו בהיררכיה הפנימית של האלילים יש מאבקי כוח, והם חלשים ופגיעים. השם "מרומת" הוא כנראה שיבוש של שם אל שמי כלשהו, או שם אירוני ("המורם" שנפל).
פסוק ג'
וַיַּחְצֹב 'מָרוּמָת' אַחֵר... בְּלֹא רֹאשׁ: תרח אינו מסיק את המסקנה המתבקשת, אלא פועל באופן טכני לחלוטין. הוא יוצר "חלקי חילוף" לאל שלו. האבסורד גובר: הוא יוצר גוף של אל ללא ראש, כדי לחבר אליו את הראש השבור. זהו תיאור גרוטסקי של עבודת האלילים כמעשה אומנות חסר כל משמעות רוחנית.
וַיַּעַשׂ מֵהֶם חֲמִשָּׁה אֱלִילִים קְטַנִּים: תרח הוא "יצרן" של אלילים. הוא ממחזר את שברי האלים הגדולים כדי לייצר "מוצרים" חדשים לשוק. הביקורת כאן היא לא רק תיאולוגית, אלא גם כלכלית: עבודת האלילים היא עסק.
פסוק ד'-ה'
וַיִּנְעַר וַיַּשְׁלֵךְ אֶת-הָאֱלִילִים: האלים, שאמורים לשלוט בעולם, אינם מסוגלים אפילו לשרוד נסיעה על חמור. הם נתונים לחסדי מקריות וטבע (נהקת הגמל, בהלת החמור).
"מַדּוּעַ לֹא אָמַרְתָּ לָנוּ?": הסוחרים הארמיים מגיבים באמונה תפלה משעשעת. לדעתם, אם רק היו יודעים שיש אלים על החמור, הם היו יכולים למנוע את האסון (אולי על ידי תפילה או טקס כלשהו). הם אינם מבינים שהאלים עצמם חסרי אונים.
בְּעָרְמָתִי הֵבֵאתִי לְךָ אֶת-כֶּסֶף הַשְּבוּרִים: אברהם הצעיר מתגלה כפיקח. הוא מבצע את העסקה, אך בליבו הוא כבר מבין את הרמאות שבבסיס כל "עסקי האלילים".
פסוק ו'-ז'
וּמַחְשְׁבוֹתַי בִּי מִתְרוֹצְצוֹת: כאן מתואר תהליך ההתפכחות הפנימי של אברהם. הוא מפעיל לוגיקה פשוטה וקטלנית: אם אל אינו יכול להציל את עצמו, כיצד יושיע אחרים? אם הוא נשבר על ידי חמור, מה כוחו? אם הוא טובע בנהר, היכן אלוהותו? זהו שימוש בהיגיון צרוף כדי להפריך את האמונה.
שֶׁרֹאשׁוֹ מֵאֶבֶן אַחַת וְגוּפוֹ מֵאֶבֶן אַחֶרֶת: אברהם מצביע על האבסורד שבאל "מורכב" מחלקים, מעשה ידי אדם.
פסוק ח'-ט'
"בָּרוּךְ אַתָּה, אַבְרָם, לֵאלֹהַי": האירוניה מגיעה לשיאה. תרח מברך את אברהם בשם האלים שאברהם יודע שהם חסרי ערך, על כסף שהושג במרמה (מכירת שני אלים במחיר חמישה).
"בִרְכָתָם – אָבְדָן, וְכֹחָם – רִיק": אברהם מטיח את האמת בפני אביו. זוהי הכרזת העצמאות הרוחנית שלו. הוא הופך את ברכת אביו על פיה.
סיפור זה הוא דוגמה קלאסית לסוגת "סיפורי חצר" או "סיפורי חוכמה" פולמוסיים, המוכרים גם מספר דניאל (דניאל והאליל בל). הוא משתמש בהומור, אירוניה ולוגיקה כדי להציג את עבודת האלילים כאיוולת. זהו הבסיס שממנו יצמח החזון הגדול של אברהם בהמשך.

חלק ב' – אברהם, האל בַּרִישָׂת והחיפוש אחר האל האמיתי (פרקים ג'-ח')
י וַיְהִי כִּי חָשַׁבְתִּי לְהִתְחַמֵּק מֵחֲרוֹן אָבִי, וָאֵצֵא הַחוּצָה. וַיִּקְרָא לִי לֵאמֹר: "אַבְרָם!" וָאֹמַר: "הִנֵּנִי". וַיֹּאמַר: "קַח וֶאֱסֹף אֶת-שִׁבְרֵי הָעֵץ שֶׁבָּהֶם עָשִׂיתִי אֱלִילִים, וְטֶרֶם תָּבוֹא, הָכֵן לִי אֲרוּחַת צָהֳרָיִם".
יא וַיְהִי כַּאֲשֶׁר אָסַפְתִּי אֶת-שִׁבְרֵי הָעֵץ, וָאֶמְצָא בְתוֹכָם אֱלִיל קָטָן, שֶׁהָיָה יָכוֹל לְהִכָּנֵס בְּכַף יָדִי. וְכָתוּב הָיָה עַל-מִצְחוֹ: "הָאֵל בַּרִישָׂת". וְלֹא הִגַּדְתִּי לְאָבִי כִּי מָצָאתִי אֶת-הָאֵל בַּרִישָׂת בֵּין הַשְּׁבָבִים.
יב וַיְהִי כַּאֲשֶׁר שַׂמְתִּי אֶת-הַשְּׁבָבִים עַל-הָאֵשׁ לְהָכִין אֶת-אֲרוּחַת אָבִי, וָאֵצֵא לְבַקֵּשׁ דְּבַר-מַאֲכָל, הִצַּבְתִּי אֶת-בַּרִישָׂת לְיַד הָאָח הַבּוֹעֶרֶת, וָאֹמַר לוֹ בְּלַעַג: "בַּרִישָׂת, שְׁמֹר פֶּן-תִּכְבֶּה הָאֵשׁ עַד אֲשֶׁר אָשׁוּב. וְאִם תִּכְבֶּה, נְשֹׁף בָּהּ וְתִתְלַקַּח".
יג וָאֵצֵא וָאֶעֱשֶׂה אֶת-עִנְיָנַי. וּבְשׁוּבִי, מָצָאתִי אֶת-בַּרִישָׂת נֹפֵל עַל-גַּחַלְתּוֹ, וְרַגְלָיו אֲחוּזוֹת אֵשׁ וְנִשְׂרָפוֹת קָשָׁה. וָאֶשְׂחַק בְּקִרְבִּי שְׂחוֹק גָּדוֹל, וָאֹמַר: "בֶּאֱמֶת, בַּרִישָׂת, יוֹדֵעַ אַתָּה לְלַבּוֹת אֵשׁ וּלְבַשֵּׁל אֲרוּחָה!". וְעוֹדֶנִּי מְדַבֵּר בִּשְׂחוֹקִי, וְהוּא מְעַט-מְעַט נִשְׂרַף בָּאֵשׁ וַיְהִי לְאֵפֶר.
יד וָאָבֵא אֶת-הָאֲרוּחָה לְאָבִי וַיֹּאכַל. וָאֶתֵּן לוֹ יַיִן וְחָלָב, וַיֵּשְׁתְּ וַיִּתְעַנַּג, וַיְבָרֶךְ אֶת-מָרוּמָת אֱלֹהָיו. וָאֹמַר לוֹ: "אָבִי תֶּרַח, אַל-תְּבָרֵךְ אֶת-מָרוּמָת אֱלֹהֶיךָ וְאַל-תְּהַלְּלֵהוּ. הַלֵּל-נָא אֶת-בַּרִישָׂת אֱלֹהֶיךָ, כִּי כְּדֵי לְהַרְאוֹת אַהֲבָתוֹ אֵלֶיךָ, הִשְׁלִיךְ עַצְמוֹ לָאֵשׁ לְבַשֵּׁל אֶת-מַאֲכָלֶךָ". וַיֹּאמֶר אֵלַי: "וְהֵיכָן הוּא עַתָּה?". וָאֹמַר: "הָיָה לְאֵפֶר בַּחֲמַת הָאֵשׁ וַיְהִי לְעָפָר". וַיֹּאמֶר: "גָּדוֹל כֹּחוֹ שֶׁל בַּרִישָׂת! אֶעֱשֶׂה אַחֵר הַיּוֹם, וּמָחָר יָכִין הוּא אֶת-אֲרוּחָתִי".
טו וַאֲנִי אַבְרָם, כְּשָׁמְעִי דְּבָרִים כָּאֵלֶּה מֵאָבִי, וָאֶשְׂחַק בְּדַעְתִּי, וָאֶאֱנַח בְּמָרַת וְחֵמַת נַפְשִׁי, וָאֹמַר: "אֵיךְ יָכֹל מַעֲשֵׂה יָדָיו לִהְיוֹת עוֹזֵר לְאָבִי? הַאִם גּוּף יִהְיֶה כָּפוּף לְנַפְשׁוֹ, וְנֶפֶשׁ נִתְּנָה לוֹ, אַךְ הוּא בְּאִוַּלְתּוֹ חָסֵר דַּעַת?".
טז וָאֹמַר: "אַחַת הִיא, אֶסְבֹּל אֶת-הָרָעָה, אַךְ אֶשְׁלַח אֶת-דַּעְתִּי אֶל-הַטָּהֳרָה, וּמַחְשְׁבוֹתַי אָשִׂים גְּלוּיוֹת לְפָנָיו". וָאֶעֱנֶה וָאֹמַר: "אָבִי תֶּרַח, אֶת-מִי מֵהֶם תְּהַלֵּל, בְּאִוֶּלֶת מַחְשַׁבְתֶּךָ? הִנֵּה אֱלֹהֵי אָחִיךָ נָחוֹר נִצָּבִים בַּהֵיכָל. הֲלֹא נִכְבָּדִים הֵם מֵאֱלֹהֶיךָ? הִנֵּה הָאֵל זוּס שֶׁל אָחִיךָ נָחוֹר נִכְבָּד מִמָּרוּמָת אֱלֹהֶיךָ, כִּי מִזָּהָב טָהוֹר מַעֲשֵׂה אָדָם הוּא. וְאִם יִתְבַּלֶּה מֵרֹב יָמִים, יְחַדְּשׁוּהוּ. אֲבָל מָרוּמָת, אִם יִשָּׁבֵר אוֹ יִשָּׂרֵט, לֹא יִתְחַדֵּש עוֹד, כִּי אֶבֶן הוּא. וּמַה בַּאֲשֶׁר לָאֵל חוֹן הָעוֹמֵד עִם זוּס? וּבַרִישָׂת עַצְמוֹ נִשְׂרַף בָּאֵשׁ וְהָיָה לְאֵפֶר וְאֵינֶנּוּ עוֹד, וְאַתָּה אוֹמֵר: 'מָחָר אֶעֱשֶׂה אַחֵר'? אָבַד לַאֲבַדּוֹן!
יז הִנְנִי אוֹמֵר: הָאֵשׁ נִכְבֶּדֶת מִן-הַפְּסִילִים, כִּי הַבִּלְתִּי-נִכְנָעִים נִכְנָעִים לָהּ, וְהִיא מִתְלוֹצֶצֶת בַּכָּלִים בְּשַׁלְהַבְתָּהּ. אַךְ גַּם אֶת-הַמַּיִם לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הֵם גּוֹבְרִים עַל-הָאֵשׁ וּמַרְוִים אֶת-הָאָרֶץ. אַךְ גַּם אוֹתָם לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הֵם נִכְנָעִים תַּחַת הָאָרֶץ וּמְחַלְחֲלִים בָּהּ. אִם-כֵּן, אֶת-הָאָרֶץ אֲכַנֶּה נִכְבֶּדֶת, כִּי הִיא גּוֹבֶרֶת עַל טֶבַע הַמַּיִם. אַךְ גַּם אוֹתָהּ לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הַשֶּׁמֶשׁ מְיַבֶּשֶׁת אוֹתָהּ. אֶת-הַשֶּׁמֶשׁ אֲכַנֶּה נִכְבֶּדֶת, כִּי בְּקַרְנֶיהָ הִיא מְאִירָה אֶת-הַתֵּבֵל כֻּלָּהּ. אַךְ גַּם אוֹתָהּ לֹא אָשִׂים לֵאלֹהַּ, כִּי בַּלַּיְלָה הַיָּרֵחַ וְהֶעָנָן מַאֲפִילִים אֶת-אוֹרָהּ. וְגַם אֶת-הַיָּרֵחַ וְהַכּוֹכָבִים לֹא אֶקְרָא אֱלוֹהַּ, כִּי גַּם הֵם בְּעִתָּם בַּלַּיְלָה נֶעְלָמִים.
יח שְׁמַע, תֶּרַח אָבִי. אֲנִי מְחַפֵּשׂ לְפָנֶיךָ אֶת-הָאֵל אֲשֶׁר בָּרָא אֶת-הַכֹּל, וְלֹא אֶת-אֵלֶּה שֶׁאָנוּ סוֹבְרִים שֶׁהֵם אֱלֹהִים. מִי הוּא, אוֹ מַה הוּא, אֲשֶׁר הִטָּה שָׁמַיִם, אֲשֶׁר קָבַע אֶת-הַשֶּׁמֶשׁ וְאֶת-אוֹר הַיָּרֵחַ וְהַכּוֹכָבִים עִמּוֹ, אֲשֶׁר יִבֵּשׁ אֶת-הָאָרֶץ בְּתוֹךְ מַיִם רַבִּים, אֲשֶׁר אוֹתְךָ עַצְמְךָ שָׂם בֵּין הָאֱלִילִים? הַלְוַאי וְיִתְגַּלֶּה לָנוּ הוּא עַצְמוֹ!".
הערות וביאורים
פסוק י"א-י"ג
הָאֵל בַּרִישָׂת: שם זה, כמו "מרומת", הוא כנראה שיבוש או שם בדוי. ייתכן קשר למילה "ברוש" או לשורש שמי הקשור לאש. הסיפור כאן הוא גרסה מתוחכמת של מדרש אברהם והפסילים המפורסם (המופיע בבראשית רבה ובמקורות מאוחרים יותר), שבו אברהם שובר את אלילי אביו. כאן, במקום לשבור אותם, הוא עורך "ניסוי מדעי" מבוקר: הוא נותן לאל העץ משימה (לשמור על האש) וצופה בכישלונו המוחלט.
"שְׁמֹר פֶּן-תִּכְבֶּה הָאֵשׁ... נְשֹׁף בָּהּ": אברהם מדבר אל הפסל באירוניה דוקרנית, כאילו היה עבד או משרת.
"בֶּאֱמֶת, בַּרִישָׂת, יוֹדֵעַ אַתָּה לְלַבּוֹת אֵשׁ וּלְבַשֵּׁל אֲרוּחָה!": השחוק של אברהם נובע מהיפוך המציאות: האל, שאמור לשלוט באיתני הטבע, נשרף בעצמו באש שאותה היה אמור לשרת.
פסוק י"ד
"גָּדוֹל כֹּחוֹ שֶׁל בַּרִישָׂת! אֶעֱשֶׂה אַחֵר הַיּוֹם": תגובתו של תרח היא שיא האבסורד והעיוורון הרוחני. במקום להבין שהאל נשרף כי הוא עשוי עץ, הוא מפרש את האירוע כהוכחה לכוחו ומסירותו של האל ("הקריב את עצמו"), ומחליט פשוט לייצר חדש. המחבר מציג את תרח לא כרשע, אלא כמי שלכוד בתוך מערכת מחשבה אטומה לחלוטין להיגיון.
פסוק ט"ז
זוּס... חוֹן: הטקסט הסלאבי מזכיר כאן שמות של אלים. "זוּס" הוא כמובן ראש הפנתיאון היווני. "חון" יכול להיות שיבוש של אל מצרי (ח'ונסו, אל הירח) או אל אחר. אזכור אלים ספציפיים מעיד על כך שהפולמוס מכוון נגד התרבות ההלניסטית-סינקרטיסטית של התקופה.
"אָבַד לַאֲבַדּוֹן!": אברהם מסכם את גורלו של ברישאת במילים נחרצות.
פסוק י"ז
הָאֵשׁ... הַמַּיִם... הָאָרֶץ... הַשֶּׁמֶשׁ...: כאן מתחיל אחד הקטעים הפילוסופיים היפים ביותר בספרות התקופה. אברהם עובר מביקורת על אלילי מעשה ידי אדם להתבוננות באיתני הטבע. הוא בוחן אותם בשיטת "שרשרת הגדלות" (Climax או Sorites): הוא מחפש את הדבר החזק והנכבד ביותר. האש נכבדת מהפסל, אך המים נכבדים מהאש. הארץ נכבדת מהמים, אך השמש נכבדת מהארץ. השמש, למרות כוחה, מוגבלת על ידי הלילה והעננים.
מבנה טיעון זה מופיע גם במקורות אחרים, למשל בתלמוד הבבלי (מסכת בבא בתרא, דף י ע"א), שם מתוארת שרשרת דומה (ברזל<אש<מים<עננים<רוח<חולי<מוות<צדקה) כדי להדגים את כוחה של הצדקה. קיומו של מבנה דומה מעיד על מסורת דרשנית-פילוסופית משותפת.
פסוק י"ח
"אֲנִי מְחַפֵּשׂ... אֶת-הָאֵל אֲשֶׁר בָּרָא אֶת-הַכֹּל": לאחר ששלל את כל האפשרויות הנראות לעין (פסלים, איתני טבע, גרמי שמיים), אברהם מגיע למסקנה הלוגית: חייב להיות כוח ראשוני, טרנסצנדנטי, שברא את כל אלה. הוא אינו יכול לראות אותו, אך הוא יכול להסיק את קיומו.
"הַלְוַאי וְיִתְגַּלֶּה לָנוּ הוּא עַצְמוֹ!": זוהי נקודת המפנה. החיפוש השכלי-פילוסופי מסתיים בקריאה נואשת להתגלות ישירה. קריאה זו היא שתענה מיד בקול מן השמיים, ותתחיל את החלק השני, האפוקליפטי, של החיבור.
