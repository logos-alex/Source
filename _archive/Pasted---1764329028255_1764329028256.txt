חלק ב' – אברהם, האל בַּרִישָׂת והחיפוש אחר האל האמיתי (פרקים ג'-ח')
ט וַאֲנִי אַבְרָם, כְּשָׁמְעִי דְּבָרִים אֵלֶּה מֵאָבִי, וָאֶשְׂחַק בְּדַעְתִּי, וָאֶאֱנַח בְּמָרַת וְחֵמַת נַפְשִׁי, וָאֹמַר: "אֵיךְ יָכֹל מַעֲשֵׂה יָדָיו לִהְיוֹת עוֹזֵר לְאָבִי? הַאִם גּוּף יִהְיֶה כָּפוּף לְנַפְשׁוֹ, וְנֶפֶשׁ אֲדוֹנָיו נְתוּנָה לְאִוֶּלֶת וּלְבוּרוּת?".
י וָאֹמַר: "אַחַת הִיא, אֶסְבֹּל אֶת-הָרָעָה, אַךְ אֶשְׁלַח אֶת-דַּעְתִּי אֶל-הַכָּבוֹד, וּמַחְשְׁבוֹתַי אָשִׂים גְּלוּיוֹת". וָאֶעֱנֶה וָאֹמַר: "אָבִי תֶּרַח, אֶת-מִי מֵהֶם תְּהַלֵּל, בְּאִוֶּלֶת מַחְשַׁבְתֶּךָ? הִנֵּה אֱלֹהֵי אָבִיךָ נָחוֹר נִצָּבִים בַּהֵיכָל. הֲלֹא נִכְבָּדִים הֵם מֵאֱלֹהֶיךָ אֵלֶּה? הִנֵּה הָאֵל זוּס שֶׁל נָחוֹר אָבִיךָ נִכְבָּד מִמָּרוּמָת אֱלֹהֶיךָ, כִּי מִזָּהָב טָהוֹר מַעֲשֵׂה אָדָם הוּא. וְאִם יִתְבַּלֶּה מֵרֹב יָמִים, יְחַדְּשׁוּהוּ. אֲבָל מָרוּמָת, אִם יִשָּׂרֵט אוֹ יִשָּׁבֵר, לֹא יִתְחַדֵּש עוֹד, כִּי אֶבֶן הוּא.
יא וּמַה בַּאֲשֶׁר לָאֵל הָאַחֵר הָעוֹמֵד עִם זוּס? הֲרֵי הוּא נִכְבָּד מִבַּרִישָׂת הָאֵל, הֶעָשׂוּי עֵץ וּמְצֻפֶּה כֶּסֶף, שֶׁגַּם הוּא נִתָּן לְהַשְׁוָאָה, נִשְׁקָל עַל-יְדֵי אָדָם לְמַרְאִית עַיִן. וְהָאֵל בַּרִישָׂת שֶׁלְּךָ, עוֹד בְּטֶרֶם נֶחְטַב, הָיָה שֹׁרֶשׁ בָּאָרֶץ, גָּדוֹל וְנִפְלָא, מָלֵא עָלִים וּפְרָחִים וְתִפְאֶרֶת. וְאַתָּה כָּרַתָּ אוֹתוֹ בְּגַרְזֶן, וּבְחָכְמָתְךָ עָשִׂיתָ מִמֶּנּוּ אֵל. וְהִנֵּה כְּבָר הִתְיַבֵּשׁ וְאָבַד, מֵרוּם קוֹמָתוֹ נָפַל לָאָרֶץ, מִגְּדֻלָּתוֹ בָּא לִקְטַנּוּת, וּמַרְאֵה פָנָיו הָיָה לְרִקָּבוֹן. וְהוּא נִצַּת בָּאֵשׁ וְהָיָה לְאֵפֶר, וְאֵינֶנּוּ עוֹד! וְאַתָּה אוֹמֵר: 'מָחָר אֶעֱשֶׂה אַחֵר'? וְכֹחוֹ אָבַד לוֹ לַאֲבַדּוֹן!".
יב וּכְשֶׁחָשַׁב אַבְרָם כָּךְ, בָּא אֶל-אָבִיו וַיֹּאמַר: "אָבִי תֶּרַח! הָאֵשׁ נִכְבֶּדֶת מֵאֱלֹהֶיךָ הַנִּכְבָּדִים, הַזָּהָב וְהַכֶּסֶף, הָאֶבֶן וְהָעֵץ, כִּי אֱלֹהֶיךָ הַיּוֹצְאִים מֵהֶם נִכְנָעִים לָאֵשׁ, וְהָאֵשׁ מִתְלוֹצֶצֶת בָּם וְאוֹכֶלֶת אֶת-אֱלֹהֶיךָ. אַךְ גַּם אוֹתָהּ לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי כְּפוּפָה הִיא לַמַּיִם. וְנִכְבָּדִים מִמֶּנָּה הַמַּיִם, כִּי גּוֹבְרִים עַל-הָאֵשׁ וּמַצְמִיחִים אֶת-הָאָרֶץ. אַךְ גַּם אוֹתָם לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הַמַּיִם נִכְנָעִים תַּחַת הָאָרֶץ. אִם-כֵּן, אֶת-הָאָרֶץ אֲכַנֶּה נִכְבֶּדֶת, כִּי גּוֹבֶרֶת עַל טֶבַע הַמַּיִם. אַךְ גַּם אוֹתָהּ לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי גַם הִיא מִיּוֹבֶשֶׁת עַל-יְדֵי הַשֶּׁמֶשׁ וּמְעֻבֶּדֶת עַל-יְדֵי הָאָדָם. אֶת-הַשֶּׁמֶשׁ אֲכַנֶּה נִכְבֶּדֶת מֵהָאָרֶץ, כִּי בְּקַרְנֶיהָ מְאִירָה אֶת-הַתֵּבֵל כֻּלָּהּ. אַךְ גַּם אוֹתָהּ לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי בְּבוֹא הַלַּיְלָה, בַּחֹשֶׁךְ תִּתְכַּסֶּה. וְגַם אֶת-הַיָּרֵחַ וְאֶת-הַכּוֹכָבִים לֹא אֲכַנֶּה, כִּי גַם הֵם בְּעִתָּם בַּלַּיְלָה נֶעְלָם אוֹרָם.
יג אַךְ הִנֵּה, שְׁמַע אָבִי תֶּרַח, אֲנִי אַגִּיד לְךָ מִי הוּא הָאֵל אֲשֶׁר בָּרָא אֶת-הַכֹּל, וְהוּא הָאֵל הָאֲמִתִּי. אֲשֶׁר צָבַע אֶת-הַשָּׁמַיִם בְּאַרְגָּמָן וְהִזְהִיב אֶת-הַשֶּׁמֶשׁ, וְהֵאִיר אֶת-הַיָּרֵחַ וְאֶת-הַכּוֹכָבִים עִמּוֹ, יִבֵּשׁ אֶת-הָאָרֶץ בְּתוֹךְ מַיִם רַבִּים, וְאוֹתְךָ עַצְמְךָ שָׂם בֵּין הָאֱלִילִים, וְאוֹתִי עַתָּה חִפֵּשׂ בְּמַחְשַׁבְתִּי הַנְּבוֹכָה. הַלְוַאי וְיִתְגַּלֶּה לָנוּ הוּא עַצְמוֹ!".
הערות וביאורים (ניתוח השוואתי)
השוואה נקודתית:
פסוק י"א: התיאור הלירי של האל ברישאת – זוהי התוספת המרכזית והיפה ביותר בחלק זה. בגרסה א', אברהם פשוט אומר שברישאת נשרף. כאן, בגרסה ב', אברהם נושא נאום שלם על גורלו הטרגי של ברישאת. הוא משתמש באירוניה פואטית כדי להדגיש את האבסורד:
"עוֹד בְּטֶרֶם נֶחְטַב, הָיָה שֹׁרֶשׁ בָּאָרֶץ, גָּדוֹל וְנִפְלָא, מָלֵא עָלִים וּפְרָחִים": אברהם מתאר את העץ במצבו הטבעי, החי והיפה, לפני שהפך לפסל. הוא היה "אל" אמיתי יותר כשהיה יציר של הטבע.
"וְאַתָּה כָּרַתָּ אוֹתוֹ בְּגַרְזֶן, וּבְחָכְמָתְךָ עָשִׂיתָ מִמֶּנּוּ אֵל": האירוניה חדה. "חכמתו" של תרח היא שהפכה יצור חי וצומח לחפץ דומם.
"מֵרוּם קוֹמָתוֹ נָפַל לָאָרֶץ, מִגְּדֻלָּתוֹ בָּא לִקְטַנּוּת": תיאור הנפילה הוא גם פיזי וגם מטאפורי. הפסל הוא צל חיוור וקטן של העץ המקורי.
"וּמַרְאֵה פָנָיו הָיָה לְרִקָּבוֹן": עוד לפני שנשרף, הפסל כבר היה בתהליך של ריקבון, כי הוא עץ מת. התוספת הזו הופכת את הטיעון של אברהם מרציונלי לאמוציונלי ופיוטי. הוא לא רק מפריך את האלוהות של הפסל, אלא גם מבכה את אובדן היופי הטבעי שהיה בו.
פסוק י"ב: "וּמְעֻבֶּדֶת עַל-יְדֵי הָאָדָם" – תוספת קטנה אך משמעותית. בשרשרת הגדלות, גרסה ב' מוסיפה שהארץ לא רק מיובשת על ידי השמש, אלא גם "מעובדת על ידי האדם". זה מחזק את הטיעון: אם האדם יכול לעבד ולשלוט בארץ, היא אינה יכולה להיות אלוהית.
פסוק י"ג: "אֲשֶׁר צָבַע אֶת-הַשָּׁמַיִם בְּאַרְגָּמָן וְהִזְהִיב אֶת-הַשֶּׁמֶשׁ" – דימויים פיוטיים נוספים. גרסה א' הייתה יותר עובדתית ("הטה שמיים, קבע את השמש"). גרסה ב' משתמשת בפעלים ציוריים ("צבע בארגמן", "הזהיב"), שהופכים את תיאור האל הבורא לעשיר ומלא הוד יותר.
פסוק י"ג: "וְאוֹתִי עַתָּה חִפֵּשׂ בְּמַחְשַׁבְתִּי הַנְּבוֹכָה" – הדגשה של החוויה האישית. בגרסה א', אברהם אומר שהאל "שם אותו בין האלילים". כאן, הוא אומר שהאל "חיפש אותו" בתוך מחשבתו הנבוכה. זה הופך את הקריאה האלוהית לאירוע פנימי ופסיכולוגי יותר, ולא רק לקביעה חיצונית. האל אינו רק בורא העולם, אלא גם זה שמעורר את הספק והחיפוש בלב האדם.
סיכום ביניים:
כמו בחלק הראשון, גם כאן גרסה ב' אינה מסתפקת בהעתקה. היא מעשירה את הטקסט באופן משמעותי:
העמקה פואטית: הטיעון נגד ברישאת הופך מטיעון לוגי לאלגיה קצרה על אובדן הטבע.
חידוד הטיעון הפילוסופי: התוספת על עיבוד האדמה על ידי האדם מחזקת את שרשרת הגדלות.
העמקה פסיכולוגית: החיפוש אחר האל מתואר כתהליך פנימי שאלוהים עצמו מעורר בלב אברהם.
נראה כי עורך גרסה ב' היה בעל רגישות ספרותית ופיוטית מפותחת יותר, וביקש להפוך את הסיפור לא רק למשכנע מבחינה לוגית, אלא גם למרגש ונוגע ללב.

חלק ג' – ההתגלות, חורבן בית תרח והציווי על הקורבן (פרקים ח'-ט')
יד וַיְהִי כַּאֲשֶׁר דִּבֵּר אַבְרָם דְּבָרִים כָּאֵלֶּה אֶל-אָבִיו תֶּרַח בַּחֲצַר בֵּיתוֹ, וְקוֹל חָזָק בָּא מִן-הַשָּׁמַיִם בְּשֶׁטֶף שֶׁל אֵשׁ, קוֹרֵא וְאוֹמֵר: "אַבְרָם, אַבְרָם!". וָאֹמַר: "הִנֵּנִי". "אֶת-הָאֵל הַיּוֹצֵר אַתָּה מְחַפֵּשׂ בְּדַעְתֶּךָ. צֵא מִתֶּרַח אָבִיךָ וּמִבֵּיתוֹ צֵא, פֶּן-תִּסָּפֶה גַּם-אַתָּה בַּחֲטָאֵי בֵּית אָבִיךָ".
טו וָאֵצֵא. וַיְהִי כְּצֵאתִי, וְטֶרֶם הִסְפַּקְתִּי לַעֲבֹר אֶת-שַׁעַר הֶחָצֵר, וְקוֹל רַעַם גָּדוֹל בָּא וַיִּשְׂרֹף אֶת-הַבַּיִת וְאֶת-כָּל-אֲשֶׁר בּוֹ, עַד לָאָרֶץ, אַרְבָּעִים אַמָּה.
טז אָז בָּא הַקּוֹל אֵלַי שְׁנֵי פְעָמִים: "אַבְרָם, אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמַר: "הִנְנִי, אַל-תִּירָא, כִּי אָנֹכִי הָאֵל הַקַּדְמוֹן וְהֶחָזָק, אֲשֶׁר בָּרָאתִי רִאשׁוֹנָה אֶת-אוֹר הָעוֹלָם. אָנֹכִי מָגֵן לָךְ וְעוֹזֵר לָךְ.
יז לֵךְ וְקַח-לִי עֶגְלָה מְשֻׁלֶּשֶׁת וְאַיִל מְשֻׁלָּשׁ וְעֵז מְשֻׁלֶּשֶׁת, וְתֹר וְגוֹזָל, וְהַעֲלֵה לִי קָרְבָּן טָהוֹר. וּבַקָּרְבָּן הַזֶּה אָשִׂים לְךָ אֶת-הָעוֹלָמוֹת, וְאַגִּיד לְךָ אֶת-אֲשֶׁר צָפַנְתִּי, וְדִבְרֵי גְדוֹלוֹת תִּרְאֶה אֲשֶׁר לֹא רָאִיתָ, כִּי אַתָּה אָהַבְתָּ לְחַפֵּשׂ אוֹתִי, וְאוֹתְךָ קָרָאתִי 'אוֹהֲבִי'.
יח אַךְ הִשָּׁמֶר לְךָ מִכָּל-מַאֲכָל הַיּוֹצֵא מִן-הָאֵשׁ, וּמִכָּל-מִשְׁתֵּה יַיִן, וּמִסִּיכַת שֶׁמֶן, אַרְבָּעִים יוֹם. וְאָז תַּקְרִיב לִי אֶת-הַקָּרְבָּן אֲשֶׁר צִוִּיתִיךָ, בַּמָּקוֹם אֲשֶׁר אַרְאֶךָּ עַל-הַר גָּבֹהַּ. וְשָׁם אַרְאֶךָּ אֶת-הָעוֹלָמוֹת, וּבְקוֹלִי אֶת-אֲשֶׁר נִבְרָא וְאֶת-אֲשֶׁר נִתְקַן, וְאַגִּיד לְךָ אֶת-אֲשֶׁר יִהְיֶה בָּהֶם, הָרָעוֹת וְהַצְדָקוֹת בְּדוֹר בְּנֵי-הָאָדָם".
יט וַיְהִי כְּשָׁמְעִי אֶת-הַקּוֹל הַמַּכְרִיז לִי דְּבָרִים כָּאֵלֶּה, וָאַבִּיט הֵנָּה וָהֵנָּה, וְהִנֵּה אֵין רוּחַ אָדָם. וַתִּבָּהֵל רוּחִי וְנִשְׁמָתִי יָצְאָה מִמֶּנִּי, וָהְיֶה כְּאֶבֶן וָאֶפֹּל אַפַּיִם אַרְצָה, כִּי לֹא-הָיָה בִי עוֹד כֹּחַ לַעֲמֹד עַל-הָאָרֶץ.
כ וְעוֹדֶנִּי נָפוּל עַל-פָּנַי אַרְצָה, וָאֶשְׁמַע קוֹל קָדוֹשׁ אוֹמֵר: "לֵךְ, אַלְתֶּזֶה, בִּשְׁמִי הַבִּלְתִּי-נֶאֱמָר, וְהָקֵם אֶת-הָאִישׁ הַזֶּה וְחַזְּקֵהוּ מֵרַעֲדוֹ".
כא וַיָּבֹא הַמַּלְאָךְ אֲשֶׁר שֻׁלַּח אֵלַי בִּדְמוּת אִישׁ, וַיֹּאחֶז בְּיַד יְמִינִי וַיְקִימֵנִי עַל-רַגְלָי. וַיֹּאמֶר אֵלַי: "קוּם, אַבְרָם, יְדִיד הָאֵל אֲשֶׁר אֲהֵבְךָ, פֶּן-יֶאֱחָזְךָ רַעַד בְּנֵי-אָדָם. הִנֵּה שָׁלוּחַ אֲנִי אֵלֶיךָ לְחַזֶּקְךָ וּלְבָרֶכְךָ בְּשֵׁם הָאֵל אֲשֶׁר אֲהֵבְךָ, יוֹצֵר הַשָּׁמַיִם וְהָאָרֶץ. חֲזַק וְהִתְאַזֵּר לִקְרָאתוֹ.
כב אָנֹכִי הוּא יָהוֹאֵל, בֵּן, הַנִּקְרָא בְשֵׁם הַמַּרְעִיד אֶת-אֵלֶּה אֲשֶׁר עִמִּי בָּרָקִיעַ הָאֶמְצָעִי, עַל-הַמִּשְׁטָח. כֹּחַ בְּתוֹךְ הַמְּתוֹכְכִים לַשֵּׁם הַנִּשְׂגָּב, הַמְשָׁרְתִים.
כג אָנֹכִי הוּא הַנִּתָּן, עַל-פִּי צִוּוּיוֹ, לְהַשְׁפִּיל אֶת-הַכְּרוּבִים, הַחַיּוֹת הַמִּתְנַצְּחוֹת זוֹ עִם זוֹ, וּלְלַמֵּד אֶת-אֵלֶּה הָעוֹלִים אֶת-שִׁירַת הַלַּיְלָה שֶׁל הָאָדָם בַּשָּׁעָה הַשְּׁבִיעִית.
כד אָנֹכִי הוּא הַמְמֻנֶּה לְהַחְזִיק בַּלִּוְיְתָנִים, כִּי בִי הֻכְנְעוּ. וְעַל כָּל-הַתַּקָּפוֹת וְהַאיוּמִים שֶׁל כָּל-רֶמֶשׂ.
כה אָנֹכִי הוּא הַמְצֻוֶּה לְהַתִּיר אֶת-הַשְּׁאוֹל וּלְהַשְׁחִית אֶת-אֵלֶּה הַמִּתְרַפִּים בַּמֵּתִים.
כו אָנֹכִי הוּא הַמְצֻוֶּה לְהַצִּית אֶת-בֵּית אָבִיךָ וְאוֹתוֹ עִמּוֹ, כִּי עָבַד לַמֵּתִים.
הערות וביאורים (ניתוח השוואתי)
השוואה נקודתית:
פסוק י"ז: סדר הקורבנות – שינוי קטן אך מעניין. בגרסה א' (ובמקור בבראשית ט"ו) הסדר הוא: עגלה, עז, איל. כאן, בגרסה ב', הסדר הוא: עגלה, איל, עז. זהו שינוי זניח, שכנראה נובע מטעות העתקה, אך הוא מראה שהמעתיק לא היה כבול באופן מוחלט לסדר המקראי.
פסוק כ': "לֵךְ, אַלְתֶּזֶה" – שינוי משמעותי בשם המלאך. בגרסה א', האל קורא למלאך יָהוֹאֵל. כאן, הוא קורא לו אַלְתֶּזֶה (Alteze). זהו שם מסתורי, שאינו נשמע שמי. ייתכן שזהו שיבוש של שם אחר, או שם ייחודי למסורת ההעתקה של כתב יד זה. למרות זאת, בפסוק כ"ב המלאך מציג את עצמו בשם יָהוֹאֵל, מה שיוצר חוסר עקביות. ייתכן ש"אלתזה" הוא כינוי או תואר, בעוד "יהואל" הוא שמו האמיתי.
פסוק כ"ב: "אָנֹכִי הוּא יָהוֹאֵל, בֵּן" – תוספת תיאולוגית טעונה. המילה "בֵּן" (сын בסלאבית) אינה מופיעה בגרסה א'. זוהי תוספת דרמטית, שכמעט בוודאות מעידה על עריכה נוצרית. הוספת התואר "בן" למלאך העליון, ששמו כשם רבו, יוצרת הקבלה ברורה למעמדו של ישוע כ"בן האלוהים". עורך גרסה זו רצה כנראה לחזק את הקשר בין דמותו של המלאך המתווך לבין ישוע.
פסוק כ"ב: "בָּרָקִיעַ הָאֶמְצָעִי" – שינוי קוסמולוגי. בגרסה א', יהואל נמצא "ברקיע השביעי" (הגבוה ביותר). כאן, הוא נמצא "ברקיע האמצעי". שינוי זה מנמיך לכאורה את מעמדו, אך ייתכן שהכוונה היא שהוא פועל מהמרכז, כציר שמחבר בין העולמות.
פסוק כ"ג: "וּלְלַמֵּד אֶת-אֵלֶּה הָעוֹלִים" – שינוי בתפקיד. בגרסה א', יהואל מלמד את הכרובים את שירת הקודש. כאן, הוא מלמד את "אלה העולים", כלומר, את הנשמות העולות לשמיים (כמו אברהם עצמו). זה הופך את תפקידו של יהואל מניהול פנימי של עולם המלאכים לתפקיד של מדריך רוחני לנשמות אנושיות.
פסוק כ"ה: "אָנֹכִי הוּא הַמְצֻוֶּה לְהַתִּיר אֶת-הַשְּׁאוֹל וּלְהַשְׁחִית אֶת-אֵלֶּה הַמִּתְרַפִּים בַּמֵּתִים" – תוספת של תפקיד חדש לחלוטין. תפקיד זה אינו קיים כלל בגרסה א'. כאן, יהואל מקבל תפקיד של שליטה על השאול ועל דין המתים. הוא "מתיר" (פותח) את השאול ומשחית את אלה ש"מתרפים" (כלומר, חוטאים או שוגים) בקרב המתים. תוספת זו מעצימה מאוד את כוחו של יהואל, והופכת אותו גם לשליט על עולם המוות, בדומה לתפקידים שיוחסו לישוע בברית החדשה (למשל, ירידה לשאול).
סיכום ביניים:
ההבדלים בחלק זה משמעותיים וחושפים מגמה ברורה:
העצמת הדמיון לישוע (כריסטולוגיזציה): התוספת של התואר "בֵּן" והתפקיד של שליטה על השאול ("להתיר את השאול") הן תוספות נוצריות מובהקות, שנועדו להפוך את דמותו של המלאך יהואל לפרה-פיגורציה (דמות מקדימה) של ישוע.
שינויים קוסמולוגיים ותפקודיים: מיקומו של יהואל משתנה מ"הרקיע השביעי" ל"רקיע האמצעי", ותפקידו כמורה לשירת קודש משתנה ממורה למלאכים למורה לנשמות אדם.
חוסר עקביות בשמות: השם "אלתזה" מופיע בקריאת האל, בעוד המלאך מציג עצמו כ"יהואל", מה שיכול להעיד על שילוב של שתי מסורות העתקה שונות.
נראה בבירור שעורך גרסה ב' לא הסתפק בתיאור המלאך כפי שהיה בגרסה א', אלא עיבד את דמותו באופן אקטיבי כדי להתאים אותה לתפיסות תיאולוגיות נוצריות ולהעצים את תפקידו כמתווך וכגואל.
