והידין גוריא דאריא אתחמת חמתא שגיאתא, ונפק בתר עיטא ושתיה ואטרדה, ואובדה מן אנטיוכיא דסוריא עדמא לתחומא דמדנחא דהי ארעא דעיטא. ונפק נמרא מן גרביא, ועמא רבא אך קמצא דאפרח נפק עמה. וסלק לנהרא פרת, וקם בעודרנא דגוריא דאריא, ומן תמן נחתו תריהון לארעא דפרס. ונפק תורא לאורעהון עם חילא רבא, וגוריא דאריא על בין תרתין קרנתה דתורא ותבר לתרתיהין. ושתה וחטפה לארעא ואוקדה בנורא. וערקו עיטא מן קדמיהון ונחתו לארעהון. וגוריא דאריא רדף בתרהון ואובדה בפתגמא דחרבא, מטל דאהפך אפה אלהא מנהון מטל טנפותהון יתירתא. ואחדה ושתי לארעה וסחפה עדמא לשתאסיה. ולא תיתב תוב לעלם, מטל דשאטו במריא ושטו בפוקדנוהי. ותהוא רגושיא רבתא בארעא זועא וכפנא ומותנא, ודחלתא ותיתורא נשלטון בעממא עדמא דנפלון ונמותון דלא כאיבין ודלא כורהנין מן דחלתא דשלטא עליהון. ויסק גוריא דאריא בחילא רבא לארעא דשובדנא ומשעבד לה למדתא, ואולצנא רבא הוא בארעא דלא הוא אכותה. ובנא שורא לפיניקיא ולחריבתא דבה. ולדרמסוק סחף עדמא לשתאסיא, מטל דמרדת בעליא. וסלק לאורשלם בטוכסא רבא, ומן תמן פנא ואזל לקריתא מלכותה.
וְאָז גּוּר הָאַרְיֵה הִתְחַמֵּם בְּחֵמָה גְּדוֹלָה, וְיָצָא אַחֲרֵי הָעוֹרְבִים וְשָׁדַד אוֹתָם וְגֵרְשָׁם, וְהֶאֱבִידָם מֵאַנְטִיוֹכְיָה שֶׁל סוּרְיָה עַד גְּבוּל הַמִּזְרָח אֲשֶׁר הִיא אֶרֶץ הָעוֹרְבִים. וְיָצָא הַנָּמֵר מִן הַצָּפוֹן, וְעַם רַב כְּאַרְבֶּה מְעוֹפֵף יָצָא עִמּוֹ. וְעָלָה אֶל נְהַר פְּרָת, וְעָמַד לְעֶזְרַת גּוּר הָאַרְיֵה, וּמִשָּׁם יָרְדוּ שְׁנֵיהֶם לְאֶרֶץ פָּרַס. וְיָצָא הַשּׁוֹר לִקְרָאתָם עִם חַיִל גָּדוֹל, וְגוּר הָאַרְיֵה נִכְנַס בֵּין שְׁתֵּי קַרְנֵי הַשּׁוֹר וְשָׁבַר אֶת שְׁתֵּיהֶן. וְשָׁדַד וְחָמַס אֶת הָאָרֶץ וּשְׂרָפָהּ בָּאֵשׁ. וְהָעוֹרְבִים בָּרְחוּ מִלְּפָנָיו וְיָרְדוּ לְאַרְצָם. וְגוּר הָאַרְיֵה רָדַף אַחֲרֵיהֶם וְהֶאֱבִידָם בְּפִי חֶרֶב, כִּי הָפַךְ הָאֱלֹהִים אֶת פָּנָיו מֵהֶם בִּגְלַל טֻמְאָתָם הַיְתֵרָה. וְלָכַד וְשָׁדַד אֶת אַרְצָם וְהָרַס אוֹתָהּ עַד לִיסוֹדוֹתֶיהָ. וְלֹא תֵשֵׁב עוֹד לְעוֹלָם, כִּי בָּזוּ לַאֲדֹנָי וְזִלְזְלוּ בְּמִצְוֹתָיו. וְתִהְיֶה מְהוּמָה גְדוֹלָה בָּאָרֶץ, רְעִידוֹת וְרָעָב וּמַגֵּפָה, וּפַחַד וּתְלָאָה יִשְׁלְטוּ בָעַמִּים עַד אֲשֶׁר יִפְּלוּ וְיָמוּתוּ בְּלִי כְאֵבִים וּבְלִי מַחֲלוֹת מִפַּחַד הַשּׁוֹלֵט עֲלֵיהֶם. וְיַעֲלֶה גּוּר הָאַרְיֵה בְּחַיִל גָּדוֹל לְאֶרֶץ הַהַבְטָחָה וּמְשַׁעְבֵּד אוֹתָהּ לַמַּס, וְיִהְיֶה מְצוּקָה גְדוֹלָה בָּאָרֶץ אֲשֶׁר כָּמוֹהָ לֹא הָיְתָה. וְיִבְנֶה חוֹמָה לְפֶנִיקְיָה וְלַחֳרָבוֹת אֲשֶׁר בָּהּ. וְאֶת דַּמֶּשֶׂק יַהֲרוֹס עַד לְיַסּוֹדוֹתֶיהָ, כִּי מָרְדָה בָּעֶלְיוֹן. וְיַעֲלֶה לִירוּשָׁלַיִם בְּמָצָד גָּדוֹל, וּמִשָּׁם יִפְנֶה וְיֵלֵךְ אֶל עִיר מַלְכוּתוֹ.
