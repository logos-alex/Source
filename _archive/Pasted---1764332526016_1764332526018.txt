חלק ו' – העלייה לשמיים וחזון המרכבה (פרקים ט"ו-כ')
מג וַיֹּאחֶז בִּי הַמַּלְאָךְ בְּיַד יְמִינִי, וַיּוֹשִׁיבֵנִי עַל-כָּנָף יְמָנִית שֶׁל הַגּוֹזָל, וְהוּא יָשַׁב עַל-כָּנָף שְׂמָאלִית שֶׁל הַתּוֹר, אֲשֶׁר לֹא נִשְׁחֲטוּ וְלֹא נִתְּחוּ. וַיַּעֲלֵנִי עַד-קְצֵה לַהֲבוֹת הָאֵשׁ. וַנַּעַל בְּגֹבַהּ, וְאוֹר חָזָק אֲשֶׁר אֵין לְסַפֵּר.
מד וְהִנֵּה בְּתוֹךְ הָאוֹר הַהוּא, אֵשׁ מִתְלַקַּחַת, וּבָהּ עַם רַב, דְּמוּת גְּבָרִים, כֻּלָּם מִשְׁתַּנִּים בְּמַרְאֵה פְנֵיהֶם, רָצִים וּמִשְׁתַּנִּים, מִשְׁתַּחֲוִים וְקוֹרְאִים בְּקוֹל שֶׁל שָׂפָה אֲשֶׁר לֹא יָדַעְתִּי.
מה וָאֹמַר אֶל-הַמַּלְאָךְ: "מַדּוּעַ הֶעֱלִיתַנִי הֵנָּה עַתָּה, כִּי אֵינֶנִּי יָכוֹל לִרְאוֹת, כִּי כְבָר נֶחֱלַשְׁתִּי וְרוּחִי עוֹזֶבֶת אוֹתִי". וַיֹּאמֶר אֵלַי: "הֱיֵה עִמִּי, אַל-תִּירָא. אֶת-אֲשֶׁר תִּרְאֶה בָּא לִקְרָאתֵנוּ בְּקוֹל גָּדוֹל – הוּא הָאֵל הַקַּדְמוֹן אֲשֶׁר אֲהֵבְךָ. אַךְ אוֹתוֹ עַצְמוֹ לֹא תִרְאֶה, פֶּן-תֶּחֱלַשׁ רוּחֲךָ, כִּי אָנֹכִי עִמְּךָ לְחַזֶּקְךָ".
מו וְעוֹדֶנּוּ מְדַבֵּר, וְהִנֵּה אֵשׁ בָּאָה לִקְרָאתֵנוּ מִסָּבִיב, וְקוֹל בְּתוֹךְ הָאֵשׁ כְּקוֹל מַיִם רַבִּים, כְּקוֹל הַיָּם בְּהִתְרַגְּשׁוֹ. וְהַמַּלְאָךְ הִשְׁתַּחֲוָה עִם מַחֲנוֹת הַמַּלְאָכִים. וַאֲנִי רָצִיתִי לִפֹּל אַפַּיִם אַרְצָה, אַךְ מְקוֹם הַגֹּבַהּ אֲשֶׁר עָמַדְנוּ בּוֹ, פַּעַם עָלָה וּפַעַם יָרַד. וַיֹּאמַר: "הִשְׁתַּחֲוֵה כָּאן, אַבְרָם, וְאָנֹכִי אֶת-הַשִּׁירָה אֲשֶׁר לִמַּדְתִּיךָ. כִּי אֵין כָּאן אֶרֶץ לִפֹּל עָלֶיהָ". וְהִשְׁתַּחֲוֵיתִי בִּלְבַד וְאָמַרְתִּי אֶת-הַשִּׁירָה אֲשֶׁר לִמְּדַנִי.
מז וַיֹּאמֶר: "אֱמֹר בְּלִי הֶרֶף". וָאֹמַר, וְהוּא עַצְמוֹ שָׁר: "קַדְמוֹן חָזָק, קָדוֹשׁ אֵל, יָחִיד בְּשִׁלְטוֹנְךָ, נוֹלָד מֵעַצְמְךָ, בִּלְתִּי-כָלֶה, טָהוֹר, בִּלְתִּי-נוֹלָד, בִּלְתִּי-נִפְגָּם, בִּלְתִּי-נִשְׁחָת, מֵאִיר מֵעַצְמְךָ, בְּלִי אֵם, בְּלִי אָב, בְּלִי שֹׁשֶׁלֶת, רָם, נִשְׂגָּב, אוֹהֵב אָדָם, נָדִיב, נַדְבָן, קַנַּאי, סַבְלָן, רַב-רַחֲמִים, אֵלִי, אֵלִי! נִצְחִי, חָזָק, קָדוֹשׁ, צְבָאוֹת, נִשְׂגָּב! אֵל, אֵל, אֵל, אֵל, יָהוֹאֵל!
מח אַתָּה הוּא אֲשֶׁר אָהֲבָה נַפְשִׁי, שׁוֹמֵר נִצְחִי, אֵשׁ זוֹהֶרֶת, שׁוֹקֵט-רַעַם, מַבְרִיק-בָּרָק, רַב-עֵינַיִם, מְקַבֵּל תְּפִלּוֹת הַמְכַבְּדִים אוֹתְךָ.
מט אַתָּה הָאוֹר הַזּוֹרֵחַ לִפְנֵי אוֹר-הַבֹּקֶר עַל-בְּרִיאָתְךָ, וּבַשָּׁמַיִם אֵין מָקוֹר אוֹר אַחֵר מִזֹּהַר פָּנֶיךָ הַבִּלְתִּי-נִתְפָּס. קַבֵּל אֶת-תְּפִלָּתִי וְאֶת-הַקָּרְבָּן אֲשֶׁר בָּרָאתָ בְּעַצְמְךָ דַּרְכִּי, הַמְחַפֵּשׂ אוֹתְךָ. קַבְּלֵנִי וְהַרְאֵנִי וְלַמְּדֵנִי, וְהוֹדַע לְעַבְדְּךָ אֶת-אֲשֶׁר הִבְטַחְתָּ לִי".
נ וְעוֹדֶנִּי אוֹמֵר אֶת-הַשִּׁירָה, וּפִיהָ שֶׁל הָאֵשׁ אֲשֶׁר עַל-פְּנֵי הַמִּשְׁטָח הִתְרוֹמֵם מַעְלָה. וָאֶשְׁמַע קוֹל כְּהִתְרַגְּשׁוּת הַיָּם, שֶׁלֹּא פָּסַק מִן-הַלַּהֲבוֹת. וּכְשֶׁהִתְרוֹמְמָה הָאֵשׁ וְנִשְׂקְפָה מַעְלָה, רָאִיתִי תַּחַת הָאֵשׁ כִּסֵּא עָשׂוּי אֵשׁ, וְאַרְבַּע חַיּוֹת אֵשׁ שׁוֹרוֹת תַּחַת הַכִּסֵּא, וּמַרְאֵה כֻּלָּן אֶחָד, וּלְכָל-אַחַת אַרְבָּעָה פָּנִים. וְזֶה מַרְאֵה פְנֵיהֶן: פְּנֵי אֲרִי, פְּנֵי אָדָם, פְּנֵי שׁוֹר, פְּנֵי נֶשֶׁר. אַרְבָּעָה רָאשִׁים. וּלְכָל-אַחַת שֵׁשׁ כְּנָפַיִם: מִכְּתֵפֵיהֶן וּמִמָּתְנֵיהֶן. בִּכְנָפַיִם אֲשֶׁר מִכְּתֵפֵיהֶן כִּסּוּ אֶת-פְּנֵיהֶן, וּבִכְנָפַיִם אֲשֶׁר מִמָּתְנֵיהֶן עָטְפוּ אֶת-רַגְלֵיהֶן. וּבִכְנָפַיִם שֶׁבָּאֶמְצַע פָּרְשׂוּ וְעָפוּ.
נא וּכְשֶׁסִּיְּמוּ לָשִׁיר, הִבִּיטוּ זוֹ בְּזוֹ וְהִתְנַצְּחוּ זוֹ עִם זוֹ. וַיְהִי כִּרְאוֹת הַמַּלְאָךְ אֲשֶׁר עִמִּי כִּי הִתְנַצְּחוּ, עֲזָבַנִי וְרָץ אֲלֵיהֶן. וַיַּפְנֶה אֶת-פְּנֵי כָּל-חַיָּה מִפְּנֵי חֲבֶרְתָּהּ, לְבַל יִרְאוּ אֶת-פְּנֵיהֶן הַמִּתְנַצְּחוֹת, וַיְלַמְּדֵם אֶת-שִׁירַת הַשָּׁלוֹם.
נב וּבְעוֹדִי עוֹמֵד לְבַדִּי וּמִסְתַּכֵּל, רָאִיתִי מֵאֲחוֹרֵי הַחַיּוֹת מֶרְכָּבָה שֶׁל אוֹפַנֵּי אֵשׁ, כָּל-אוֹפַן מָלֵא עֵינַיִם מִסָּבִיב. וּמֵעַל הָאוֹפַנִּים הָיָה הַכִּסֵּא אֲשֶׁר רָאִיתִי, וְהוּא מְכֻסֶּה אֵשׁ, וְהָאֵשׁ סוֹבֶבֶת אוֹתוֹ. וְהִנֵּה אוֹר שֶׁאֵין לְתָאֲרוֹ סוֹבֵב אֶת-עַם הָאֵשׁ. וָאֶשְׁמַע קוֹל קְדֻשָּׁתָם כְּקוֹל אִישׁ אֶחָד.
הערות וביאורים
פסוק מ"ג
עַל-כָּנָף יְמָנִית שֶׁל הַגּוֹזָל... עַל-כָּנָף שְׂמָאלִית שֶׁל הַתּוֹר: תיאור סימטרי ומדויק של העלייה. אברהם והמלאך רוכבים על העופות שלא בותרו, והופכים אותם למרכבה שמימית.
פסוק מ"ד
עַם רַב, דְּמוּת גְּבָרִים... מִשְׁתַּנִּים בְּמַרְאֵה פְנֵיהֶם: זהו תיאור של מחנות המלאכים הנמוכים. הם מתוארים כיצורים דינמיים, משתנים, שאינם שומרים על צורה קבועה.
פסוק מ"ז
הַשִּׁירָה: זוהי תפילה או המנון ליטורגי המורכב משורה ארוכה של תארים לאל. סגנון זה, המכונה "ויה נגטיבה" (דרך השלילה – "בלי אם, בלי אב") ו"ויה פוזיטיבה" (דרך החיוב – "חזק, קדוש"), אופייני לספרות המיסטית והפילוסופית של התקופה (למשל, בכתבי פילון האלכסנדרוני).
אֵל, אֵל, אֵל, אֵל, יָהוֹאֵל!: שיא התפילה. השם "אל" חוזר ארבע פעמים (כמספר חיות הקודש) ומסתיים בשמו של המלאך המדריך, יָהוֹאֵל, שמכיל בתוכו את שם האל. זה מדגיש את תפקידו של יָהוֹאֵל כמתווך עליון.
פסוק נ'
כִּסֵּא עָשׂוּי אֵשׁ, וְאַרְבַּע חַיּוֹת...: זהו תיאור קלאסי של חזון המרכבה, השואב ישירות מיחזקאל (פרק א') וישעיהו (פרק ו').
אַרְבָּעָה פָּנִים: אֲרִי, אָדָם, שׁוֹר, נֶשֶׁר: אלו ארבעת הפנים של חיות הקודש ביחזקאל א'.
שֵׁשׁ כְּנָפַיִם: מספר הכנפיים לקוח מתיאור השרפים בישעיהו ו', ולא מתיאור החיות ביחזקאל (שם יש להן ארבע כנפיים). המחבר יוצר סינתזה של שני החזיונות.
פסוק נ"א
הִתְנַצְּחוּ זוֹ עִם זוֹ: הרעיון שהחיות, נושאות הכיסא, מתנצחות ביניהן, הוא רעיון ייחודי לטקסט זה. הוא מרמז על מתח פנימי בלב הקדושה, אולי בין מידת הדין למידת הרחמים. תפקידו של יָהוֹאֵל הוא להשכין שלום ביניהן, ובכך לאפשר למרכבה האלוהית לתפקד בהרמוניה.
פסוק נ"ב
אוֹפַנֵּי אֵשׁ, כָּל-אוֹפַן מָלֵא עֵינַיִם: תיאור האופנים, המלאכים הנעים לצד החיות, לקוח ישירות מיחזקאל א' וי'.
קוֹל קְדֻשָּׁתָם כְּקוֹל אִישׁ אֶחָד: למרות המתח הפנימי, בסופו של דבר, שירת כל מחנות המלאכים מתאחדת לקול אחד של הלל.
הגענו לשיא החזון השמימי. כעת, אברהם יופנה להביט מטה ולראות את העולמות שתחתיו.

חלק ז' – חזון הבריאה וגן העדן (פרקים כ"א-כ"ד)
נג וַיָּבֹא אֵלַי קוֹל מִתּוֹךְ הָאֵשׁ, אוֹמֵר: "אַבְרָם, אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמַר: "הַבֵּט בַּמִּשְׁטָחִים אֲשֶׁר תַּחַת הָרָקִיעַ שֶׁעָלָיו אַתָּה נִצָּב, וּרְאֵה כִּי אֵין עַל-אַף מִשְׁטָח אַחֵר זוּלַת זֶה אֲשֶׁר חָפַשְׂתָּ וַאֲשֶׁר אֲהֵבְךָ". וְעוֹדֶנּוּ מְדַבֵּר, וְהִנֵּה נִפְתְּחוּ הַשָּׁמַיִם אֲשֶׁר תַּחְתַּי.
נד וָאֵרֶא עַל-הָרָקִיעַ הַשְּׁבִיעִי אֲשֶׁר עָמַדְתִּי עָלָיו, וּבוֹ אוֹר וְטַל וְהָמוֹן מַלְאָכִים, וְכֹחַ שֶׁל כָּבוֹד בִּלְתִּי-נִרְאֶה מֵאֵלֶּה שֶׁרָאִיתִי לְמַעְלָה, וְאֶת-חַיּוֹת הַקֹּדֶשׁ, אַךְ לֹא רָאִיתִי שָׁם אִישׁ.
נה וָאַבִּיט מִן-הָהָר אֲשֶׁר עָמַדְתִּי עָלָיו אֶל-הָרָקִיעַ הַשִּׁשִּׁי, וָאֵרֶא שָׁם הָמוֹן מַלְאָכִים בְּלִי גוּף, הַמְּבַצְּעִים אֶת-צִוּוּיֵי מַלְאֲכֵי הָאֵשׁ אֲשֶׁר בָּרָקִיעַ הַשְּׁבִיעִי.
נו וָאֵרֶא שָׁם, בָּרָקִיעַ הַחֲמִישִׁי, כֹּחוֹת שֶׁל כּוֹכָבִים, וְהֵם מְבַצְּעִים אֶת-צִוּוּיֵיהֶם. וְהֵם נִשְׁמָעִים לְכֹחוֹת הַשָּׁמַיִם.
נז וַיֹּאמֶר אֵלַי הַקַּדְמוֹן הֶחָזָק: "אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמֶר: "הַבֵּט לְמַעְלָה בַּכּוֹכָבִים אֲשֶׁר מִתַּחְתֶּיךָ, וּסְפֹר אוֹתָם, וְהַגֵּד אֶת-מִסְפָּרָם". וָאֹמַר: "אֵיךְ אוּכַל, וַאֲנִי אָדָם?". וַיֹּאמֶר לִי: "כְּמִסְפַּר הַכּוֹכָבִים וּכְכֹחָם אָשִׂים לְזַרְעֲךָ עַם וְגוֹי, הַנִּבְדָּלִים לִי בְּגוֹרָלִי, עִם עֲזָאזֵל".
נח וָאֹמַר: "נִצְחִי וְחָזָק! יֹאמַר-נָא עַבְדְּךָ לְפָנֶיךָ, וְאַל-יִחַר אַפְּךָ עַל-בְּחִירֶךָ. הִנֵּה, טֶרֶם הֶעֱלִיתַנִי, חֵרֵף אוֹתִי עֲזָאזֵל. וְאֵיךְ עַתָּה, כְּשֶׁאֵינוֹ לְפָנֶיךָ, קָבַעְתָּ אוֹתוֹ עִמָּם?". וַיֹּאמֶר אֵלַי: "הַבֵּט-נָא תַּחַת רַגְלֶיךָ, אֶל-הַמִּשְׁטָח, וְהָבֵן אֶת-הַבְּרִיאָה שֶׁנּוֹסְדָה מִקֶּדֶם, וְאֶת-הַיְצוּרִים שֶׁבָּהּ, וְאֶת-הָעוֹלָם הַמּוּכָן אַחֲרֶיהָ".
נט וָאַבִּיט תַּחַת מִשְׁטַח רַגְלַי, וָאֵרֶא אֶת-הָרָקִיעַ הַשִּׁשִּׁי, וְאֶת-אֲשֶׁר בּוֹ. וּמִשָּׁם אֶת-הָאָרֶץ וּפְרִיָּהּ, וְאֶת-הַנָּעִים בָּהּ, וְאֶת-הַנּוֹשְׁמִים בָּהּ, וְאֶת-כֹּחַ בְּנֵי-הָאָדָם, וְאֶת-רִשְׁעַת נַפְשׁוֹתֵיהֶם, וְאֶת-צִדְקָתָם. וְאֶת-הַתְּהוֹמוֹת וְאֶת-אֲבַדּוֹנָם, וְאֶת-הַשְּׁאוֹל וְאֶת-עִנּוּיֶיהָ.
ס וָאֵרֶא שָׁם אֶת-הַיָּם וְאֶת-אִיָּיו, וְאֶת-חַיּוֹתָיו וְאֶת-דָּגָיו, וְאֶת-הַלִּוְיָתָן וְאֶת-מֶמְשַׁלְתּוֹ, וְאֶת-מִשְׁכָּבוֹ וְאֶת-מְאוּרוֹתָיו, וְאֶת-הָעוֹלָם הַנִּשְׁעָן עָלָיו, וְאֶת-חָרְבָּנוֹ בַּעֲבוּרוֹ. וָאֵרֶא שָׁם אֶת-הַנְּהָרוֹת וְאֶת-מַעְיְנוֹתֵיהֶם וְאֶת-סִבּוּבֵיהֶם.
סא וָאֵרֶא שָׁם אֶת-גַּן-עֵדֶן וְאֶת-פִּרְיוֹ, וְאֶת-הַמַּעְיָן הַמּוֹצִיא מִמֶּנּוּ נָהָר, וְאֶת-עֲצֵי הַגָּן וְאֶת-פִּרְחֵיהֶם, וְאֶת-עוֹשֵׂי הַצֶּדֶק. וָאֵרֶא בּוֹ אֶת-מְזוֹנָם וְאֶת-מְנוּחָתָם.
סב וָאֵרֶא מִחוּץ לוֹ עַם רַב, גְּבָרִים וְנָשִׁים וִילָדִים, וַחֲצִי מֵהֶם מִצַּד יָמִין שֶׁל הַתְּמוּנָה, וַחֲצִי מִצַּד שְׂמֹאל. וָאֹמַר: "נִצְחִי וְחָזָק! מָה הִיא הַתְּמוּנָה הַזֹּאת שֶׁל הַבְּרִיאָה?".
סג וַיֹּאמֶר אֵלַי: "זֶהוּ רְצוֹנִי בַּאֲשֶׁר קַיָּם בָּאוֹר, וְכָךְ הָיָה טוֹב בְּעֵינַי. וְאָז, אַחֲרֵי כֵן, צִוִּיתִי לָהֶם בִּדְבָרִי, וְהָיוּ. וְכָל-אֲשֶׁר קָבַעְתִּי לִהְיוֹת, כְּבָר נֶחֱרַט וְעָמַד לְפָנַי, טֶרֶם נִבְרָא כָּל-אֲשֶׁר רָאִיתָ".
סד וָאֹמַר: "אֲדוֹנִי, חָזָק וְנִצְחִי! מִי הוּא הָעָם הַזֶּה שֶׁבַּתְּמוּנָה, מִצַּד זֶה וּמִצַּד זֶה?". וַיֹּאמֶר אֵלַי: "אֵלֶּה שֶׁמִּצַּד שְׂמֹאל הֵם הָמוֹן הַגּוֹיִים שֶׁהָיוּ לְפָנֶיךָ וְשֶׁיִּהְיוּ אַחֲרֶיךָ. אֵלֶּה מוּכָנִים לַדִּין וְלַסֵּדֶר, וְאֵלֶּה לִנְקָמָה וּלְהַשְׁמָדָה בְּקֵץ הָעוֹלָם. וְאֵלֶּה שֶׁמִּצַּד יָמִין שֶׁל הַתְּמוּנָה – הֵם הָעָם הַנִּבְדָּל לִי מִן-הָעַמִּים, עִם עֲזָאזֵל. אֵלֶּה הֵם אֲשֶׁר הֲכִינוֹתִי לְהִוָּלֵד מִמְּךָ וּלְהִקָּרֵא עַמִּי".
הערות וביאורים
פסוק נ"ג-נ"ו
הַבֵּט בַּמִּשְׁטָחִים...: אברהם מצווה להביט מטה ולראות את ההיררכיה של הרקיעים. הוא מתאר את הרקיע השביעי (מקום הכבוד), השישי (מקום המלאכים חסרי הגוף), והחמישי (מקום הכוכבים). תיאור זה של קוסמולוגיה מדורגת אופייני לספרות חנוך ולספרות ההיכלות המאוחרת יותר.
פסוק נ"ז
"כְּמִסְפַּר הַכּוֹכָבִים... אָשִׂים לְזַרְעֲךָ": שימוש בהבטחה המקראית מבראשית ט"ו, אך בהקשר חדש. אברהם אינו רואה את הכוכבים בשמי הלילה, אלא רואה אותם מתחתיו, כשהוא עצמו נמצא מעליהם.
הַנִּבְדָּלִים לִי בְּגוֹרָלִי, עִם עֲזָאזֵל: משפט מפתח וקשה. האל אומר שזרעו של אברהם "נבדל" לו, אך הוא מזכיר את עזאזל באותה נשימה. הפירוש הסביר הוא שהעולם מחולק לשני "גורלות" או תחומים, כמו בטקס יום הכיפורים ("גּוֹרָל אֶחָד לַה', וְגוֹרָל אֶחָד לַעֲזָאזֵל", ויקרא ט"ז). זרע אברהם שייך לגורל ה', בעוד שאר העמים שייכים לגורל עזאזל.
פסוק נ"ח
"אֵיךְ... קָבַעְתָּ אוֹתוֹ עִמָּם?": אברהם שואל את שאלת קיומו של הרוע. אם עזאזל הוא אויב, מדוע נתת לו שליטה על חלק מהבריאה? זוהי שאלה תיאולוגית נועזת.
"הָבֵן אֶת-הַבְּרִיאָה שֶׁנּוֹסְדָה מִקֶּדֶם": התשובה אינה ישירה. האל מפנה את אברהם להתבונן ב"תמונה" או ב"תוכנית" הכוללת של הבריאה, כדי להבין את מקומו של הרוע בתוכה.
פסוק נ"ט-ס'
וָאַבִּיט... וָאֵרֶא: אברהם מקבל חזון פנורמי של כל המציאות הארצית: הארץ, הים, התהומות, השאול, וגורל בני האדם. הוא רואה הכל בו-זמנית, מנקודת מבט אלוהית.
פסוק ס"א
גַּן-עֵדֶן... וְעוֹשֵׂי הַצֶּדֶק: גן העדן מתואר לא רק כמקום מיתולוגי מהעבר, אלא כמקום הקיים בהווה, שבו שוכנים הצדיקים לאחר מותם, נהנים ממזון רוחני וממנוחה.
פסוק ס"ב-ס"ד
הַתְּמוּנָה: אברהם רואה את כל ההיסטוריה האנושית כ"תמונה" או "דיוקן" (בסלאבית: образованиіе), המחולקת לצד ימין (הצדיקים, זרע אברהם) וצד שמאל (הרשעים, הגויים).
"כְּבָר נֶחֱרַט וְעָמַד לְפָנַי, טֶרֶם נִבְרָא": זהו רעיון של פרה-דסטינציה (גזירה קדומה) מוחלטת. כל מה שהיה וכל מה שיהיה כבר תוכנן ו"נחרט" במחשבה האלוהית לפני הבריאה. ההיסטוריה אינה אלא התגלגלות של תוכנית קבועה מראש.
הָעָם הַנִּבְדָּל לִי... עִם עֲזָאזֵל: חזרה על הרעיון הדואליסטי. העולם מחולק בין ממלכת האל (שבה שולט זרע אברהם) לבין ממלכת עזאזל (שבה שולטים הגויים).
הגענו לנקודה שבה אברהם עומד לראות את האירוע המכונן של ההיסטוריה האנושית – חטא גן עדן.
