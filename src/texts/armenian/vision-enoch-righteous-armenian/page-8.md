וְהַמֶּלֶךְ הַשִּׁשִּׁי, אֲשֶׁר יָשַׁב אָבֵל עַל נָשִׁים וּבָנִים, עָלָיו נֶאֱמַר: "אֵשׁ יוֹצֵאת מִן הַוִּישַׁאפ וְשָׂרְפָה אֶת כָּל הָאָרֶץ".

כִּי אַחֲרֵי מוֹת הַמֶּלֶךְ אֲשֶׁר שְׁמוֹ הֵרַקְלִיּוּס, תִּתְעָרֵב הַמַּלְכוּת וְתֵחָלֵק לַעֲשָׂרָה בְּשָׁנָה אַחַת. וְאַחֲרֵי כֵן יָקוּם הַמּוֹרֵד (הָאַפּוֹסְטַט), וְיַעֲלֶה עַל עֲשֶׂרֶת הַמְּלָכִים, וְיִלָּחֵם בִּשְׁנַיִם מֵהֶם, וְיַשְׁמִיד מֶמְשֶׁלֶת הָעֲשָׂרָה. וְיִקַּח לוֹ הַמַּלְכוּת, וְיִמְשֹׁל בְּשָׁלוֹם עַל פְּלַשְׁתִּינָה וְעַל מֶמְשַׁל בְּנֵי יִשְׁמָעֵאל. וְשׁוּב, בְּמִלְחָמָה וּבְכֹחַ גָּדוֹל, יָבוֹא עַם הַדָּרוֹם עַל פְּלַשְׁתִּינָה וְיַכֶּהָ; וְתִתְלַהֵט חֲמָתוֹ כְּנָחָשׁ עַל כָּל הָאָרֶץ. אָז יִקְרָא לְעַצְמוֹ אֱלֹהִים, וִידַבֵּר בְּגָאוֹן כְּאֵל.

הָאֵשׁ שֶׁיָּצְאָה מִן הַוִּישַׁאפ וְשָׂרְפָה אֶת הָאָרֶץ — כִּי לֹא תִמָּצֵא בָּעֵת הַהִיא צְדָקָה בָּאָדָם, אֶלָּא הַכֹּל כְּקַשׁ לִשְׂרֵפָה. בִּגְלָלוֹ תִּתְלַהֵט הַנְּקָמָה בָּאֲנָשִׁים: רָעָב, חֶרֶב, שְׂרֵפָה, מַדְוִים וְיִסּוּרִים. וּבָהֶם יְעַנֶּה אֶת עֹבְדֵי הָאֵל, וּבְיִסּוּרִים יְעַנֶּה אֶת קְדוֹשֵׁי עֶלְיוֹן, אֵלֶּה שֶׁלֹּא עָלְתָה עַל לִבָּם לַעֲבָדוֹ כֵּאל.

וְתִהְיֶה רַחֲמִים מִן הָעֶלְיוֹן עַל יְרֵאָיו. וּכְכָל אֲשֶׁר יְעַנּוּ אוֹתָם יוֹתֵר, כֵּן יוֹסִיפוּ לְהָאִיר, כְּגַן עֵדֶן פּוֹרֵחַ, חַיִּים וּפְרָחִים: מִקְצָתָם לְבָנִים, מִקְצָתָם אֲדֻמִּים, מִקְצָתָם אַרְגְּמָנִיִּים.

יְמֵי מֶמְשֶׁלֶת הַמּוֹרֵד — מָאתַיִם שִׁשִּׁים וְחָמֵשׁ. וְאַחֲרָיו יָקוּם מֶלֶךְ יְרֵא שָׁמַיִם מֵהֹדּוּ, וְיִקָּבְצוּ אֵלָיו כָּל הַנּוֹתָרִים מֵאֹהֲבֵי אֱלֹהִים. וְיַעַשׂ מִלְחָמָה, וְיִנְצַח אֶת הַמּוֹרֵד וְאֶת חֵילוֹ. וַה' יִנְקֹם מִמֶּנּוּ; כִּי הָאֵשׁ יוֹצֵאת מִמֶּנּוּ תַּהֲרֹג אוֹתוֹ וְאֶת צְבָאוֹ, וְתֹאכַל אֶת כָּל הַחוֹטְאִים. וְלֹא תִגַּע הָאֵשׁ בְּעַבְדֵי עֶלְיוֹן, כִּי יֵרְדוּ בְתוֹכָהּ כְּמוֹ לְתוֹךְ גֶּשֶׁם עָז.

וְהָאֵשׁ תֹּאכַל בְּשַׂר בְּנֵי אָדָם, תִּשְׂרֹף הָרִים וּגְבָעוֹת, תְּיַבֵּשׁ מַעְיָנוֹת וּנְהָרוֹת. וּבַיּוֹם הַשְּׁלִישִׁי, בַּשָּׁעָה הַשְּׁלִישִׁית לַבֹּקֶר, יֵרָאֶה אוֹת עֵץ, כְּאוֹר זַךְ מִמִּזְרָח. וְיַעֲמֹד יוֹמַיִם. וּבַיּוֹם הַשְּׁלִישִׁי, עִם אוֹר הַבֹּקֶר, יָבוֹא ה' עִם צִבְאוֹת מַלְאָכָיו, לְהַעֲמִיד אֶת כָּל בְּנֵי הָאָדָם וְלִשְׁפֹּט אֶת יוֹשְׁבֵי הָאָרֶץ.

וְיִתְקְעוּ הַמַּלְאָכִים בַּשּׁוֹפָר שָׁלֹשׁ פְּעָמִים, וְיִקָּבְצוּ כָּל עַמֵּי אָדָם לַעֲמֹד לִפְנֵי ה'. בָּעֵת הַהִיא יֵאָסֵר הַמּוֹרֵד וְצְבָאוֹ, וְכָל הָאָרֶץ תְּבַקֵּשׁ מִמֶּנּוּ דְּמוּת. וְיִקְחוּ חֹמֶר בִּלְתִּי נִרְאֶה שֶׁל אֵשׁ וּמַיִם, וְיַשְׁלִיכֻהוּ לְטַרְטָרוֹס שֶׁל כֵּפֶסְטוֹס בְּצַד דָּרוֹם־מַעֲרָב שֶׁל הַיָּם. וּבָאֵשׁ וּבַמַּיִם יִשְׂרְפוּ אֶת כָּל הַחוֹטְאִים, וְלֹא יוֹתִירוּ.

בָּעֵת הַהִיא כָּל הַקְּדוֹשִׁים עִם הַמַּלְאָכִים יַעַמְדוּ בְּשִׂמְחָה לִפְנֵי אֱלֹהִים עֶלְיוֹן, מְהַלְּלִים אֶת אֱלֹהִים לְעוֹלְמֵי עוֹלָמִים.

אָמֵן.
