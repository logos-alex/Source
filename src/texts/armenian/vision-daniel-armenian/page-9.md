עמוד כט
וְיָנוּס לְבֵית הַקֹּדֶשׁ אֲשֶׁר גֵּרֵשׁ מִלְּפָנִים וְאִישׁ לֹא יְקַבְּלוֹ כִּי רַעֲעָמִים שָׁמַיִם יִהְיוּ בוֹ
וְיַגִּיעַ הַנַּעַר לְסַלָּמַנְדַּר וְלִפְנֵי יַגִּישׁ אֶת הַגֵּר לִפְנֵי הַוִישָׁאפ וְיִרְאֵה הַוִישָׁאפ וְיָנוּס מִמֶּנּוּ וְהַגֵּר לֹא יַהֲרֹג אֶת הַוִישָׁאפ יַבְדִּיל אוֹתוֹ מִפָּנָיו וְיִפֹּל הַנַּעַר בִּנְפִילָה גְדוֹלָה וְהָאִישׁ לֹא יִחְיֶה וְיַרְאוּ לוֹ אֶת הַגֵּר וְאֶת שֻׁתָּפוֹ וְיִהְיוּ מְשַׁרְתִּים בְּאֹהֶל אֲשֶׁר בָּרָא אוֹתוֹ וְיְבַקֵּשׁ דַּם אָבִיו עַל כִּסְאוֹתָיו וְהַגֵּר יִהְיֶה גָדוֹל בְּמְעַט זְמַן וְיִהְיֶה מְלֹא בָּךְ שְׁבִיעִית וּמָוֶת רַב לִבְנֵי אָדָם אֲשֶׁר נֶאֶסְפוּ אֵלֶיךָ מֵעָרִים וּמִמְּדִינוֹת
בָּעֵת הַהִיא מִמָּקוֹם לְמָקוֹם רְעָדִים יִהְיוּ וְיִשְׁמְעוּ אֶת קוֹל הָרַבִּים וְלֹא יַאֲמִינוּ וְעֹשֶׁר אֶתְכֶם וְכָבוֹדְךָ יַעֲצִיבוּ אוֹתָךְ לִהְיוֹת גֵּאֶה וְגַאֲוָה וְכְבוֹדְךָ הִכְנִיעַ אוֹתָךְ אַךְ מְלֹא וְהוֹדְךָ רַב יִהְיֶה בָּךְ
קֵץ הַזְּמַן יַגִּיעַ לָךְ וְיֵצֵא מִמֶּךָ מַטֵּה הַצַּיָּד אֲשֶׁר בַּחַיָּה וְיֵלְכוּ לְגֵרִים לִמְקוֹמוֹת וּבְנוֹתֵיךָ יְזַהֲרוּ אֶת עַצְמָן לְמַלְכוּת נְעוּרִים לְהַשְׁחִית רַבִּים וּגְדוֹלֶיךָ יִפֹּלוּ צָרָה וּמְהוּמָה רַבּוֹת יִהְיוּ

עמוד ל
בָּעֵת הַהִיא מַלְאָךְ הַמָּוֶת יִדְרֹשׁ אֶתְכֶם בְּכָל הֶעָרִים וְהַמְּדִינוֹת וּמָוֶת חָזָק יִשָּׁלַח מִן הַשָּׁמַיִם
פִּתְאֹם יִתְגַּבֵּר עַל הָאָרֶץ תִּרְעַשׁ הָאָרֶץ וְיִפְּלוּ הַבִּנְיָנִים וּבָתֵּיהֶם יִהְיוּ לְקִבְרוֹתָם וְיִתְבּוֹעַע הַיָּם אֶת גַּלָּיו מִלְמַעְלָה וִיכַסֶּה אֶת הָאָדָם וְיִהְיוּ אֲשֶׁר יִבְרְחוּ וְיִחְיוּ
בָּעֵת הַהִיא רַבּוֹת מַלְאָכִים יִהְיוּ וְיַעַמְדוּ לִפְנֵי וְיִתְחַנְּנוּ לִפְנֵי הַכִּסֵּא
בָּעֵת הַהִיא שְׁבִיעִית בָּבֶלָה תִּבְכֶּה עַל עַצְמָהּ וְתַשְׁלִיךְ שַׂק וְתִזְרֶה אֵפֶר עַל רֹאשָׁהּ לְרֹאוֹת אֶת הָאָדָם כִּי בְּעַד חַטֹּאת וְעַוֹנוֹת נָפָלוּ אֲנָשִׁים וּנְעָרִים וְנָשִׁים עִם זְקֵנִים יִבְכּוּ כִּי זַעַם יְהוָה עַל
וְחוֹמוֹתֵיךָ יִתְפָּרְדוּ וְיִפְּלוּ אַרְצָה אַרְמוֹנוֹתֵיךָ נַעֲרִים עִם זְקֵנִים נִגְרָרִים בְּפִיהֶם יִזְעֲקוּ לֵאלֹהִים וְכֹהֲנֵי אֶתְכֶם יִזְרְעוּ וְיִתְאַבְּלוּ אֶת מַטּוֹתֵיךָ גְּדוֹלֶיךָ יִבְכּוּ וְעָרֵי אֶתְכֶם יִתְאַבְּלוּ מְאֹד וְדַּרְכֵּי אֶתְכֶם בְּצָרָה יִהְיוּ וְאֶבְלֶיךָ
