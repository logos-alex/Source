**15.** אוֹי לָךְ בָּעֵת הַהִיא, בָּבֶל הַשְּׁבִיעִית, כַּאֲשֶׁר תִּמְלֹךְ אַלְמָנָה, וְהַתַּנִּין יָדוּף אֶת־הַנָּכְרִי, וְהַנָּכְרִי --- אֲשֶׁר נִקְרָא סַלַמַנְדְרָה --- בִּמְנוּסָה יָשׁוּב, מֵשִׁיב פָּנָיו אֶל־אִיִּים, זָהָב וָכֶסֶף וַאֲבָנִים יְקָרוֹת וַעֲטֶרֶת הַמַּטּוֹת וַאֲשֶׁר בְּחֵיק הַחַיָּה, מְשֻׁלָּשִׁים מִזֶּרַע פָּרָס, בְּכַלְקֵדוֹן; לֹא יִפֹּלוּ, וּבָנַיִךְ לֹא יִמְעָטוּ; בְּרַעַשׁ תִּבָּקַע הָאָרֶץ וְתַשְׁחִית אֶת־בְּנֵי־אָדָם עַד־עוֹלָלִים וְעַד־שְׁסָעִים. אַךְ עַד־כָּלֶה לֹא יַשְׁחִיתְךָ יְיָ, הַשְּׁבִיעִית, כִּי עוֹד לֹא־בָא עֵת אָבְדָנֵךְ; אֶלָּא בִּשְׁחִיתוּת לְכַסּוֹת אֶת־בְּנֵי־אָדָם, וְלֹא יְכַסֶּה, כִּי עוֹד לֹא־הִגִּיעַ קִצֵּךְ. מוּכָנוֹת לָךְ מַכּוֹת, כִּי כָּל־רֶשַׁע עָשִׂית בְּקִרְבֵּךְ בְּקַבְּלֵךְ אֶת־הָאָרֶץ אֶל־חֵיקֵךְ. מְבוּכוֹת מַחְשְׁבוֹת בְּנֵי־אָדָם רַבּוֹת תִּהְיֶינָה, עַד אֲשֶׁר יְיָ מִמָּרוֹם יְצַוֶּה לְהַשְׁחִית אֶת־הָאָדָם, אַךְ עֲצַת יְיָ תִּגָּלֶה מֵאַיִן תָּבוֹא הַפֻּרְעָנוּת, וְהַנִּסְתָּרִים יְקַבְּלוּ מַכּוֹת; הָעָם אֲשֶׁר בַּמִּדְבָּר לְצִדֵּךְ, וּמַטֵּה הַצַּיִד יַאֲבִידֵךְ, וְגוּר אֶת־גּוּר יַבְרִיחַ בְּבַקְּשׁוֹ דַּם אֲבוֹתָיו, וּבִרְצוֹן נַפְשׁוֹ יִמָּסֵר לַכְּלָבִים הַנָּכְרִים וַחֲבֵרָיו, וְיָקִים לְשׁוֹנוֹת וְעַמִּים, וְיִמְשֹׁל עַל־עַמִּים רַבִּים, וּשְׁנֵי הַכְּלָבִים יִלָּחֲמוּ אִישׁ בְּאָחִיו, וְאִישׁ אֶת־רֵעֵהוּ יַשְׁחִיתוּ.

**16.** אוֹי כַּאֲשֶׁר תִּמְלֹךְ אַלְמָנָה, וּבְמִרְמָה תִּשְׂחַק עַל־הַנָּכְרִי, וְהַנָּכְרִי הָאַחֵר יְבַקֵּשׁ אֶת־אָבְדָנוֹ, וּבְמִרְמָה יַכְחִידֵהוּ, וְהָיוּ הַיָּמִים הָהֵם קָשִׁים וְרָעִים. בְּאָנָחָה יִקַּח אֶת־רוּחוֹ, וְהוּא מָלַךְ יִתְפֹּשׂ אֶת־הַמַּטֶּה אֲשֶׁר בָּךְ, הַשְּׁבִיעִית, וְיִהְיֶה גָּדוֹל בְּכִסְאוֹת גְּדוֹלִים וּבְתִפְאָרוֹת, וְהָאִישׁ הַהוּא אַמִּיץ מִיָּמִין וּמִשְּׂמֹאל, וּבְיָמָיו רָעָב גָּדוֹל יִהְיֶה וְלֹא קָטָן, וְתִשָּׁחֵת הָאָרֶץ מֵאָדָם, וַאֲפִיקֵי מַיִם יִזְרֹמוּ, וְרוּחוֹת שְׁחֹרוֹת יִתְעָרְבוּ, וּכְרָמַיִךְ יִמְעָטוּ, וְתִפְאַרְתֵּךְ תִּצְטַמְצֵם, וְיִתְקַצְּרוּ הַיָּמִים, וְהָיָה הַיּוֹם בֶּן־שֵׁשׁ שָׁעוֹת. אוֹי לַאֲנָשִׁים אֲשֶׁר יִהְיוּ בָּעֵת הַהִיא; וְיָשִׁיב הַמֶּלֶךְ פָּנָיו אֶל־הַמַּעֲרָב. אָז אוֹי לָךְ, הַשְּׁבִיעִית, כַּאֲשֶׁר מַלְכֵּךְ בָּחוּר. בָּעֵת הַהִיא תִּהְיֶה לָךְ צָרָה גְדוֹלָה: אִישׁ אֲשֶׁר יִקַּח אֶת־אֵשֶׁת אָחִיו, וּבֵן אֶת־אִמּוֹ, וּבַת תֵּצֵא מִמִּשְׁכַּב אָבִיהָ, אָח אֶת־אֲחוֹתוֹ; וְיִרְבּוּ גִדּוּפִים, רְצִיחוֹת, שְׁבוּעוֹת, דִּבָּה רָעָה, שְׁקָרִים, נִאוּפִים, כְּאֵבִים, גְּזֵלוֹת, שִׂנְאַת אַחִים, מְרִיבוֹת, שְׁפִיכוּת דָּם וּמִקְדְּשֵׁי מְשָׁרְתֵי קֹדֶשׁ; וּמְלָכִים בַּמְּלָכִים, שָׂרִים בַּשָּׂרִים, חָזָק עַל־דָּל יָקוּם; וּמַפָּלָה תִּהְיֶה לֶעָשִׁיר וְלָדָל.

**17.** וּבִיתִינְיָא אֲשֶׁר עַל־שְׂפַת הַיָּם, בְּרַעַשׁ תִּשָּׁקֵעַ, וְיִתְנַשְּׂאוּ גַּלֵּי הַיָּם וִיכַסּוּ אֶת־יְסוֹדוֹת בִּיתִינְיָא עַד־רֹאשׁ הָעִיר נִיקוֹמִידְיָא.

**18.** וְעוֹד מֶלֶךְ אַחֵר, וִימֵי מַלְכוּתוֹ מְעַטִּים, רָעִים וְאָיֹם מְאֹד, וּמֵאָז וָהָלְאָה לֹא־עוֹד עֵת טוֹבָה כִּי אִם־רָעָה; וְעִמּוֹ יִלָּחֵם בְּנוֹ וְיַכֵּהוּ בֶּחָרֶב. וּמֶלֶךְ אַחֵר יָקוּם מִפֻּלְחָן אַחֵר, אֲשֶׁר הוּא אַרְיָנוֹס, וְיִמְשֹׁךְ אֵלָיו אֶת־כֻּלָּם. בָּעֵת הַהִיא אוֹי לָךְ, הַשְּׁבִיעִית, כִּי מִכֻּלָּם תִּתְאַבְּלִי עַל־נַפְשֵׁךְ וְעַל־גְּבוּלָיִךְ. וְאָז יָקוּמוּ מְלָכִים וְשָׂרִים וּנְגִידִים וּנְדִיבִים עַל־עָרִים וּמְדִינוֹת וּמְקוֹמוֹת, וְיִהְיוּ מְבוּכוֹת וּמְהוּמוֹת רַבּוֹת בִּבְנֵי־אָדָם. וְעַם בַּרְבָּרִים יָקוּם עַל־עָרִים וּמְדִינוֹת, וְתִכָּנַע הָאָרֶץ מִפְּנֵי הֲמוֹן עַמִּים מִשְׁעַת קוֹמָה שִׁבְעִים וְשָׁלוֹשׁ; אַל־תִּלָּחֲצִי אָז, הַשְּׁבִיעִית, מֵהֶם, אֶלָּא מִלְחָמָה מְיֻחֶדֶת תְּיַסְּרֵךְ, וְתִפְאֶרֶת הָאָרֶץ תִּמְעַט בָּךְ; חֲרוֹן אַף מִן־הַשָּׁמַיִם יִהְיֶה בָּךְ, מַטּוֹת הַשִּׁלְטוֹן בִּמְבוּכָה יִהְיוּ, נִלְחָמִים אִישׁ בְּאָחִיו; וְאָז צָרָה רַבָּה וְרָעָה תִּהְיֶה לִבְנֵי־אָדָם כַּאֲשֶׁר לֹא־הָיְתָה מֵעוֹלָם, וְאַל־יִבְטַח שַׂר מַאֲמִין בְּסַלַמַנְדְרָה, כִּי נָכְרִי הוּא, וּמִזַּרְעוֹ יְקַבְּלוּ אֶת־אֲשֶׁר לֹא יְבַקֵּשׁוּ; וְיַחְפֹּץ סַלַמַנְדְרָה לִבְרֹחַ וְלֹא יוּכָל, וְיַשִּׂיגוּהוּ וְיַהַרְגֵהוּ פְּלַקִיטַס.

**19.** וְיִמְשֹׁל מַטֶּה אַחֵר בָּךְ, הַשְּׁבִיעִית, וְיִהְיוּ יָמִים רַבִּים; יָקוּם אִישׁ דָּל אֲשֶׁר לֹא יַכִּיר טוֹבָה, גֵּאֶה, יָהִיר, אֹהֵב כֶּסֶף, אֹהֵב מִלְחָמָה, וּשְׁמוֹ אוֹרוֹזִיוֹס, וְכִסְאוֹ מִסְפַּר יָמִים מְעַט, וְרַגְזָן, וּגְדוֹלָיו יִשְׂנְאוּהוּ, וּבְנֵי עִירוֹ נִלְחָצִים מִמֶּנּוּ; וּבְיָמָיו תִּהְיֶה חֵמָה יוֹרֶדֶת, פְּעָמִים רַבּוֹת יָצִיק לְבָבֶל; בִּימֵי מַלְכוּתוֹ קוֹלוֹת שֵׁבֶר יַגִּיעוּ אֵלַיִךְ, וְחֵילוֹת בַּרְבָּרִים יַרְעִישׁוּךְ וְלֹא יִלָּחֲמוּ. וְיָקוּם מֶלֶךְ אַחֵר עָרִיץ, וּבְהִתְגָּלוּת נִלְחָם יְגָרְשֵׁהוּ, וּבְעֶצֶב רַב יָנוּס אֶל־הַהֵיכָל הַקָּדוֹשׁ אֲשֶׁר גָּזַל לְפָנִים, וְאֵין מְקַבְּלוֹ, כִּי מַכּוֹת שָׁמַיִם יִהְיוּ בוֹ.

**20.** וְהַבָּחוּר יַגִּיעַ אֶל־סַלַמַנְדְרָה, וְיַעֲמִיד אֶת־הַנָּכְרִי לִפְנֵי הַתַּנִּין; וְכִרְאוֹת הַתַּנִּין יָנוּס מִמֶּנּוּ, וְהַנָּכְרִי לֹא יַהֲרֹג אֶת־הַתַּנִּין, אֶלָּא יַבְדִּילֵהוּ מִפָּנָיו, וְהַבָּחוּר יִפֹּל מַפָּלָה גְדוֹלָה, וְאַלְמָנָה תִּחְיֶה, וְיַרְאוּ לָהּ אֶת־הַנָּכְרִי וְאֶת־שׁוּתָפוֹ לַכִּסֵּא, וְיִהְיוּ מַעֲלֵי מַס בְּאָהֳלֵי עֹשֵׂהוּ, וּבְכִסְאוֹתָיו יְבַקֵּשׁ דַּם אָבִיו, וְהַנָּכְרִי יִהְיֶה גָּדוֹל לִזְמַן קָצָר, וְיִהְיֶה שֶׁפַע בָּךְ, הַשְּׁבִיעִית, וּמִיתוֹת רַבּוֹת לִבְנֵי־אָדָם אֲשֶׁר נִקְבְּצוּ אֵלַיִךְ מֵעָרִים וּמִמְּדִינוֹת.

**21.** אָז בִּמְקוֹמוֹת מְקוֹמוֹת רְעָשׁוֹת יִהְיוּ, וְיִשְׁמְעוּ קוֹל רַבִּים וְלֹא יַאֲמִינוּ; וְעָשְׁרֵךְ וּתְהִלָּתֵךְ יַעֲצִיבוּךָ בִּהְיוֹתֵךְ גֵּאָה וִיהִירָה, וּכְבוֹדֵךְ הִשְׁפִּילֵךְ; אַךְ שֶׁפַע וָעֹשֶׁר רַב יִהְיֶה בָּךְ. קֵץ הָעִתִּים יַגִּיעַ אֵלַיִךְ, וְיֵצֵא מִמֵּךְ מַטֵּה הַצַּיִד אֲשֶׁר מִן־הַחַיָּה, וְיֵלְכוּ אֶל־הַנָּכְרִים בַּמְּקוֹמוֹת, וּבְנוֹתַיִךְ תִּתְקַשֵּׁטְנָה לִמְחוֹלוֹת; מַלְכוּת בַּחוּרִים לְהַשְׁחִית רַבִּים, וּגְדוֹלַיִךְ יִפֹּלוּ, מְצוּקוֹת וּמְבוּכוֹת רַבּוֹת תִּהְיֶינָה.
