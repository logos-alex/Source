עמוד יז

לְכָל הָעָרִים וְהַמְּדִינוֹת

בְּרָעָב הֲתִשָּׁחֲתִי עִירֵךְ הַמְּפֹאָרָה בְּזָהָב וָכֶסֶף וְעַם מְחֻלָּל וּמְזֹהָר

מְרִיבוֹת רַבּוֹת יִהְיוּ בָךְ וּבָנֶיךָ יִתְהַלְּכוּ בְּזָהָב וְאַחַר יִשָּׁחֲתוּ בָּרָעָב

שְׁמֵי בִּיזְיוֹן וּבָבֶלָה יִסָּגְרוּ יִלָּכְדוּ בַּמְּדִינָה וְיְסוֹדוֹתֶיהָ יִשָּׁחֲתוּ וְחוֹמוֹתֶיהָ תִּפֹּלְנָה

בְּרוֹמָא יִהִי שַׂר אֶחָד בָּעֵת הַהִיא אֲבָל יוֹמוֹ יְחֻדַּד וְחִצּוֹ יֶחֱרַד וּתְרוּמָתוֹ יַעֲמִיק

פְּעָמִים רַבּוֹת יָקוּם שַׂר וְיָשׁוּב יִפֹּל וְתִהְיֶה מְלוּכָה מְשֻׁלֶּשֶׁת

וְרַבִּים יֵלְכוּ אֵלֶיךָ בְּעַד רֹב שָׂרִים מְהֻדָּרִים וּבְעַד צַוַּאר גֵּאֶה וְיֶתֶר עֹשֶׁר

בְּנֵי בִּיזְיוֹן יִהְיוּ בִּלְבוּשֵׁי הוֹד וְתַפְאֶרֶת כִּי יֵצֵא כָמוֹהוּ בַּשֵּׁם הַנִּקְרָא בַּשְּׁבִיעִית אִישׁ בִּיזַנְטִי וְיָקִים אֶת יְסוֹדוֹתֶיהָ

וְיִהְיֶה שְׁמוֹ עַל כָּל יוֹשְׁבֵי תֵבֵל עַד קְצֵה הַלָּשׁוֹן

וְיָשׁוּב יְבָנֶה אוֹתָהּ אִישׁ מְסֻמָּן נִכְסָף בְּאִשָּׁה צַדֶּקֶת וּבְעֵתוֹ יִשְׁלַם חֵשֶׁק לִבּוֹ וְיִמְצָצֵא אֶת עֵץ הַחַיִּים וְיָשֵׂם מַטֵּהוּ גָדוֹל וְיָבוֹא עַד קְצֵה

עמוד יח

א

אֲשֶׁר בַּסִּימָנִים וְיָשֵׂם אוֹתוֹ לְרֶסֶן בְּעַד מִלְחֲמוֹת רַכּוֹת לִנְצֹחַ וְקַרְנוֹ גְּבוֹהָה וְחָזָקָה וּשְׁמוֹ עַל כָּל לָשׁוֹן וְיִנָּתֵן לָעִיר זֵכֶר לְעוֹלָם

וְאַחֲרָיו יָקוּם מַטֶּה שְׁלִישִׁי אֲשֶׁר יִשְׁפֹּךְ דַּם צַדִּיקִים יִקְרָא אֶת הַחֻקִּים וְלֹא יֵדַע אֶת הָאֱלֹהִים וְיִבְלַע אֶת הַסְּפָרִים הַקְּדוֹשִׁים בְּלֵב נִכְסָף

וְאַחֲרָיו יָבוֹא אִישׁ חָכָם אֵלֶיךָ בַּשְּׁבִיעִית בָּבֶלָה וְיָקִים צֶלֶם בָּךְ בְּגִלָּלָהּ לֹא תּוּכַל לִשְׁפֹּט אוֹתוֹ

וְאַחֲרֵי כָּל זֹאת יָקוּמוּ קַרְנוֹת גְּבוֹהוֹת וּמַטּוֹת גְּדוֹלוֹת וְיָעָמְדוּ עָלַיִךְ וְזִכְרָם יִהְיֶה גָּדוֹל מְאֹד

וְְ

וְאַחֵר גַּם מַטֶּה וְגָבוֹהַּ יִתְאַמֵּץ לְפֶלֶא וְגָבוֹהַּ בְּעַד הַשֵּׁם וְהַגָּלוּי לַיהוָה כִּי חֶרֶב אֱלֹהִים בּוֹ

בְּיָדוֹ יִהְיוּ חַיִּים גְּדוֹלִים וּבְשִׂמְחָתוֹ יִשְׂמְחוּ כָּל בְּנֵי אָדָם וּבֶעָרִים וּבִמְדִינָה יִתְקַבְּצוּ אֵלֶיךָ וְיַגִּיעוּ אֵלֶיךָ בַּשְּׁבִיעִית וְאִישׁ אִישׁ יַעֲשֶׂה אֶת מְלַאכְתּוֹ

וְיַעֲמֹד מַטֵּה הַמַּלְכוּת עַד יִכְלֶה

וְיֵצְאוּ שְׁנֵי מַטּוֹת מִמַּטֶּה אֶחָד וְיִהְיֶה מַטֶּה אֶחָד מִכֶּם חַיָּה וְהַמַּטֶּה הַשֵּׁנִי מִכֶּם חָכָם בִּרְחוֹבוֹת בַּשְּׁבִיעִית
