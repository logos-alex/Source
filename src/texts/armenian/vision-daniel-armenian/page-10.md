עמוד לא
וְרַעַשׁ גָּדוֹל וְיִהְיֶה וְיֵרָאֶה עַמּוּד אֵשׁ מִן הַשָּׁמַיִם עַל הָאָרֶץ
וּבָעֵת הַהִיא תִּשָּׁנֶה מַלְכוּתֵךְ וְתֵרֵד הַנִּשְׁאֶרֶת לְרִשְׁעָה לְחֵלֶא וּלְרָעָב
בָּעֵת הַהִיא כָּכָה תִּתְפָּרֵד וְיִקַּח אֶת הַמַּטֶּה הַשֵּׁנִי לְעִיר אַחֶרֶת מִשָּׁר לֹא יִמְצָא וְיָשִׂים אֵשׁ אֵשׁ מִן הַשָּׁמַיִם אוֹתוֹת וְכַלָּה רַבּוֹת יִהְיוּ
וּבָעֵת הַהִיא יִמְלֹךְ הַנֵּרֶא וְיִתְעוּ אֲנָשִׁים מִן הָאֱמוּנָה לְלֹא אֱמוּנָה בְּבִיאַת אֲשֶׁר לֹא יְבַקֵּשׁוּ וְלֹא הָיָה לָהֶם עַיִן הַהוּא אֲשֶׁר לֹא הֶאֱמִינוּ בּוֹ אֲשֶׁר הָיָה מְנַגֵּד לַכֹּל
הוּא יִתְיַחֵד וְיָנִיעַ אֵת אֲשֶׁר לֹא יְבַקֵּשׁוּ מִבְּתוּלָה טְטָמֵא וְיַחֲזִיק אֶת מַטֵּה הַתַּרְמִית עַל עַם הָאָדָם בְּזְמַנִּים שָׁלוֹשׁ וְחֵצִי זְמַן יַנִּיחַ אֶת נַפְשׁוֹת הָרַבִּים לַאֲבַדְּתָּן לִהְיוֹת שֻׁתָּפִים לַגֵּיהִנָּם לָעַד
בָּעֵת הַהִיא מְהוּמַת מַלְאָכִים תִּהְיֶה לְרֹאוֹת אֶת הָאוֹת אֲשֶׁר מִלְּפָנִים הוּא הֶרְאָה
וְזֹאת יֵדְעוּ אַנְשֵׁי אֲשֶׁר אֱלֹהִים אוֹהֲבִים יַכִּירוּ וְיִרְאוּ כָּל אֶת הַתַּרְמִית אֲשֶׁר לְעַם הָאָדָם אֲשֶׁר לוֹ הָאוֹתוֹת הָאֵלֶּה
הוֹי הַנּוֹלָד מִבֵּין הַהוֹרִרים חֲשׁוּךְ שֵׂעָר רֹאשׁ מְכֻסֶּה שֵׂעָר גַּבְהוּת מְשׁוּחַ שֵׂטָן

עמוד לב
וְיִרְדֹּף כֶּלֶב שֵׁנִי וְיָסֹב אֶת לְשׁוֹנוֹ לְצוּרוֹת אֲחֵרוֹת עַד חֲצַר הָאֲרִי וְיִפָּרֵד מִמֶּנּוּ כְּאִלּוּ לֹא יוּכַל לוֹ
וְיַמְלִיךְ אֶת הַכֶּלֶב בִּזְמַנִּים וּזְמַן וְחֵצִי זְמַן וּבְעֵתּוֹ יִמְלֹךְ וִיהִי שְׁמוֹ כֶּלֶב אֲשֶׁר מְתֻרְגָּם נִקְרָא מַטֵּה עַמִּים
וְהַחַיָּה תִּמְחֶה אֶת זֵכֶר הַכֶּלֶב וּמִקְּדֹם יָשִׁירוּ עָלָיו דִּבְרֵי רָע וּבְעִירוֹ יִמְחוּ אֶת צַלְמוֹ וְרַבִּים בְּמִקְדַּשׁ הַחַיָּה יַחְשְׁבוּ לַהֲרֹג אוֹתוֹ וְלֹא יוּכְלוּ לוֹ וְשֻׁתָּפָיו בְּעִיר אַחַת בַּחֶרֶב יָמוּתוּ וְיֵדְעוּ מְאֹד אֶת הַמְבֻקָּשׁ לְכֹהֲנִים עַל שֵׁם הַהוּא
וְאִישׁ אֶחָד מִן הַגְּדוֹלִים נַעַר מִבֵּין חֵיל הַחַיָּה הָרַבִּים יַאֲסֹר וְיָבִיא וּבְיָדָיו יִשְׁלַח לְהַחַיָּה וְאַחֲרֵי כֵּן כַּאֲשֶׁר יְקַבֵּל הַחַיָּה אֶת הָרַבִּים מֵעַם עַמִּים יִקְרָא לְעַצְמוֹ כְּאָדוֹן וְיַעֲמֹד אִישׁ אַחֵר לְנֹכַח וְיְקַבֵּל בְּגוֹגוֹ וְיִרְדֹּף אֶת הַלּוֹחֲמִים וְהוּא יִרְדֹּף מֵאַחֲרֵיהֶם וְיָנוּס מֵהֶם וְאִישׁ לֹא יַשִּׂיגֵהוּ כִּי
