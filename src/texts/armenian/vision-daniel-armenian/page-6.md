עמוד כג

יִרֹדֵף אוֹתוֹ מִבֵּין בְּנֵי אָדָם בְּעַד אַהֲבַת הַוִישָׁאפ וְהוֹד גָּדוֹל אֲשֶׁר הָיָה בָּךְ וְיִסַּח אֶת פָּנָיו מֵאֶרֶץ לָאָרֶץ וְיִצֶּר לְאֻמִּים

וְאֻמִּים נִדְחִים וְיַאֲמִינוּ בְּבוֹרְאָם כֻּלָּם

וּמַטֵּה הַצַּיָד אֲשֶׁר הָיָה בַּחַיָּה יִתְיַחֵד עִם הַכֶּלֶב וְיַחֲזִיק הַוִישָׁאפ אֶת מְקוֹם הַתַּרְמִית וְיָנוּחַ עִם כִּסְאוֹת רֵיקִים וְיַרְדֹּף אֶת מְקוֹם הַקְּדֻשָּׁה וְיְנַעְנֵעַ אֶת הַוִישָׁאפ בִּלְשׁוֹנוֹ עַל אַנְשֵׁי קֹדֶשׁ וְעַל שְׁרִידֵי קֹדֶשׁ וְיְכַסֶּה אֶת תִּפְאֶרֶת הָאָבוֹת וְיִשְׁבֶּה אֶת כְּנֵסִיּוֹת הַקֹּדֶשׁ

וַאֲשֶׁר בְּיָד וּבָאָב וּבְסָבָא יִשְׁכְּנוּ בָּאָרֶץ יָבוֹאוּ אֵלֶיךָ שְׁבִיעִית נִרְדָּפִים מִן הַוִישָׁאפ וְאַחֲרֵי כָּל זֹאת יְנַעֲרוּ אֶת אֲבַק רַגְלֵיהֶם עָלַיִךְ מְעִידִים בָּךְ

בָּעֵת הַהִיא יִבְעַעַר אֵשׁ יֹפִי הוֹדְךָ וְאִישׁ אֶחָד יָנוּס בְּמַטֵּה הַצַּיָּד וְיָבִיא אֶת הָרִאשׁוֹן מַטֶּה אֲשֶׁר נִקְרָא סַלָּמַנְדַּר

וְאַחַר כֵּן בָּעֵת הַהִיא יִצֶּר הַוִישָׁאפ וְיִלָּכֵד בְּיָדֵי יוֹרְשָׁיו וְיִקָּשֵׁר בְּיַד הָרֶשָׁעִים וְאִישׁ לֹא יְקַבְּלוֹ כִּי שָׁפַךְ דַּם קְדוֹשִׁים בֶּעָרִים וּבִמְדִינוֹת

עמוד כד

כִּי רָגְלוֹ רַגְלַיִם וְיָנוּס אֶל הַחַיָּה וְאֶל אֲשֶׁר בְּגוֹגָהּ וְיָבוֹא אֵלָיו וְיֵצֵא הַחַיָּה קָטָן וְיִתְגַּבֵּר וְיָשִׁיב הַחַיָּה גְדוֹלָה אֶת הַקָּטָן עַל כִּסֵּאוֹ וְיַעַשׂ אֶת שֻׁתָּף וְכִסֵּא שֻׁתָּף תַּחַת עַצְמוֹ

שְׁתֵּי חַיּוֹת בְּאִישׁ אֶחָד שֹׁכְנוֹת וְיְגַבֵֵּר הַקָּטָן בְּמִלְחֶמֶת וְיָשׁוּב הַגָּדוֹל לְאַרְצוֹ אֲשֶׁר מִשָּׁם וְיַחֲזִיק הַקָּטָן אֶת מְקוֹמוֹ עַד עֵת אַחַת לֹא מִמִּשְׁפָּט מָלַךְ כִּי מֵרֹשׁ יָצָא יָבוֹא לְמִקְדַּשׁ הַמַּלְכוּת

וְיַעֲשֶׂה הַחַיָּה הַקַּטַנָּה לְהַחַיָּה הַגְּדוֹלָה וְיַנַּח אֶת מְקוֹם יָדוֹ הַיְצִירָתִי וְיִשְׁלֹט בָּךְ שְׁבִיעִית וְיַחֲזִיק בְּצָרָה

שִׂמְחַת הַקָּטָן גְּדוֹלָה מְאֹד יְשַׂחֵק בְּשַׂחַק וְיִתְאַהֵב בִּגְדוֹלִים וְיִשְׂנָא אֶת הָרַרְבֵּה וְיִתֵּן חַיִּים וּמָלוֹן וְיְשַׂחֵק הַוִישָׁאפ עִמּוֹ וּבְמַטֵּה הַצַּיָּד יִגְרֹר אֶת הָאַרְיֵה

אוֹי לָךְ בָּעֵת הַהִיא שְׁבִיעִית בָּבֶלָה כִּי אִישׁ יִמְלֹךְ וְהַוִישָׁאפ יִרְדֹּף אֶת הַגֵּר וְהַגֵּר אֲשֶׁר נִקְרְאֶה סַלָּמַנְדַּר יָנוּס וְיָשִׂים פָּנָיו לְאִיִּם זָהָב וְכֶסֶף וְאַבְנֵי חֵפֶץ וְכֶתֶר הַמַּטּוֹת וַאֲשֶׁר בְּגוֹג הַחַיָּה תְּלָתִי מֵעַם פָּרְסִי בְּכַלְקֵדוֹן לֹא
