עמוד טו

בְּגַלַטְיָה יֵרָאֶה אֵשׁ מִן הַשָּׁמַיִם וּרְעָמִים וּבְרָקִים יְבַעֲתוּהָ וְכִסְאוֹת שָׂרֶיהָ יִפְּלוּ אַרְצָה וְצַד דָּרוֹם שֶׁלָּהּ יִבְעַר בְּדָם וּבְאֵשׁ וְרַבִּים יִבְרְחוּ בָּעֵת הַהִיא אֶל רוֹמִי

בְּקַפַּדּוֹקְיָה הַקְּטַנָּה בְּנֵיהֶם יִשְׁחֲטוּ זֶה אֶת זֶה וְיַשְׁבּוּ זֶה אֶת זֶה וְשָׂרֶיהָ יִמָּצְאוּ לִתְבוּסָה וְהַסְּבִיבוֹתֶיךָ יִהְיוּ לְצַעַר וְלַאֲנָחָה בִּקְטַנָּה בָּבֶלָה

בְּקַרְפַּתְיָה בְּנֵיהֶם יִהְיוּ לְצָרָה יִרְאוּ שְׂרֵפָה וְלֹא יַאֲמִינוּ פְּצָלִים יִהְיוּ הַכֹּל עַד שְׁאוֹל תַּחְתִּית רַבִּים יִבְרְחוּ בַּשְּׁבִיעִית

בִּסְמִירְנָה יִרְבֶּה זַעַם יִמָּלֵא כַכּוֹס דָּם וְנֶפֶשׁ מֵרָמִים תִּפּוֹל

שָׂרֶיךָ יַעֲלוּ וּגְדוֹלֶיךָ יִפֹּלוּ כִּי יוֹם זַעַם יְהוָה בָּךְ

בְּאַנְטְיוֹכְיָה בָּנֶיהָ יִשָּׁחֲתוּ וּבְנוּיוֹתֶיהָ יִחָרְבוּ וְשָׂרֶיהָ לֹא יְחָמְלוּ

נְפִילַת רַעַשׁ יִהְיֶה בָךְ וְרֹב גְּדֻלָּתֵךְ יְאַבֵּדֵךְ

עמוד טז

בַּאֲלֶכְּסַנְדְּרִיָּה יִרְבּוּ מִלְחֲמוֹת מְהוּמָה וְלֵבָבָם יַגִּיעַ עַד קְצֵה חוֹמֹתֶיהָ

שָׂרֶיהָ יֵרָדְפוּ

בְּנֵי מִצְרַיִם יִבְרְחוּ נִרְדָּפִים בְּרָעָב

נְחָלָתֵךְ תִּהְיֶה לְאַיִן וְהַיְאוֹר יִיבַשׁ וְשָׂרֶיךָ יֹאבֵדוּ

בְּנוֹת נִקְיָה יִהְיוּ לְאֵבֶל וּלְמַרְת נֶפֶשׁ בְּעַד בְּנוֹתֵיהֶם וְגֵרוּשָׁן מִפְּתחֵי בְנֵי אָדָם וְשָׂרֶיךָ יַעַבְדוּ אֶת אֲשֶׁר לֹא יֵדָעוּן

אוֹי לָךְ נִיקוֹמֵדְיָה אֲשֶׁר הֲרִימוֹת קַרן וְאָכַלְתְּ אֶת בְּשַׂר קְדוֹשֶׁיךָ אֲשֶׁר הָיוּ בָךְ הֲתִשָּׁחֲתִי בְּדַם אֲנָשִׁים צַדִּיקִים אֲשֶׁר יְפָאֲרוּ אֶת יִחוּדְךָ וְתֵרְדִי שְׁאוֹל תַּחְתִּית

וּלְכִי וּבְכִי אֲבֵלָה כִּי תִמָּחִי עִם בָּנַיִךְ

שָׂרֵי אֶתְכֶם שָׂרֵי הָרֶשַׁע וְכֹהֲנֵי אֶתְכֶם אוֹהֲבֵי זָהָב וְכֶסֶף וְיֹפִי הוֹדְךָ יִכָּחֵד

כַּלְקֵדוֹן וְעַם פָּרַס מַה יִּקְרָאֶךָ בְּיוֹם קֵּאַחֲרִית הַיָּמִים אַתָּה לֹא תֵדָע וּבְקֵץ הָעוֹלָמִים לָעַד אֵי זֶה יִהְִהְיֶה הָעֵת
