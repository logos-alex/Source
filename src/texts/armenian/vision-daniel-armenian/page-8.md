עמוד כז
וּבִיתִינְיָה אֲשֶׁר עַל יַם תִּרְעַשׁ וְתִּפֹּל וְיַעֲלוּ גַּלֵּי הַיָּם וִיכַסּוּ אֶת יְסוֹדוֹת בִּיתִינְיָה עַד רֹאשׁ הָעִיר נִיקוֹמֵדְיָה
וְאַחַר כֵּן יָקוּם מֶלֶךְ אַחֵר וְעֵתּוֹ מְעַט יָמִים רָע וְכֻלּוֹ נוֹרָא וְאַחֲרָיו לֹא עוֹד יִהְיֶה עֵת טוֹב כִּי רַע וְעִם זֹאת יִלָּחֵם הַבֵּן וְיִמְכֹּר אוֹתוֹ בַּחֶרֶב
וְיָקוּם מֶלֶךְ אַחֵר מִמִּדָּה אַחֵר אֲשֶׁר הוּא אֲרִיאָנוֹס יִמְשֹׁךְ אֶל עַצְמוֹ אֶת הַכֹּל
בָּעֵת הַהִיא אוֹי לָךְ שְׁבִיעִית כִּי מִכָּל תִּתְאַבְּלִי עַל עַצְמֵךְ וְעַל גְּבוּלוֹתָיִךְ
וְאַחֲרֵי כֵּן יָקוּמוּ מְלָכִים וְשָׂרִים וּמַנְהִיגִים וּמַרטִירִים עַל הֶעָרִים וְעַל הַמְּדִינוֹת וְעַל הַמְּקוֹמוֹת וְיִהְיוּ מְהוּמוֹת וּמְבוּכוֹת רַבּוֹת לִבְנֵי אָדָם
וְעַם בַּרְבָּרִי יָקוּם עַל הֶעָרִים וְהַמְּדִינוֹת וְתִּכְרַע הָאָרֶץ מֵרֹב עַמִּים עַם שִׁבְעִים וְשִׁבְעָה
וְלֹא תִצְּרִי בָּעֵת הַהִיא שְׁבִיעִית מֵהֶם כִּי מִלְחָמָה לְבַדָּה תְּיַסֵּר אוֹתָךְ וְהוֹד הָאָאָרֶץ יִמְעַט בָּךְ וְזַעַם מִן הַשָּׁמַיִם יִהְיֶה בָּךְ

עמוד כח
וְחַיּוֹת הַמַּטּוֹת בִּמְבוּכָה יִהְיוּ זֶה עִם זֶה לְהִלָּחֵם וּבָעֵת הַהִיא רַב צָּה צָרָה וָרָע יִהְיֶה לִבְנֵי אָדָם כַּאֲשֶׁר לֹא הָיָה מֵעוֹלָם וְלֹא יַאֲמִין הַשַּׂר הַמֵּאֲמִין לְסַלָּמַנְדַּר כִּי גֵּר הוּא וּמֵעַמּוֹ יְקַבְּלוּ אֵת אֲשֶׁר לֹא יְבַקֵּשׁוּ וְיַחְפֹּץ לִבְרֹחַ סַלָּמַנְדַּר וְלֹא יוּכַל וְיַגִּיעוּ לוֹ וְיַהֲרְגוּ אוֹתוֹ פְּלַקִיטָס
וְיִמְלֹךְ מַטֶּה אַחֵר בָּךְ שְׁבִיעִית וִיהְיוּ רְעָדִים רַבּוֹת יָקוּם אִישׁ עָנִי אֲשֶׁר לֹא יַכִּיר אֶת הַחֶסֶד עַז פָּנִים גֵּאֶה אוֹהֵב זָהָב אוֹהֵב מִלְחָמָה וִיהִי שְׁמוֹ אוֹרוֹזְיוֹס וְכִסְאוֹ מְעַט יָמִים וְזַעְפָּן וּגְדוֹלָיו יִשְׂנְאוּ אוֹתוֹ וְעָרֵי אֶתְכֶם נִלְאוּ מִמֶּנּוּ וְיֵרֵד וְיִהְיֶה זַעַם בְּיָמָיו רַבּוֹת יְצַעֵר אֶת בָּבֶלָה בְּיָמָיו צְעָקַת גְּוִיעָה יַגִּיעוּ אֵלֶיךָ וְחֵילוֹת בַּרְבָּרִים יְבַהֲלוּ אוֹתָךְ וְלֹא יִלָּחֲמוּ
וְיָקוּם מֶלֶךְ אַחֵר חָזָק לְנֹכַח מִלְחָמָה יִרְדֹּף אוֹתוֹ וְרַבָּה מָרָה
