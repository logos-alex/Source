חזון דניאל השביעי

**טו.** אוֹי לָךְ בָּעֵת הַהִיא, בָּבֶל הַשְּׁבִיעִית, כַּאֲשֶׁר תִּמְלֹךְ אַלְמָנָה, וְהַתַּנִּין יָדוּף אֶת־הַנָּכְרִי, וְהַנָּכְרִי – אֲשֶׁר נִקְרָא סַלַמַנְדְרָה – בִּמְנוּסָה יָשׁוּב, מֵשִׁיב פָּנָיו אֶל־אִיִּים, זָהָב וָכֶסֶף וַאֲבָנִים יְקָרוֹת וַעֲטֶרֶת הַמַּטּוֹת וַאֲשֶׁר בְּחֵיק הַחַיָּה, מְשֻׁלָּשִׁים מִזֶּרַע פָּרָס, בְּכַלְקֵדוֹן; לֹא יִפֹּלוּ, וּבָנַיִךְ לֹא יִמְעָטוּ; בְּרַעַשׁ תִּבָּקַע הָאָרֶץ וְתַשְׁחִית אֶת־בְּנֵי־אָדָם עַד־עוֹלָלִים וְעַד־שְׁסָעִים. אַךְ עַד־כָּלֶה לֹא יַשְׁחִיתְךָ יְיָ, הַשְּׁבִיעִית, כִּי עוֹד לֹא־בָא עֵת אָבְדָנֵךְ; אֶלָּא בִּשְׁחִיתוֹת לְכַסּוֹת אֶת־בְּנֵי־אָדָם, וְלֹא יְכַסֶּה, כִּי עוֹד לֹא־הִגִּיעַ קִצֵּךְ. מוּכָנוֹת לָךְ מַכּוֹת, כִּי כָּל־רֶשַׁע עָשִׂית בְּקִרְבֵּךְ בְּקַבְּלֵךְ אֶת־הָאָרֶץ אֶל־חֵיקֵךְ. מְבוּכוֹת מַחְשְׁבוֹת בְּנֵי־אָדָם רַבּוֹת תִּהְיֶינָה, עַד אֲשֶׁר יְיָ מִמָּרוֹם יְצַוֶּה לְהַשְׁחִית אֶת־הָאָדָם, אַךְ עֲצַת יְיָ תִּגָּלֶה מֵאַיִן תָּבוֹא הַפֻּרְעָנוּת, וְהַבּוֹטְחִים בָּךְ יְקַבְּלוּ מַכּוֹת; הָעָם אֲשֶׁר בְּצִפִּיָּה בָּךְ, וּמַטֵּה הַצַּיִד יַאֲבִידֵךְ, וְגוּר אֶת־גּוּר יַבְרִיחַ בְּבַקְּשׁוֹ דַּם אֲבוֹתָיו, וּבִרְצוֹן נַפְשׁוֹ יִמָּסֵר לַכְּלָבִים הַנָּכְרִים וַחֲבֵרָיו, וְיָקִים לְשׁוֹנוֹת וְעַמִּים, וְיִמְשֹׁל עַל־עַמִּים רַבִּים, וּשְׁנֵי הַכְּלָבִים יִלָּחֲמוּ אִישׁ בְּאָחִיו, וְאִישׁ אֶת־רֵעֵהוּ יַשְׁחִיתוּ.

**טז.** אוֹי כַּאֲשֶׁר תִּמְלֹךְ אַלְמָנָה, וּבְמִרְמָה תִּשְׂחַק עַל־הַנָּכְרִי, וְהַנָּכְרִי הָאַחֵר יְבַקֵּשׁ אֶת־אָבְדָנוֹ, וּבְמִרְמָה יַכְחִידֵהוּ, וְהָיוּ הַיָּמִים הָהֵם קָשִׁים וְרָעִים. בַּאֲנָחָה יִקַּח אֶת־רוּחוֹ, וְהוּא מָלַךְ יִתְפֹּשׂ אֶת־הַמַּטֶּה אֲשֶׁר בָּךְ, הַשְּׁבִיעִית, וְיִהְיֶה גָּדוֹל בְּכִסְאוֹת גְּדוֹלִים וּבְתִפְאָרוֹת, וְהָאִישׁ הַהוּא אַמִּיץ מִיָּמִין וּמִשְּׂמֹאל, וּבְיָמָיו רָעָב גָּדוֹל יִהְיֶה וְלֹא קָטָן, וְתִשָּׁחֵת הָאָרֶץ מֵאָדָם, וַאֲפִיקֵי מַיִם יִזְרֹמוּ, וַאֲוִירִים שְׁחֹרִים יִתְעָרְבוּ, וּכְרָמַיִךְ יִמְעָטוּ, וְתִפְאַרְתֵּךְ תִּצְטַמְצֵם, וְיִתְקַצְּרוּ הַיָּמִים, וְהָיָה הַיּוֹם בֶּן־שֵׁשׁ שָׁעוֹת. אוֹי לַאֲנָשִׁים אֲשֶׁר יִהְיוּ בָּעֵת הַהִיא; וְיָשִׁיב הַמֶּלֶךְ פָּנָיו אֶל־הַמַּעֲרָב. אָז אוֹי לָךְ, הַשְּׁבִיעִית, כַּאֲשֶׁר מַלְכֵּךְ בָּחוּר. בָּעֵת הַהִיא תִּהְיֶה לָךְ צָרָה גְדוֹלָה: אִישׁ אֲשֶׁר יִקַּח אֶת־אֵשֶׁת אָחִיו, וּבֵן אֶת־אִמּוֹ, וּבַת תֵּצֵא מִמִּשְׁכַּב אָבִיהָ, אָח אֶת־אֲחוֹתוֹ; וְיִרְבּוּ גִדּוּפִים, רְצִיחוֹת, שְׁבוּעוֹת, דִּבָּה רָעָה, שְׁקָרִים, נִאוּפִים, כְּאֵבִים, גְּזֵלוֹת, אַחְוָה, מְרִיבוֹת, שְׁפִיכוּת דָּם וּמִקְדְּשֵׁי מְשָׁרְתֵי קֹדֶשׁ; וּמְלָכִים בַּמְּלָכִים, שָׂרִים בַּשָּׂרִים, חָזָק עַל־דָּל יָקוּם; וּמַפָּלָה תִּהְיֶה לֶעָשִׁיר וְלָדָל.
