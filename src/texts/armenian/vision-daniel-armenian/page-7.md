חזון דניאל השביעי

**יג.** הַזְעִיקוּ חֵילוֹת מְזֻיָּנִים וּבַעֲלֵי חֲרָבוֹת מִתְּרַקְיָא וּמִקִּילִיקְיָא. וּבָעֵת הַהִיא יִשְׁלַח הַחַיָּה אֶל־הַמִּזְרָח אֶת־חֵילוֹ, וְלֹא יוּכַל לְנַצֵּחַ, וְיִבֶז לוֹ אִישׁ דְּמוּת תַּנִּין מִן־הַתָּוֶךְ וָמָעְלָה, וְיִשְׁלַח בְּיָדוֹ אֶת־הַגְּדוּד הַשֵּׁנִי אֶל־הַמַּעֲרָב, וְלֹא יוּכְלוּ לְנַצְּחוֹ, וּבְבִזָּיוֹן יִבֶז הַתַּנִּין מִן־הַכֶּלֶב; בְּמַתָּנוֹת רַבּוֹת וּבְזָהָב וּבִקְנִיָּנוֹת רַבִּים יְרַמֶּה הַכֶּלֶב אֶת־הַחַיָּה, וְיָקוּם הַכֶּלֶב עַל־הַחַיָּה, וְיַעֲמִיד אֶת־גּוּרָיו עַל־כִּסְאוֹת הַחַיָּה וּמַטּוֹתֶיהָ, לְהַאֲבִיד אֶת־הַחַיָּה מִן־הַכֶּלֶב וְלָקַחַת אֶת־מְדִינוֹתֶיהָ; וְהַכְּלָבִים יַעַמְדוּ מִחוּץ לַחַיָּה, וְיֵדְעוּ כָּל־בְּנֵי אָדָם כִּי־כֶלֶב רָדַף אַרְיֵה. וְעוֹד יָשׁוּב הָאַרְיֵה אָחוֹר וְיַהֲרֹג אֶת־הַכֶּלֶב וְאֶת־גּוּרָיו, וְיִשְׁאַג הָאַרְיֵה שְׁאָגָה גְדוֹלָה מְאֹד, וְשַׁאֲגָתוֹ תִּשָּׁמַע בְּכָל־עָרִים וּמְדִינוֹת; וּפַחְדּוֹ עַל־דּוֹרוֹתָיו, וְהָיוּ בְּנֵי־אָדָם נְבוֹכִים אִישׁ מֵרֵעֵהוּ מִקּוֹל שַׁאֲגַת הָאַרְיֵה וּמִמּוֹת הַכֶּלֶב.

**יד.** יִרְדְּפֵהוּ מִפְּנֵי אָדָם בַּעֲבוּר אַהֲבַת הַתַּנִּין וּגְדוֹת הָעֹשֶׁר אֲשֶׁר הָיוּ בָךְ; וְיִנּוֹעַ מֵאֶרֶץ אֶל־אֶרֶץ, וְיִלָּחֵץ מִן־הַנָּכְרִים; וְהַנָּכְרִי הַנִּלְחָץ מִקְּרוֹבָיו וּמִמְּסִלּוֹתָיו, יַאֲמִין בְּעֹשֵׂהוּ בְּכָל־אֲשֶׁר־לוֹ. וּמַטֵּה הַצַּיִד אֲשֶׁר הָיָה מִן־הַחַיָּה, יִתְחַבֵּר אֶל־הַכֶּלֶב, וְהַתַּנִּין יִתְפֹּשׂ מְקוֹם מִרְמָה וּמָנוֹחַ, אֶל־כִּסְאוֹת שָׁוְא יִדְבַּק, וְאֶת־מְקוֹם הַקֹּדֶשׁ יַהֲרֹס, וִיבַלְבֵּל אֶת־הַתַּנִּין בִּלְשׁוֹנוֹ עַל־אֲנָשִׁים קְדוֹשִׁים וְעַל־עַצְמוֹת קְדוֹשִׁים, וְיָפֵר תְּהִלַּת הָאָבוֹת, וְיִשְׁבֶּה אֶת־קְהִלּוֹת הַקֹּדֶשׁ, וַאֲשֶׁר בַּסְּדָקִים וּבַמְּעָרוֹת וּבְחַגְוֵי הָאָרֶץ יֹאחֲזוּ, בַּמִּדְבָּר תּוֹעִים, יָבוֹאוּ אֵלַיִךְ בַּשְּׁבִיעִית נִכְפִּים מִן־הַתַּנִּין; וְאַחֲרֵי כָל־אֵלֶּה אֶת־אֲבַק רַגְלֵיהֶם יְנַעֲרוּ בָךְ לְעֵדוּת בָּךְ. אָז בָּאֵשׁ תִּשָּׂרֵף יְפִי תִפְאַרְתֵּךְ, וְאֶחָד נִמְלָט, בָּחוּר, בְּתַחְבֻּלוֹת מַטֵּה הַצַּיִד, לְהָבִיא אֶת־הַמַּטֶּה הָרִאשׁוֹן אֲשֶׁר נִקְרָא סַלַמַנְדְרָה; וְאָז בָּעֵת הַהִיא יִלָּחֵץ הַתַּנִּין, וּבְיַד יוֹרְשׁוֹ יִתָּפֵשׂ; וּבַחֲבָלִים אֵיתָנִים יֵאָסֵר, וְאֵין מְקַבְּלוֹ, כִּי שָׁפַךְ דַּם קְדוֹשִׁים בְּעָרִים וּבַמְּדִינוֹת, כִּי קַל בְּרַגְלָיו; וְיָנוּס אֶל־הַחַיָּה וַאֲשֶׁר בְּחֵיקָהּ, וְיָבוֹא אֵלֶיהָ, וְתֵצֵא הַחַיָּה הַקְּטַנָּה וְתִתְחַזֵּק, וְהַחַיָּה הַגְּדוֹלָה תּוֹשִׁיבֵהוּ עַל־כִּסְאָהּ, וַתַּעֲשֵׂהוּ שֻׁתָּף לַמַּטֶּה וְלַכִּסֵּא תַּחְתֶּיהָ; שְׁתֵּי חַיּוֹת בְּאִישׁ אֶחָד תָּגֹרְנָה, הַגּוּר יִתְחַזֵּק בִּמְלֶאכֶת מִלְחָמָה, וְהַחַיָּה הַגְּדוֹלָה תָּשׁוּב אֶל־אַרְצָהּ אֲשֶׁר מִשָּׁם הָיְתָה, וְהַחַיָּה הַקְּטַנָּה תִּתְפֹּשׂ מְקוֹמָהּ עַד־עֵת מְסֻיָּמָה, לֹא מֵאִישׁ מָלְכָה, כִּי אִם מִנֶּפֶשׁ עָרִיצִים יָצְאָה, וַתָּבוֹא אֶל־חַדְרֵי הַמַּלְכוּת, וְהַחַיָּה הַקְּטַנָּה תַּעֲשֶׂה לִפְנֵי הַחַיָּה הַגְּדוֹלָה, וּבְעָזְבָהּ אֶת־חֲדָרֶיהָ לְעֹשֵׂהָ – תִּמְשֹׁל בָּךְ, הַשְּׁבִיעִית, וְתִהְיֶה בַּמֵּצָר; שִׂמְחָתָהּ גְּדוֹלָה מְאֹד, שָׂחוֹק תִּשְׂחַק, וּבַגְּדוֹלִים תֵּאָהֵב, וּמֵרַבִּים תִּשָּׂנֵא, חַיִּים וְשֶׁפַע תִּתֵּן, וְהַתַּנִּין יִשְׂחַק בָּהּ, וּבְמַטֵּה הַצַּיִד יִמְשֹׁךְ אֶת־הַחַיָּה אַחֲרָיו.
