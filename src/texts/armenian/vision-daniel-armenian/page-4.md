עמוד יט
בְּרוֹמָא וּבֵין אֵלֶיךָ יַגִּיעַ רֹחַב זַעַם וְכוֹס מְלֹא וְעַד יַגִּיעַ עִתֵּךְ
רִאשׁוֹן יַמְעִיט עִיר חָכְמַת תּוֹרָה וְשֵׁנִי עֹשֶׁר יִהְיֶה מְקוֹר לַכֹּל כְּכַלָּה מְזֹהָרָה וּכְאִישׁ תֵּרָאֶה
וְאֶבְלֶיךָ הַמְּתוּקִים אֲשֶׁר הָיוּ לְפָנִים יַמְעִיטוּ וּכְבוֹדְךָ גָדוֹל יְחֻלַּק וְיִפֹּל
מַלְכוּת אֲשֶׁר בָּךְ וְגַם מַטֶּה אַחֵר יָקִים אֲשֶׁר יִקָּרֵא תֵּאוֹדוֹסְיוֹס מָשִׁיחַ בְּקְדֻשָּׁה וּשְׁמוֹ יְהִי מֻכְרָז בָּךְ בַּשְּׁבִיעִית
וּבְלִדְתּוֹ בָּנֶיךָ יִשְׂמְחוּ וְיַעֲשֶׂה אִישׁ אֶת אֲשֶׁר יַחְפֹּץ
יְשָׁרֶתְךָ כָּל עִיר וּמְדִינָה וּבְמִלּוּאֲךָ יִהְיֶה הוֹד בָּאָרֶץ רַב
וְיַאוֹר יִשְׁתֶּה אוֹתְךָ וְחוֹמַת כְּנֵסִיּוֹת תַּהֲפֹךְ אַתָּה
יִרְאָה וְרַעַד יִהְיֶה בְמַטֵּךְ מְאֹד וְזֹאת תְּחִלַּת הָרֶשַׁע בְּמַלְכוּתְךָ וְעֵת מַטֵּהוּ

עמוד כ
רַב וְגָדוֹל מְאֹד
וְיַחֲזִיק מַטֵּהוֹ עַד קְצֵה כָּל הָאָרֶץ מִמִּזְרָח וּמַעֲרָב מִצָּפוֹן וּמִדָּרוֹם וְצַוָּארוֹ קָיֹם וְיָמִינו חֲזָקָה וְשָׁנָיו רַבּוֹת כַּאֲשֶׁר לֹא הָיָה לַמְּלָכִים
וְאַחֲרֵי כָּל זֹאת יָשִׁיב פָּנָיו אֶל אָבִיו וּמִסְפַּר שָׁנָיו רַב וְשְׁמוֹ נוֹרָרָא וּמַלְכוּתוֹ מְהֻדֶּרֶת
וְשְׁלִישִׁי גַּם מֶלֶךְ יִהְיֶה לָךְ בְּרוֹמָא בַּשְּׁבִיעִית
וְתִקָּרֵא שְׁבִיעִית בָּבֶלָה כִּי כָּל עַם פָּרַס יָבוֹאוּ אֵלֶיךְ רוֹמָא וּמָלַךְ אֶחָד יִמְלֹךְ עַד קֵץ הָעוֹלָמִים
וְהַשֵּׁנִי יִמְלֹךְ עֲלֵיכֶם וּשְׁמוֹ מַרְקִיָּנוֹס וּמַטֵּהוּ מְעַט יָמִים מִן הָרִאשׁשׁוֹן בְּנוֹרְאוּת מַלְכוּתוֹ מְאֹד
וִיהִי בָּעֵת הַהִיא לְאֵיזֶה לְטוֹב וּלְאֵיזֶה לְרָע וּמַלְכוּתוֹ תִּהְיֶה לְעִתִּים וּלְזְמַנִּים וְלַחֲצִי זְמַן וְנִתַּן לוֹ מִבִּנְיָן אוֹתָךְ אוֹ שְׁבִיעִית לְהַעֲלוֹת כָּרֹאשׁ אֶת אֱמוּנַת הַכָּתוּב
וִיהִי קֶרַע גָּדוֹל בְּמַלְכוּתוֹ כֹּהֲנִים יִפְּלוּ מִכִּסְאוֹתָם נְפִילַת עָרִים רַבּוֹת תִּהְיֶה וּתְמוּרוֹת בְּנֵי אָדָם רַבּוֹת וְהוֹד
