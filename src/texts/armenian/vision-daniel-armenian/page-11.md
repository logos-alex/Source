עמוד לג
רֹאשׁ נָחָשׁ חָנוּן גָּדוֹל חָכָם רַךְ לֵב רַךְ יָדַיִם חָזֶה חָזֶה עָמֹק מַחְשָׁבָה שׁוֹמֵר יָדַיִם רַךְ עוֹשֵׂה אוֹתוֹת מָוֹצִיא לֶחֶם מֵאֲבָנִים לְעִוְרִים רְאוּת לְפִסְחִים הִלּוּךְ וְהֶהָרִים יַעֲבִיר מִמָּקוֹם לְמָקוֹם וְלִפְנֵי עֵינַיִם יַעֲשֶׂה אֶת כָּל זֹאת וְרַבִּים יַאֲמִינוּ בּוֹ
אוֹי לָהֶם אֲשֶׁר יַאֲמִינוּ בּוֹ וְיִקְחוּ אֶת אוֹתוֹ יְמִינָם קָשׁוּר אֲשֶׁר לֹא יָשׁוּב אַחוֹר מֵאֲשֶׁר קִוּוּ מִלְּפָנִים
בָּעֵת הַהִיא רָעָב גָּדוֹל מְאֹד יִהְיֶה לֹא יִתְֵּן הַשָּׁמַיִם מָטָר וְהָאָרֶץ לֹא תַצְמִיחַ פִּרְיָהּ וְיִהְיוּ כָּל הַפֵּבֶשׁ יָבֵשׁ וּבָעֵת הַהִיא כָּל הֶעָרִים וְהַמְּדִינוֹת יִתְאַבְּלוּ עַל עַצְמָן וְיָנוּסוּ וְלֹא יִבְרְחוּ מִמִּזְרָח לְמַעֲרָב וּמֵמַעֲרָב לְמִּזְרָח וַאֲשֶׁר יִשְׁכְּנוּ בֶּהָרִים וּבְמְעָרוֹת וּבְזִקְנֵי הָאָרֶץ וּבְמִּדְבָּרוֹת הֵם לְבַדָּם יוּכְלוּ לְהִמָּלֵט עַד בִּיאַת הַשֵּׁנִי אֲשֶׁר נוֹלַד מִבְּתוּלָה קְדוֹשָׁה
בָּעֵת הַהִיא נִבְחָרָיו יִגָּלוּ לְרֹאוֹת אֶת בִּיאַת יְהוָה לָעַד יַעֲמֹד לִפְנֵי וְרַבִּים יִשָּׁפֵטוּ מִן הַשָּׁמַיִם מְהוּמָה תִּהְיֶה
וּבָעֵת הַהִיא צָרָה גְדוֹלָה תִּהְיֶה עַל כָּל הַיְקוּם אוֹי לְמַעֲבִרוֹת וְלַמֵּנִיקוֹת בַּיָּמִים הָאַחֲרוֹנִים אוֹי לְ

עמוד לד
זְמַן אֲשֶׁר יַאֲמִינוּ בְּאַנְטִיכְרִיסְטוֹס אוֹי לָהֶם אֲשֶׁר יִשְׁתַּחֲווּ לוֹ וּבְעַד בִּיאָתוֹ יִגָּלוּ
וְאַחֲרֵי כָּל זֹאת וְלָשֵׂאת אֶת הַיִּסּוּרִים אֲנָשִׁים קְדוֹשִׁים וְצַדִּיקִים אֲשֶׁר נִלְחֲצוּ מֵרָשׁ וּמֵעֹשֶׁק יַגִּיעַ אָז הַקֵּץ
וּלְאֵיזֶה מִבְּנֵי אָדָם אוֹת נִתַּן יַכִּירוּ אֶת שְׁבִיעִית וְיֹאמְרוּ הֲעִיר הָיְתָה זֹאת
וְאִשָּׁה אַחַת תְּחַפֵּשׂ מִזְרָח מַעֲרָב צָפוֹן וְדָרוֹם וְלֹא תִמְצָא פְּרִי כִּי אִם תִּמְצָא אֶחָד זַיִת וְתַשְׁלֵךְ אֶת זֵיתֵי הַזַּיִת וְתַּעֲבֹר נַפְשָׁהּ וְתֹאמַר אַשְׁרֵי הַנּוֹפֵל אֶת הָעֵצִים הָאֵלֶּה
וְרוּחָהּ תַּעֲלֶה מִשָּׁם לְמָקוֹם אַחֵר
בָּעֵת הַהִיא הַשֶּׁמֶשׁ תֵהָפֵךְ לְחֹשֶׁךְ וְהַיָּרֵחַ לְדָם וְהַכּוֹכָבִים כֶּעָלֶה יִפֹּלוּ וְהַשָּׁמַיִם כַּמְּגִלָּה יִתְגֹּלְלוּ וְהַיָּם יִתְבּוֹעַ מִתְּהוֹמוֹתָיו לְכַסּוֹת אֶת הָאָדָם וְכָל יֵחָרֵב בָּאֵשׁ וְיִיבַשׁ
מַלְאָכִים אֵשׁ יֵרְדוּ מִן הַשָּׁמַיִם וְאֵשׁ תִּתְלַהֵט עַל כָּל הַיְקוּם מַחֲשַׁבִּים אֵשׁ וּנְחשֶׁת וְיִהְיוּ וְכָמוֹהֶם
