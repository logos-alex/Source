עמוד כא

אֵלֶיךָ שְׁבִיעִית לֹא יֶחְסַר

וְאַחֲרָיו גַּם הוּא יִכְלֶה עִם כַּלָּה

וְיִמְלֹךְ מַטֶּה אַחֵר בָּךְ וְהוּא יִהְיֶה חַיָּה וְיְקַבֵּל אֶת הָרִאשׁוֹן מַטֶּה אֲשֶׁר לְפָנִים יִדְחֵק לַכֶּלֶב

וִיהִי הַחַיָּה הַזֹּאת כָּרִאשׁוֹן גָּדוֹל וְחָזָק בִּדְבָרִים וּבְחָכְמָה וְלֹא יִזָּחֵל מִגְּדוֹלָיו

וְצַוָּארוֹ כְּצַוַּאר הַפַּר וְעֵינָיו כְּעֵינֵי אַרְיֵה בְּאֵימָה יִזְעַק וּמִקרְנוֹתָיו יִפְרְצוּ כָּל הֶעָרִים וְהַמְּדִינוֹת

וּבְיָמָיו יֵרָאֶה קֶשֶׁת בַּשָּׁמַיִם וְאוֹתוֹת לְעַם עַם בַּשָּׁמַיִם וּבָאָרֶץ

קוֹלוֹת רְעָמִים וּנְפִילַת עָרִים רַבּוֹת תִּשָּׁמַע וְתִתְפָּרֵד הָאָרֶץ וְיִהְיוּ בְּנֵי הַבִּנְיָן לְחָרְבָּן

דַּרְכָּיו דַּרְכֵי אֵשׁ מִלְחָמוֹת תִּהְיֶינה בּוֹ וּבְךָ שְׁבִיעִית וּבָעֵת הַהִיא יִבְעֲרוּ בְּאֵשׁ בִּנְיְנֵי הו הוֹדְךָ וּגְבוֹהֶיךָ תַּחְתִּית וּבְנֵי אֶתְכֶם יִבְכּוּ עָלַיִךְ וְרַב תִּשְׂמַחְתֵּךְ תֵהָפֵךְ לְאֵבֶל וּבָנֶיךָ יִגְרְרוּ אֶת גְּוִיּוֹת גְּדוֹלֶיךָ עַל הָאָרֶץ

וּבְפִתְאֹם סְעָרָה תִּפֹּל מִן הַשָּׁמַיִם וּתְכַסֶּה אֶת הָאָרֶץ וְעַם דְּמוּת וִישָׁאפ יֵרָאוּ בָּאָרֶץ וְרַבִּים יִדַּלּוּ

עמוד כב

וְרַבִּים יַעֲשִׂירוּ וּמֶהוּמָה גְדוֹלָה תִּהְיֶה בָּךְ

זַעֲקוּ חֵילוֹת חֲמוּשִׁים וַחֲרָבוֹת בְּתִרָקְיָה וּבְקִילִיקְיָה

וּבָעֵת הַהִיא יִשְׁלַח הַחַיָּה אֶת חֵילוֹ לְמִזְרָח וְלֹא יוּכַל לְנַצֵּח וְיִבְזֶה אוֹתוֹ אִישׁ אֶחָד דְּמוּת וִישָׁאפ מִבֵּין וּמִלְמַעְלָה

וְיִשְׁלַח בְּיָדוֹ אֶת הַדַּרְגָּה הַשֵּׁנִי לְמַעֲרָב וְלֹא יוּכַל כָּל הָעֵת לְנַצֵּח אוֹתוֹ וּבִזּוֹת יִבְזֶה אֶת הַוִישָׁאפ מִן הַכֶּלֶב

בְּרֹב קָרְבָּנוֹת וְזָהָהָב וְכֶסֶף יְפַתֶּה אֶת הַחַיָּה מִן הַכֶּלֶב וְיָקוּם הַכֶּלֶב עַל הַחַיָּה וְיָרִים אֶת דָּמָיו עַל כִּסְאוֹת הַחַיָּה וְהַמַּטּוֹת וּלְאַבֵּד אֶת הַחַיָּה מִן הַכֶּלֶב וְלָקַחַת אֶת מְדִינוֹתָיו

וְיַחֲזִיקוּ הַכְּלָבִים מִבַּחוּץ אֶת הַחַיָּה וְיֵדְעוּ כָּל בְּנֵי אָדָם כִּי כֶּלֶב רֹדֵף אֶת הָאַאֲרִי

וְיָשׁוּב הָאֲרִי מֵאַחֲרָיו וְיַהֲרֹג אֶת הַכֶּלֶב וְאֶת דָּמוֹ וְיִזְעַק הָאֲרִי זְעָקָה גְדוֹלָה מְאֹד וְיִשָּׁמַע זַעֲקָתוֹ בְּכָל הֶעָרִים וְהַמְּדִינוֹת

וְיִרְאוּ מִמֶּנּוּ וְיִהְיוּ בְנֵי אָדָם מְבֻהָלִים זֶה אֶת זֶה בְּעַד זַעֲקַת הָאֲרִי וּבְעַד מוֹת הַכָּלֶב
