    **⁴⁵ הַנֶּפֶשׁ אֵינָהּ נִכְנֶסֶת לְבֶן-אָדָם אֶלָּא מִן-הַגּוּף** — מוצא-הנפש. **הקשר פילוסופי**: ניקטס מעלה את השאלה הפילוסופית הקלאסית: האם הנפש נוצרה מן-הגוף (ואז אינה נצחית) או מן-הבורא (ואז כיצד היא מוגבלת למקום)? **חז"ל**: ראה תלמוד-בבלי, סנהדרין צ"א ע"ב: "נשמה שנתת בי — טהורה אתה בראתה"; ובנדה ל"א ע"א (נוסח-מקובל): "נשמה — מלמעלה בראה הקב"ה"; ובבראשית-רבה י"ד,ח (נוסח-מקובל): "ואלה תולדות השמים והארץ — זו הנפש". **רמב"ם**, הלכות יסודי-התורה ד,ח-ט: "ודע, כי אותו הכח המצויר באדם הוא אשר יודע השכל הנבדל והוא הנקרא אצלם 'נפש' ואינו נפש ההווית בכל בעל-הנפש". **בהקשר הקלמנטיני**: הקלמנטינה מעלה את הדילמה בלי להכריע בה — דיאלקטיקה אופיינית לפילוסופיה היוונית. **הערה ביקורתית**: הרמב"ם מכריע כי הנפש השכלית אינה מן-הגוף — הקבלה לעמדת פטרוס בקלמנטינה, אם כי הקלמנטינה מתארת את הדילמה ואינה מכריעה.

### תרגום עברי

### פרק יג — דִּין וָדָבָר עִם נִיקְטֵס עַל הַנֶּפֶשׁ

(א) וַיְפַתַּח וַיֹּאמַר אֶל-הַנֶּפֶשׁ: נֶפֶשׁ אֲשֶׁר אֵינָהּ דּוֹאֶגֶת לַגּוּף — מֵתָה הִיא; וּבַסְּתָרִים נִקְרְאָה מוּמִית.

(ב) עָשִׂיתָ חַיִל לְךָ, כְּדֵי שֶׁתַּסִּיעַ הַכֹּל — אַךְ אֲנִי צִוִּיתִיךָ; וּכְשֶׁאָמַרְתִּי לָהּ: כֵּיצַד תּוּכַל נֶפֶשׁ לַעֲשׂוֹת כָּכָה? — וַיָּשָׁב אֵלַי דָּבָר.

(ג) חָפַצְתִּי כִּי-תִכָּנְעוּ בִּשְׁתֵּי אֲלֹהוּת; וְאַתָּה, נֶפֶשׁ כָּל-בֶּן-אָדָם תָּסִיעַ — הֲלֹא-זֹאת חֻשְׁכָא רַבָּה שֶׁל-גּוּף, וְיֶשׁ-לָהּ יְדִיעָה קַדְמוֹנִית.

(ד) כִּי-מִפְּנֵי זֹאת גַּם לַנְּקִירוֹמַנְטְיָא (νεκυομαντία — נְבִיאוּת-הַמֵּתִים) יֵשׁ-לָהּ קֵץ, כִּי-מֵתָה הִיא; וַיְהִי-נָא סִלּוּק הַנֶּפֶשׁ, וַאֲנִי שַׁבְתִּי וּפָנִיתִי אֵלֶיהָ.

(ה) וָאֹמַר: וּמִפְּנֵי מָה אֵלֶּה נֶהֱרָגוּ, וְלֹא-שָׁתְקוּ מִן-הָרִאשׁוֹן? כִּי-אֵלֶּה הֲרֵי נֶהֱרָגוּ; וַיֹּאמֶר סִימוֹן: וּמַה-הוֹדַעְתִּי לְךָ? כִּי הַדָּם הַיּוֹצֵא מִן-הַגּוּף גַּם-לוֹ יְדִיעָה קַדְמוֹנִית יֶשׁ-לוֹ.

(ו) וָאֹמַר לוֹ: הוֹדַעְתִּי אֲנִי. וַיָּשָׁב אֵלַי דָּבָר: כִּי-מִן-הַגּוּף יוֹצֵא בְּרִשְׁעָתוֹ, מִפְּנֵי שֶׁכָּל-דָּבָר קֶדֶם יוֹדֵעַ; וְהוּא אוֹבֵד כִּי-דִין הוּא, וְכָל-אִישׁ עָבַר בָּרָעָה הַזֹּאת, וַיִּסְעָרֵם וַיִּדְרֹשֵׁם וַיִּתְּנֵם — מִפְּנֵי זֹאת אֵינֶנּוּ יָכוֹל לִבְקֹשׁ אֶת-הָעִיר מִן-הַנִּכְנָסִים בָּהּ, מִפְּנֵי שֶׁגַּם-עַל-הָרָעָה הַזֹּאת סְעָרָה חֲלָלָה הִיא, וְעִמָּהֶם הִיא, וְגַם אֵינָם מַפְסִידִים אוֹתָם אוֹ עוֹשִׂים אוֹתָם כְּמַלְאָכִים — הֵם אֲשֶׁר מִפְּנֵיהֶם עוֹמְדִים הֵם.
