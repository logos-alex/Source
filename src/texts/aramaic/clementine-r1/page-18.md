אמרת דצבנא דאלפ הו דינ פטרוס בהנינ פני לי פתגמא
ואמר בפסיקתא או קלמס יהבנא לכ ידעתא דהלינ דאיתיהנ
דהשא מנ דוכתה שמע צבינה דאלהא איתוהי הוא כד לא

### תרגום עברי

### פרק טז — עַל קַבָּלַת הָאֱמֶת בֶּאֱמוּנָה וְעַל כְּתִיבַת הַדְּבָרִים

(א) וַאֲנִי אָמַרְתִּי: אָנֹכִי הִנָּחֹתִי לִבְדֹּק לִי אֵת אֲשֶׁר מִמֶּנּוּ יֵשׁ מְצִיאוּת, וְאֵיךְ יִמָּצֵא, וּבֶאֱמֶת נָתַן לִי לִמְצֹא אוֹתוֹ וְיֶתֶר מֵאֵלֶּה, כִּי לְעֵינַי נִגְלָה אֱמֶת הַדָּבָר אֲשֶׁר עַל הַנָּבִיא בְּאוֹר גָּדוֹל.

(ב) וּבְשָׁמְעִי, כָּכָה נְהַרְתִּי וְנִפְלָאתִי אֵיךְ, וְאִם גַּם אִישׁ מֵבִין מֵאֵלֶּה, מִכָּל אִישׁ נִדְרָשִׁים וְעָשׂוּנִי אָכֵן לִכְתֹּב, לְדַבֵּר עַל הַנָּבִיא בְּמִצְוָתוֹ וּלְשַׁלְּחָהּ אֵלֶיךָ מִקֵּיסָרְיָא סְטְרָטוֹן.

(ג) כַּאֲשֶׁר צִוִּיתָ, שֶׁכָּל הַדִּבּוּר וְהַתְּשׁוֹקָה אֲשֶׁר תִּתְעוֹרֵר בְּכָל שָׁנָה יִכָּתֵב וְיִשָּׁלַח אֵלֶיךָ. אַךְ בַּיּוֹם הָרִאשׁוֹן אֲשֶׁר עָשָׂה הַשַּׂר עַל הַנָּבִיא אֶת הָאֱמֶת, הִגִּיד לִי עַל כָּל דָּבָר, וַאֲנִי נָתַתִּי וְשָׁקַלְתִּי וְאָמַרְתִּי כֹּה: חָשׁוּב הַדָּבָר וְנִבְדַּק בָּעֵת אֲשֶׁר תִּהְיֶה מִמֶּנִּי לִפְנֵי אוֹתָם אֲשֶׁר לִפְנֵיהֶם יֵשׁ לְהִתְיַצֵּב.

(ד) וְאִם גַּם בִּמְעַט יִמָּצֵא, אַל תִּירָא, פֶּן תִּפָּרֵד מִן הָאֱמֶת אֲשֶׁר לְךָ, כִּי אֲנִי בָּטוּחַ שֶׁנִּצְדַּקְתִּי, וְלֹא הַבֶּלֶבֶל הוּא אֲשֶׁר בְּיַד הַנָּבִיא הָאֱמֶת, כִּי אִם מִן הַסְּבָרָה נִשְׁלְמָה לָנוּ הָאֱמֶת, שֶׁאַף לֹא בִּדְבַר שְׁאֵלָה מִתְצַדֵּק אָדָם.
