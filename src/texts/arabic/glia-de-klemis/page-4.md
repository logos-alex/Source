…עָשׂוּ. וְאָמְנָם כֹּהֲנַי — הִנֵּה אוֹשִׁיבֵם בְּאוֹרֵי הָאוֹר וְהָאֱלֹהוּת, וְאֶתֵּן לְכָל אֶחָד מֵהֶם מָקוֹם מְפֹאָר כְּפִי מַעֲלוֹתָם וְחַסְדָּם וְחַסְדִּי. וּלְךָ, בֶּן־פֶּטרוֹס, נָתַתִּי שְׁאָר רַחֲמַי, וּלְךָ אֶהְיֶה מַמְלִיץ בְּכָל מִי שֶׁתְּבַקֵּשׁ מִמֶּנִּי. הוֹי בֶּן־פֶּטרוֹס! הֲרֵי רַחֲמַי תְּהַלֵּל מַלְאֲכַי וְתֹאמַר בִּלְשׁוֹנָם. וְעוֹד אֶעֱשֶׂה אֶת הַמַּאֲמִינִים בִּי בְּיוֹם הַתְּקוּמָה כַּמַּלְאָכִים אֲשֶׁר אֵין לָהֶם זָכָר וְלֹא נְקֵבָה, עַד שֶׁיִּהְיוּ כֻלָּם טְהוֹרִים, לֹא יַעַבְרוּ וְלֹא ⟨יִוָּלְדוּ⟩, כִּי כְחֵזוּת הָעַיִן ⟨תִּהְיֶה אֲחִיזָתָם⟩. לֹא יֵדְעוּ מְרִירוּת הַמָּוֶת, וְלֹא יִהְיוּ לָהֶם אֹרֶךְ וְלֹא רֹחַב, אֶלָּא כִשְׁנֵי מַלְאָכִים אֲשֶׁר יְכָלְתָּם לְהִתְרַחֵק לַאֲשֶׁר יִרְצוּ; וְיֵלְכוּ וְיִשְׁמְעוּ וְיִרְאוּ, וְלֹא יָנוּסוּ מִפְּנֵי מוֹרָא מֵאִישׁ מִכָּל בָּאֵי עוֹלָם.

וַאֲנִי, בֶּן־פֶּטרוֹס, אוֹדִיעֲךָ אֶת אֲשֶׁר יִהְיֶה בְּיוֹם הַתְּקוּמָה אַחַר הָעֵת הַזֹּאת. וְהוֹחֵל מְעַט, עַד תַּשִּׂיג הַמַּעֲלָה הַמְפֻרְסֶמֶת קֹדֶם הַגְּדוֹלָה, וּשְׁאַל קֹדֶם שְׁאֵלָתְךָ עַל אֲשֶׁר יִהְיֶה בַתְּקוּמָה וּמַה שֶׁאַחֲרֶיהָ — עַל אֲשֶׁר יִהְיֶה קֹדֶם לָזֶה, וְעַל אַחֶיךָ וִידִידֶיךָ הַמַּאֲמִינִים, וְעַל עִנְיַן הַכּוֹפְרִים.

הוֹי בֶּן־פֶּטרוֹס! יִוָּסְפוּ הַצָּרוֹת וְהָאֵימוֹת הַגְּדוֹלוֹת אַחַר עֲלוֹתִי לַשָּׁמַיִם, וְתְקַבְּלוּ ⟨אוֹתָן⟩.

[בשולים:] …בִּשְׁאָר קַצְוֵי הָאָרֶץ, וְיִשָּׂרְפוּ בָּאֵשׁ, וְיִקָּצְצוּ אֶבְרֵיהֶם זֶה מִזֶּה, וְיִגֹּרְשׁוּ מִמּוֹשְׁבוֹתָם. אַךְ לֹא אֶעֶזְבֵם, וְאֶתֵּן כֹּחִי בָהֶם, וְאֲגַלֶּה בָהֶם נִפְלְאוֹתַי.

דַּע, בֶּן־פֶּטרוֹס, כִּי נִשְׁאֲרוּ מִשִּׁלְטוֹן הַתַּחְתּוֹנִים עֶשְׂרִים שָׁבוּעַ וְאֶחָד. וְכִי תִכְלֶינָה הַשָּׁבוּעוֹת, אַרְאֶה כֹחִי וְחַסְדִּי בְּכָל הַמַּאֲמִינִים בִּי. וְיִגַּף אֶחָד מֵהֶם אֶלֶף, וּשְׁנַיִם — רְבָבוֹת, עַד שֶׁיָּשׁוּבוּ הָאֲנָשִׁים אֶל יְשׁוּעָתִי וְקִדּוּשַׁי וִיכֻבְּדוּנִי.

הוֹי בֶּן־פֶּטרוֹס! דַּע כִּי כַּאֲשֶׁר אֶעֱלֶה אֶל אָבִי וְאֶעֱמֹד בַּתִּפְאֶרֶת וְהַכָּבוֹד אֲשֶׁר לֹא חָדַלְתִּי מֵהֶם — כִּסֵּא תִפְאַרְתִּי וּתְהִלָּתִי — אֶשְׁלַח אֵלֶיךָ וְאֶל כָּל תַּלְמִידַי אֶת רוּחַ הַפַּרַקְלִיט אֲשֶׁר הוּא רוּחַ הַקֹּדֶשׁ, לְמַעַן יִהְיֶה עִמָּכֶם וְלֹא יִפָּרֵד מִכֶּם, וְיִהְיֶה כֹחִי אָהוּב לָכֶם, וְאָשִׂים כָּל הָאֻמּוֹת תַּחַת רַגְלֵיכֶם, וְתִהְיֶה רַחֲמָנוּתִי מְהֻדֶּרֶת לָכֶם עַד סוֹף הַנֶּצַח.

וְאַל תֵּעָצֵב, בֶּן־פֶּטרוֹס, וּשְׁמֹר אֶת כָּל אֲשֶׁר אָמַרְתִּי לְךָ וְאֹמַר לְךָ, וְהָנַח אוֹתָם בְּלִבְּךָ. וְדַע כִּי הֲרֵי אַתָּה ⟨הַשְּׁלִישִׁי⟩ בְמַלְכוּת הַשָּׁמַיִם, וְאַתָּה תִּצָּלֵב, וְתִפְגְּשֵׁנִי. ויעקב — נֶהֱרָג, וְיוֹחָנָן — נֶהֱרָג מִן הָאֲרָיוֹת. וְתִפְגְּשֻׁנִי כָּל הַתַּלְמִידִים, וְהֵם נֶהֱרָגִים…
