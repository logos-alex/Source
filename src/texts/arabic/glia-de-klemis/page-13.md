…וְיֹאבְדוּ כָּל הַחוֹלְקִים עַל מַלְכוּתִי וְהַמַּכְזִיבִים אוֹתָהּ. וְאַגַּלֶּה אֶת הַיְּשׁוּעָה בָּאָרֶץ בְּגָלוּי. וְיֵרוֹמַם שְׁמִי אֵצֶל הַמַּאֲמִינִים בִּי. וְיִתְוַדְעוּ הַשְּׁבָטִים, וְיָשׁוּבוּ אִישׁ אֶל אָחִיו, וְיִתְגַּלֶּה כָּל הַנִּסְתָּר, וְיַעֲלֶה הָעָנָו מִן הַשְּׁפָלִים, וְיֵצֵא בָאָרֶץ הֶעָלוּב — גִּבּוֹר רָצוּי.

אַחַר־כֵּן יִתְעוֹרֵר גּוּר הָאַרְיֵה, וְיַכְנִיעַ אֶת כָּל הַמְּלָכִים, וְיַעַזוֹב לוֹ — לְמִי שֶׁנָּתַתִּי הַשִּׁלְטוֹן. וִיהֵא הַסִּבָּה לְהִתְעוֹרְרוּת גּוּר הָאַרְיֵה וְהִתְעוֹרְרוּתוֹ מִשְּׁנַת שְׂגִיאָתוֹ: אֲשֶׁר יִקַּח לוֹ הָאוֹיֵב מִמַּלְכוּתוֹ עִיר בְּצַד הַמַּעֲרָב. וְיִכָּנֵס אֲחֵרִים תַּחְתָּיו לְכָךְ, וְיִזְעֹף. וְיִגְבַּר עַל מִי שֶׁפָּתַח אֶת הָעִיר אֲשֶׁר מִמַּלְכוּתוֹ בַּמַּעֲרָב. וְיִשְׁאֲלוּ מִמֶּנּוּ אַמָּן — וְלֹא יַעֲנֵם בּוֹ. וְיִשְׁלַח צְבָאָיו אֶל הַמִּזְרָח וְהַמַּעֲרָב, וְיִכְבְּשֵׁן, וְתִקְוֶה כָּל הַמַּמְלָכוֹת בְּקַבְעוֹ.

וְיַחֲנֶה אֶל יְרוּשָׁלַיִם וְיִכָּנֵס בָּהּ שָׂמֵחַ וַעֲלִיז, וִיחַדֵּשׁ אֵת אֲשֶׁר יִמְצָא בָּהּ חָרוּב. וְיִבְנֶה אֶת חוֹמוֹת מְדִינָתָהּ וּצְלָמֶיהָ, וִיעַמֵּר אֶת מַעֲמָרֶיהָ. וִיחַדֵּשׁ חוֹמוֹת קוּנְיָה, כִּי יִמְצָאֵן חֲרֵבוֹת. וְיִהְיוּ, בֶּן־פֶּטרוֹס, כָּל הַמַּאֲמִינִים בָּעֵת הַהִיא בְּשִׂמְחָה וְשָׂשׂוֹן.

וְיַחֲרִיב גּוּר הָאַרְיֵה אֶת דַּמֶּשֶׂק, וִיתַשְׁתָה אֶת יְסוֹדוֹתֶיהָ — אִם לֹא יָשׁוּבוּ יוֹשְׁבֶיהָ מִמַּעֲשֵׂי הָרָע אֲשֶׁר עָשׂוּ, כִּי הֵם רוֹדְפֵי הַמַּאֲמִינִים בָּעִיר הַזֹּאת.

וְאַחַר זֶה יָשׁוּב אֶל עִיר מַלְכוּתוֹ בְּהוֹד גָּדוֹל וְרַב, וְיֵשֵׁב עַל כִּסֵּא מַלְכוּתוֹ שָׁנִים רַבּוֹת, כִּי יַעֲשֶׂה בְּמִצְוָתִי וְיֵלֵךְ אַחֲרַי, וְלֹא יְרַחֵם עַל הַכּוֹפֵר בִּי. וְיִתְמַשֵּׁךְ הַמַּלְכוּת לַמַּאֲמִינִים שִׁבְעִים שָׁבוּעַ. וְהַמֶּלֶךְ אֲשֶׁר יֵשֵׁב אֲשֶׁר נוֹתָר מִשְּׁנוֹתָיו — רֹאשׁ שְׁמוֹ אָלֶ"ף — יִהְיֶה בָּעֵת אֲשֶׁר יִגָּמֵר הָעִנְיָן הַזֶּה — מְסַלֵּק אֶת מַלְכוּתוֹ. וְיִדָּבֵק זֶה הַמֶּלֶךְ הַמְּבֹרָךְ בַּבָּנִים הַיְּגוֹנִים, וְתִוָּצֵא מֵהֶם עֵין הַמַּחֲלִישָׁה אֶת כָּל גּוּר וְהַלְקִיחָה אוֹתָם מִן הַסְּחוֹחִים.
