…וְיִתְפַּלְּאוּ הָאֲנָשִׁים מִזַּעַם גּוּר הָאַרְיֵה וּמֵעֲלִילוֹתָיו, כִּי יִהְיוּ בְּלִי גְעִיָּה, מֵתִים וְלֹא קָמִים. וּמֵעֵת הַהוּא לֹא יִירָא אִישׁ מִבְּנֵי הַזְּאֵב נֶשֶׁק חוּחֵי הַשָּׂפָה לְעוֹלָם, וְלֹא יוּכַל לִגְזֹל אֶת הָעָרִים הַחֲרוּבוֹת. וְיִהְיֶה הַלֶּחֶם כַּחוֹל אֲשֶׁר עַל שְׂפַת הַיָּם בְּרֹב. וְיִהְיוּ כָּל הַמַּאֲמִינִים בְּטוֹבָה רַבָּה אֲשֶׁר לֹא תְתֹאַר.

וְדַע, בֶּן־פֶּטרוֹס, כִּי הַדְּבָרִים הָאֵלֶּה יִהְיוּ קֹדֶם בּוֹאִי הַשֵּׁנִי לָעוֹלָם — בִּשְׁנַיִם וְעֶשְׂרִים שָׁבוּעַ וְשָׁנָה וָחֵצִי גְּדוֹלִים, וְשִׁבְעִים שָׁבוּעַ קְטַנִּים.

הוֹי בֶּן־פֶּטרוֹס! מַלְכוּת בְּנֵי הַחֲמוֹר — יָפֶה. וּכְבָר גִּלִּיתִי לִדָנִיֵּאל מִדְּבָרִים אֵלֶּה דָּבָר מְעַט, וְהוֹדַעְתִּיו אֶת הַמַּלְכוּת, וּמְשַׁלְתִּי אוֹתָהּ לוֹ בְּאַרְבַּע תְּמוּנוֹת מִן הַבְּהֵמוֹת אֲשֶׁר עָלוּ מִן הַיָּם הָרִאשׁוֹן — בְּדִמְיוֹן הָאַרְיֵה, אֲשֶׁר הוּא הַגָּדוֹל וְהָאָיֹם בַּחַיּוֹת.

הָרִאשׁוֹן הַדּוֹמֶה לָאַרְיֵה — הוּא מֶלֶךְ בָּבֶל; וְהַשֵּׁנִי הַדּוֹמֶה לַזְּאֵב — הוּא מֶלֶךְ בְּנֵי הַחֲמוֹר; וְהַמֶּלֶךְ הַדּוֹמֶה לַנָּמֵר — הוּא מֶלֶךְ הַיְּוָנִים; וְהָרְבִיעִי הַדּוֹמֶה לְגוּר הָאַרְיֵה — הוּא מֶלֶךְ רוֹמִי, וְהוּא נִכְבָּד וְגָדוֹל מִכָּל הַמְּלָכִים.

וְאָמְנָם מֶלֶךְ בָּבֶל — יִמְלֹךְ חֲמֵשׁ מֵאוֹת שָׁנָה. וּמֶלֶךְ בְּנֵי הַחֲמוֹר — יִמְלֹךְ, כְּמוֹ שֶׁאָמַר יְשַׁעְיָה הַנָּבִיא, שְׁנוֹת הַשָּׂכִיר. וּמֶלֶךְ הַיְּוָנִים — יִמְלֹךְ שְׁלֹשָׁה־חֲמִישִׁיּוֹת הַכֶּפֶל. וְאָמְנָם מֶלֶךְ רוֹמִי — קַיָּם עַד בּוֹאִי הַשֵּׁנִי. וְאַתָּה, בֶּן־פֶּטרוֹס — מָסַרְתִּי הַמַּלְכוּת מִמֶּנּוּ.

וְאָמְנָם מֶלֶךְ בְּנֵי הַחֲמוֹר — יֵצְאוּ מִמֶּנּוּ שְׁנֵים־עָשָׂר מֶלֶךְ בָּאָרֶץ. וְיִהְיוּ גְּדוֹלִים נִכְבָּדִים, וְשִׁבְעָה ⟨מֵהֶם⟩ חוּטֵי שָׁנִי, יְפַתּוּ אֶת הַמַּלְכוּת. וְתִהְיֶינָה שְׁנוֹתֵיהֶם מְהֵרֵי הִתְמַשְּׁכוּת, וְיָמֵיהֶם קְצָרִים. וְיִהְיֶה כִּלְיוֹנָם מַר מֵהֶם. וְאַרְבָּעָה מֵהֶם — יִשְׁמְרוּ מַלְכוּתָם, וְיִהְיֶה בָהֶם רוּחַ פֶּלֶא עַד הַסּוֹף.

וְיִהְיוּ בָהֶם שְׁלֹשָׁה מְלָכִים, שְׁלֹשָׁה אַחִים: הָרִאשׁוֹן מֵהֶם — וְלֹא בָּאַחִים הַשְּׁלוֹשָׁה — יֵהָרֵג עַל מִטָּתוֹ. וְהַשֵּׁנִי מִן הַשְּׁלֹשָׁה — יִמְלֹךְ חֲמֵשׁ שָׁנִים וָחֵצִי, וְיֵצֵא עַל מִי שֶׁהֲרָגוֹ עָם רַב, וְכִלָּיוֹנָם.
