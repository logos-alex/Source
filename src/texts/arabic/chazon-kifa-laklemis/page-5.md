וְיֻתַּן בַּנִּשּׂוּאִין לָקַחַת שְׁתֵּי אֲחָיוֹת, וְיִהְיוּ רֹב חֲבֵרֵיהֶם כּוֹפְרִים
בִּילְדֵיהֶם. וְהַשּׁוֹמֵר מִן הַמַּאֲמִינִים עַל הַשִּׁפְלוּת וְהַבִּזָּיוֹן ― הוּא יִחְיֶה,
כִּי אִמַּת הָאִישׁ הַזֶּה לֹא תְּקַצֵּר בְּמוּסַר עַמִּי. וְאִם יַהֲרֹג אִישׁ
מֵהֶם אֶחָד מִן הַמַּאֲמִינִים יִשְׁפֹּט שֶׁמְּקָרֵב הוּא קָרְבָּן, וְיֹאמַר: הָרוֹצֵחַ
וְהַנֶּהֱרָג בַּגַּן־עֵדֶן. וְיַעֲזְבוּ אֵלּוּ הָעָם אֶת הַטְּבִילָה אֲשֶׁר צִוִּיתִי [בְּגִלָּיוֹן: בָּהּ],
וְלֹא תְּהֵא תְּפִלָּתָם דּוֹמָה לִתְפִלּוֹת אֶחָד מִן הַנְּבִיאִים אֲשֶׁר שָׁלַחְתִּי
אֶל בְּנֵי יִשְׂרָאֵל הַחוֹטְאִים, וִיכַוְּנוּ מְקוֹם הָאוֹרוֹת בְּעֵת תְּפִלָּתָם,
וְיָחִישׁוּ רַגְלֵיהֶם לְבוֹאָהּ. פֶּטְרוֹס, לוּלֵא מַחֲצִית הָאֶחָד מֵהֶם ―
הַהַבְטָחָה אֲשֶׁר הִבְטַחְתִּי אֶת אַבְרָהָם, וְהָאַחֶרֶת רֹב חַטֹּאותֵיהֶם שֶׁל עַמִּי ― דְּבָרָם
מִקְצָתָם הַמְכֹעָר מִן הָאֳמָנוּת ― לֹא הָיִיתִי שׁוֹלֵחַ אֶת אַרְבֶּה פַּרְעֹה
לְהַשְׁחִית הָאָרֶץ הַקְּדוֹשָׁה הַזֹּאת. פֶּטְרוֹס, רְעֵה צֹאנִי, רְעֵה כִּבְשׂוֹתַי, רְעֵה
לִי כָּל טָלֶה הַנִּסְקָר אֶל הָאֳמָנוּת. אַל תְּאַבֵּד דָּבָר מִמַּה שֶּׁנָּתַתִּי לְךָ. וְדַע
כִּי בֶּן הָאֲבַדּוֹן אֲשֶׁר הוֹדַעְתִּיךָ עָלָיו יַעֲסֹק בְּכִשּׁוּפָיו בְּמִצְרַיִם, וְיִלְמַד הַמְשֻׁפָּל וְהַמָּלֵא
וְהַמָּאוּס. כִּי אֲשֶׁר רֹאשׁ הַמְכַשְּׁפִים הָיָה ― הוּא יִהְיֶה תַּלְמִידוֹ. הַמְכַשְּׁפִים
יַגִּיעוּ אֶת הָעָם הָאֵלֶּה אֶל הָעָוֶל, וְהַמַּאֲמִינִים בְּמַה שֶּׁיְּקַצְּרוּ ―
