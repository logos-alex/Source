בְּנֵי הָאֲבַדּוֹן ― אֲנִי אֶמְנַע מֵהֶם רַחֲמַי וְחֶמְלָתִי וּסְלִיחָתִי. בָּעֵת הַהִיא
הִשְׁתַּחֲוֵיתִי לִפְנֵי אֲדוֹנִי יֵשׁוּעַ הַמָּשִׁיחַ וְאָמַרְתִּי לוֹ: בֶּן הָאֱלֹהִים הַיָּחִיד
אֲשֶׁר בָּא לָעוֹלָם לְהַצָּלַת אַנְשֵׁי הָעוֹלָם, הוֹדִיעַ עַבְדְּךָ עִנְיַן הַפַּרְדֵּס, וְאֵיזֶה
מָקוֹם הוּא בָּאָרֶץ ― כְּמוֹ שֶׁנֶּאֱמַר ― אוֹ בַּשָּׁמַיִם, אוֹ מַה זֶה הָאִילָן אֲשֶׁר אָכַל
מִמֶּנּוּ אָדָם וְחַוָּה, וְנִמְנַע שֶׁלֹּא יֹאכַל מִמֶּנּוּ, וּמַה זֶה הָאִילָן אֲשֶׁר צִוִּיתָם
שֶׁלֹּא יֹאכְלוּ מִמֶּנּוּ, וּמַה הִיא עֵץ הַחַיִּים, וּמַה הָיְתָה הַסִּבָּה בְּהַשְׁמָדַת אָדָם וְחַוָּה
אֲשֶׁר בַּפַּרְדֵּס. וְהַאִם בּוֹ נְהָרוֹת, וְהַאִם בּוֹ מַאֲכָל וּמַשְׁקֶה? וְאֵיךְ עֵצָיו ―
הַאִם כְּעֵצֵי הָעוֹלָם הַזֶּה וּפֵרוֹתָיו כְּפֵרוֹתָם, כְּאוֹתָם הָעֵצִים אֲשֶׁר בָּעוֹלָם הַזֶּה?
הַאִם בּוֹ קַר אוֹ חֹם אוֹ רָעָב אוֹ צָמָא? וְהַאִם הַשּׁוֹכֵן בּוֹ צָרִיךְ
לִכְסוּת אִם אוֹ לֹא? וּמֵאַיִן יוֹצְאִים אֵלֶּה אַרְבַּעַת הַנְּהָרוֹת ― סִיחוֹן וְגִיחוֹן
וְהַחִדֶּקֶל וּפְרָת ― הַזּוֹרְמִים בָּעוֹלָם, וְכַמָּה שָׁבְלוּ אוֹתָם מִן הַפַּרְדֵּס?
וְכַמָּה אֹרֶךְ הַפַּרְדֵּס וְכַמָּה רָחְבּוֹ? וְהַאִם יָבוֹא קֵץ עִם תֹּם הַזְּמַנִּים אַחַר[יו]?
וַיֹּאמֶר אֵלַי בֶּן הָאֱלֹהִים: פֶּטְרוֹס, מָה־רַחֲבָה תְּאֵבָתְךָ עַל בַּקָּשַׁת עִמָּקוֹתַי. וְאַחַר כֵּן אָמַר אֵלַי:
הִנֵּה הַפַּרְדֵּס ― יְצַרְתִּיהוּ בְּעֵדֶן הַנֹּעַם אַחַר שֶׁיָּצַרְנוּ הַשָּׁמַיִם וְהָאָרֶץ,
בַּגַּן עֵדֶן הַפַּרְדֵּס אֲשֶׁר יְצַרְנוּ עִם בְּרִיאַת הַשָּׁמַיִם וְהָאָרֶץ.
אֲפֵלָה מְאִירָה הוּא, כְּעֵדֶן כָּבוֹד גָּבוֹהַּ מִן הָעוֹלָם הַזֶּה כְּרֹחַק הַמִּזְרָח
מִן הַמַּעֲרָב.
