‏וּבֶאֱמֶת אֲנִי, פֶּטְרוֹס, אֵין נֶאֱבָד אֶלָּא בֶּן הָאֲבַדּוֹן. שָׁלַחְנוּ אֶת הַחַיָּה
‏הַסְּמוּחָה לְהוֹצִיאוֹ מִקִּבְרוֹ. וְקֹדֶם מוֹתוֹ הוּא הוֹרֵג אֶת מְלַמְּדוֹ אֲשֶׁר יִסֵּד
‏אֶת בִּנְיָנוֹ בְּשֵׁשׁ שָׁנִים, וְאוֹסֵר עַל עַצְמוֹ בִּשְׁבִיל זֶה דְּבָרִים רַבִּים
‏מִן הַמַּאֲכָלִים וְהַמַּשְׁקִים. וְיִהְיֶה בַּנָּשִׁים אֲשֶׁר יִשָּׂא מֵהֶן אִשָּׁה, רֵאשִׁית
‏שְׁמָהּ צַ״דִי, וְהִיא חוּגֶּרֶת מַשְׁכֶּלֶת ― יִלְמַד מִמֶּנָּה אֶת הַכִּשּׁוּף. וְאַחֲרֵיהָ ― כִּלָּיוֹן רֹאשׁ
‏הַכִּלָּיוֹן בְּאַרְצוֹת הַסַּעַף. וְאַחֲרָיו יָקוּם אִישׁ רֵאשִׁית שְׁמוֹ עַ״יִן, יִהְיֶה אַחֵר
‏בִּדְעוֹתָיו מִמַּה שֶּׁקָּדַם זִכְרוֹ. אֲבָל לוֹ מִן הַמְּלָכִים אֲשֶׁר יִהְיוּ מִן הָעָם
‏הַזֶּה אַחֲרָיו: לֹא יֹאהַב שְׁפֹךְ דָּמִים, וְיִטֶּה אֶל הַשָּׁלוֹם, וְיִקְרָא אֶת הַקָּהָל
‏אֶל אֱמוּנָתוֹ, וִישַׁנֶּה מִן הָעִנְיָנִים אֲשֶׁר הָיוּ קְבוּעִים, וְלֹא יִתָּאֵם לְאִישׁ לְהָסִיר עָלָיו
‏[?] הַיָּדוּעַ; וְלֹא תֶּאֱרַךְ תְּקוּפַת מַלְכוּתוֹ, וְיֹאבַד מִן הָאָרֶץ הַחֲשׁוּכָה אֲשֶׁר הָיָה בָּהּ, כִּי
‏יָמוּת עַל מִטָּתוֹ. וּכְשֶׁיֹּאבַד ― יִתְכּוֹנֵן הָעָם לְהוֹסִיף עַל אֲשֶׁר תִּקֵּן.
‏וְלֹא חָדַל יֵשׁוּעַ הַמָּשִׁיחַ לְהַשְׁמִיעַ לִי, אֲנִי הֶחָלוּשׁ פֶּטְרוֹס, מַלְכוּת
‏הָעָם הַזֶּה, וְכַמָּה יִחְיֶה כָּל אֶחָד מֵהֶם, וּמַה יִּקְרֶה וְאֵיךְ יִהְיֶה
‏מִמֶּנּוּ בִּזְמַנּוֹ מִן הַקְּרָב וְהָרַע. רָאוּי לִמְנוֹת אַרְבָּעִים מֶלֶךְ, וְאַחֲרוֹן הַמְּלָכִים
‏אֲשֶׁר יָקוּם בָּהֶם ― הוּא יִהְיֶה אֲשֶׁר עַל יָדָיו תַּעֲבֹר תְּקוּפַת אַנְשֵׁי בֵיתוֹ. וּבָעֵת
‏הַהִיא תִּשָּׁבֶה מִצְרַיִם שָׁלֹשׁ פְּעָמִים, וְאַחַר תֵּחָרֵב; וְיוֹפִיעוּ אַחֲרָיו מְלָכִים
‏שְׁלֹשָׁה.
