‏פֶּטְרוֹס, אֲשֶׁר סָבְלוּ עַל הַיְגוֹנוֹת בִּגְלָלִי וְנִשְׁבּוּ וְנֶאֱבָדוּ
‏מִבְּנֵי הַזְּמַן ― טוּבָם, כִּי הֵם עִמִּי בְּחַסְדִּי אֲשֶׁר לֹא יִכְלֶה וְלֹא יָסוּר.
‏וְאוּלָם הִתְעוּ בַּעַבְדוּת ― הֵם עֵדַי, וְהֵם עַמּוּדֵי הַפַּרְדֵּס, וְעִנְיָנָם
‏אֶצְלִי כָּתוּב, וּשְׁמוֹתָם בְּסֵפֶר הַחַיִּים רְשׁוּמִים בְּאוֹצְרוֹתַי הַזּוֹהֲרִים. וְאֶשְׁלַח
‏מַלְאָכַי, וְיַקְבִּילוּם, וַאֲעַבְּרֵם בְּלוֹעַ יָרֹק, וְלֹא יִשָּׂרֵף אַף לֹא שַׂעֲרָה אַחַת
‏מֵרָאשֵׁיהֶם. וְעִם מַלְאָכַי יִתְהַלְּלוּ וְיִשְׂמְחוּ בְּלֹא הֲפָרָה וְלֹא הִתְפָּרְדוּת,
‏כִּי הֵם מְבֹרָכִים אֲשֶׁר נֶהֶרְגוּ בִּגְלַל שְׁמִי. וְתִשְׁתֶּה הָאָרֶץ מִ־
‏דְּמֵיהֶם, וְתִהְיֶה לָהֶם הָאָרֶץ הַהִיא עֵדוּת עַל מִי שֶׁהֲרָגָם. וְאֶצֹר לָהֶם
‏כְּתָרִים וְאֵאַזְּרֵם מִן אוֹרִי וְזָהֳרִי, וְאַלְבִּישֵׁם הַתָּאגוֹת. יִהְיוּ כְּכוֹכָבִים
‏בַּשָּׁמַיִם, יִתְעַנְּגוּ בְּ[?]. לֹא, אַף מְבֹרָכִים אֲשֶׁר יֹאבְדוּ מָמוֹנָם בִּגְלָלִי, וְיִשְׁחֲטוּ
‏אֶת בְּנֵיהֶם וּבְנוֹתֵיהֶם מִפְּנֵי הַטָּפַת הַשָּׁוְא אֲשֶׁר בִּגְלָלִי תָּבֹאנָה עֲלֵיהֶם הַיְגוֹנוֹת, וְיִתְחָרְטוּ
‏עַל אֲשֶׁר עָבַר מֵהֶם, עַל אֲבֵדַת מָמוֹנָם אֲשֶׁר עָבַר מֵהֶם, וְסָבְלוּ, וְהֻחֲרְפוּ בִּגְלָלִי,
‏לִהְיוֹת לָהֶם הַשֶּׁבַח עִם שְׁלִיחַי. אֲנִי כִּוַּנְתִּי אוֹתָם וְכִבַּדְתִּי נַפְשׁוֹתֵיהֶם אֶצְלִי,
‏וּמַלְאָכַי אֶעֱשֶׂה לִקְהָלָם, וְיִשְׁתַּקְּעוּ גּוּפֵיהֶם וְנַפְשׁוֹתֵיהֶם בְּקַבָּלָתִי בַּיּוֹם
‏הָאַחֲרוֹן. מְבֹרָכִים אֲשֶׁר בַּהֲרוּסָה וְהָאֳמָנוּת נִשְׁפְּכוּ בַּדָּם הַשָּׁפוּךְ הַרּוֹ־
