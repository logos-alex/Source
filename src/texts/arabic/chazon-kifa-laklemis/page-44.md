כָּל הַמַּאֲמִינִים בִּשְׁמִי עַד אַחֲרִית הַדּוֹר. וְכַאֲשֶׁר יִהְיוּ שְׁנַיִם עַל שְׁמִי ―
אֶהְיֶה אֲנִי עִמָּם וּשְׁלִישָׁם. וְלֹא אַעֲלִים יָדַי מֵהֶם עַד הַדּוֹר הָאַחֲרוֹן. וְאַל[?]
הָאֳמָנוּת הַזֹּאת, פֶּטְרוֹס. וְהַסְתֵּר אֶת מַאֲמָרִי זֶה מִן הַחִיצוֹנִיִּים, פֶּטְרוֹס.
וְאִם עָלִיתִי לִתְהִלָּתִי וּכְבוֹדִי אֵצֶל הָאָב עִם הָאָב ― אֶשְׁלַח לְךָ
וְלַאֲחָיֶיךָ אֶת רוּחַ הַקֹּדֶשׁ הַפַּרַקְלִיט וְהוּא יְלַמֶּדְכֶם בְּמַה שֶּׁכְּבָר נִשְׁאַר לָכֶם,
וְהַנְּסִתָּרוֹת יְגַלֶּה לָכֶם. וְהוּא יִתֵּן לָכֶם מַתָּנָה, וְתִתְחַדֵּשׁ לְשׁוֹנוֹת לֵדַע אֶת
הַנֶּעֱלָם, וּתְדַבְּרוּ הָאֻמּוֹת, וִהְיוּ אֱלָהוּת קְדוֹשָׁה, וְיוֹתֵר מִמַּה שֶּׁעָשִׂיתִי
מִן הַסַּמִּים וְהַפְּלָאוֹת תְּדַבְּרוּ, וּתְרַפְּאוּ הַחוֹלִים. אֶת הַמֵּתִים תָּקִימוּ, וְהַמְּלָכִים
וְהָעַמִּים יִשְׁתַּחֲווּ לָכֶם, וְהַמַּיִם יִתְכַּנְּסוּ לָכֶם, וְהָאָרֶץ כִּסֵּא, וְהַמַּלְכוּת
לֹא תִפָּרֵד מִכֶּם לֹא בְּחַיֵּיכֶם וְלֹא בְּמוֹתְכֶם. וַאֲנִי עִמָּכֶם מַלְבּוּשִׁי, אֲנִי מְדַבֵּר
פֶּטְרוֹס, וְלֹא תָסוּר בַּקְּדוֹשִׁים עִמְּךָ וְעִם עַמִּי שֶׁכְּבָר תָּעוּ הַזֵּזָנִים
מֵהַגּוֹרָל, וְהֵעֲרַב שֶׁל עַמִּי בָּאֻמָּה הָאַחֲרוֹנָה. שֵׁבֶט תְּלוּנָתְךָ ― שֵׁבֶט הַחֵמָה
לַמִּתְרַחֲקִים מִן הָרַחֲמִים, פֶּטְרוֹס. יְרִידָתָהּ אֵינָהּ עַל הַטּוֹבִים. וְהִצַּלְתִּי עַמִּי עִם
שִׁפְטִי, עַד שֶׁיְּתֻוַּכְּחוּ עַל הָאֳמָנוּת, וְהַזּוֹעֲקִים עַל הָאֳמָנוּת ― דְּבָרַי וְאֵלֶיךָ.
הֵם הַזּוֹכִים, וְהֵם יוֹרְשַׁי בָּאָרֶץ. וְאַחַר הַתִּינֻק אֲצִילֵם כְּמוֹ שֶׁהִצַּלְתִּיךָ
