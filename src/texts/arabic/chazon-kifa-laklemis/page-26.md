‏עַל הָאֱמֶת הַבְּהִירָה, פֶּטְרוֹס. דַּע כִּי כְּשֶׁיֻּשְׁלְמוּ אַרְבָּעָה הַכִּסְּאוֹת
‏וְתֵיעָשֶׂה אֱמוּנָה אֶחָת, בַּעֵת הַהוּא תִּשְׁתַּנֶּה הַזֹּהֲרוּת וְיִבְטַל אוֹר הַלְּבָנָה
‏בַּמִּזְרָח, וְתִתְחַזֵּק הַשֶּׁמֶשׁ בְּזָהֳרָהּ וְתָאִיר כְּפַלַיִם מִמַּה שֶּׁהָיְתָה
‏בִּימֵי קֶדֶם. וְאִם הָיָה זֶה ― אוּמַר יוֹם הָעוֹלָם. וְאִם תְּפַלְּצוּ הָעַוְלָה הָעַמִּים,
‏וְהִתְקַפְּצוּ עַל הַמַּאֲמִינִים בָּרֹעַ, וְהָיוּ הַיְלָדִים מִתְקוֹמְמִים זֶה לָזֶה ― אֲזַי הִכִּירוֹ מִ־
‏פֶּטְרוֹס הַצַּעַר וְהַכְּאֵב לְעַמְּךָ. כִּי יוֹם הַגְּאֻלָּה אֲשֶׁר הוֹדַעְתִּיךָ עָלָיו
‏יִהְיֶה תְהִלּוֹת לִי מִפִּיּוֹת הָעוֹלָלִים וְהַיּוֹנְקִים, וּמֵהַנְּעָרִים זוּלָתָם,
‏שֶׁכֻּלָּם יִשְׁתַּחֲווּ וִישַׁבְּחוּ שְׁמִי וִיכַבְּדוּנִי, זוּלָתָם וּמֵהֶם.
‏וְתָחֵל הַתְּשׁוּבָה וְיִתַּם יוֹם הַגְּאֻלָּה. וְאִם הָיָה יוֹם הַגְּאֻלָּה ― הִתְפָּאֲרָה הָעוֹלָם וְהִתְבָּהֲרָה
‏מִכָּל קַצְוֵי הָאָרֶץ, וְתִתְמַלֵּא מִשִּׁבְחִי וּכְבוֹדִי. אַחֲרֵי כֵן יִמְלֹךְ
‏אִישׁ דַּ״לֶת שְׁמוֹ, יְכֻנֶּה הָאַמָּן בַּמַּאֲמִינִים. אַחֲרָיו יִמְלֹךְ אִישׁ דַּ״לֶת שְׁמוֹ,
‏מְעֻרְפָּל. וְאַחֲרָיו יִמְלֹךְ אִישׁ עָשִׁיר, עַד שֶׁיַּחְזְרוּ אַרְבָּעִים מֶלֶךְ. כִּי אֵלֶּה
‏הַמְּלָכִים יִהְיוּ מִן הַמַּאֲמִינִים לִפְנֵי הַבִּיאָה הַשְּׁנִיָּה. וְתִהְיֶה תְּקוּפַת עַמְּךָ,
‏פֶּטְרוֹס, שִׁבְעִים שָׁבוּעַ. אַחַר זֶה יָבוֹא בֶּן הַתַּנִּין בִּזְמַנּוֹ
‏הָרָע, אֲשֶׁר יִקְרָא לְשֵׁמוֹ בִּשְׁמִי. וְכַאֲשֶׁר יַגִּיעַ ― הִנֵּה הוּא עַד סוֹף הַפֶּרֶק.
