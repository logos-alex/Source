זוֹ הִיא יְדִיעַת הַצִּנְצֶנֶת שֶׁל הַשֶּׁמֶן הַטּוֹב אֲשֶׁר שָׁפְכָה אוֹתָהּ מִרְיָם
הַחוֹטֵאת עַל רַגְלֵי הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ בְּבֵית שִׁמְעוֹן.

שִׁמְעוֹן זֶה כְּבָר הִזְמִין אֶת אֲדוֹנֵנוּ אֶל בֵּיתוֹ. וְשָׁמְעָה בּוֹ מִרְיָם הַחוֹטֵאת, וְכִי הוּא
שָׁם בְּבֵית שִׁמְעוֹן. וְהָלְכָה וְקָנְתָה אֶת הַצִּנְצֶנֶת מֵאֵצֶל הָרוֹקֵחַ וֶהֱבִיאַתָּה.
וְכִי מוֹצָאָהּ הָיָה ― שֶׁהָיְתָה קְרוֹבָה לַמָּשִׁיחַ, אֲהוּבָה מִזְּמַן מֶלֶךְ נְבוּכַדְנֶצַּר אֲשֶׁר שָׁבָה אֶת יְרוּשָׁלַיִם
אֶל בָּבֶל. וּבָעֵת אֲשֶׁר נוֹלַד הַמָּשִׁיחַ מִמִּרְיָם הַבְּתוּלָה בְּבֵית לֶחֶם, בַּמְּעָרָה ― הָיְתָה גְּנוּזָה
בָּאָרֶץ. וּכְשֶׁבִּקְּשׁוּ שֶׁמֶן לִמְשֹׁחַ בּוֹ אֶת הַמָּשִׁיחַ בְּעֵת לֵדָתוֹ ― הוֹפִיעַ מַלְאַךְ הָאֱלֹהִים לַמְּיַלֶּדֶת
שָׁלוֹמִי, וְתַחַת אֵימָה מוֹלִיךְ הַשֶּׁמֶן. אָמַר לָהּ מַלְאַךְ הָאֱלֹהִים יִתְעַלֶּה: קְחוּ אוֹתוֹ. בָּא אֶל הַמָּקוֹם
אֲשֶׁר בּוֹ הַקֶּרֶן בְּעֵת שֶׁהָיָה גָנוּז. אָמַר לָהּ: קְחִי זֶה הִינָה מִזֶּה. וְכַאֲשֶׁר פָּתְחוּ הַצִּנְצֶנֶת ―
יָצָא מִמֶּנָּה רֵיחַ בָּאוֹתָהּ הָאֶרֶץ כֻּלָּהּ, וְהִרְגִּישׁוּ מִמֶּנָּה רֵיחַ גָּדוֹל, וְלֹא הֵרִיחוּ כָּמוֹהוּ. וְאָז
מָשְׁחוּ אוֹתוֹ בוֹ, וְאַחַר רָחֲצוּ אוֹתוֹ. וּמֵרֵיחוֹ הַטּוֹב נִשְׁאַר עַל פְּנֵי הַמַּיִם. וְאָז
לָקְחָה הַמְּיַלֶּדֶת שָׁלוֹמִי צֶמֶר וְלִקְּטָתוֹ, וְשָׂמַתּוּ בְּצִנְצֶנֶת, וְהָיָה נוֹדֵף מִמֶּנָּה רֵיחַ
טוֹב ― אֲשֶׁר לֹא הָיָה לְאִישׁ כָּמוֹהוּ וְלֹא כְּרֵיחוֹ, וְלֹא הָיָה זִיף שֶׁלּוֹ ― וְהָיָה מִתְמַלֵּא הַמָּקוֹם מִן רֵיחוֹ
הַזַּכּוּת שֶׁלּוֹ. וְלָקְחָה שָׁלוֹמִי אֶת הַצִּנְצֶנֶת וְנָתְנָה לְבַעֲלָהּ, וְהָיָה רוֹקֵחַ, וְשָׂם אוֹתָהּ בַּחֲנוּתוֹ
וְהָיָה מוֹכֵר יוֹתֵר מִכָּל אֲשֶׁר בַּשּׁוּק בִּגְלַל רֵיחַ הַשֶּׁמֶן הַטּוֹב. וְאָז מֵת
בַּעֲלָהּ.



                          סוֹף הַתַּעְתִּיק הַמָּצוּי
