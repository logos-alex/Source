‏שָׁתַק מִדְּבָרוֹ, וְהִשְׁתַּחֲוֵיתִי אֲנִי הֶחָלוּשׁ פֶּטְרוֹס לִפְנֵי אֲדוֹנִי וּבוֹרְאִי
‏וְרִבּוֹנִי וּמוֹשִׁיעִי יֵשׁוּעַ הַמָּשִׁיחַ, וְאָמַרְתִּי: אֲדוֹנִי וֵאלֹהֵינוּ, קַבֵּל תַּחֲנוּן עַבְדְּךָ
‏וְתַלְמִידְךָ אֲשֶׁר בָּחַרְתָּ בּוֹ וְקָבַעְתָּ אוֹתוֹ לִדְבָרְךָ הַטָּהוֹר מִלִּפְנֵי הַדּוֹרוֹת וְהַזְּמַנִּים,
‏וְקֻיַּם דִּבּוּרְךָ הַטָּהוֹר וְאִמְרָתְךָ הָאֱמֶת הַקְּדוֹשָׁה. וְהָשֵׁב אוֹתִי בְּקַבָּלָתְךָ, וְאַל תַּעֲלִם
‏יָדְךָ מִמֶּנִּי וְלֹא מֵעַמְּךָ. וְהוֹדִיעֵנִי הַיּוֹם אֲשֶׁר בּוֹ תִּהְיֶה הָרִיסַת וְכִלָּיוֹן
‏הַמְקֻלָּל. וְהוֹדִיעֵנִי אֶת אָבִיו וְאֶת אִמּוֹ וּמֵאֵיזֶה שֵׁבֶט יֵצֵא וְכַמָּה הָאוֹת בִּכְבוֹדוֹ
‏וְכַמָּה תְּקוּפַת מַלְכוּתוֹ בָּעוֹלָם, וּמַה יַּעֲשֶׂה מִן הָאוֹתוֹת וְהַפְּלָאוֹת, וְאֵיפֹה יִוָּלֵד וּבְ־
‏אֵיזֶה מָקוֹם יִמְלֹךְ, וְאֵיפֹה יֵשֵׁב בְּאֵיזֶה מִן הָאֲרָצוֹת תִּהְיֶה יְשִׁיבָתוֹ, וּמַה הָאוֹת
‏אֲשֶׁר תִּהְיֶה לִפְנֵי יְצִיאָתוֹ, עַד אֲשֶׁר יִשָּׁמְרוּ הַמַּאֲמִינִים. לְכָל זֶה הוֹדִיעֵנִי דְּבָרוֹ וְאַגֶּה
‏עִנְיָנוֹ. וְאָמַר לִי אֲדוֹנִי וֵאלֹהַי וְרִבּוֹנִי יֵשׁוּעַ הַמָּשִׁיחַ: פֶּטְרוֹס, הִנֵּה זֶה הַ־
‏מְקֻלָּל יִוָּלֵד מִשֵּׁבֶט דָּן הַתַּנִּין הָרוֹבֵץ בַּדֶּרֶךְ, כְּמוֹ שֶׁאָמַר מֹשֶׁה
‏הַנָּבִיא. מוֹלַדְתּוֹ בְּכוֹרָזִין וּבֵית צַיְדָה, וּבִכְפַר־נַחוּם, וְיֵשֵׁב בִּירוּשָׁלַיִם בְּבֵית דָּוִד
‏הַנָּבִיא. וְיִהְיוּ רֹב עַמּוֹ הַיְּהוּדִים, כִּי יִקַּח שִׁלְטוֹנוֹ וְיַהֲרֹג אֶת כֹּהֲנַי
‏וְיִתְנַקְּמוּ מֵהֶם. וְאֶמְחֹל גַּם בְּיָמָיו מִן הַחֲטָאִים לְאוֹתָם מֵעַמִּי הַהוֹלְכִים אַחֲרָי.
‏כִּי אֻמָּה אֲשֶׁר לֹא תַשְׁרֶנָּה אִמְּהוֹת חֲמוֹר הַפֶּרֶא וְהַזְּאֵב יְלָדֶיהָ,
‏וְיַעַבְדוּ לוֹ הַיְּהוּדִים וּבְנֵי הַצְּפָעִים וְהַשּׁוֹמְרֹנִים וְהַצַּאבִּים וְהַמָּגוֹשִׁים וְשֵׁאֲרִית בְּנֵי
‏הַזְּאֵב. וְיִהְיֶה רֹאשׁוֹ נוֹגֵעַ בְּעָבִים, וְיַעֲנוּ לוֹ הַמַּטָּרוֹת, וְיַגִּיעַ
‏עַד הַיָּם
