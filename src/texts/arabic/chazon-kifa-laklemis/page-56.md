‏בִּגְדֵי הַטְּבִילוֹת, וְחִנַּט בּוֹ אֶת גּוּף הַמָּשִׁיחַ, וְהוּא הַחַי בֶּן הָאֱלֹהִים
‏הַחַי הָעוֹלָמִי, אֲשֶׁר נָתַן לִי אֶת יָדוֹ בִּמְצוּלַת הַיָּם וְלֹא טָבַעְתִּי, וְלֹא יָסוּר כִּדְבָרוֹ. כִּי
‏אָמַר לִי: פֶּטְרוֹס, לֹא תִקַּח חֻלְשַׁת עֵינַי גַּם בְּעֵת טְבִילָתוֹ
‏בַּשִּׁשָּׁה לְכָנוֹן הַשֵּׁנִי. כְּזֵכֶר ― נִמְסַר אֵלֵינוּ יוֹחָנָן בֶּן זְכַרְיָה אֲשֶׁר הִטְבִּיל אוֹתוֹ בִּנְהַר
‏הַיַּרְדֵּן, וְהוּא אֲשֶׁר הֵעִיד כִּי רָאָה אֶת הָרוּחַ יוֹרֶדֶת מִן הַשָּׁמַיִם, וְהָרוֹאֶה לוֹ עַל דְּמוּת
‏הַזֹּהַר ― יָרְדָה עָלָיו כִּדְמוּת יוֹנָה לְבָנָה, וְהָיְתָה נוֹכַחַת. וְעוֹד חַג הָאֻמָּה ― אַרְבָּעָה־
‏עָשָׂר לֶאֱלוּל. וְהָיָה וְעָבַר אַחֲרֵי עֲלִיָּתוֹ אֶל הַשָּׁמַיִם אַחֲרֵי יוֹם תְּקוּמָתוֹ, אַחֲרֵי
‏תְּקוּמָתוֹ בַּיּוֹם הַשְּׁמִינִי, אֲשֶׁר נִכְנַס אֵלֵינוּ וְהָיִינוּ עִם הָעֲלִיָּה וְהַשְּׁעָרִים סְגוּרִים,
‏מְקֻבָּצִים. כְּבָר נִכְנַס אֵלֵינוּ, וְתוֹמָא שָׂם אֶצְבָּעוֹ בְּצִדּוֹ וְהֻגַּח בַּחֲנִית הַמְּטַיֶּלֶת, הֵיטֵיב,
‏וְהֶרְאָה אֶצְבְּעוֹתָיו עַד אֲשֶׁר אָמַר לוֹ: שֶׁתִּהְיֶה מַאֲמִין הִכַּרְתָּ. אָמַר לוֹ: אֲדוֹנִי וֵאלֹהַי. וְ־
‏חוֹזֵר עוֹד הוֹדַעְתָּנוּ מַה שֶּׁהוֹרַד עָלֵינוּ ― הַמַּשִּׁיכָה מִן הָאֵשׁ אֲשֶׁר עָבַרְנוּ בָּהּ וְהִכַּרְנוּהָ קֹדֶם. וְכָךְ
‏הַיּוֹם וּבַיּוֹם הָעֲשִׂירִי אַחַר עֲלִיָּתוֹ, וְהוּא מִן הָאַרְבָּעִים אַחַר טְבִילָתוֹ, וּבִגְלַל הַשִּׁלְטוֹן עַל הָאֲנָשִׁים
‏וְהָאֹכֶל וְחַג גּוּף הָאָדוֹן וּלֵדָתוֹ וְהַשִּׁבְעָה לוֹ, וְהַבְּהִירוּת לוֹ. וְיִהְיֶה מִקְרָאֲךָ
‏הַחַיִּים בְּבֹקֶר וְעֶרֶב, וְחֵצִי הַיּוֹם כָּל יוֹם רִאשׁוֹן, עַד שֶׁיִּתְאַוּוּ כָּל הַמַּאֲמִינִים אֶת אָהֳלָם.
‏וּדְבָרוֹ וְהַנְהָגַת תְּשׁוּעָה לְאָדָם וּלְזַרְעוֹ. וְהַבֵּט עַתָּה, בְּנִי קְלֵמֶנְס, וְהֵא צַוָּאתִי, וּשְׁמֹר
‏אֶת כָּל מַה שֶּׁצִּוִּיתִיהוּ, כְּמוֹ שֶׁשָּׁמַר יְהוֹשֻׁעַ בִּן נוּן צַוָּאַת רַבּוֹ מֹשֶׁה. וִיהִי גַּם
‏בְּנִי קְלֵמֶנְס מַזְכִּיר אֶת כָּל הֵיכָל שֶׁתִּבְנֶהוּ. וְחָתוֹם עַל דִּבְרִי. וְהָאָב וְהַבֵּן וְרוּחַ הַקֹּדֶשׁ ― אֵל אֶחָד, אָמֵן.



‏נִסְפָּח ― טֶקְסְט נוֹסָף (אֵינוֹ מִסֵּפֶר הַמְּגִלּוֹת ― טֶקְסְט עַרְבִי נוֹצְרִי אַחֵר:
              אַגָּדַת בֹּשֶׂם הַמּוֹר / מַעֲשֵׂה צִנְצֶנֶת הַשֶּׁמֶן שֶׁל מִרְיָם הַחוֹטֵאת)
