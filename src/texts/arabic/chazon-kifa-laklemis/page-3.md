וּבְלֵדָתוֹ יִהְיֶה לוֹ מְבַשֵּׂר בְּכָל יוֹם, וְהַמְּלָכִים אֲשֶׁר יָבִינוּ דָּבָר זֶה יִכָּנְעוּ
וְיִשְׁתַּחֲווּ בְּכָל יוֹם, וִיהַלְלוּהוּ וִיגַדְּלוּ זִכְרוֹ, וְיַעֲשׂוּ אֶת זִכְרוֹ בֵּינֵיהֶם,
וְיִכְתְּבוּ שְׁמוֹ עַל קִירוֹתֵיהֶם וְעַל בָּתֵּיהֶם, וְיַעֲלוּ לְרֶגֶל אֶל מְקוֹם קִבְרוֹ. וְעוֹד הַבַּיִת
שֶׁל הַפֶּסֶל הַגָּדוֹל הַנִּקְרָא ׳סוּחַ׳, כִּי בְּבֵית מָקוֹם זֶה שְׁלֹשָׁה פְּסִילִים, וִיהִי
הָאֵלֶּה בּוֹנִים בָּתִּים לַעֲבוֹדָתוֹ וְיִשְׁתַּחֲווּ לוֹ כְּנֶגֶד אֲשֶׁר צִוִּיתִי. וְיֵלֵךְ אַחַר חֲמוֹר הַפֶּרֶא,
מִן הַיְּהוּדִים, עַם אֲשֶׁר יְבַקֵּשׁ בּוֹ הַשְׁחָתָה, וְיַעֲשׂוּ עָוֶל וְהַשְׁחָתָה רַבָּה, וְיַשְׁמִידוּ אֶת הַכִּכָּר וְאֶת [?],
אֲשֶׁר יִהְיֶה עִמּוֹ בְּרֵאשִׁית עִנְיָנוֹ; וְיֵלְכוּ אַחֲרָיו שְׁנֵי אֲנָשִׁים מִן הַיְּהוּדִים, רֵאשִׁית
שֵׁם הָאֶחָד כָּ״ף, כָּ״ף, שָׁלוּחַ הָעָם. שְׁמוֹ אַחַר
מוֹתוֹ, וִיסַפְּרוּ אֶת קוֹרוֹתָיו וִיחַפְּשׂוּ אֶת דְּבָרָיו הַכְּתוּבִים אֵצֶל אֲחֵרִים. וּשְׁמוֹ
שָׁמִין, וּמְקוֹר מוֹצָאוֹ מִן הַמִּזְרָח. וְיִהְיוּ אֵלּוּ הַשְּׁנַיִם מְחַבְּרִים לַחֲמוֹר הַפֶּרֶא
סֵפֶר מְחֻבָּר מִכָּל הַסְּפָרִים, וְיִהְיוּ הָאֵלֶּה הַשְּׁנֵי יְהוּדִים
כּוֹפְרִים מִצַּד אֶחָד וּמַאֲמִינִים בִּסְפָרִי מִצַּד אַחֵר. וְיִתְאַחֵד הָאִישׁ הַזֶּה עִם
שְׁנֵים־עָשָׂר תַּלְמִידִים מְדֻמִּים בִּדְמוּת. אוֹי לַחֲכָמִים הַחֲנֵפִים הַיְּהוּדִים אַחֲרָיו,
אוֹי לָאִישׁ הָרָשָׁע אֲשֶׁר יִהְיֶה לְפָנָיו. וְאַחַר כֵּן יֵעָשׂוּ מִסַּפְּרֵי שִׁבְחוֹ הָאֵלֶּה אֲשֶׁר
יִקָּרְאוּ עַל שְׁמוֹ, שְׁלֹשָׁה רָעֵי הַחֲנֵפִים הָאַרְדִיָּא; כִּי הֵם יַשְׁחִיתוּ אֶת אֲשֶׁר הָיָה
