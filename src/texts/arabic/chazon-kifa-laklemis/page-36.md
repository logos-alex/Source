עַצְמוֹתֵיהֶם זֶה מִזֶּה, וְיֻגְלוּ מִמִּשְׁכְּנוֹתָם, וְלֹא אֶעְזֹב אֶתְכֶם.
וְאָשִׂים כֹּחִי בָּהֶם וְאַרְאֶה פְּלָאוֹתַי בָּהֶם. וּכְשֶׁעֻלִּיתָ, פֶּטְרוֹס, לְ־
הִתְקַבֵּל ― עֲלִיתִי אֶל אָבִי וְצִוִּיתִי לְרוּחַ הַקֹּדֶשׁ, אֲשֶׁר כְּשֶׁעָזַב בְּנָהּ הוֹשַׁבְתִּיהָ עַל
כִּסֵּא תְּהִלָּתִי, וְשִׁלַּחְתִּי אֵלֵיכֶם וְאֶל כָּל תַּלְמִידַי וּמְכַתְּבֵיכֶם, שֶׁתְּהֵא
עִמָּכֶם וְלֹא תָסוּר מֵעֲלֵיכֶם, וְתִהְיֶה כֹּחִי מַקִּיף אֶתְכֶם עַד אַחֲרִית הַזְּמַן. אַל תַּעֲצֵב, פֶּטְרוֹס.
הַכֹּהֲנִים אֲשֶׁר שָׁפְטוּ בָּאֱמֶת, וְאֲשֶׁר אֲנִי תֵּאַרְתִּי לְךָ מִן הַפַּטְרִיאַרְכֵים וְהַמִּטְרוֹפוֹלִיטִים וְהָאֶפִּיסְקוֹפִּים,
וְעָשׂוּ אֶת חֻקַּתִי וְהִכְרִיזוּ בִּבְשׂוֹרָתִי, וְקָרְאוּ אֶת עַמִּי אֶל מַה שֶּׁחוֹבָה עֲלֵיהֶם מִן הָרְעוֹת
וְשָׁפְטוּ בָּאֱמֶת, וַאֲשֶׁר תֵּאַרְתִּי לְךָ מִן הַצֶּדֶק וְהָאֲמוּנָה, וְלֹא נָטוּ לְאַהֲבַת
הַזָּהָב וְהַכֶּסֶף, וְהָיוּ בָהֶם רַחֲמִים עַל הָאֶבְיוֹנִים, וְהַמַּאֲמִינִים בִּי עִמָּכֶם יֵשְׁבוּ
עַל נְעִימֵי הַכִּסְּאוֹת הַמְּאִירִים בִּירוּשָׁלַיִם הַשְּׁמֵימִית, אֲשֶׁר הֵם עִמִּי
בְּמַלְכוּתִי יִתְעַנְּגוּ, פֶּטְרוֹס. כָּל מִי שֶׁנִּטְבָּל בַּמַּיִם וּבָרוּחַ, וְלֹא הִשְׁתַּמֵּשׁ
בַּהַתְעַסְּקוּת הַמַּפְרִידָה אֶל שׂבַע חֲטָאִים ― אֲנִי אֲבָרֵךְ אוֹרְךָ פָּנַי, וְאַרְאֶה לוֹ אוֹר
אֱלָהוּתִי, וְיָאִיר פָּנָיו, כְּמוֹ שֶׁהִתְרָאֵיתִי לְךָ, פֶּטְרוֹס, וּלְיַעֲקֹב וְלִיהוֹחָנָן עַל הַר
תָּבוֹר, כְּשֶׁכִּנַּסְתִּי אֶתְכֶם עִם מֹשֶׁה וְאֵלִיָּהוּ בָּהָר בַּעֲנָנָה, וְשָׁמַעְתֶּם קוֹל
צִדְקְךָ, וּכְמוֹ שֶׁהִבְרִיק אוֹרִי עַל הַקֶּבֶר עַל מִרְיָם הַמַּגְדָּלִית כְּשֶׁהִיא מִשְׁתַּחֲוָה מִמְּשַׁךְ
הָאֲבַדּוֹן. כֵּן אַרְאֶה לָהֶם אוֹרִי, וְיִשְׂמְחוּ שִׂמְחָה גְּדוֹלָה. וְאֵין, פֶּטְרוֹס,
