‏יִלְכוּ וְיֵעָמְרוּ, וְהַחַיָּה וְהָעוֹף ― כְּשֶׁיִּקְרָאֵם יָבֹאוּ אֵלָיו בְּרָצוֹן, וִירַפֵּא
‏אֶת הָאִלְּמִים, וְהַמְצֹרָע יְטַהֲרֵהוּ מִצַּרַעְתּוֹ, וְיָקִים אֶת בַּעַל־הַזְּמַן כָּאַיָּל,
‏וְיֵלֵךְ עַל הַמַּיִם, וְיוֹצִיא הַמַּיִם מִן הָאָרֶץ הַיְּבֵשָׁה, וְיֹאמַר: אֲנִי אֱלֹהֵי
‏אַבְרָהָם וֵאלֹהֵי יִצְחָק וֵאלֹהֵי יַעֲקֹב, וְיִשְׂרֹף בָּאֵשׁ מִי שֶׁמַּכְעִיסוֹ. [בֵּין הַשּׁוּרוֹת: פֶּטְרוֹס]
‏הַצֵּל כְּמוֹ אֶת הַמַּאֲמִינִים בִּי מִן הַיָּמִים הָאֲפֵלִים הָהֵם. וְאִם רָאִיתָ הַשֶּׁמֶשׁ שֶׁכְּבָר נֶחְשְׁכָה [בְּגִלָּיוֹן: זְהִירוּת]
‏וְהַיָּרֵחַ ― לֹא יִתֵּן אוֹרוֹ. אִם יֵאָמֵר: הִנֵּהוּ יוֹצֵא בַּמִּדְבָּר ― אַל תֵּלְכוּ אֵלָיו [בֵּין הַשּׁוּרוֹת: זְהִירוּת]
‏עַם מִבֶּן הָאֲבַדּוֹן ― סָר רַע, יִמְלֹךְ. לוּלֵא אֲנִי אֶעֱשֶׂה אוֹתָם
‏הַיָּמִים שֶׁיִּתְקַצְּרוּ ― לֹא הָיָה חַי אֶחָד מִן הַמַּאֲמִינִים. אֶלָּא אֲנִי, בִּגְלַל הַנִּבְחָרִים,
‏עָשִׂיתִי אֶת הַיָּמִים קְצָרִים. כְּבָר שָׂמַח הָאָדָם אֲשֶׁר יִסְבֹּל עַל הַמִּלְחָמָה בְּבֶן הָאֲבַדּוֹן
‏וְיוֹכִיחוֹ. מְבֹרָךְ הוּא הַנֶּהֱרָג עַל יָדָיו, כִּי הוּא מַעֲלָה מִן הָעֵדִים
‏אֲשֶׁר נֶהֶרְגוּ וְנִשְׁחֲטוּ מִקֶּדֶם. אַשְׁרֵיהוּ [בֵּין הַשּׁוּרוֹת: פֶּטְרוֹס]. סוֹד, וְכִי אֵיךְ הָעֱלַמְתִּי [בֵּין הַשּׁוּרוֹת: מִמְּךָ]
‏דָּבָר מִסּוֹדוֹתַי, וְאַל תַּגֶּה לְךָ עָלָיו. הִיא הוֹדִיעַ הַמְקֻלָּל הַמְלֻעָז אֲשֶׁר הוֹדַעְתִּי
‏עָלָיו, חֲבֵרְךָ [בֵּין הַשּׁוּרוֹת: יֵשׁוּעַ הַמָּשִׁיחַ]. וְאַחַר כֵּן אָמַר חֲבֵרִי [בֵּין הַשּׁוּרוֹת: פֶּטְרוֹס]: זֶה הוּא מִן הָרָע
‏הַשְּׁלוּחִים לַמַּאֲמִינִים, הַמֻּפְקָדִים בְּמִי שֶׁנִּטְבָּל בְּשֵׁם הָאָב וְהַבֵּן וְרוּחַ הַקֹּדֶשׁ.
‏מֵעֵת בְּרִיאָתִי הָיִיתִי קוֹבֵעַ בְּמִי שֶׁנִּטְבָּל בַּטְּבִילָה ― מַלְאָכִים, וּמַלְאָךְ אֶחָד מִן
‏יְמִינוֹ וְהָאַחֵר מִשְּׂמֹאלוֹ, וְשׁוֹמְרֵיהֶם וְהַמַּקִּיפֵיהֶם מִכָּל הַכּוֹחוֹת. וְאוּלָם
