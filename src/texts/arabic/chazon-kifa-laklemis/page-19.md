וְהוּא יִפְתַּח אֶת שַׁעֲרֵי יְרוּשָׁלַיִם בְּיוֹם הַשִּׁשִּׁי, וְיִכָּנֵס בָּהּ בְּשָׁלֹשׁ
שָׁעוֹת מִיּוֹם הַשַּׁבָּת. וְאִם יִהְיֶה יוֹם הָאֶחָד, וְהוּא הַשְּׁלִישִׁי שֶׁל שְׁתֵּי פְּעָמִים שְׁתַּיִם ―
יִכָּנֵס לַהֵיכָל, וְיַעֲמִיד עֵץ צְלִיבָתִי בִּמְקוֹם גָּלְגָּלְתָּא אֲשֶׁר
שָׁם הִצְלִיבוּנִי הַיְּהוּדִים, וְיָשִׂים אֶת כֶּתֶר מַלְכוּתִי עַל הָעֵץ. וְאוֹתוֹ הַיּוֹם
יִהְיֶה יוֹם הָאָדוֹן, כִּי בּוֹ תִּשְׂמַחְנָה שְׁאָר הָאֻמּוֹת וּתְרַקַּדְנָה, וְיִתְקַבְּצוּ אֶל
שָׁם מַלְכֵי הָאָרֶץ, וְיִשְׂמְחוּ וְיִשְׁתּוּ וִיחַבְּקוּ זֶה אֶת זֶה בְּכַפַּיִם, וְתַכִּיר
הָאֻמּוֹת זוֹ אֶת זוֹ, וְתָגֵל בּוֹ הַשָּׁמַיִם וְהָאָרֶץ. וְתִשָּׁמַע
קוֹלוֹת הַמַּלְאָכִים בְּהִלּוּל וְזַמְּרָה, וְיִתְעָרְבוּ קוֹלוֹתֵיהֶם בְּשִׁבְחֵי הַמַּאֲמִינִים,
וְיֹאמְרוּ כֻּלָּם: זֶה הוּא יוֹם הָאָדוֹן אֲשֶׁר הָיָה צָפוּן. בּוֹאוּ עִמָּנוּ,
נִשְׂמַח וְנָגִילָה בוֹ, וְנִכְבָּד הַכֹּל מִי שֶׁהוּא מַאֲמִין וְחַי [?] הַיּוֹם
שֶׁיַּבִּיט אֶל זֶה הָעוֹמֵד בְּעֵינָיו ― אוֹר אֲשֶׁר עוֹלֶה עַל אוֹר הַשֶּׁמֶשׁ וְזָהֳרוֹ,
עַד שֶׁיִּהְיֶה בַּהַזְהָרָה כְּקֶרֶן הַלְּבָנָה הַבְּהִירָה. וְאוֹתוֹ הַיּוֹם תִּתְקַבֵּץ
בִּרְצוֹנִי אֶל מַלְכוּת צִיּוֹן כָּל הָאֻמּוֹת וְיִהְיוּ לִי בְּשִׂמְחָה רַבָּה,
כִּי בּוֹ אֶמְחֹל הַחֲטָאִים לִשְׁאָר הַמַּאֲמִינִים, וְאָאִיר פְּנֵיהֶם, וְאֶעֱשֶׂה לָהֶם
שָׁם סֻכַּת מַאֲכָל וּמַשְׁקֶה, וְיִשְׂמְחוּ בָּהּ וִיחַבְּקוּ זֶה אֶת זֶה בְּכַפֵּיהֶם
כְּמוֹ אֲשֶׁר הִלְלוּ
