בְּהַשְׁכֵּל מְקוֹמָהּ אֶצְלִי, אוֹרִי כְּבָר הֲכִינוֹתִיהָ לְנַפְשׁוֹת הַצַּדִּיקִים [?] כְּבָר
הֶעֱתַקְתִּי לְךָ מַפְתְּחוֹת מַלְכוּת הַשָּׁמַיִם וְהָאָרֶץ, וְנָתַתִּי כֻלָּן בְּיָדְךָ.
כִּי הָיִיתָ עָשׂוּי בַּעַל הַבַּיִת אֲשֶׁר קָרָאתִי לְהִכָּנֵס בּוֹ ― זֶה הַעֵדֶן אֲשֶׁר תֵּאַרְתִּי
לְךָ, וְהַמָּדוֹרוֹת הַמְּאִירוֹת אֲשֶׁר בְּמַלְכוּת הַשָּׁמַיִם ― מוּכָנוֹת כֻּלָּן לְמִי שֶׁמַּאֲמִין
בִּי וְהוֹלֵךְ אַחֲרַי, פֶּטְרוֹס. כְּשֶׁכָּלָה אֲדוֹנִי מִדִּבּוּרוֹ ― הִשְׁתַּחֲוֵיתִי
אֲנִי, פֶּטְרוֹס, לְפָנָיו וְאָמַרְתִּי: רַחֵם עַל עַבְדְּךָ וְאַל תִּדֹּם אוֹתִי מִדִּבּוּרוֹ. וַיֹּאמֶר אֵלַי יֵשׁוּעַ
הַמָּשִׁיחַ בֶּן הָאֱלֹהִים: דַּבֵּר בְּמַה שֶּׁאָהַבְתָּ, כִּי אֵין דָּבָר שֶׁתִּשְׁאָלֵנִי עָלָיו. וַאֲמַרְתִּי לוֹ: אֲדוֹנִי,
הוֹדִיעֵנִי מַה אֹמַר לְרֹב מַעֲלוֹת הַמַּאֲמִינִים בְּךָ, מִן הַצַּדִּיקִים וְהַנְּבִיאִים וְזוּלָתָם, וּמַה
יִהְיֶה מִן הַמַּאֲמִינִים בְּךָ לִפְנֵי הַבִּיאָה הַשְּׁנִיָּה. וַיֹּאמֶר אֵלַי רִבּוֹנִי וֵאלֹהַי יֵשׁוּעַ הַמָּשִׁיחַ:
הִנֵּה, פֶּטְרוֹס, אֲנִי עוֹנֶה לִשְׁאֵלָתְךָ זוֹ, וְאוֹדִיעֵךָ מַעֲלוֹת הַמַּאֲמִינִים מִן
הַנְּבִיאִים וְהַצַּדִּיקִים וְזוּלָתָם, אַחֲרֵי עֲלִיָּתִי מִן הָאָרֶץ אֶל מְקוֹמִי, אֶל תְּהִלָּתִי וּכְבוֹדִי אֶל
הַשָּׁמַיִם. אַשֶׁר מַעֲלוֹת הַטְּהוֹרִים הַצַּדִּיקִים אֲשֶׁר נִטְבְּלוּ בִּשְׁמִי וְהֶאֱמִינוּ בִּי וְשָׁבוּ מֵ־
חֲטָאֵיהֶם לִפְנֵי פְּטִירָתָם ― הַפַּרְדֵּס אֲשֶׁר נְטַעְתִּיו בְּאֶרֶץ הַנֹּעַם, שָׁם
אוֹשִׁיבֵם, וּבוֹ יִהְיֶה נֹעַמָם לְעַד אֲשֶׁר אֵין לוֹ קֵץ. וְאוּלָם אַתֶּם אֲשֶׁר
הִנַּחְתֶּם כָּל מַה שֶּׁהָיָה דַּ״ל לִהְיוֹת ― עִמִּי בְּמַלְכוּת הַשָּׁמַיִם אֲשֶׁר אֵינָהּ מַשִּׂיגָהּ
אֵימָה גְּדוֹלָה. אַחַר עֲלִיָּתִי לַשָּׁמַיִם ― יֵהָרְגוּ בָּאָרֶץ וְיִשָּׂרְפוּ בָּאֵשׁ וְיִקְצְצוּ
עַצְמוֹתֵיהֶם
