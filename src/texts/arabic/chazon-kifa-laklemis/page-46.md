וּבַשָּׁעָה הַשִּׁשִּׁית ― בָּהּ עָבַר עַל הַמִּצְוָה וְהִתְהוֹלֵל לַחֲשֹׁק
הָאֱלָהוּת, וְשָׁבַר אוֹתוֹ הַשָּׂטָן הַמִּשְׂתַּטֵּן. וּבַשָּׁעָה הַתְּשִׁיעִית
גֵּרַשְׁתִּי אוֹתוֹ מֵחַסְדִּי, וְשַׂמְתִּי שִׁכְנוֹ עִם הַחַיּוֹת הַשּׁוֹמְמוֹת, כִּי
לֹא הֵבִין אֶת הַכָּבוֹד וְעָבַר עַל הַמִּצְוָה. וּבַשָּׁעָה הָאַחַת־עֶשְׂרֵה, בְּהַזְמָנָתִי
וּבְרֹב אַהֲבָתִי לִתְשׁוּעָה ― בַּלַּיְלָה הוּא, יַעַל הַמַּלְאָכִים בְּ־
שֶׁבַח, רוֹצֶה לוֹמַר: כִּשּׁוּף; כִּי בָּהּ הִתְחַלְתִּי בְּבָרֹא הַזֹּהַר. וּכְבָר בָּחַרְתִּיךָ
פֶּטְרוֹס, חַסְרֵה־לִי חֵיק הַכּוֹכָבִים וְהַחֻקָּה הָעַתִּיקָה אֲשֶׁר חִלַּקְתִּיהָ
לְאוֹתָם אֲשֶׁר מָרְדוּ וְעָבְרוּ עָלַי, וְלֹא יַאֲמִינוּ בִּי. וְכִי כְבָר בִּטַּלְתִּי מִן הָעָם הַשַּׁבָּת
אֲשֶׁר נָתַתִּי לָהֶם, וְשַׂמְתִּי תַּחְתֶּיהָ אֶת יוֹם הָרִאשׁוֹן, וְהוּא הַיּוֹם אֲשֶׁר הִצַּלְתִּים בּוֹ
מִן הַגֵּיהִנּוֹם, וְחִדַּשְׁתִּי חַיָּיו וְאַהֲבָתוֹ אֵצֶל מַסַּל הַחֵטְא, פֶּטְרוֹס. הָעֵר
אֶת הָעָם, שֶׁיִּהְיֶה עֵרוּי, פֶּן יָבוֹא בַּמְּצוּקוֹת ― הַנִּסְיוֹנוֹת רַבּוֹת הַיְּגִיעוֹת,
פֶּן יָבוֹא בְּעַבְדוּת הַזֵּזָנִים, שֵׁבֶט הַחֵמָה. טוּב לָעָם אֲשֶׁר לֹא יַשִּׂיגֵהוּ.
וְטוּב לְמִי שֶׁנִּמְלַט מִמֶּנּוּ וְהֻצַּל מִמַּלְכוּת הַזֵּזָן הָרָע. הַטּוּב לְמִי
שֶׁסָּבַל עַל עִנּוּי הַזֵּזָן, וְעָמַד וְלֹא נִכְשַׁל בְּמַה שֶּׁבּוֹ ― כִּי אֶתֵּן לוֹ אֶת הַטּוֹבוֹת
