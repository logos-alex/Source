‏מִצְּלִיבָתִי ― אוֹרִישֵׁם הָאֵשׁ אֲשֶׁר לֹא תִכְבֶּה שַׁלְהֲבוֹתֶיהָ. שָׁם יְעֻנּוּ הַיְּהוּדִים,
‏בְּלֹא תַם. וְאַשֶׁר בְּנֵי הָעִנּוּי אֲשֶׁר הֲכִינוֹתִי לָהֶם בְּאוֹצְרוֹת זַעֲמִי ― הֲרֵי
‏אוֹדִיעֲךָ, פֶּטְרוֹס: הֲכִינוֹתִי לָהֶם מִן הַתּוֹלָע אֲשֶׁר אֵינוֹ מָלֵא הַסְּדוּדָה,
‏וְלֹא יָמוּת, וְלֹא יֹאכַל אֶת בְּשָׂרָם עַד שֶׁיַּגִּיעַ עַל עֶצֶם. כִּדְלִיָּתוּ יָמִית גּוּפוֹ ―
‏הוּא בּוֹ, וּמֵהֶם, פֶּטְרוֹס, וּמֵהֶם מִי שֶׁאֲנִי מְגָרְשֵׁם אֶל הַחֲשֵׁכָה הָרְחוֹקָה אֲשֶׁר אֵין
‏בָּהּ אוֹרִי. רֹאשׁ עַ״יִן בָּהָר ― הִיא חֲשֵׁכָה כֻלָּהּ, וְתַחְתֶּיהָ זִפְתָה וּמְכֻעֶרֶת
‏הַמַּהֲרֶגֶת מִי שֶׁמְּרִיחָהּ. וּמֵהֶם מִי שֶׁאֲנִי דָן בְּדִינִים רְחוֹקִים, וְאִם יִרְצֶה כְּלוֹתָם
‏אֲשֶׁר אַחֲרֵיהֶם ― יִוָּלֵד עָלָיו מַחֲשֶׁבֶת אֲחֶרֶת, וְיִהְיֶה בְּעִנּוּי אֵין לוֹ קֵץ. וּמֵהֶם
‏מִי שֶׁיִּהְיֶה קָשׁוּר יָדַיִם וְרַגְלַיִם בַּחֲבָלִים שֶׁל אֵשׁ, לֹא יִנָּצֵל מֵהֶם לְעוֹלָם, וְאֵין לְעִנּוּיָם
‏קֵץ. וְהָאַחֵר מֵהֶם מִי שֶׁאֲנִי מוֹצִיאוֹ מִן הַשְּׁעָרִים וְהִקִּיפַתּוּ הַחֹרֶף כְּמוֹ
‏קֶרֶן הַשֶּׁמֶשׁ לָרֹעַ. וּמֵהֶם מִי שֶׁאֲנִי מַשְׁלִיכוֹ בְּתוֹךְ הָרַעַד הָאָיֹם, וְהוּא בְּ־
‏עִנּוּי אֵין סוֹף. וּמֵהֶם מִי שֶׁעֲנָנִי לוֹ בְּגֵיהִנּוֹם בָּאֵשׁ אֲשֶׁר לֹא תִכְבֶּה, כִּי כָּפְרוּ וְנָאֲחוּ
‏אַחַר הָאֱמוּנָה, וְלֹא טְבִילָה וְלֹא יִהְיֶה לָהֶם מָנוֹחַ מֵעִנְיָנִי לְעוֹלָם, וְלֹא מֵלִיץ
‏אֵצֶל הִשְׁתַּמְּטוּת הַטְּבִילָה מֵהֶם וּכְלָל הַמִּירוֹן אֲשֶׁר אֲנִי אֶגְזֹר לָהֶם בּוֹ,
‏וְיִשָּׁאֲרוּ בָּעִנּוּי הַגָּדוֹל כֻּלָּם עַד תַּכְלִית, לֹא יִכְלוּ לְעוֹלָם.

‏פֶּטְרוֹס,
