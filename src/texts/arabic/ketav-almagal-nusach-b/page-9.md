הוּא בֵּית הַתְּפִלָּה (כְּנֶסֶת) שֶׁל הָאֵל הַפּוֹנֶה מִן הַמִּזְרָח לְרוּם שְׁמוֹנֶה מַעֲלוֹת מִמַּעֲלוֹת הַשֶּׁמֶשׁ. וְהוּא רַחֲמֵי הָאֵל אֲשֶׁר הָבְטְחוּ בָּם בְּנֵי הָאָדָם, וּשֶׁתִּהְיֶה יְשׁוּעָתָם בָּהֶם. כִּי הָאֵל יִתְבָּרַךְ שְׁמוֹ יָדַע מִקֶּדֶם בִּידִיעָתוֹ מַה שֶּׁעָתִיד הַשָּׂטָן לַעֲשׂוֹת בְּאָדָם, וְהוֹשִׁיב לְאָדָם בְּמִשְׁכַּן רַחֲמָיו — כְּמוֹ שֶׁנִּבָּא דָּוִד וְאָמַר: "אֲדוֹנָי, אַתָּה הִחְיֵיתָנוּ אֶל בֵּיתֵנוּ בְּתוֹךְ רַחֲמֶיךָ. עָשִׂיתָ זֹאת לְהוֹשִׁיבֵנוּ בּוֹ, וְהוּא מִקֶּדֶם הַדּוֹרוֹת". וְאָמַר עוֹד בִּתְפִלָּתוֹ עַל בְּנֵי הָאָדָם: "זְכֹר ה' אֶת עַמְּךָ אֲשֶׁר הִכְשַׁרְתָּ לְהַכָּרָתְךָ, וְעָשִׂיתָ אוֹתָם בְּיָדְךָ הַקְּדוֹשָׁה" (= הַכַּוָּנָה לְדָוִד עַל בְּנֵי הָאָדָם). כִּי הָאֵל אָהַב לְהַשְׁפִּיעַ רַחֲמָיו עַל כָּל בְּנֵי הָאָדָם וְעַל מִינֵנוּ הֶחָלוּשׁ הַזֶּה.

וְזֹאת הִיא כְּנֶסֶת הָאֵל, וְהַפַּרְדֵּס אֲשֶׁר בָּהּ הוּא מִזְבַּח הַנְּחָמוֹת אֲשֶׁר נִמְשָׁכִים בּוֹ הַחַיִּים אֲשֶׁר הֵכִין הָאֵל לַמַּאֲמִינִים וּלְכָל הַקְּדוֹשִׁים, נֶצַח הַדּוֹרוֹת.

וּמִפְּנֵי שֶׁאָדָם הָיָה מֶלֶךְ וְכֹהֵן וְנָבִיא, הִכְנִיסוֹ הָאֵל אֶל תּוֹךְ הַפַּרְדֵּס לְשָׁרֵת בְּתוֹךְ עֵדֶן — בֵּית הַתְּפִלָּה הַקָּדוֹשׁ שֶׁל הָאֵל. כְּמוֹ שֶׁהֵעִיד עַל זֶה מֹשֶׁה בְּאָמְרוֹ: אַשְׁרֵי לָאִישׁ אֲשֶׁר יְזַכֵּהוּ הָאָדוֹן לְהִתְבַּשֵּׂם בַּשֵּׁם הָאָצִיל הָנֶהְדָּר, וְיִשְׁמֹר אֶת הַמִּצְווֹת אֲשֶׁר בִּגְלָלָן יִכָּנֵס אֶל רַחֲמֵי הָאֵל אֲשֶׁר נִכְנְסוּ בָּהֶן אָדָם וְחַוָּה — וְהִיא כְּנֶסֶת הָאֵל.

וְהָאֵל נָטַע אֶת עֵץ הַחַיִּים בְּתוֹךְ הַפַּרְדֵּס, וְהוּא דְּמוּת הַצְּלָב אֲשֶׁר פָּרַשׂ עָלָיו אֶת יָדָיו מוֹשִׁיעַ הָעוֹלָם הָרִבּוֹן הַמָּשִׁיחַ.

וְהֶחֱטִיא [הַשָּׂטָן] אוֹתָהּ [אֶת חַוָּה] לְמַעַן יִמְשֹׁךְ אֵלָיו אֶת חַוָּה. וְאָמַר לָהּ וְהִדְבִּיק אוֹתָהּ בָּהֶן בַּתּוֹעֲלוֹת: "אִם אַתְּ תֹּאכְלִי מִמֶּנּוּ — תִּהְיִי כֵאלֹהִים יוֹדַעַת טוֹב מֵרָע". וְהִלְהִיב לָהּ אֶת הַטַּעַם, וְעָשָׂה לָהּ כְּלָלִים בַּמַּעֲשִׂים וּבַתֹּאַר, וְחָשְׁקָה בָּהֶן וּבַתֹּאַר אֲשֶׁר תֵּאַר אוֹתָהּ בּוֹ. וְנֶחְפְּזָה לְקַטֵּף מִמֶּנּוּ וְלִקְצֹר אוֹתוֹ — לַאֲכִילָה. וּמְהָרֶגַע שֶׁאָכְלָה — נִתְעָרְטָלָה מִכְּבוֹד הָאָדוֹן, וְנִגְלָה גּוּפָהּ כֻּלָּהּ מְגֻלֶּה.

אַחַר כָּךְ נָטְלָה מֵאוֹתוֹ הָעֵץ — רוֹצֶה לוֹמַר מִן הַפְּרִי — וְנָסְעָה אֶל אָדָם, וְנָתְנָה לוֹ מִן הַפְּרִי, וְהוֹדִיעַתּוּ שֶׁאִם יֹאכַל מִמֶּנּוּ יִהְיֶה אֵל.

    [שׁוּלַיִם]: נִפְרֶדֶת הַמַּעֲלָה הַזַּכָּה כְּשֶׁאוֹכֵל מֵעֵץ זֶה. כָּאן צָרִיךְ הָאָדָם לְהִתְבּוֹנֵן שֶׁהַשָּׂטָן בַּתַּחְבּוּלוֹת הִדִּיחַ
