קַבְּלִיו וְשִׁמְרִיו, כִּי בְּהֶכְרֵחַ יִשָּׁמֵר וְיִגָּרֵס עִם הַזְּמַן, וְיַסְפַּק וְיֻגַּשׁ הַכֹּל לְבֶן הָאֵל כַּאֲשֶׁר יָבוֹא אֶל הָעוֹלָם, לְמַעַן יִהְיֶה הַזָּהָב סִימָן לַמַּלְכוּת, וְהַלְּבוֹנָה לִקְטֹרֶת לְפָנָיו, וְהַמֹּר לְ"חֲנִיטַת" גּוּפוֹ הַלָּקוּחַ מִמֶּנּוּ. וְיִהְיֶה זֶה עֵדוּת עַל הַבְּרִית אֲשֶׁר בֵּינִי וּבֵינֵךְ, בְּמוֹשַׁב נוֹכְחוּת מוֹשִׁיעֵנוּ, כַּאֲשֶׁר יוֹפִיעַ בְּעוֹלָמֵנוּ זֶה. וְקָרָא אָדָם לְאוֹתָהּ הַמְּעָרָה "מְעָרַת הָאוֹצָרוֹת" — כִּי הִפְקִיד בָּהּ אֶת אוֹצָרוֹ וְהִשְׁמִיר אוֹתוֹ.

וְכַאֲשֶׁר בָּאוּ לְאָדָם מֵאָה שָׁנָה אַחֲרֵי צֵאתוֹ מִן הַפַּרְדֵּס, וְהוּא מִתְעַצֵּב בּוֹכֶה מִתְמַרְמֵר עַל חַטֹּאתוֹ, בְּסוֹף הַמֵּאָה שָׁנָה יָרְדוּ אָדָם וְחַוָּה מֵעַל לָהָר אֶל הָאַחֲרִית שֶׁלּוֹ. וְשָׁם יָדַע אָדָם אֶת חַוָּה, וְהָרְתָה. וְכַאֲשֶׁר נִשְׁלְמוּ יְמֵי הָרֵיהָ, יָלְדָה תְּאוֹמִים: קַיִן וּלְיוֹדָה אֲחוֹתוֹ. וְאַחַר כָּךְ שָׁב אָדָם אֶל חַוָּה, וְהָרְתָה וְיָלְדָה אֶת הֶבֶל וְאֶת אִקְלִימְיָה אֲחוֹתוֹ תְּאוֹמִים.

וְאוּלָם הַנְּעָרִים וְהַנְּעָרוֹת זָכוּ לְהַשָּׂגַת שִׂכְלָם. וְאָמַר אָדָם: כְּבָר הִגְדִּיל הָאֵל אֶת הַנְּעָרִים וְהַנְּעָרוֹת. וְהַנָּכוֹן שֶׁיִּזְדַּוֵּג קַיִן עִם אִקְלִימְיָה אֲחוֹת הֶבֶל, וְיִזְדַּוֵּג הֶבֶל עִם לְיוֹדָה אֲחוֹת קַיִן. וְאַחַר כָּךְ, בַּיּוֹם אֲשֶׁר נֶעֱרְכוּ הַנִּשּׂוּאִים עַל פִּי זֶה הַחֹק שֶׁחָקַק אָדָם בְּבָנָיו — אָמַר קַיִן לְאִמּוֹ: אֲנִי זוֹכֶה יוֹתֵר בַּאֲחוֹתִי אֲשֶׁר הָיִיתִי אֲנִי וְהִיא תְּאוֹמִים, וְתִמָּסֵר אֵלַי אִשָּׁה, וְתִמָּסֵר לְהֶבֶל אֲחוֹתוֹ אֲשֶׁר נוֹלְדָה עִמּוֹ.

וְהַסִּבָּה הָיְתָה שֶׁלְּיוֹדָה אֲחוֹת קַיִן יָפָה הָיְתָה בַּצּוּרָה יוֹתֵר מֵאִקְלִימְיָה אֲחוֹת הֶבֶל. וְכַאֲשֶׁר הִגִּיעַ דָּבָר זֶה לְאָדָם מִפִּי קַיִן — קָשָׁה עָלָיו וְנֶעֱצַב מְאֹד, וְאָמַר אָדָם לְקַיִן: אֲשֶׁר שָׁאַלְתָּ, בְּנִי, חוּצָה מִן הַתּוֹרָה, כִּי לֹא יֻתַּר לְךָ לָקַחַת אֶת אֲחוֹתְךָ אֲשֶׁר נוֹלְדָה עִמְּךָ. וְנִכְנְסָה בְּקַיִן מִדָּבָר זֶה הַקִּנְאָה לְהֶבֶל אָחִיו, וְהִתְכַּוֵּן לְהָרְגוֹ.

אַחַר כָּךְ צִוָּה אָדָם אֶת בָּנָיו קַיִן וְהֶבֶל שֶׁיִּקְחוּ עִמָּם מִפְּרִי הָאָרֶץ וּמִוּלְדֵי הַצֹּאן, וְיַעֲלוּ שְׁנֵיהֶם אֶל הָהָר הַקָּדוֹשׁ, וְיָבוֹאוּ אֶל מְעָרַת הָאוֹצָרוֹת וְיִתְפַּלְּלוּ שָׁם לִפְנֵי הָאֵל.
