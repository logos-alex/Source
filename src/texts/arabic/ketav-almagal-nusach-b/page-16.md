…וְהַהִלּוּל וְהַתִּשְׁבָּחָה לַיָּחִיד הַיָּחִיד הַמְדַבֵּר בִּבְנוֹ הַיָּפֶה. וְכַאֲשֶׁר יִתְפַּלֵּל הָאָדָם בְּאוֹתָהּ הָעֵת — תְּהִי תְּהִלָּתוֹ מִתְעָרֶבֶת בְּתִשְׁבְּחוֹת הַמַּלְאָכִים, וְתִהְיֶה אֵצֶל הָאֵל תְּפִלָּתוֹ נִשְׁמַעַת.

וּבַשָּׁעָה הַשְּׁמִינִית — עוֹבְדִים הַשֵּׁדִים אֶת הָאֵל יִתְבָּרַךְ. וְכַאֲשֶׁר יִתְפַּלֵּל הָאָדָם בְּאוֹתָהּ שָׁעָה — לֹא יִתְחִיל לוֹ בְּלִבּוֹ מַחֲשָׁבָה רָעָה וְלֹא וִסְוּוּס מִצַּד הַשָּׂטָן, כִּי הַשֵּׁדִים טְרוּדִים בַּעֲבוֹדַת נֵצַח הָאֵל.

וּבַשָּׁעָה הַתְּשִׁיעִית — עוֹבְדִים הַמַּלְאָכִים הַשּׁוֹמְרִים אֶת כִּסֵּא הָאֵל, וְאֵלֶּה נִקְרָאִים "הַשְּׁמִירָה הַמְּאִירָה".

וּבַשָּׁעָה הָעֲשִׂירִית — תְּפִלַּת הַטָּהֳרוֹת. וְאוֹתָהּ שָׁעָה עוֹמֵד הָאָדָם מִחַיָּיו וּמִתְיַשֵּׁב, וְעוֹלָה עָלָיו רוּחַ הַקֹּדֶשׁ וּמְרַחֶפֶת עָלָיו, וְיִטָּהֵר מִכְּנִיסַת הַשֵּׁדִים אֵלָיו, וּתְעַטְּרֵהוּ. וְלוּלֵא שֶׁרוּחַ הָאֵל הַקְּדוֹשָׁה בְּכָל יוֹם מַזְהֶרֶת עַל כָּל הָאֻמּוֹת וּמְרַחֶפֶת עֲלֵיהֶן רַחֲמִים מֵהָאֵל — הָרוּחַ הַמַּחְיָה הַמְדַבֶּרֶת הַתְּדִירָה לַחַיִּים, הַבּוֹרֵא לְכָל חַי — לֹא הָיָה יְכוֹלֶת לְמִי שֶׁבָּרָא הָאֵל לִשְׁתּוֹת מִן הַמַּיִם הַמֻּעָטִים מִבְּלִי שֶׁיַּהֲרֹג עַצְמוֹ וְיֹאבַד נַפְשׁוֹ.

וּבַשָּׁעָה הָאַחַת-עֶשְׂרֵה — מִתְפַּעֲלוֹת וְשָׂמוֹת נַפְשׁוֹת הַקְּדוֹשִׁים בְּמַה שֶּׁמּוֹצְאוֹת מִן הַמְּנוּחָה וְהַשַּׁלְוָה וְהַשַּׁלְוָה אֲשֶׁר מוֹשִׁיב לָהֶן הָאֵל עַל יְדֵי מַלְאָכָיו הַמְקֹרָבִים. וְכָךְ הַמְשַׁבְּחִים וְהָעֵדִים וְהַמַּאֲמִינִים.

וּבַשָּׁעָה הַשְּׁתֵּים-עֶשְׂרֵה — תְּפִלַּת הַקּוֹרְאִים וְהִפָּרְדָם מִן הַיּוֹם הָעוֹבֵר מִן הַיוֹם, וְקָרְבָּן לָאֵל וְהִתְחַנְּנוּת וּבַקָּשָׁה נִשְׁמַעַת.
תְּפִלּוֹת שָׁעוֹת הַלַּיְלָה

וְאוּלָם שְׁעוֹת הַלַּיְלָה — בַּשָּׁעָה הָרִאשׁוֹנָה — עֲבוֹדַת הַשֵּׁדִים אֶת הָאֵל.

וּבַשָּׁעָה הַשְּׁנִיָּה — עֲבוֹדַת הַתַּנִּינִים וְהַדָּגִים וְחַיּוֹת הַמַּיִם.

וּבַשָּׁעָה הַשְּׁלִישִׁית — עֲבוֹדַת הָאֵשׁ אֲשֶׁר בַּשָּׁמַיִם הַשְּׁנִיִּים מֵעַל לָרָקִיעַ.

וּבַשָּׁעָה הָרְבִיעִית — קְדֻשַּׁת הַמַּלְאָכִים הַשְּׂרָפִים לָאֵל יִתְבָּרַךְ.

וּבַשָּׁעָה הַחֲמִישִׁית — עֲבוֹדַת הָאֻמּוֹת כֻּלָּן אֲשֶׁר מֵעַל הַשָּׁמַיִם. כִּי הָיִיתִי שׁוֹמֵעַ אֶת הֲמוֹן הַמַּלְאָכִים כַּהֲמוֹן הַסְּפִינוֹת הָעוֹבְרוֹת בַּיָּם, וְרַעַשׁ הַגַּלִּים וְזַעַמָּם בִּתְשְׁבָּחוֹת לָאָדוֹן.

וּבַשָּׁעָה הַשִּׁשִּׁית — תַּחֲנוּת הֶעָנָן אֶל הָאֵל, וְהוּא נִפְחָד.

וּבַשָּׁעָה הַשְּׁבִיעִית — עוֹמֵד כֹּחַ הַשֶּׁמֶשׁ בָּאָרֶץ וְשׁוֹקֵט וּמִתְיַצֵּב וּמְשַׁבֵּחַ לָאָדוֹן. וְיָשְׁנִים הַמַּיִם וְנָחִים. וְאִם יַחְטֹף הָאָדָם דָּבָר מֵאוֹתָם הַמַּיִם בְּאוֹתָהּ שָׁעָה, וְיִתְעָרְבוּ בָּהֶם שֶׁמֶן זַיִת אֲשֶׁר נִתְבָּרַךְ עַל יְדֵי הָאֵל, וִימָשַׁח בְּאוֹתוֹ שֶׁמֶן וּמַיִם מִי שֶׁיֵּשׁ לוֹ כְּאֵבִים וּמִי שֶׁאֵינוֹ יָשֵׁן בַּלַּיְלָה — יָנוּחַ מֵעִנּוּי חֹדֶשׁ, וִירַפְּאֵהוּ הָאֵל מִמַּחֲלָתוֹ וּמִכְּאֵבוֹ.
