אָמַר אָדָם: שְׁמַע, בְּנִי שֵׁת, אֵת אֲשֶׁר אֲנִי מְצַוֶּה אוֹתְךָ, וְהָבֵן זֹאת, וְצַוֵּה בוֹ אֶת בָּנֶיךָ, יֶלֶד אַחַר יֶלֶד, וְדוֹר אַחַר דּוֹר, וְעַם אַחַר עַם, בְּמַה שֶּׁנִּגְלָה לִי וְנִשְׁמַר עִמְּדִי. עָלֶיךָ — כְּשֶׁתִּרְאֶה אֶת הַמַּלְאָךְ וּכְבָר קָבַץ נַפְשִׁי מִגּוּפִי — שֶׁתִּרְחַץ אֶת גּוּפִי בַּמַּיִם הָרוֹתְחִים הַפּוֹשְׁרִים, בְּרֵיחַ הָאַסָּא, שֶׁבַע פְּעָמִים, וְתַנִּיחֵנִי בְּאָרוֹן, וְתָשִׂים תַּחְתַּי וּמֵעָלַי וְעַל שְׁאָר גּוּפִי אֶת הַמֹּר וְהַסְּלִיכָה. וְתַעֲמֹד אַתָּה וּבָנֶיךָ וְתִתְפַּלֵּל עָלַי, וְתַקְטִירוּ עָלַי לְבוֹנָה עַד אֲשֶׁר תֻּשְׁלַם הַתְּפִלָּה. וְתַעֲבִירוּנִי וְתַנִּיחוּנִי בִּמְעָרַת הָאוֹצָרוֹת בְּהָהָר הַקָּדוֹשׁ.

וְצַוֵּה אֶת כָּל בָּנֶיךָ שֶׁיַּעֲשׂוּ כְּמוֹ שֶׁהוֹדַעְתִּיךָ מֵעִנְיַן הַמָּוֶת. וְכָל מִי שֶׁיָּמוּת מִזַּרְעֲךָ — יָשׁוּב וְיִפְתַּח אֶת הַמְּעָרָה וְיָשִׂים גּוּפוֹ עִם הָאַחֵר בַּמְּעָרָה. וְתַעֲשׂוּ אוֹתָהּ בֵּית לִתְפִלּוֹתֵיכֶם וּלְבִקּוּרֵיכֶם — בָּזֶה בִּלְבַד, לֹא בְּזוּלָתוֹ. וְתַזְהִיר אֶת בָּנֶיךָ, וְתָקֵף בָּם הַצַּוָּאָה. וְזֶה יִוָּדַע בַּצָּעִיר וְהַגָּדוֹל מִכֶּם, וְיַעֲבִירוּהוּ בִּירֻשָּׁה עַד הָעֵת הָאַחֲרוֹנָה.

כִּי בְּעוֹלָם נִתָּנָה לָכֶם רְשׁוּת לַעֲבֹר מִן הָהָר הַזֶּה אֶל הָאָרֶץ הַתַּחְתּוֹנָה, וְלִשְׁכֹּן בָּהּ. וְאוֹתָהּ אֶרֶץ — הִיא אֲשֶׁר בָּרָאַנִי הָאֵל מִמֶּנָּה, וּבְהֶכְרֵחַ אֵלֶיהָ יְשִׁיבֵנִי. וּמִמֶּנָּה מְקַוֶּה אֲנִי לַיְשׁוּעָה עַל יְדֵי מוֹשִׁיעַ הָעוֹלָם.

כְּבָר הוֹדַעְתִּיךָ, בְּנִי שֵׁת, אֱמֶת אֲשֶׁר רוֹצֶה לִהְיוֹת לִפְנֵי שֶׁיִּהְיֶה. וּלְמַעַן כַּאֲשֶׁר תִּקְרַב הָעֵת מֵאוֹתָם הַיָּמִים — תִּהְיֶה אֵצֶל בָּנֶיךָ יְדִיעַת הַצַּוָּאָה אֲשֶׁר צִוִּיתִיךָ. וְכַאֲשֶׁר הֵם יַעַבְרוּ מִן הָהָר הַקָּדוֹשׁ וְיֵרְדוּ אֶל הָאָרֶץ — יִקְחוּ אֶת גּוּפִי עִמָּהֶם וְיִקְבְּרוּהוּ בְּאֶמְצַע הָאָרֶץ, מָקוֹם אֲשֶׁר הָאֵל יֵדַע אוֹתוֹ. כִּי הָאֵל לֹא יְעַזְּבֵכֶם, וְלֹא תָּזוּז עֵינוֹ עִמָּכֶם, כַּאֲשֶׁר תִּקְרְאוּהוּ בְּלִבָּבוֹת זַכִּים נְקִיִּים. כִּי יִתְבָּרַךְ זִכְרוֹ יִעֲנֶה מִכֶּם וִיכַלְכֵּל צָרְכֵיכֶם וִירוֹמֵם מַעֲלָתְכֶם…
