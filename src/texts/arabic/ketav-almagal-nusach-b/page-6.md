לָחֲשׁוּ אִישׁ אֶל אָחִיו וְאָמְרוּ: מַה הַדִּבּוּר הַמֻּפְלָא הַגָּדוֹל אֲשֶׁר לֹא נִשְׁמַע כְּמוֹתוֹ מִן הָאָדוֹן מֵעוֹלָם? הִנֵּה דָּבָר נִשְׂגָּב, וְכֵיצַד יִתָּכֵן שֶׁתּוֹפַע צוּרַת אֱלֹהֵינוּ אֵלֵינוּ? וְהָאֲרִיכוּ בָּזֶה הַשִּׂיחוֹת בֵּינֵיהֶם.

וּבְעוֹדָם הַמַּלְאָכִים מְשׂוֹחֲחִים בְּדִבּוּר זֶה — וְהִנֵּה רָאוּ אֶת יַד הָאֵל יִתְבָּרַךְ מֵעַל לַטְּהוֹרִים, וּכְבָר נִשְׁתַּחְפָה וְהִקִּיפָה אֶת כָּל הַשָּׁמַיִם וְהָאֲרָצוֹת. וּכְבָר קָמְצָה מִן הָאָרֶץ קֹמֶץ עָפָר, וּמִן הַמַּיִם נְטָף מַיִם, וּמִן הָאֲוִיר נְשִׁימַת רוּחַ, וּמִן הָאֵשׁ כֹּחַ חֲמִימוּת.

וְכַאֲשֶׁר נִכְנְסוּ בִּקְמִיצַת הָאָדוֹן אַרְבָּעָה הַיְסוֹדוֹת — וְהֵם הַחֲמִימוּת וְהַקְּרִירוּת וְהַטִּבְעִיּוּת וְהַיַּבָּשׁוּת — קָבַע אוֹתָם הָאֵל יִתְבָּרַךְ בְּחָכְמָתוֹ צוּרַת הָעֶלֶם אָדָם, צוּרָה מֻשְׁלֶמֶת תְּמִימָה מִכָּל אַרְבָּעָה הַיְסוֹדוֹת הַחֲלָשִׁים, אֲשֶׁר יִשָּׁמַע לוֹ וְיִשְׁמְעוּ לוֹ כָּל הַבְּרִיּוֹת וְהָאֲנָשִׁים אֲשֶׁר נִבְרְאוּ מֵהֶם. כִּי הֶעָפָר אֲשֶׁר בָּרָא הָאֵל אֶת אָדָם מִמֶּנּוּ — נִתְכַּוֵּן בָּזֶה שֶׁיִּהְיוּ נִשְׁמָעִים לוֹ בְּנֵי בְּרִיאָתוֹ.

וְלָמָּה הוֹסִיף אֵלָיו אֶת הָאֲוִיר וְהַמַּיִם וְהָאֵשׁ? כִּי הַמַּיִם — לְהִכָּלֵל בָּהֶם הֶרְכֵּבוֹ, וְלִשְׁמֹעַ לוֹ כָּל הַנּוֹלָדִים מִמִּינֵי הַמַּיִם — מַיִם גְּדוֹלִים וּקְטַנִּים. וְתוֹעֲלוֹת הָאֲוִיר — כִּי הָאֲוִיר הוּא מַחֲיֶה אֶת הַכֹּל, וּבְרָאוֹ הָאֵל עַל גַּב הָאָרֶץ לְמַעַן יִשְׁאַף אוֹתוֹ הָאָדָם וְיִחְיֶה, וּלְמַעַן יִשָּׁמַע לוֹ נִשְׁמָע. וְאוּלָם חֲמִימוּת הָאֵשׁ — תּוֹעַלְתָּהּ לְגוּף אָדָם לְהָגֵן עָלָיו מִטִּבְעוֹ, בִּהְיוֹתוֹ נִבְרָא מִן הָאָרֶץ אֲשֶׁר מֻטְבֵּעַת מִן הַיַּבָּשׁוּת וְהַקֹּר. וְהַחֲמִימוּת מְחַזֶּקֶת תְּנוּעָתוֹ, וְנוֹלָד בּוֹ הַדָּם הָעוֹזֵר לְחַיָּיו. וּלְמַעַן יִהְיוּ הַכֹּחוֹת הַנִּבְרָאִים מִן הָאֵשׁ נוֹכְחִים לוֹ, מַעֲזָר לְהַרְגָּשָׁתוֹ.

וְאוּלָם הַסִּבָּה לִבְרִיאַת הָאֵל יִתְבָּרַךְ אֶת אָדָם בְּיָדָיו הַקְּדוֹשׁוֹת עַל צוּרָתוֹ וְצַלְמוֹ וּדְמוּתוֹ — שֶׁכֵּן עָשָׂה זֹאת הָאֵל לְמַעַן יְקַבֵּל אָדָם מֵחַסְדִּי הַחָכְמָה וְהַדִּבּוּר וְהַתְּנוּעָה הַחַיָּתִית וְהַיְדִיעָה בִּדְבָרִים.

וְכַאֲשֶׁר הִבִּיטוּ הַמַּלְאָכִים אֶל הַצּוּרָה הַמְפֹאֶרֶת מְצֻיֶּרֶת מִבְּרִיאַת אָדָם — נִזְדַּעְזְעוּ, וְהֵבְעִיתָם הַזֹּהַר הָעָצוּם אֲשֶׁר אֶל סְבִיבוֹתָיו צִוָּה הָאֵל אֶל אָדָם וְהִלְבִּישׁוֹ אוֹתוֹ וְכִבְּדוֹ בִּשְׁכִינָתוֹ עָלָיו, כִּי שָׁפַע הָאֵל עַל אָדָם אוֹרוֹת אֱלֹהִיִּים
