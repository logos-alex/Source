…וְזִכְרוּ אֶת הַיָּמִים אֲשֶׁר הוֹרִישׁוּ אֶתְכֶם אֲבוֹתֵיכֶם אָדָם וְשֵׁת. שֶׁכָּל מִי שֶׁשּׁוֹמֵר אֶת מִצְוַת יֶרֶד אֶת עֲבוֹדַת אָבִינוּ הָרִאשׁוֹן בָּהָר הַקָּדוֹשׁ — לֹא יִתָּכֵן לוֹ עוֹד לַעֲלוֹת אֵלָיו לְעוֹלָם. וְלֹא הִשְׁתַּמְּעוּ לִדְבָרוֹ, וְלֹא שָׁמְרוּ אֶת מִצְוַת יֶרֶד אֲבִיהֶם, וְלֹא הִקְשִׁיבוּ לְמַה שֶּׁאָמַר חֲנוֹךְ — וּמָרְדוּ וְהִתְפַּקֵּרוּ וְהֵטּוּ מֵהַמִּצְוָה, וְיָרְדוּ מִן הָהָר הַקָּדוֹשׁ.

וְכַאֲשֶׁר רָאוּ אֶת בְּנוֹת קַיִן, וְהֵן יְפוֹת מַרְאֶה וְעַרְבוֹת תַּכְלִית, וְהֵן זוֹעֲקוֹת עֲרוּמוֹת בְּלִי בּוּשָׁה — הֻצְּתוּ הַגִּבּוֹרִים בְּנֵי שֵׁת בְּאֵשׁ הַתַּאֲוָה, וְהִשְׁתַּלֵּט בָּהֶם הַשָּׂטָן.

וְכֵן בְּנוֹת קַיִן, כְּשֶׁרָאוּ אֶת הַגְּבָרִים הַגִּבּוֹרִים בְּנֵי שֵׁת וְיָפְיָם וַעֲדִינוּתָם — קְנוּ אוֹתָם בַּעֲדָיִים הָרָעִים הַטְּמֵאִים. וְטִמְּאוּ גּוּפוֹתֵיהֶם עִם בְּנוֹת קַיִן, וְצָבְעוּ עַצְמָם בִּזְנוּת עִמָּהֶן, וְעָשׂוּ בְּמַעֲשֵׂיהֶם הַמְגֻנִּים.

וּכְשֶׁרָצוּ לַעֲלוֹת אֶל הָהָר הַקָּדוֹשׁ וְהֵם מְלֻכְלָכִים בְּסִיא הַזְּנוּת וְהַשְׁחָתָה — לֹא יָכְלוּ לַעֲלוֹת. כִּי נַעֲשׂוּ אַבְנֵי הָהָר אֵשׁ מְלֻהֶבֶת בִּפְנֵיהֶם, וְנִמְנְעוּ מִלַּעֲלוֹת אֶל אוֹתוֹ הַמָּקוֹם הַקָּדוֹשׁ.

אַחַר כָּךְ עוֹד יָצְאוּ אֲחֵרִים מֵהֶם וְיָרְדוּ חֲבוּרָה אַחַר הָאֵלֶּה, וְהֵטּוּ מֵהַמִּצְוָה, וְנָפְלוּ אַף הֵם בַּזְּנוּת עִם בְּנוֹת קַיִן — וְהֵם יוֹדְעִים מַה שֶּׁאֵרַע לְאַחֵיהֶם הַמּוּטִים ⁑

וּכְשֶׁחָיָה יֶרֶד תְּשַׁע מֵאוֹת וּשְׁתַּיִם וְשִׁשִּׁים שָׁנָה — וְהִגִּיעַ עֵת מוֹתוֹ — נֶאֱסַף אֵלָיו חֲנוֹךְ בְּנוֹ, וּמְתוּשֶׁלַח, וְלֶמֶךְ, וְנֹחַ. וְהִתְפַּלֵּל עֲלֵיהֶם וְאָמַר לָהֶם: יוֹדֵעַ אֲנִי, בָּנַי, שֶׁיִּתְרַגֵּז הָאֵל עַל זֶה הָעָם, וְיִשְׁפֹּט בְּלִי רַחֲמִים — יִכְרֹת עֲלֵיהֶם. וְאַתֶּם — שְׁאֵרִית וִיתֶר זֶה הָעָם. וּמֵעַתָּה לֹא יִוָּלֵד שְׁנֵי-זָכָר-וּנְקֵבָה בָּהָר הַקָּדוֹשׁ הַזֶּה — עַד אֲשֶׁר יִהְיֶה רֹאשׁ עָם. וְאַתֶּם תֵּרְדוּ מִן הָהָר הַקָּדוֹשׁ בַּעֲבוּר אֲשֶׁר עֲבַרְתֶּם עַל מִצְוֹת הָאָבוֹת, וְתֵצְאוּ אֶל הָאֶרֶץ הֶעָפָרִית הַסְּרוּחָה — קוֹצִים וְדַרְדַּרִים. וּסְבִיב הַפַּרְדֵּס תִּהְיוּ קְרוֹבִים.

אַךְ יַשְׁגִּיחַ הַיּוֹצֵא מִכֶּם מִן הַמָּקוֹם הַקָּדוֹשׁ הַזֶּה, וְיִקַּח אִתּוֹ אֶת הַגּוּף הַטָּהוֹר שֶׁל אָבִינוּ אָדָם. וְאִם יוּכַל — יִקַּח אֶת אֶגֶד אֲבוֹתֵינוּ כֻּלָּם. וְיִקַּח אַף…

    [שׁוּלַיִם]: וְקֵינָן וּמַהֲלַלְאֵל וְהֶעֱבִירָם חֲנוֹךְ בֶּן שֵׁת.
