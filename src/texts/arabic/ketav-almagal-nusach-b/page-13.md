…וְיַקְרִיבוּ מַה שֶּׁהֵבִיאוּ עִמָּהֶם מִן הַפֵּרוֹת וּמוּלְדֵי הַצֹּאן. וְאָמַר לָהֶם אָדָם: אִם אַתֶּם עוֹשִׂים זֹאת — יִקַּח כָּל אֶחָד אֶת אִשְׁתּוֹ אֶל כַּנְפוֹ וִיכַלְכֵּל אוֹתָהּ. וְעָשׂוּ שְׁנֵיהֶם כְּכָל אֲשֶׁר צִוָּם אֲבִיהֶם אָדָם, עָלָיו הַשָּׁלוֹם, מִלָּקַחַת הַפֵּרוֹת וּמוּלְדֵי הַצֹּאן. וְעָלָה אָדָם מִתַּחַת הָהָר אֶל רֹאשׁ הָהָר, וּשְׁנֵי בָנָיו עוֹלִים אַחֲרָיו. וְנִכְנַס הַשָּׂטָן בְּלִבּוֹ שֶׁל קַיִן וּלְחָשַׁל לוֹ וְיִפָּה לוֹ הֶרֶג אָחִיו.

אַחַר כָּךְ הִקְרִיבוּ קָרְבְּנוֹתֵיהֶם לִפְנֵי הָאֵל. וְקִבֵּל אֶת קָרְבַּן הֶבֶל, וְלֹא קִבֵּל אֶת קָרְבַּן קַיִן — כִּי הָאֵל יוֹדֵעַ הָיָה בְּלִבּוֹ שֶׁל קַיִן מִן הָרָע וְהַזָּדוֹן וְהָעַרְמוּמִיּוּת וְהַקִּנְאָה וְהַשִּׂנְאָה לְאָחִיו הֶבֶל, וְעַל כֵּן לֹא קִבֵּל מִמֶּנּוּ אֶת קָרְבָּנוֹ.

וְכַאֲשֶׁר רָאָה קַיִן אֶת קַבָּלַת הָאָדוֹן לְקָרְבַּן הֶבֶל וְלֹא לְשֶׁלּוֹ — נוֹסַף לוֹ קִנְאָה לְהֶבֶל, וּמִלֵּא אוֹתוֹ זַעַם. וְכַאֲשֶׁר יָרְדוּ מִן הָהָר, נָטַל קַיִן אֶבֶן מְחֻדֶּדֶת וְהִכָּה בָּהּ אֶת הֶבֶל וַהֲרָגוֹ.

וְקִלֵּל הָאֵל אֶת קַיִן, וְהוֹרִיד עָלָיו אֶת מִשְׁפָּטוֹ. וְהָיָה מֵאוֹתָהּ שָׁעָה מֻפְחָד פַּחַד נָע, כָּל יְמֵי חַיָּיו. וְגֵרְשׁוֹ הָאֵל מֵרֹאשׁ הָהָר אֶל מָקוֹם הַנִּקְרָא "הַ'בֻּקְעָה'", וְהוֹשִׁיבוֹ הָאֵל שָׁם — לוֹ וּלְאִשְׁתּוֹ, שֶׁהִיא אֲחוֹתוֹ. וְהִתְעַצְּבוּ אָדָם וְחַוָּה עַל הֶבֶל מֵאָה שָׁנָה.

אַחַר כָּךְ יָדַע אָדָם אֶת חַוָּה, וְהָרְתָה וְיָלְדָה אֶת שֵׁת — הָאִישׁ הַיָּפֶה בַּקּוֹל, הָעָצוּם, הַתָּם הַשָּׁלֵם — כִּי הָיָה בְּכָל הַדְּבָרִים מֻשְׁלָם כִּדְמוּת אָדָם אָבִיו. וְהֶעֱנִיקוֹ הָאֵל בְּהַגִּיעוֹ לְמַעֲלַת הָאֲנָשִׁים, וְעָשָׂהוּ אָב לְכָל גִּבּוֹרֵי הָאָרֶץ. וְהוֹלִיד שֵׁת הַגִּבּוֹר אֶת אֱנוֹשׁ, וֶאֱנוֹשׁ הוֹלִיד אֶת קֵינָן, וְקֵינָן הוֹלִיד אֶת מַהֲלַלְאֵל. כִּי הָאָבוֹת הָאֵלֶּה כֻּלָּם נִשְׁתָּלְשְׁלוּ בְּחַיֵּי אָבִינוּ אָדָם.

וְאוּלָם יְמֵי חַיֵּי אָדָם הָיוּ תְּשַׁע מֵאוֹת וּשְׁלוֹשִׁים שָׁנָה — עַד הָעֵת שֶׁבָּה בָּאוּ לְמַהֲלַלְאֵל בֶּן קֵינָן בֶּן אֱנוֹשׁ בֶּן שֵׁת מֵחַיָּיו מֵאָה וּשְׁמוֹנֶה וּשְׁלוֹשִׁים שָׁנָה.

וּכְשֶׁהִגִּיעַ עֵת מִיתַת אָדָם, הִקְרִיב אֶת בָּנָיו: שֵׁת, וֶאֱנוֹשׁ, וְקֵינָן, וּמַהֲלַלְאֵל בְּנֵי שֵׁת, וְהִתְפַּלֵּל עֲלֵיהֶם, וְקָרָא לָהֶם, וּבֵרְכָם, וְצִוָּה לְשֵׁת אֶת הַצַּוָּאָה הַזֹּאת.
