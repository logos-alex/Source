הַפְּרִי. וְכַאֲשֶׁר אָכַל מֵאוֹתָהּ אֲשֶׁר מִטִּבְעָהּ הַמָּוֶת — נִשְׁבַּר גּוּפוֹ וְנִתְעָרְטֵל מִכְּבוֹד הָאֵל, וְנִשָּׁא הָאוֹר אֲשֶׁר הָיָה לְבוּשׁוֹ, וְנִתְגַּלּוּ לִשְׁנֵיהֶם פְּגָמֵי גּוּפוֹתֵיהֶם. וְנֶחְפְּזוּ אֶל הָאִילָנוֹת, וְנִסְתַּתְּרוּ בַּעֲלֵי הַתְּאֵנָה. וַעֲלָה עֲלֵיהֶם הָעֶצֶב וְהַבְּכִי וְהַיְּלָלָה. וְלֹא יָשְׁבוּ בַּמַּלְכוּת וּבַנְּעִימוּת אֲשֶׁר הֶעֱנִיק לָהֶם הָאֵל אֶלָּא שָׁלוֹשׁ שָׁעוֹת. וְהוֹרִידָם הָאֵל מִן הַפַּרְדֵּס אֶל מַטָּה לַפַּרְדֵּס בְּתֵשַׁע שָׁעוֹת מִן הַיּוֹם הַשִּׁשִּׁי — אֲשֶׁר בּוֹ, בִּתְחִלָּתוֹ, מָרַד אִבְלִיס בָּאֵל, וְתָבַע אֶת הָאֱלוֹהוּת, וְהֻגְלָה מִמַּעֲלָתוֹ וְהוּרַד מִכִּסְאוֹ.

וּבוֹ, בְּשָׁלוֹשׁ שָׁעוֹת מִן הַיּוֹם, נִבְרָא אָדָם וְהֵיטִיב הָאֵל עִמּוֹ וְכִבְּדוֹ וְהֶעֱלָה אֶת מַעֲלָתוֹ וְהוֹשִׁיבוֹ בַּפַּרְדֵּס וְחָנַן אוֹתוֹ בַּמַּלְכוּת וְהַכְּהֻנָּה וְהַנְּבוּאָה. וּבוֹ — בָּרָא הָאֵל אֶת חַוָּה מִצַּלְעוֹ שֶׁל אָדָם בַּפַּרְדֵּס. וּבוֹ — זִוֵּג הָאֵל לְאָדָם אֶת חַוָּה בִּפְנֵי כָּל הַמַּלְאָכִים. וּבוֹ — חָטְאוּ חַוָּה וְאָדָם, וּבִתְמַם תֵּשַׁע שָׁעוֹת מִמֶּנּוּ יָצְאוּ אָדָם וְחַוָּה מִן הַנְּעִימוּת אֶל הַצַּעַר וְהַיְּגִיעָה וְהָעֳנָּה.

וּפִיֵּס לָהֶם הָאֵל אַחַר צֵאתָם מִן הַפַּרְדֵּס תַּחַת הָאוֹרוֹת אֲשֶׁר הָיוּ לְבוּשָׁם — עוֹרוֹת אֲשֶׁר נִמְתַּחוּ עַל גּוּפוֹתֵיהֶם, וְהֵם פְּחָתָה הַכְּאֵבִים וְהַחֳלָאִים. וּכְשֶׁיָּצְאוּ מִן הַפַּרְדֵּס לֹא יָצְאוּ אֶלָּא בִּבְכִי עָז וּבִילָלָה וּבְחֶמְלָה גְּדוֹלָה עָלֵיהֶם. וְכַאֲשֶׁר הִגִּיעוּ אֶל מִתַּחַת לַפַּרְדֵּס, נָפְלָה עֲלֵיהֶם הַשֵּׁנָה וְיָשְׁנוּ בְּמִזְרְחֵי הַפַּרְדֵּס. וְכַאֲשֶׁר נֵעוֹרוּ מִשְּׁנָתָם, דִּבֵּר הָאֵל לְאָדָם וְנִחֲמוֹ וְאָמַר לוֹ: יִתְקַדַּשׁ שְׁמִי — אָדָם, אַל תִּתְעַצֵּב, כִּי אָשִׁיב אוֹתְךָ אֶל יְרֻשָּׁתְךָ אֲשֶׁר הוּצֵאתָ מִמֶּנָּה בְּמִרְיֵךָ. וְדַע כִּי מֵאַהֲבָתִי אוֹתְךָ קִלַּלְתִּי אֶת הָאָרֶץ, וְלֹא חַסְתִּי עָלֶיהָ, בִּגְלַל חֶטְאָתְךָ. וְאוּלָם הַנָּחָשׁ אֲשֶׁר הִדִּיחֲךָ וְהֶחֱטִיאֲךָ וְנִכְנַס אֶל הַגַּן — שַׂמְתִּי רַגְלָיו בְּבִטְנוֹ וְהוֹשַׁבְתִּיו בְּקַרְקַע הָאָרֶץ, וּמַאֲכָלוֹ עָפָר, וְקִלַּלְתִּיו וְלֹא קִלַּלְתִּיךָ.

וְהֶחֱנַנְתִּי אֶת חַוָּה בְּצַיְּתָנוּתֵךְ, וְתִהְיֶה תַּחַת עוֹל שֵׁרוּתֵךְ. וְהֱיֵה בָּטוּחַ שֶׁכַּאֲשֶׁר תַּשְׁלִים יְשִׁיבָתְךָ בַּיִּשּׁוּב — וְהִיא הָאָרֶץ אֲשֶׁר מוּל פַּרְדֵּסְךָ — אֶשְׁלַח אֶת בְּנִי לָרֶדֶת
