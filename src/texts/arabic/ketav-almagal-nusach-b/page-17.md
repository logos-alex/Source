וּבַשָּׁעָה הַשְּׁמִינִית — בִּרְשׁוּת הָאֵל יִתְבָּרַךְ עוֹלֶה הָעֵשֶׂב מִן הָאָרֶץ. וּבַשָּׁעָה הַתְּשִׁיעִית — תְּשַׁבֵּחַ הַמַּלְאָכִים וּכְנִיסַת הַתְּפִלּוֹת לִפְנֵי הָאֵל.

וּבַשָּׁעָה הָעֲשִׂירִית — נִפְתָּחוֹת שַׁעֲרֵי הַשָּׁמַיִם. וְכַאֲשֶׁר תַּעֲלֶה מִבָּנַי תְּפִלָּה בְּאוֹתָהּ שָׁעָה — תִּהְיֶה נִשְׁמַעַת אֵצֶל הָאֵל, מְקֻבֶּלֶת. וּבְאוֹתָהּ שָׁעָה מַכְּפוֹת כַּנְפֵי הַשְּׂרָפִים, וּבְאוֹתוֹ זְמַן צוֹעֵק הַתַּרְנְגוֹל.

וּבַשָּׁעָה הָאַחַת-עֶשְׂרֵה — תְּהִי שִׂמְחָה וְעֲלִיצוּת בְּשְׁאָר אֶרְצוֹת הָעוֹלָם, וְזֶה — שֶׁהַשֶּׁמֶשׁ נִכְנֶסֶת בַּפַּרְדֵּס וְזוֹרְחִים זִיוָהּ בַּקְּצָווֹת הָאָרֶץ, וּמְאִירִים הַבְּרוּאִים כֻּלָּם בְּנְפִילַת קַרְנֵי הַשֶּׁמֶשׁ עֲלֵיהֶם.

וּבַשָּׁעָה הַשְּׁתֵּים-עֶשְׂרֵה — נָאוֶה, בְּנִי, לְצַוּוֹת אֶת בָּנֶיךָ שֶׁיַּקְטִירוּ לִפְנֵי הָאָדוֹן בִּלְבוֹנָה. כִּי רֵיחַ הַלְּבוֹנָה מְשַׂמֵּחַ אֶת הַמַּלְאָכִים, וּמִמֶּנּוּ בּוֹרְחִים הַמּוֹרְדִים וְהָאִבְלִיסִים. וְשׁוֹקְעוֹת בְּשַׁלְוָה נַפְשׁוֹת הַצַּדִּיקִים, וְחַיּוֹת רוּחוֹת בְּנֵי הָאָדָם וְנוֹחוֹת.
[נְבוּאַת אָדָם עַל בּוֹא הַמָּשִׁיחַ]

וְדַע כִּי הָאֵל יִשְׁלַח אֶת דְּבָרוֹ הַקָּדוֹשׁ הַיּוֹרֵד מִן הַשָּׁמַיִם אֶל הָאָרֶץ, וְיַשְׁכִּין הָאֵל אֶת הַדָּבָר בִּמְעֵי בְּתוּלָה מִבָּנַי, וְיִתְאַנֵּשׁ בְּצוּרָה אוֹרִית, יִתְגַּשֵּׁם בִּלְבוּשׁ הָאֱנוֹשִׁיּוּת, וְיַשְׁכִּין הָאֵל הָאָדוֹן אֶת דְּבָרוֹ בָּהּ, וְיוֹפִיעֶנּוּ אֶל הָאָרֶץ בְּרַחֲמִים מִמֶּנּוּ וּבְחֶמְלָה עַל כָּל בְּרוּאָיו, לְהַצִּיל בּוֹ אֶת הָאָדָם מִמּוֹת הַחֵטְא אֲשֶׁר יָרְשׁוּ אוֹתוֹ מִמֶּנִּי בִּגְלַל מִרְיִי בּוֹ. וִיקוֹמֵם אֶת גּוּפוֹתֵיהֶם הַבָּלִים אֲשֶׁר עָטְפָה אוֹתָם הָאָרֶץ וְעָשְׂאַתָּם רֶמֶשׁ — בַּעֲבוּר שֶׁאֲנִי אָכַלְתִּי מֵעֵץ הַמָּוֶת אֲשֶׁר נְהָנִי הָאֵל מִלֶּאֱכֹל מִמֶּנּוּ, וְאָכַלְתִּי, וְנִתְחַיַּבְתִּי בְּמִרְיִי לָאֵל יִתְבָּרַךְ — לָמוּת מָוֶת רִאשׁוֹן וּמָוֶת שֵׁנִי — מוֹת הַחֵטְא הָרִאשׁוֹן, וְהַמָּוֶת אֲשֶׁר מִצַּד אוֹתָהּ הַפְּרִי הַתִּשׁוּכָה, אֲשֶׁר חָשַׁבְתִּי שֶׁהִיא פְּרִי הַחַיִּים הַנִּצְחִיִּים, וְהָיְתָה הִיא פְּרִי הַמָּוֶת.

וְכַאֲשֶׁר יוֹפִיעַ הָאָדוֹן בִּלְבוּשׁוֹ הַקָּדוֹשׁ — יוֹפִיעַ אוֹתוֹת וּפְלָאִים וּמוֹפְתִים וּמַעֲשִׂים נוֹרָאִים, וִיחַיֶּה אֶת הַמֵּתִים, וְיַהֲפֹךְ צוּרַת מִי שֶׁיֹּאהַב לְהַפְכוֹ, וְיִכְבֹּשׁ אֶת כָּל בָּרוּא בְּכֹחוֹ וִיכָלְתּוֹ וְעַצְמָתוֹ וְכָבוֹדוֹ וְאֵימָתוֹ. וְיִשָּׁמַע לוֹ כָּל חַי בַּיַּבָּשָׁה וּבַיָּם וּבָאֲוִיר וּבָאֵשׁ, וְכָל אֲשֶׁר תָּכִיל אוֹתוֹ הַדְּבָרִים. וְיִשָּׁמְעוּ לוֹ הַשֵּׁדִים, וְיִפְקַח עֵינֵי הָעִוְרִים, וִיסַפֵּר אֶת הָאִלְּמִים, וְיַשְׁמִיעַ
