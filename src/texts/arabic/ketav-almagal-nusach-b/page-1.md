בְּשֵׁם הָאֵל הַחַי, הַקַּיָּם, הַקַּדְמוֹן אֲשֶׁר אֵין לוֹ רֵאשִׁית.

סֵפֶר הַתּוֹעָלוֹת הַנִּבְחָרוֹת בַּחָכְמָה הַבָּלוּגָה, מֵאֵת הֶחָכָם הַבָּקִי הַמַּעֲלֶה הַקַּדְמוֹן, הָאֶחָד אֲשֶׁר אֵין בִּלְעָדָיו, רוּחַ הָאֵל, הַיָּחִיד הָעֶלְיוֹן הַמְהֻלָּל הָרַחוּם הַטּוֹב, מְחַדֵּשׁ כָּל הַנִּבְרָאִים הָעוֹלִים אֶל הַמְּצִיאוּת בְּלִי כָל הַצְעָדָה קוֹדֶמֶת. הוּא בַּעַל הַגּוֹן (אלאיר־מָאשָׁא) הָחַי בְּחַיֵּי נֶצַח, מְחַדֵּשׁ מִי שֶׁיָּשׁוּב וְיֵדַע גְּמוּל בּוֹ ⁂

בּוֹ פָּתַח הָאָב בִּבְרִיאַת כָּל דָּבָר שׁוֹתֵק וְכָל דָּבָר מְדַבֵּר, וּבְפָנָיו הִחְיָה הוּא אֶת הָרוּחוֹת וְאֶת הַשְּׁלֵמִים שֶׁבַּמְּאוֹרוֹת וְהַשְּׂמָחוֹת, וְצָר אֶת צוּרַת הַמַּלְאָכִים בַּחֲשֵׁכָה. וַיְהִי כַּאֲשֶׁר הוֹפִיעַ הָאוֹר, הִכִּירוּ הֵם בַּהוֹדָאָה וּבַתְּהִלָּה וּבָרוֹמְמוּת, וְהֵשִׁיר הוּא כָּל אֲשֶׁר רָצָה ⁂ וְקָרָא הוּא לְבַדּוֹ שִׁיר בַּהִתְבּוֹדְדוּת ⁂

נוֹסַד הָאֵל וְזָרַם קֹדֶשׁ לִבּוֹ. סֵפֶר הָאָב הַמְּעֻלֶּה הַיּוֹדֵעַ, הַטָּהוֹר בְּמַנְהִיגוּתוֹ, [הָאָב] הָאוֹחֵז בְּצַעַר, הוֹי תַּלְמִיד אֶל הָאָב הַגָּדוֹל אַב הָאָבוֹת, מַנְהִיג הָרָאשִׁים, אוֹחֵז מַפְתְּחוֹת מַנְעוּלֵי הַשָּׁמַיִם, סוֹחֵב פֶּטְרוּס, עָלָיו [הַשָּׁלוֹם], מִכָּל בְּנֵי בָנָיו בְּנֵי הַטְּבִילָה אֲשֶׁר בָּנָה [טַ...] קוֹרְאִים אֵלָיו, וּמַעֲשֵׂיהֶם תַּחַת מַלְאָךְ, עָלָיו הַשָּׁלוֹם וְהַבְּרָכָה וְהַכָּבוֹד וְהַתְּהִלָּה וְהָרוֹמְמוּת וְהַהוֹקָרָה וְהַהֲדָרָה וְהָאֲצוּלָה וְהָאֲנָוָה — בַּעֲבוּר שֶׁנֶּעֱטָר בִּשְׁמִירַת עֶדְרֵי הַצֹּאן אֲשֶׁר מָסַר אוֹתָם בְּיַד הָרִבּוֹן הָאָדוֹן, וְעַל הַשָּׁלוֹם ⁂

וְהָיָה לָנוּ לְאָב, נְאַחֵז בְּהוֹדָאָה לוֹ וְלֹא יָמוּת, נֶצַח עֲדֵי עַד, מִפִּינוּ זִכְרוֹ. שׁוֹאֵל כָּל אֶחָד מִמֶּנּוּ עַל יְדֵי מַתָּתוֹ בְּעוֹלָמֵנוּ וּבָאַחֲרִיתֵנוּ. מְקַוִּים אָנוּ לִשְׁפִיעַת חַסְדּוֹ אֶל מִי שֶׁאֵינֶנּוּ רָחוֹק מִמֶּנּוּ, וְהוּא נוֹכֵחַ בְּכֻלָּנוּ, מַרְבֶּה מַה שֶּׁאָנוּ חוֹשְׁקִים, בִּהְיוֹתוֹ אֲדוֹנֵנוּ אֱלֹהֵינוּ עִמָּנוּ ⁂

לֹא יָסוּר מִקַּבֵּץ אוֹתָנוּ כָּל יְמוֹתֵינוּ, בְּהַשְׁגָּחַת רַחֲמָיו — מֵקִיף אוֹתָנוּ, חוֹבֵק אוֹתָנוּ, שׁוֹמְרֵנוּ, מוֹצְאֵנוּ, מַרְפְּאֵנוּ, מַצִּילֵנוּ מֵעַרְמַת אוֹיְבֵינוּ ⁂ וּמַפְלִיטֵנוּ מֵחֲשֵׁכַת מַסְטִינֵינוּ. וִיכָלְתּוֹ נִגְלֵית בֵּינֵינוּ וּבְקִרְבֵּנוּ ⁂ שׁוֹמַעַת תְּפִלּוֹתֵינוּ וּמִיָּד נֶעֱנֵית לָנוּ ⁂

שָׁלַח הוּא אֶת תַּלְמִידוֹ לְהָקִים אֶת הַקְּרִיאָה לִדְבָרוֹ, וְלֹא הֶחֱטִיא אוֹתוֹ בִּגְבוּרָתוֹ וּבְעֶזְרָתוֹ, וְשָׁלַחוֹ אֶל קְצוֹת הַתֵּבֵל לְהָשִׁיב בּוֹ אֶת בְּנֵי הַחֲנֻפָּה. וְהֶעֱמִידוֹ בְּעוֹלָמוֹ וְרָצָה בוֹ ⁂ רוֹצֶה הוּא בְּכָל אֲשֶׁר עָבַר עָלָיו הַגְּזֵרָה — אֲדוֹנִי אֲשֶׁר הֶעֱמִידוֹ וְאֲשֶׁר אִשֵּׁר עָלָיו ⁂ וְיָצָא הַתַּלְמִיד אֶל מַה שֶּׁהָפְקַד עָלָיו ⁂ בִּמְכוּנַת רִבּוֹנוֹ וְאֲדוֹנוֹ ⁂ וְעִמּוֹ עֲשָׂרָה תַּלְמִידִים אֲשֶׁר נִקְּתָה לְבָבָם וּמֵרָעַת הָעוֹלָם הַזֶּה נִמְלָטוּ. כָּל אֶחָד מֵהֶם בַּהַשְׁפָּלָה כְּמוֹתוֹ ⁂ וּבִשְׂאֵתוֹ וּבַנְּבוּאָה וּבָרֵאשׁוּת — מַתָּת נָתַן מֵחַסְדּוֹ ⁂ אָכֵן נָתַן לוֹ ⁂ וּבְמַעֲשֵׂה הַמּוֹפְתִים וְהָאוֹתוֹת, וּבְמַרְפֵּא הַחֳלָאִים וְהַכְּאֵבִים וְהַתַּחֲלוּאִים
