…וְהָ"אִשְׁרָאקִיִּין" (הַזּוֹהֲרִים) וְהָאוֹר וְהַיּוֹם וְהַלַּיְלָה וְהָרוּחוֹת וְהַמַּיִם וְהָאֲוִיר וְהָאֵשׁ, וְכָל מַה שֶּׁהָיָה סִבַּת אֵלּוּ הַיְסוֹדוֹת — אֶת כָּל זֶה הִוָּה הָאֵל בִּדְבָרוֹ הַקָּדוֹשׁ ⁑

וּבְיוֹם רִאשׁוֹן רִחֲפָה רוּחַ הַקֹּדֶשׁ עַל פְּנֵי הַמַּיִם, וּבְרִחוּפָהּ נִתְבָּרְכוּ הַמַּיִם וְנִתְקַדְּשׁוּ, וְנוֹלַד בָּהֶם הַחִמּוּם אֲשֶׁר מִמֶּנּוּ נוֹלָדִים טִבְעֵי הַמַּיִם. וְהִתְעָרְבוּ בָּהּ הֶחֳמָרִים אֲשֶׁר בַּבְּרִיאָה, כָּעוֹף הָדוֹגֵר עַל הַבֵּיצִים בִּכְנָפָיו, וְאָז יוֹצֵא מֵאוֹתוֹ עוֹף בַּעַל חַי. כִּי דֶּרֶךְ הַחֲמִימוּת הַטִּבְעִית — הָאֵשׁ הַמְלַהֶטֶת — לְהוֹלִיד חֲמִימוּת. וּכְנָפֵי הָעוֹף מַה שֶּׁבָּהֶן נִכְנֶסֶת הַבֵּיצָה — מְצֻיָּרִים בָּהּ הַגּוֹזָלִים. וְלֹא הָיְתָה סִבַּת רִחוּף הַ"פַארַקְלִיט" (= פָרַקְלֵיט, רוּחַ הַקֹּדֶשׁ) עַל הַמַּיִם בִּדְמוּת הָעוֹף — אֶלָּא כְּדֵי שֶׁיִּתְהַוֶּה כָּל עוֹף בִּשְׁתֵּי כְנָפַיִם עַל אוֹתוֹ הַתֹּאַר ⁑

וּבַיּוֹם הַשֵּׁנִי בָּרָא הָאֵל אֶת הַשָּׁמַיִם הַתַּחְתּוֹנִים הַנִּקְרָאִים "הַגַּלְגַּל" (הָרָקִיעַ), אֲשֶׁר עָלָיו יִפֹּל מַבָּט בְּנֵי הָאָדָם לְמַעַן יֵדְעוּ אֶת טִבְעָהּ שֶׁל זוֹ הַאַחְרוֹנָה. וּבָרָא הָאֵל מֵעָלָיו שָׁמַיִם סוֹבְבִים וּמְעֻגָּלִים מִבְּטָנָם. וּמֵעַל לְשָׁמַיִם אֵלֶּה — אֵשׁ מְלֻהֶבֶת מִמֶּנָּה הַחֹם, וְשָׁמַיִם הַמִּתְרוֹמְמִים מֵעַל שָׁמַיִם הָאֵשׁ, וּשְׁנֵי שְׁמֵי שָׁמַיִם מְלֵאִים זִיו וְאוֹר אֲשֶׁר לֹא יְכוֹלוֹת הָעֵינַיִם הַנִּבְרָאוֹת לְהִסְתַּכֵּל בָּהֶם. וְהִפְרִיד הָאֵל הָאָדוֹן בֵּין הַמַּיִם אֲשֶׁר מֵעַל לָרָקִיעַ וּבֵין הַמַּיִם אֲשֶׁר מִתַּחַת לָרָקִיעַ. וְהַמַּיִם אֲשֶׁר לְמַעְלָה — עֲלִיָּתָם הָיְתָה כְּאֵד מְצֻמְצָם וְקָרוּשׁ, וְאוּלָם הַמַּיִם שׁוֹכְנִים בָּאֲוִיר וְלֹא יִזּוֹל מֵהֶם דָּבָר אֶל שׁוּם רוּחַ ⁑

וּבַיּוֹם הַשְּׁלִישִׁי הוּא יוֹם שְׁלִישִׁי בַּשַּׁבָּת ⁑ צִוָּה הָאֵל אֶת הַמַּיִם אֲשֶׁר מִתַּחַת לָרָקִיעַ שֶׁיֵּאָסְפוּ אֶל מָקוֹם אֶחָד, לְמַעַן תִּיבַשׁ הָאָרֶץ וְתִתְחַזֵּק וְתִתְמַצָּק. וְכַאֲשֶׁר נִפְרְדוּ הַמַּיִם וְהוֹפִיעוּ הַ"אִסְקוֹרָה" (יַבָּשׁוֹת) וְהִתְפַּתְּחוּ — וְהִיא עָפָר רָטֹב מִן הַמַּיִם הַמְעֹרָבִים בָּהֶם וּמֵעֲלֵיהֶם וּמִתַּחְתֵּיהֶם — הָיְתָה הָאָרֶץ נְקוּבָה לְהַחְדָּרָה, וְהָרוּחוֹת בָּהּ. כִּי הָאֲוִיר הָיָה עוֹלֶה מִן הָאָרֶץ וְשׁוֹכֵן בִּמְצוּלוֹתֶיהָ, וְאָז מִתְעַלִּים הָאֵדִים מֵאוֹתָהּ הַחֲמִימוּת אֲשֶׁר נוֹלְדָה בָּהּ, וְשׁוֹכֶנֶת הַקְּרִירוּת, לְמַעַן תִּתְבַּסֵּס הָאָרֶץ וְיֻצַּב שִׁמְעָהּ — כִּי הָאָרֶץ דְּחוּסָה (רוֹצֶה לוֹמַר, מִן הַמַּיִם דְּחוּסָה) עוֹמֶדֶת מֵעַל לַמַּיִם ⁑ וְצִוָּה הָאֵל אֶת הָאָרֶץ שֶׁתַּצְמִיחַ אֶת הַדֶּשֶׁא וְהַזֶּרַע וְהָאִילָנוֹת וְהָעֲשָׂבִים וְזוּלָתָם ⁑

וּבַיּוֹם הָרְבִיעִי הוּא יוֹם רְבִיעִי בַּשַּׁבָּת ⁑ הִוָּה הָאֵל אֶת הַשֶּׁמֶשׁ וְאֶת הַיָּרֵחַ וְאֶת הַכּוֹכָבִים, לְמַעַן תִּתְפַּשֵּׁט חֲמִימוּת הַשֶּׁמֶשׁ עַל פְּנֵי הָאָרֶץ וְתַשְׁפִּיעַ עָלֶיהָ מֵרִפְיוֹנָהּ, וְתִיבַּשׁ רְטִיבוּתָהּ…
