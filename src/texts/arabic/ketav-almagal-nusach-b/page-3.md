…מַכִּירִים אוֹתִי בַּהֱיוֹתִי בַּעַל יְדִיעָה בַּחָכְמָה עַד שֶׁאַשִּׁיב בִּתְשׁוּבָה. נִפְגַּשְׁנוּ יַחַד, וְהֶעֱמִיק וּפָגַע בּוֹ הַחֲדָשׁוֹת מֵחֲמָתִי. וְאָמַר: בְּנִי אַפְלָמִיס (= קְלֶמֶנְס), אַל יִכָּנֵס לִבְּךָ סָפֵק מֵאֵלֶּה הָאֲרוּרִים אוֹיְבֵי הָאֵל וּמְשִׁיחוֹ. וַאֲנִי אוֹדִיעֲךָ כָּל דָּבָר בִּשְׁלֵמוּת, בְּהַתְחָלָה וּבְסוֹף, וְאַעֲמִידְךָ עַל הָעִנְיָן מִתְּחִלַּת הַבְּרִיאָה עַד תַּכְלִית כָּל אֲשֶׁר נִכְתַּב בַּתּוֹרָה, וְעַל אֲשֶׁר לֹא נֶחְשַׂף לֹא בַּתּוֹרָה וְלֹא בְּסִפְרֵי הַנְּבִיאִים — כִּי הִפְקִיד אוֹתוֹ עִמִּי וְהוֹדִיעוֹ לִי אֲדוֹנִי וְאוֹהֲבִי וּמַשְׂאַת נַפְשִׁי הַמָּשִׁיחַ, וְדִבֵּר אֵלָיו הָאֵל פֶּה אֶל פֶּה אַחֲרֵי שֶׁשְּׁאַלְתִּיו עַל כָּךְ, וְהֵיטִיב עִמִּי בְּכָל אֲשֶׁר חָפַצְתִּי מִמֶּנּוּ. וַאֲנִי אוֹדִיעֲךָ אֶת יִחוּס הַגְּבִירָה הַטְּהוֹרָה אֵם טַהֲרַת הַטְּהוֹרִים, וְאֶמֶת הֱיוֹתָהּ בְּלִי סָפֵק מִיהוּדָה בֶּן יַעֲקֹב בֶּן יִצְחָק בֶּן אַבְרָהָם, מִשֵּׁבֶט יְהוּדָה. וְאוֹדִיעֲךָ אֶת הַסִּבָּה אֲשֶׁר בִּגְלָלָהּ הָיְתָה נְפִילַת הַמּוֹרֵד הַסּוֹטֶה — הַשָּׂטָן — מִן הַשָּׁמַיִם.

אַחַר כָּךְ פָּתַח וְאָמַר: בְּנִי אַפְלָמִיס, צָרִיךְ אַתָּה תְּחִלָּה לַעֲמֹד עַל הָאַחֲדוּת אֲשֶׁר בִּידִיעָתָהּ תַּשִּׂיג אֶת הַדְּבָרִים כֻּלָּם. דַּע וְהָבֵן כִּי הָאֵל הָאָדוֹן, הַיָּחִיד בְּעַצְמוּתוֹ, הַנֶּעֱבָד מִמֶּנּוּ בְּשִׁלּוּשׁ מְיֻחָד — הוּא הַמִּתְרוֹמֵם עַל הַמְּרוֹמִים, הַשָּׁוֶה עִם הַמְּרוֹמִים, וְאֵין לוֹ — מַה שֶּׁיִּהְיֶה — שׁוּם הַזְכָּרָה: לֹא לְמַטָּה וְלֹא בַּחוּץ וְלֹא בַּפְּנִים. הוּא אֲשֶׁר לִפְנֵי הַשֵּׂכֶל, הָעֶצֶם הַקַּדְמוֹן אֲשֶׁר אֵין לוֹ אַחֲרִית, וְאֵין מַשִּׂיג אוֹתוֹ שֵׂכֶל כָּל מַשִּׂיג מִן הַנִּבְרָאִים, וְלֹא יוּקַם לוֹ הֶבְדֵּל וְלֹא תֹּאַר. מְהַוֶּה הֶהֱוָיוֹת הָעֶלְיוֹנוֹת וְהַתַּחְתּוֹנוֹת, וְהוּא לְמַעְלָה מִן הַהֱוָיָה וְעִם הַהֱוָיָה וְתַחַת הַהֱוָיָה. הָעֶצֶם הַבּוֹרֵא, הָאוֹר הַנִּצְחִי אֲשֶׁר לֹא תַשִּׂיגֵהוּ הַחֲשֵׁכָה, הָאוֹרָה אֲשֶׁר לֹא תָרִים אוֹתָהּ הָעֵינַיִם, הַשּׁוֹכֵן בָּאוֹר.

לִפְנֵי הַנִּבְרָאִים הָיָה הוּא בּוֹנֵה הַיְּשָׁרוֹת — אֲשֶׁר מִמֶּנּוּ וּבוֹ נַעֲשׂוּ הַשָּׁמַיִם וְהָאָרֶץ. וּבָרָא לִפְנֵי הַפְרָדַת הַדְּבָרִים מַלְאָכִים הַמְשַׁבְּחִים אוֹתוֹ: עֲשָׂרָה גְּדוּדִים מִינִיִּים, רוֹצֶה לוֹמַר עֶשֶׂר מַעֲלוֹת. וְהָיְתָה הַמַּעֲלָה הָעֶלְיוֹנָה שֶׁבָּהֶן, הַקְּרוֹבָה לְכִסֵּא הָאֵל, הַשּׁוֹפַעַת לַתִּשְׁבָּחוֹת — מַעֲלַת "סַטַנָאיִיל" (שׂטנאל), אֲשֶׁר הוּא הָאַרְכוֹן, וְהָיוּ הַתִּשְׁבָּחוֹת עוֹלוֹת אֶל הָאֵל מִכָּל הַמַּלְאָכִים עַל יָדָיו.

וּבַתְּחִלָּה, בַּיּוֹם הָרִאשׁוֹן הוּא יוֹם רִאשׁוֹן בַּשַּׁבָּת, רֹאשׁ הַיָּמִים וּבְכוֹרָם — בָּרָא הָאֵל אֶת הַשָּׁמַיִם הָעֶלְיוֹנִים וְאֶת הָאָרֶץ וְאֶת הַמַּעֲלָה הָעֶלְיוֹנָה שֶׁל הַמַּלְאָכִים, וְהִיא מַעֲלַת סַטַנָאיִיל, וְאֶת הַמַּעֲלוֹת הַמְיֻחָדוֹת לְרָאשֵׁי הַמַּלְאָכִים — הַכֹּחוֹת וְהַשִּׁלְטוֹנוֹת וְהָאֱלֹהִיִּים וְהַשְּׂרָפִים.
