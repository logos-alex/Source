…אֶת הַחֵרְשִׁים, וִיקַיֵּם אֶת הַפִּסְּחִים, וִיטַהֵר אֶת הַמְצֹרָעִים, וִירַפֵּא אֶת הַכְּאֵבִים, וִידַבְּרוּ הָאִלְּמִים בְּצִוּוּיוֹ וְאִסּוּרוֹ. וְהוּא הוֹלֵךְ עַל הָאָרֶץ, וְהַשָּׁמַיִם מְלֵאִים מִמֶּנּוּ. וְלֹא הִתְעַלֵּם הָאֵל מֵעוֹלָם מִבְּרוּאָיו אֶלָּא לְמַעַן יוֹפִיעַ בִּלְבוּשׁוֹ וְיֵלֵךְ בֵּין בָּנַי עַל פְּנֵי הָאָרֶץ — בְּחֶשְׁקוֹ וּרְצוֹנוֹ, בִּהְיוֹת הָאֵל הַפּוֹעֵל מַה שֶּׁיִּרְצֶה בְּיָדוֹ, וְהוּא מַקִּיף אֶת בְּרוּאָיו, וְכָל בְּרוּאָיו אֵינָם עוֹמְדִים עַל סוֹדוֹ.

לוֹ מִמֶּנּוּ הַתִּשְׁבָּחוֹת וְהַשִּׁבּוּחִים, הָאָדוֹן הַמָּשִׁיחַ הַמּוֹפִיעַ בְּכֹחַ מֵרֶחֶם הַבְּתוּלָה — טַהֲרָתוֹ אֵין לְהַשִּׂיג אֵיכוּתָהּ — בִּלְבוּשׁוֹ תִּינוֹק יוֹנֵק מִדַּד אוֹתָהּ אֲשֶׁר נִתְמַלָּה מִמֶּנּוּ רַחְמָהּ, וְהִיא טְהוֹרָה כְּטַהֲרַת הַקַּדְמוֹן, לֹא כְּטַהֲרַת הַהֶרְגֵּל. וְיִתְגַּדֵּל בֵּין הַבָּנוֹת וְהַבָּנִים, וְיוֹפִיעַ אֶת פְּלָאָיו הַשְּׁמֵימִיִּים בֵּין בְּנֵי הָאָדָם עַל פְּנֵי הָאָרֶץ, וְיַשְׁלִים אֶת רְצוֹנוֹ בְּעוֹלָמוֹ, וְיַעֲמִיד מִי שֶׁיִּרְצֶה תַּחְתָּיו בָּאָרֶץ.

וְיָרִים אֶת גְּוִיָּתוֹ הַשְּׁכוּחָה כְּגוּפִי, מִן הַבָּשָׂר וְהַדָּם — עַל עֵץ הַצְּלָב. וּבְאוֹתוֹ הַיּוֹם תִּהְיֶה יְשׁוּעָתִי מֵחֶטְאִי, כְּמוֹ שֶׁהִבְטִיחַ לִי הָאָב הָאֵל הָרַחוּם בְּיוֹם עָרְטָלוֹ אוֹתִי בְּצֵאתִי מִן הַפַּרְדֵּס אֶל הָהָר הַקָּדוֹשׁ הַזֶּה.

עַל כֵּן צָרִיךְ לָדַעַת שֶׁכֹּל אֲשֶׁר הִפְקַדְתִּי בְּיָדְךָ מִצַּוָּאָתִי זוֹ — בְּהֶכְרֵחַ יִהְיֶה, כִּי הָאֵל הוֹדִיעֵנִי אוֹתוֹ. וְדַע כִּי מוֹשִׁיעִי מֵרִים גְּוִיָּתוֹ בְּאוֹתָם הַיָּמִים עַל עֵץ הַצְּלָב, וְיַשְׁלִים מוֹתוֹ וּקְבוּרָתוֹ. וְהָאָדוֹן הַדָּבָר הַשּׁוֹכֵן בּוֹ אֲשֶׁר בָּחַר לוֹ לְבוּשׁ דּוֹבֵר מִמֶּנּוּ מְדַבֵּר לִבְנֵי הָאָדָם — מֵרִים אוֹתוֹ בְּכֹחוֹ מִקַּבְרוֹ, וִיטַהֲרֵהוּ אֶל אֵלֶּה שֶׁהִפְקִיד אֶצְלָם סוֹדוֹתָיו, לְמַעַן יִשְׂמְחוּ לְבָבוֹתָם, וְיִתְאַמֵּת אֶצְלָם שֶׁהָאֵל הָאָדוֹן הַדָּבָר יָרַד מִשְּׁמָיו וְהִתְגַּשֵּׁם בִּדְמוּת גּוּפֵיכֶם מִבָּשָׂר וְדָם כִּבְרִיאָתִי, לְמַעַן יְקַיֵּם דִּבּוּר הָאֵל בְּאָמְרוֹ לִי לִפְנֵי שֶׁבְּרָאַנִי אֶל הַמְּצִיאוּת: "נִבְרָא אָדָם כִּדְמוּתֵנוּ וְכִצְלָמֵנוּ" — וּבָרָאַנִי בַּצּוּרָה בִּלְבוּשׁ מוֹשִׁיעִי, הָאָדוֹן הַצַּדִּיק, הָאֵל הַדָּבָר אֲשֶׁר בָּרָא הָאָב בּוֹ אֶת כָּל בְּרוּאָיו בַּשָּׁמַיִם וּבָאֲרָצוֹת.

וּבְהֶכְרֵחַ יָבוֹא עַל הָאָרֶץ מַבּוּל הַמַּיִם וְיַשְׁמִיד
