## חזון אברהם – חלק ז' (חזון הבריאה וגן העדן)

מו וַיִּשְׁמַע קוֹלִי הַמַּלְאָךְ: "כִּנְדַּעְתִּי כִּי הַשָּׁמַיִם וַהָאָרֶץ בִּנְיַן מִשָּׁמֶּן וַיֵּשׁ לָהֶם סוּף. וַחֲנֹנָם עֲדַיִן יִשְׁמוּ בִּשְׁמִי וַיִּשְׁמְרוּ אֶת-דָּבְרִי עַד עִת קִצָּם. וַאֲחַר קִצָּם יִהְיוּ אֲחֵרִים חֲדָשִׁים וַיִּשְׁמוּ בִּשְׁמִי וַיִּשְׁמְרוּ אֶת-דָּבְרִי וַיִּשְׁמְרוּ אֶת-בְּרִיתִי".

מז "וַיִּשְׁמַע קוֹלִי יִשְׂרָאֵל הַלְאוֹם הַנִּבְחַר, אֲשֶׁר בָּחַרְתִּי לַאֲבוֹתֶיךָ מִרְאשִׁית הָעוֹלָם. וַיִּהְיוּ שׁוֹמְרִים אֶת-דָּבְרִי וּמִלִּים אֶת-דְּרָכַי. וּלְאַחֲרִית הַיָּמִים יִשְׁלַח הַקַּדְמוֹן אֶת-שְׁלִיחוֹ לְהָשִׁיב אוֹתְכֶם וּלְנַחֲמֵיכֶם עַל-כָּל-הָרָעוֹת אֲשֶׁר בָּאוּ עֲלֵיכֶם. וִישׂוּ אַתֶּם רַבִּים בִּמְמָלוֹ שֶׁל הַגָּאוּל אֲשֶׁר יִשְׁלַח לָכֶם".

מח וַיִּשְׁמַע קוֹלִי וּמַרְאֵה לִי אֶת-הַגִּנָּה הַנִּשְׁמֶרֶת, גַּן עֵדֶן, וּמַרְאֵה לִי כָּל-הָעֵצִים אֲשֶׁר בוֹ וַעִם זִנָּה אֲשֶׁר בַּעֵצִים, וְנַחֲלוֹתֶיהָ וַחַיּוֹתֶיהָ הַטּוֹבוֹת. וַיִּשְׁמַר עַל־גַּן עֵדֶן שִׁמּוּר גָּדוֹל, וּמַלְאָכִים עוֹמְדִים בִּפִּתְחוֹ.

מט וַיִּשְׁמַע קוֹלִי: "הִנְּנִי מַגִּיד לְךָ אֶת-כָּל-אֲשֶׁר בַּגִּנָּה. וּדִיתוֹ נְתוּנָה לִי לִרְאוֹת אוֹתוֹ וּלְיַדֵּע אֶת-כָּל-מִלְאוֹ. וַחֲנֹנָם בִּשְׁמִי וַלִּמּוּדִים בִּי. וּמִזְוּוֹת הַגַּן אֲשֶׁר בִּתְחִלָּה נִשְׁמְרוּ בַּחֳרַב וּמִשְׁמָר עַל־פִּתְחוֹ הַמַּלְאָךְ הַנִּשְׂמָר אוֹתוֹ בְּנִשְׁמַר גַּם הוּא עַתָּה לִי וּלְזַרְעֲךָ".

נ "וַחֲנֹנָם הָעֵצִים אֲשֶׁר בַּגִּנָּה, וְאֵלֶּה הֵם: עֵץ הַחַיִּים בְּתוֹךְ הַגִּנָּה, עִם זִנָּה וְעִם אִשָּׁה עָשִׂיתִי לָהּ הַשְׁקָיָה מֵעַיִן בַּגִּנָּה. וְהַנַּחַל נִשְׁמַר הַנִּשְׁפַּךְ מִמֶּנָּה לְזַוְּוּוֹת הַגַּן וּלְעִטּוּף הָאָרֶץ כֻּלָּהּ. וּמִשָּׁם הִוא נִשְׁמַר בִּשְׁלֹשָׁה עַנְפִּים נִשְׁמָרוּ לַרְחוֹק".

נא וּמִשַּׁם הִשְׁתַּלַּח הַנָּהָר הַגָּדוֹל בַּמִּדְבַּר וַיִּתְּנוּ לוֹ שִׁמּוּר. וַעָלָיו הַחַיּוֹת הַטּוֹבוֹת שָׁמוּ שִׁמּוּר. וַחֲנֹנָם הַיִּדְעוּ אִתִּי וַשּׁוֹמְרִים דָּבְרִי בַּגִּנָּה וַעִם הַחַיּוֹת אֲשֶׁר בָּהּ. וּתַחַת הַנַּחַל יֵשׁ וִזְרוּעִי שְׁמוּרִים בַּמַּיִם הַנִּשְׁמָרִים לַעִיר הַנִּשְׁמַרֶת הַנִּשְׁמֶרֶת לִירוּשָׁלַיִם בַּמִּדִּין וּבַמִּשְׁמָר".

נב וַיִּשְׁמַע קוֹלִי הַמַּלְאָךְ וַיִּשְׁמַע קוֹלִי הַקָּדוֹשׁ מִן הַכִּסֵּא: "הִנְּנִי מַרְאֶה לְךָ אֶת-כָּל-הַדָּבָר. וְנַחֲלוֹת הַגַּן וַהַיִּדְעוּ מִן הַנַּחַל הַגָּדוֹל אֶת-דָּבְרִי וַשׁוֹמְרִים אֶת־בְּרִיתִי וּנִשְׁמָרוּ מִן־הַנּוֹתְנוּ לָהֶם סִנְדוּן הָאִשָּׁה הַנִּשְׁמֶרֶת בַּחִנְנוֹן הַנִּשְׁמָר בַּגִּנָּה".
