## חזון אברהם – חלק ו' (העלייה לשמיים וחזון המרכבה)

מ וַיִּשָּׂא אוֹתִי עַל-כַּנְפֵיו וַיֹּלִיךְ אוֹתִי עַל-הָרַקִּיעִים. וַיִּשְׁמַע קוֹלִי: "אַבְרָם! רְאֵה אֶת-אֲשֶׁר תִּשְׁמָע וְאֶת-אֲשֶׁר תִּרְאֶה. כִּי לִי לִבְדִּי לִשְׁמוֹר עַל יִשְׂרָאֵל, הַלְאוֹם הַנִּבְחַר, וּלְךָ אִשְׁמֹר אַתָּה בְּבִרִיתִי. רְאֵה אֶת-כִּסְאִי וְאַתָּה רָאִיתָ אֶת כִּסֵּא כְּבוֹדִי. וְאֶת-זִיו עַצְמוּתִי וְאַתָּה תֵּדַע. וְאֶת-הַחִידוֹת שֶׁשֶּׁאַל דִּבַּרְתִּי, אַתָּה תֵּדַע. וְאֶת-כָּל-מִסְתְּרוֹת הַשָּׁמַיִם וַהָאָרֶץ, וְאֶת-בּוֹא הַמִּשִׁיחַ וְאַחֲרִית הַיָּמִים, אַתָּה תִּרְאֶה וְתֵדַע".

מא וַיְהִי כִּי שָׁמְעוּ דְּבָרִים אֵלֶּה, וַיִּשְׁמַר אַבְרָם, וַיִּשְׁמַע קוֹלִי הַמַּלְאָךְ: "אַל-תִּלְחַץ עַל-שְׁמוֹ הַגָּדוֹל וְהַנּוֹרָא. הִנָּחֵם הִצַּבְתִּי בִּצְלוֹ שְׁמוֹ הַגָּדוֹל, הַגָּדוֹל וְהַנִּשְׁגָּב. וַיִּשְׁמַר אַבְרָם מִשֵּׁם זֶה וַיִּבָּתֵר מִן-הַשֵּׁם הַנִּשְׁמַר בִּכְלִי קַדּוֹשׁ".

מב וַיִּשְׁמַע קוֹלִי מִן הַכִּסֵּא קוֹל גָּדוֹל וְנִישׁ, קוֹל כְּקוֹל דַּמּוּיוֹת רַבּוֹת וּמִתְנַצְּחוֹת בַּשָּׁמָיִם: "זֶה הוּא הַדָּבָר אֲשֶׁר אָמַרְתִּי לְךָ בִּתְחִלָּה, בִּהְיוֹת אַתָּה בָּאָרֶץ. הִנָּחֵם כִּי בָחַרְתִּי בְּךָ וּבִזְרָעְךָ לְבָרִית עוֹלָם. וּלְךָ תִּתֵּן הָאָרֶץ אֲשֶׁר דִּבַּרְתִּי לְךָ לִנְחָלָה. וַאֲנִי אִהְיֶה לְךָ לֵאלֹהִים וּלְזַרְעֲךָ אַחֲרֶיךָ. וְשִׁמְךָ יִקָּרֵא בָּאָרֶץ כִּשְׁמִי, וִירוּשָׁלַיִם תִּהְיֶה לִמְקוֹם שְׁמִי, וְהַבַּיִת אֲשֶׁר יִבָּנֶה שָׁם יִהְיֶה קָדוֹשׁ לַשְּׁמִי".

מג וַיִּשְׁמַע קוֹלִי הַמַּלְאָךְ: "בַּיָּמִים הַהֵם יִהְיֶה לְךָ כָּל-אֲשֶׁר אָמַרְתִּי. וְתִשְׁמֹר אַתָּה אֶת-בְּרִיתִי וְאֶת-דְּרָכַי. וַיִּהְיוּ בָנֶיךָ וִבְנֵי בָנֶיךָ שׁוֹמְרִים אֶת-הַבְּרִית וּמִלִּין אֶת-דְּרָכַי. וּלְאַחֲרִית הַיָּמִים יִשְׁלַח הַקַּדְמוֹן אֶת-שְׁלִיחוֹ לְשׁוּב אֶתְכֶם מִן-הַשֵּׁבוּת וּלְהִתְחַדֵּשׁ הַבַּיִת הַנִּשְׁרַף בָּאֵשׁ".

מד וַיְהִי כִּי דִבֵּר אֵלַי, וַיְהִי קוֹל מֵהַשָּׁמַיִם הִנּוֹצֵץ בְּבִרְקִים וְרִשְׁרוּשׁ מַיִם וּדְקִירוֹת אוֹר חִוַּור מֵעַל הַכִּסֵּא. וַיִּשְׂמַח לִבִּי וַתִּשְׂמַח נַפְשִׁי וְרוּחִי בִּרְגָעִים בּוֹ. וַאֲנִי אַבְרָם הִצְטַלַּחְתִּי בְּדִרְכִּי אָחוֹר וַעֲמַדְתִּי בִּשְׁמַחַת לִבִּי.

מה וַיִּשְׁמַע קוֹלִי הַמַּלְאָךְ: "עַכְשָׁיו יָדַעְתָּ אֶת-הַדְּבָרִים הַנִּשְׁמָרִים לִךְ וּלְזַרְעֲךָ. קוּם עַכְשָׁיו וִלַּח מֵהֶם. וַאֲנִי אֶהְיֶה עִמְּךָ עַד כִּלָּיוֹן יָמֶיךָ".
