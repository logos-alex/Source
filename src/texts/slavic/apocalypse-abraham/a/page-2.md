## חזון אברהם – חלק ב' (אברהם והאל בַּרִישָׂת)

י וַיְהִי כִּי חָשַׁבְתִּי לְהִתְחַמֵּק מֵחֲרוֹן אָבִי, וָאֵצֵא הַחוּצָה. וַיִּקְרָא לִי לֵאמֹר: "אַבְרָם!" וָאֹמַר: "הִנֵּנִי". וַיֹּאמַר: "קַח וֶאֱסֹף אֶת-שִׁבְרֵי הָעֵץ שֶׁבָּהֶם עָשִׂיתִי אֱלִילִים, וְטֶרֶם תָּבוֹא, הָכֵן לִי אֲרוּחַת צָהֳרָיִם".

יא וַיְהִי כַּאֲשֶׁר אָסַפְתִּי אֶת-שִׁבְרֵי הָעֵץ, וָאֶמְצָא בְתוֹכָם אֱלִיל קָטָן, שֶׁהָיָה יָכוֹל לְהִכָּנֵס בְּכַף יָדִי. וְכָתוּב הָיָה עַל-מִצְחוֹ: "הָאֵל בַּרִישָׂת". וְלֹא הִגַּדְתִּי לְאָבִי כִּי מָצָאתִי אֶת-הָאֵל בַּרִישָׂת בֵּין הַשְּׁבָבִים.

יב וַיְהִי כַּאֲשֶׁר שַׂמְתִּי אֶת-הַשְּׁבָבִים עַל-הָאֵשׁ לְהָכִין אֶת-אֲרוּחַת אָבִי, וָאֵצֵא לְבַקֵּשׁ דְּבַר-מַאֲכָל, הִצַּבְתִּי אֶת-בַּרִישָׂת לְיַד הָאָח הַבּוֹעֶרֶת, וָאֹמַר לוֹ בְּלַעַג: "בַּרִישָׂת, שְׁמֹר פֶּן-תִּכְבֶּה הָאֵשׁ עַד אֲשֶׁר אָשׁוּב. וְאִם תִּכְבֶּה, נְשֹׁף בָּהּ וְתִתְלַקַּח".

יג וָאֵצֵא וָאֶעֱשֶׂה אֶת-עִנְיָנַי. וּבְשׁוּבִי, מָצָאתִי אֶת-בַּרִישָׂת נֹפֵל עַל-גַּחַלְתּוֹ, וְרַגְלָיו אֲחוּזוֹת אֵשׁ וְנִשְׂרָפוֹת קָשָׁה. וָאֶשְׂחַק בְּקִרְבִּי שְׂחוֹק גָּדוֹל, וָאֹמַר: "בֶּאֱמֶת, בַּרִישָׂת, יוֹדֵעַ אַתָּה לְלַבּוֹת אֵשׁ וּלְבַשֵּׁל אֲרוּחָה!". וְעוֹדֶנִּי מְדַבֵּר בִּשְׂחוֹקִי, וְהוּא מְעַט-מְעַט נִשְׂרַף בָּאֵשׁ וַיְהִי לְאֵפֶר.

יד וָאָבֵא אֶת-הָאֲרוּחָה לְאָבִי וַיֹּאכַל. וָאֶתֵּן לוֹ יַיִן וְחָלָב, וַיֵּשְׁתְּ וַיִּתְעַנַּג, וַיְבָרֶךְ אֶת-מָרוּמָת אֱלֹהָיו. וָאֹמַר לוֹ: "אָבִי תֶּרַח, אַל-תְּבָרֵךְ אֶת-מָרוּמָת אֱלֹהֶיךָ וְאַל-תְּהַלְּלֵהוּ. הַלֵּל-נָא אֶת-בַּרִישָׂת אֱלֹהֶיךָ, כִּי כְּדֵי לְהַרְאוֹת אַהֲבָתוֹ אֵלֶיךָ, הִשְׁלִיךְ עַצְמוֹ לָאֵשׁ לְבַשֵּׁל אֶת-מַאֲכָלֶךָ". וַיֹּאמֶר אֵלַי: "וְהֵיכָן הוּא עַתָּה?". וָאֹמַר: "הָיָה לְאֵפֶר בַּחֲמַת הָאֵשׁ וַיְהִי לְעָפָר". וַיֹּאמֶר: "גָּדוֹל כֹּחוֹ שֶׁל בַּרִישָׂת! אֶעֱשֶׂה אַחֵר הַיּוֹם, וּמָחָר יָכִין הוּא אֶת-אֲרוּחָתִי".

טו וַאֲנִי אַבְרָם, כְּשָׁמְעִי דְּבָרִים כָּאֵלֶּה מֵאָבִי, וָאֶשְׂחַק בְּדַעְתִּי, וָאֶאֱנַח בְּמָרַת וְחֵמַת נַפְשִׁי, וָאֹמַר: "אֵיךְ יָכֹל מַעֲשֵׂה יָדָיו לִהְיוֹת עוֹזֵר לְאָבִי? הַאִם גּוּף יִהְיֶה כָּפוּף לְנַפְשׁוֹ, וְנֶפֶשׁ נִתְּנָה לוֹ, אַךְ הוּא בְּאִוַּלְתּוֹ חָסֵר דַּעַת?".

טז וָאֹמַר: "אַחַת הִיא, אֶסְבֹּל אֶת-הָרָעָה, אַךְ אֶשְׁלַח אֶת-דַּעְתִּי אֶל-הַטָּהֳרָה, וּמַחְשְׁבוֹתַי אָשִׂים גְּלוּיוֹת לְפָנָיו". וָאֶעֱנֶה וָאֹמַר: "אָבִי תֶּרַח, אֶת-מִי מֵהֶם תְּהַלֵּל, בְּאִוֶּלֶת מַחְשַׁבְתֶּךָ? הִנֵּה אֱלֹהֵי אָחִיךָ נָחוֹר נִצָּבִים בַּהֵיכָל. הֲלֹא נִכְבָּדִים הֵם מֵאֱלֹהֶיךָ? הִנֵּה הָאֵל זוּס שֶׁל אָחִיךָ נָחוֹר נִכְבָּד מִמָּרוּמָת אֱלֹהֶיךָ, כִּי מִזָּהָב טָהוֹר מַעֲשֵׂה אָדָם הוּא. וְאִם יִתְבַּלֶּה מֵרֹב יָמִים, יְחַדְּשׁוּהוּ. אֲבָל מָרוּמָת, אִם יִשָּׁבֵר אוֹ יִשָּׂרֵט, לֹא יִתְחַדֵּש עוֹד, כִּי אֶבֶן הוּא. וּמַה בַּאֲשֶׁר לָאֵל חוֹן הָעוֹמֵד עִם זוּס? וּבַרִישָׂת עַצְמוֹ נִשְׂרַף בָּאֵשׁ וְהָיָה לְאֵפֶר וְאֵינֶנּוּ עוֹד, וְאַתָּה אוֹמֵר: 'מָחָר אֶעֱשֶׂה אַחֵר'? אָבַד לַאֲבַדּוֹן!

יז הִנְנִי אוֹמֵר: הָאֵשׁ נִכְבֶּדֶת מִן-הַפְּסִילִים, כִּי הַבִּלְתִּי-נִכְנָעִים נִכְנָעִים לָהּ, וְהִיא מִתְלוֹצֶצֶת בַּכָּלִים בְּשַׁלְהַבְתָּהּ. אַךְ גַּם אֶת-הַמַּיִם לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הֵם גּוֹבְרִים עַל-הָאֵשׁ וּמַרְוִים אֶת-הָאָרֶץ. אַךְ גַּם אוֹתָם לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הֵם נִכְנָעִים תַּחַת הָאָרֶץ וּמְחַלְחֲלִים בָּהּ. אִם-כֵּן, אֶת-הָאָרֶץ אֲכַנֶּה נִכְבֶּדֶת, כִּי הִיא גּוֹבֶרֶת עַל טֶבַע הַמַּיִם. אַךְ גַּם אוֹתָהּ לֹא אֲכַנֶּה אֱלוֹהַּ, כִּי הַשֶּׁמֶשׁ מְיַבֶּשֶׁת אוֹתָהּ. אֶת-הַשֶּׁמֶשׁ אֲכַנֶּה נִכְבֶּדֶת, כִּי בְּקַרְנֶיהָ הִיא מְאִירָה אֶת-הַתֵּבֵל כֻּלָּהּ. אַךְ גַּם אוֹתָהּ לֹא אָשִׂים לֵאלֹהַּ, כִּי בַּלַּיְלָה הַיָּרֵחַ וְהֶעָנָן מַאֲפִילִים אֶת-אוֹרָהּ. וְגַם אֶת-הַיָּרֵחַ וְהַכּוֹכָבִים לֹא אֶקְרָא אֱלוֹהַּ, כִּי גַּם הֵם בְּעִתָּם בַּלַּיְלָה נֶעְלָמִים.

יח שְׁמַע, תֶּרַח אָבִי. אֲנִי מְחַפֵּשׂ לְפָנֶיךָ אֶת-הָאֵל אֲשֶׁר בָּרָא אֶת-הַכֹּל, וְלֹא אֶת-אֵלֶּה שֶׁאָנוּ סוֹבְרִים שֶׁהֵם אֱלֹהִים. מִי הוּא, אוֹ מַה הוּא, אֲשֶׁר הִטָּה שָׁמַיִם, אֲשֶׁר קָבַע אֶת-הַשֶּׁמֶשׁ וְאֶת-אוֹר הַיָּרֵחַ וְהַכּוֹכָבִים עִמּוֹ, אֲשֶׁר יִבֵּשׁ אֶת-הָאָרֶץ בְּתוֹךְ מַיִם רַבִּים, אֲשֶׁר אוֹתְךָ עַצְמְךָ שָׂם בֵּין הָאֱלִילִים? הַלְוַאי וְיִתְגַּלֶּה לָנוּ הוּא עַצְמוֹ!".
