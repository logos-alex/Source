## חזון אברהם – חלק ה' (המסע, הקורבן, והעימות עם עזאזל)

לב וַנֵּלֵךְ שְׁנֵינוּ יַחְדָּו, אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה. וְלֶחֶם לֹא אָכַלְתִּי וּמַיִם לֹא שָׁתִיתִי, כִּי מַאֲכָלִי הָיָה לִרְאוֹת אֶת-הַמַּלְאָךְ אֲשֶׁר עִמִּי, וּמִשְׁתִּי – דְּבָרָיו אֲשֶׁר דִּבֵּר אֵלַי.

לג וָאָבֹא אֶל-הַר חוֹרֵב, הַר הָאֱלֹהִים. וָאֹמַר אֶל-הַמַּלְאָךְ, שַׂר הַקַּדְמוֹן: "הִנֵּה אֵין עִמִּי קָרְבָּן, וְאֶת-מְקוֹם הַמִּזְבֵּחַ עַל-הָהָר לֹא יָדַעְתִּי. אֵיךְ אֶעֱשֶׂה אֶת-הַקָּרְבָּן?". וַיֹּאמַר: "הַבֵּט אֲחוֹרֶיךָ". וָאַבִּיט, וְהִנֵּה הוֹלְכִים אַחֲרֵינוּ כָּל-הַקָּרְבָּנוֹת שֶׁצֻּוּוּ: הָעֶגְלָה, וְהָעֵז, וְהָאַיִל, וְהַתּוֹר, וְהַגּוֹזָל.

לד וַיֹּאמֶר אֵלַי הַמַּלְאָךְ: "אַבְרָם!". וָאֹמַר: "הִנֵּנִי". וַיֹּאמֶר לִי: "שְׁחַט אֶת-כֻּלָּם וְנַתַּח אוֹתָם. אֶת-הַנְּתָחִים שִׂים זֶה מוּל זֶה, אַךְ אֶת-הָעוֹף לֹא תְבַתֵּר. וְתֵן אוֹתָם לָאֲנָשִׁים אֲשֶׁר אַרְאֶךָּ עוֹמְדִים לְצִדְּךָ, כִּי הֵם הַמִּזְבֵּחַ עַל-הָהָר לְהַקְרִיב קָרְבָּן לַקַּדְמוֹן. וְאֶת-הַתּוֹר וְאֶת-הַגּוֹזָל תֵּן לִי, וַאֲנִי אֶעֱלֶה עַל-כַּנְפֵי הָעוֹפוֹת לְהַרְאוֹתְךָ אֶת-אֲשֶׁר בַּשָּׁמַיִם וּבָאָרֶץ וּבַיָּם וּבַתְּהוֹמוֹת, וּבְגַן-עֵדֶן וּבִנְהָרוֹתָיו, וְאֶת-מִלְאוֹ הַתֵּבֵל כֻּלָּהּ. וְאֶת-סוֹבֵב הָעוֹלָם תִּרְאֶה בְּכֻלּוֹ".

לה וָאֶעֱשׂ כְּכֹל אֲשֶׁר צִוָּה הַמַּלְאָךְ. וָאֶתֵּן לַמַּלְאָכִים אֲשֶׁר בָּאוּ אֵלֵינוּ אֶת-הַבְּהֵמוֹת הַמְנֻתָּחוֹת. וְאֶת-הָעוֹפוֹת לָקַח הַמַּלְאָךְ. וָאֲחַכֶּה לְמִנְחַת עָרֶב.

לו וַיֵּרֶד עַיִט טָמֵא עַל-הַפְּגָרִים, וָאַבְרֵחַ אוֹתוֹ. וַיִּקְרָא אֵלַי הָעַיִט הַטָּמֵא וַיֹּאמַר: "מַה-לְּךָ, אַבְרָם, בַּמְּרוֹמִים הַקְּדוֹשִׁים, אֲשֶׁר שָׁם לֹא אוֹכְלִים וְלֹא שׁוֹתִים, וְאֵין בָּהֶם מַאֲכַל אָדָם, אֶלָּא הַכֹּל בָּאֵשׁ נֶאֱכָלִים וְאוֹתְךָ יְכַלּוּ? עֲזֹב אֶת-הָאִישׁ אֲשֶׁר עִמָּךְ וּבְרַח, כִּי אִם תַּעֲלֶה לַמָּרוֹם, יַשְׁחִיתוּךָ!".

לז וַיְהִי כִּרְאוֹתִי אֶת-הָעוֹף הַמְדַבֵּר, וָאֹמַר אֶל-הַמַּלְאָךְ: "מֶה-הוּא זֶה, אֲדֹנִי?". וַיֹּאמַר: "זֶהוּ חֹסֶר-הַצֶּדֶק, זֶהוּ עֲזָאזֵל. כִּי בָּחַר לוֹ לָגוּר עַל-הָאָרֶץ, וְחֶלְקוֹ עַל-הָאָרֶץ, וְאַתָּה חֶלְקְךָ בַּשָּׁמָיִם". וַיֹּאמֶר לוֹ: "חֶרְפָּה לְךָ, עֲזָאזֵל! כִּי בָּחַרְתָּ וְאָהַבְתָּ אֶת-הַמָּקוֹם הַמְטֻנָּף. עַל-כֵּן נְתָנְךָ הָאָדוֹן הַקַּדְמוֹן, הֶחָזָק, לִהְיוֹת שׁוֹכֵן עַל-הָאָרֶץ. וּבְךָ רוּחַ הַשֶּׁקֶר הָרָעָה, וּבְךָ חֵמָה וּמַכְאוֹבִים עַל-דּוֹרוֹת הָרְשָׁעָה. כִּי לֹא נָתַן הָאָדוֹן הַקַּדְמוֹן, הֶחָזָק, אֶת-גּוּפוֹת הַצַּדִּיקִים בְּיָדְךָ, לְמַעַן יִתְחַזְּקוּ בָּהֶם חַיֵּי הַצַּדִּיק וְאַבְדַן הָרָשָׁע. שְׁמַע, יְדִידִי, וְהִכָּלֵם מִמֶּנִּי! כִּי לֹא נִתַּן לְךָ לְפַתּוֹת אֶת-כָּל-הַצַּדִּיקִים. סוּר מֵאִישׁ זֶה! אֵינְךָ יָכוֹל לְהַטְעוֹתוֹ, כִּי אוֹיֵב הוּא לְךָ, וְגַם לַהוֹלְכִים בְּעִקְּבוֹתֶיךָ וְלָאוֹהֲבִים אֶת-אֲשֶׁר תִּרְצֶה. הִנֵּה הַלְּבוּשׁ אֲשֶׁר הָיָה לְךָ בַּשָּׁמַיִם נִטַּל מִמְּךָ מִקֶּדֶם, וְהַכָּבוֹד אֲשֶׁר הָיָה עַל-גּוּפְךָ, עָבַר אֵלָיו".

לח וַיֹּאמֶר הַמַּלְאָךְ לְאַבְרָם: "אֱמֹר לוֹ: 'הֱיֵה גֶּחָלִים לַאֲדָמָה! סוּר, עֲזָאזֵל, אֶל-מְקוֹמוֹת הַשְּׁאוֹל הַבִּלְתִּי-נִכְנָסִים!'". וַיֹּאמֶר לִי הַמַּלְאָךְ: "אַל-תַּעֲנֶה לוֹ, כִּי נִתְּנָה לוֹ הָרְשׁוּת לְפַתּוֹת אֶת-אֵלֶּה הָעוֹנִים לוֹ". וְכַאֲשֶׁר דִּבֵּר אֵלַי, לֹא עָנִיתִי לוֹ.

לט וַיְהִי כַּאֲשֶׁר הַשֶּׁמֶשׁ בָּאָה, וְהִנֵּה עָשָׁן כְּעֶשַׁן כִּבְשָׁן. וַיַּעֲלוּ הַמַּלְאָכִים מֵעַל הַנְּתָחִים, מֵרֹאשׁ הַכִּבְשָׁן הַמַּעֲלֶה עָשָׁן.
