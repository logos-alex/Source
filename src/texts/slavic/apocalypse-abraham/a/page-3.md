## חזון אברהם – חלק ג' (התגלות האל וחורבן בית תרח)

יט וּלְמָחֳרָת, בַּבֹּקֶר, קַמְתִּי וָאֵצֵא לְהַרְאוֹת לְאָבִי אֶת-הַמִּזְבֵּחַ אֲשֶׁר עִם הָאֱלִילִים. וְכָל-הָעִיר נִשְׁמְעוּ קוֹלִי כִּי שׁוֹחַק אֲנִי בְּלִבִּי. וָאֹמַר בְּקוֹל גָּדוֹל לַעַד: "הִנֵּה אֲנִי עוֹשֶׂה אֶת-מַעֲשַׂי לַמַּלְאָכִים הַשּׁוֹמְרִים עִם הַקָּדוֹשׁ בַּשָּׁמַיִם, וְכִי הָעִיר כֻּלָּהּ תְּשׁוֹמַע וְתִדַּע כִּי אֱלִילִים לֹא הֵם".

כ וַיְהִי כַּאֲשֶׁר הִשְׁמַעְתִּי אֶת-קוֹלִי, וְכָל-הָעִיר רַעַם וַתִּשְׁמַע קוֹלִי וַתִּלְהַט בְּאוּר עַל-הַשּׁוֹחֲרִים בִּשְׁחוֹקָם הָמִּים. וּמִשַּׁחַק עִיר בִּשְׁחוֹקָם הָמִּים, קַמוּ כֻּלָּם וַיִּדְמוּ לִי כָּל-דִּבְרִי אֲשֶׁר דִּבַּרְתִּי.

כא וַיְהִי כַּאֲשֶׁר שָׁמְעוּ אֶת-הַדְּבָרִים אֵלֶּה, וַיַּרְגִּיזוּ וַיִּקְנְאוּ לְאָבִיהֶם וַיֹּאמְרוּ: "מִי יֵשׁ בְּיִשְׂרָאֵל הַמּוֹעִיל אֱלֹהִים?" וָאִישׁ קוּם וַיִּלְכֹּד אֶת-הַמִּזְבֵּחַ וּבֵית אָבִי וַיִּשְׁרְפוּ אוֹתוֹ בָּאֵשׁ וַיִּשְׁלְחוּ אַחֲרַי.

כב וַתִּדְלַק אֵשׁ גְּדוֹלָה בְּבֵית אָבִי וַיִּשְׁרְפוּ אֶת-כָּל-הַפְּסִילִים בִּלְהַבִּים. וַיִּשְׁרְפוּ גַּם אֶת-תַּרַח אָבִי בִּלְהַבִּים עַל־הַמִּזְבֵּחַ, עַם-הַפְּסִילִים שֶׁעָבַד בָּהֶם. וּנַשְׁמַת אָבִי וַתֵּצֵא מִן-הַלְּהָבוֹת וַתַּעַל.

כג וַאֲנִי אַבְרָם בָּרַחְתִּי מִלִּפְנֵי הָאֵשׁ, וַחְתִּי חַיִּים מִמֶּנָּה. וַתִּשְׁמַע קוֹלָם אֶת-קוֹלִי, וַיִּקְרְאוּ אֵלַי: "הִנְּנָה הִשְׁמְרוּ מִשָּׁם!" וָאִישׁ קוּם וַיִּרְדֹּף אַחֲרַי עַל־קְדִימוּת קַל־רַגְלַיִם.

כד וַיְהִי כַּאֲשֶׁר נִלְחַמוּ בִּי וְרָדְפוּ אַחֲרַי, וַיִּשְׁמַע קוֹלִי שֵׁם הַקָּדוֹשׁ וַיִּקְרָא אֵלַי: "אַבְרָם! אַבְרָם!" וָאֹמַר: "הִנֵּנִי". וַיֹּאמֶר: "צֵא מֵהֶם וִלַּח מִן-הַשּׁוֹלְחִים אַחֲרֶיךָ, כִּי לֹא יִשְׂגּוּ עַד-הַנְּחַל. וּבְעָבֹר הַנָּהָר, אַתָּה יִהְיֶה בְּאֶרֶץ כִּי הִנִּי עִמָּךָ לִשְׁמוֹר וּלְהַצִּיל אוֹתְךָ מִדִּבְרוֹת הָרָעָה. וְאַבִּיךָ הִנּוֹ אִתִּי וְאַל-תִּדְאַג עַל-נַפְשׁוֹ".

כה וְעוֹדֶנִּי שׁוֹמֵעַ אֶת-הַקּוֹל, וָרַצְתִּי אַל-הַנָּהָר בְּמִהִירוּת גְּדוֹלָה וַאַבְרַח מִפְּנֵיהֶם. וַאֲנִי אַבְרָם בָּרַחְתִּי וְעָבַרְתִּי אֶת-הַנָּהָר וַאֱהִי בַּמִּדְבָּר וַאּּוּד לִי מְגוּרִים בְּמִדְבַּר קְדוּמוֹ.
