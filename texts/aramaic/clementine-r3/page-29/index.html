ומצחא וסוגאא אגר בטריפולס וכלה דינ עותדה הנא הוא דמא דאתחזי הו פטרוס אפ לא אנש משמעתה נתל לה איכ דלחרשא ודלא אלהא ומצערא וחרעא והדיוטא ודמשתודא הנינ דלא משכחנ דנהוינ ואמר דמיתא קימינ הדא למ דלא משכחא דתהוא ומנסא למ דלהנונ דמכסינ לה כסיאית נקטול ביד אנשינ מסיענוהי דמטל הדא למ אנא זכיתה ופרסיתה ומנ נכלה דחלת וערקת דדמ בחרשוהי נקטלני או כסיאית לקטלא נשלמני והו הכיל פטרוס כד הוא פצחא קדמ יומא אחריא קרה לאגרתא לה לכנשא ומלל עמהונ איכנא דלכל מדמ לזכי נתטפיסונ הו דמנה אפסקופא אתסימ להונ והידינ אגעל להונ לקשישא עמ משמשנא אפ להונ לה לעמא ואיכ דולא הוא וזדק מלל עמהונ ופקד אנונ ואמר להונ דבטריפולס מסתא והנינ דילה דנמוסא איכנא דהו כורסיה דמושא משלמ והו דינ דתרינ על רשנותא ודאיכנא חדא איתיה או לא ודלא טעא לה לנמוסא דבית עבריא הו דמנ כל דוכא לא מסתיכ והו דינ דתלתא על אלהא ועל הנינ דמנה אתחרק והו דינ דארבעא דכד סגיאא אתקריו אלהא חד הו איתוהי הו דבשררא אלהא הו איכנא דמסהדינ כתבא אלהיא והו דינ דחמשא דתרינ איתיהונ בלחוד שמיא דמנהונ חד איתוהי הו דמתחזא הו דאפ עבר לה והו דינ אחרנא דלעלמ לא מתחזא והו פרק כ״ז — שְׁמִירַת הַלִּמּוּד וְשִׁקִּידַת הַלַּיְלָה

(א) וַיֵּצְאוּ עִמּוֹ מֵאֵלֶּה הַמְעַטִּים אֲשֶׁר-טוֹב — אֵלֶּה אֲשֶׁר-אַף-כִּי-נִכְפּוּ הָיוּ; וְאֵלֶּה כֻּלָּם — אַחֲרֵית-אַחֲרִית בְּפֶטְרוֹס נֶאֶסְפוּ.

(ב) וְעַל-בִּרְכֵּיהֶם בֵּרְכוּ; וּמֵהֶם אֵלֶּה אֲשֶׁר-בְּכָל-מִין-חוֹלְיָם וּמַכְאוֹבִים הָיוּ; וּמִצְלוֹתָיו שֶׁל פֶּטְרוֹס נִתְרַפּוּ וְנִטְהֲרוּ וְנִשְׁתַּחְרֲרוּ בְּשִׂמְחָה.

(ג) כַּאֲשֶׁר-בִּלְמוּד-הָאֱלֹהִים אֲשֶׁר-הַשַּׂרְרָה בְּרַעֲיוֹנֵיהֶם נִתְמַלֵּאוּ — וַיִּפְשְׁטוּ בְּנֵי-עֵת וְרַחֲמִים; וְעַל-הָאָרֶץ שָׂמוּ-לָהֶם; וַיִּסָּמְכוּ אִישׁ-אִישׁ בִּמְקוֹמוֹ אֲשֶׁר-יָדַע כִּי-מִלְּפָנִים יָמִים נִסְמַךְ בּוֹ.

(ד) וַיִּשְׂאוּ תִּקְוָה וַיּוֹדוּ וַיְשַׁבְּחוּ, וַיַּעֲלוּ פְנִימָה וַיִּשְׁכָּבוּ.

(ה) וּלְמָחֳרָת — כְּעֵדָה בָּעֵת — קָם פֶּטְרוֹס בַּלַּיְלָה; וַיִּמְצָא אֶת-כֻּלָּם מֵיטִיבִים אוֹתָנוּ; וַיָּחֵל לוֹמַר:

(ו) כָּכָה חָפֵץ אָנִי וּמִתְחַנֵּן אָנִי מִכֶּם, אַחַי וּקְנָאָי — כִּי-יֵשׁ אִישׁ אֲשֶׁר-לֹא-יִמְצָא לִשְׁהוֹת — לֹא-מִפְּנֵי כִּי-נִכְפֶּה מִמֶּנּוּ-עַל-כֹּחוֹ, כִּי-מִדֵּי-בַעֲגַל-אִישׁ לֹא-יִמְצָא לַעֲשׂוֹת-עַצְמוֹ — בַּזְּמַן נִגְרָה מְצָאָה לִסְעֹר.

(ז) מִפְּנֵי כִּי-מֵאֵת-הָעֵדָה אֲנַחְנוּ מֻפְרָשִׁים — בַּזְּמַן מֻפְרָשִׁים וְנִגְרִים אֲשֶׁר-יִשְׁתַּנּוּ וְיִתְעָרְבוּ — מוֹצְאִים אֲנַחְנוּ עֵדָה; כִּי-כֵן-הוּא — שְׁנַיִם מְעִידִים בִּי — כִּי-הָאֱלֹהִים לֹא-יַעֲיֵף אוֹתִי עַל-זֹאת.

(ח) כִּי-אִישׁ לֹא-שָׁהָה — אֶלָּא-כִּי-אִישׁ כֻּלּוֹ-לַיְלָה נִרְדַּם, וּבַיּוֹם לֹא-נִתְמַלֵּא מִן-הַלַּיְלָה; וְגַם-לֹא-צִדֶּקֶת הַהִלּוּךְ-בַּלַּיְלָה — בַּלִּמּוּד-אוֹר.

(ט) כֵּיצַד רַעֲיוֹנוֹ מְאֻחָד שֶׁל-הָאֱלֹהִים לְבַדּוֹ יִהְיֶה בּוֹ; כִּי-בַּאֲשֶׁר-מְאֻחָד שֶׁל-הָרָע בּוֹ — גַּם-לֹא-יִמְצָא לְהִגָּזֵל.

(י) וַיֹּאמֶר פֶּטְרוֹס: כֻּלָּנוּ אָמְנָם וְחָפֵצִים לְהַפְסִיק וְלוֹמַר כִּי-רַב-קֹדֶם יָקְטִיל-שֵׁנָה נִשְׂבַּעְנוּ וְנֵעוֹר וְנִכְפֶּה לְהָעִירְכֶם; כִּי-אַף-לֹא-יָפֶה לִתְלַמִּיד לְהִפָּקַד מֵעֲשׂוֹת-טוּבָם — כְּאַשְׁרֵי-הַמְתְאַלְּצִים הָיִינוּ.

(יא) טוֹב לָנוּ לְהִתְמַרֵח; כִּי-סַגִּיא הוּא-לָבֶן וְרַעֲיוֹנֵנוּ — מִשִּׂמְחָתֵנוּ אֲשֶׁר-תְּמַלֵּא מְשׁוֹר-לָנוּ; וְהִיא-לַשָּׁנָה תָּרַד וּתְדַחֵק; וּמִפְּנֵי חוֹבֵנוּ לַלִּמּוּד הַזֶּה — נַקְדִּים-נָּא יִקְרֶה מַפְסִיק-הוּא-לָנוּ — וְנָעִירְכֶם-נָּא.

(יב) וַיֹּאמֶר פֶּטְרוֹס: מִפְּנֵי-כָךְ — מִן-הַחוֹבָה אֲשֶׁר-לָנוּ — מָלֵאתֶם עֵדוּת כִּי-שָׁהִיתֶם וְהֶעֱרַתֶּם חֶפְצֵי-מָכִיל — לֹא-בִּלְבַד דָּבָר עַל-אֵלֶּה אֲשֶׁר-נִתְמַלֵּא נִתְמַלֵּל וְנֶעֱדַד וְנֶאֱהָב בְּכָל-תִּכְסִית-צֶדֶק נִשְׁלַם לָכֶם.

(יג) אֶלָּא-אַף-מַלֵּא כִּי-בְּכָל-זְמַן בַּיּוֹם — הָיִינוּ בַּעֲתִיד-נָפַל לָנוּ בְּכָל-לֵילוֹתֵינוּ מַה-שֶּׁיֶּתֶר לְהִתְמַלֵּל עִם-נַפְשֵׁנוּ — מוֹצְאִים אֲנַחְנוּ לְהִתְעַדֵּד בְּתִכְסִיתֵנוּ וּמַה-שֶּׁחָסַר בָּהֶם נַמְלִיא.

(יד) וַיְהִי כְּאָמְרוֹ-זֹּאת — וַיָּחֶל לָנוּעַ עַל-אֵלֶּה אֲשֶׁר-מִלְּפָנִים יוֹם — בֹּאֵנוּ תִּכְסִית-בַּעֲתֵנוּ — מִתְחַיֶּבֶת הָיְתָה לִהְיוֹת וְלֹא-הָיְתָה — מִפְּנֵי-יֹפִי-הַשָּׂבִיר וְרַחֲמַת.

הערות וביאורים

  • פרק כז — שְׁמִירַת הַלִּמּוּד וְשִׁקִּידַת הַלַּיְלָה ⁴⁸ נִכְפֶּה לְהָעִירְכֶם — קם להעיר. ההקבלה המקראית: ראה תהילים נ"ז,ט: "עוּרָה כְבוֹדִי עוּרָה הַנֵּבֶל וְכִנּוֹר אָעִירָה שָּׁחַר"; וכן תהילים ק"ח,ג: "עֲרִיצִים עֲרַבְתִּי אֶקְדָּמֶךָּ". חז"ל: ראה תלמוד-בבלי, ברכות ג ע"א: "דוד היה קם בחצות לילה"; ובסוכה כ"ו ע"ב: "קריאת-שמע על המיטה — בכל עת". רמב"ם, הלכות תפילה ו,ג (נוסח-מקובל): "יש להעיר בלילה לקרות קריאת-שמע". בהקשר הקלמנטיני: פטרוס מתאר את שקידת-הלילה של התלמידים — הקבלה לדוד ולמוסד "תיקון-חצות" המאוחר. הערה ביקורתית: "תיקון-חצות" אינו מוזכר בתלמוד במפורש, אך קם-בחצות לתורה מוזכר רבות בחז"ל. הקלמנטינה מתארת מנהג דומה. ⁴⁹ רַעֲיוֹנוֹ מְאֻחָד שֶׁל-הָאֱלֹהִים לְבַדּוֹ יִהְיֶה בּוֹ — איחוד-המחשבה באלהים. ההקבלה המקראית: ראה דברים ו,ה: "וְאָהַבְתָּ אֵת ה' אֱלֹהֶיךָ בְּכָל-לְבָבְךָ וּבְכָל-נַפְשְׁךָ וּבְכָל-מְאֹדֶךָ"; וכן דברים י,יב: "לְיִרְאָה אֶת-ה' אֱלֹהֶיךָ לָלֶכֶת בְּכָל-דְּרָכָיו וּלְאַהֲבָה אֹתוֹ". חז"ל: ראה תלמוד-בבלי, ברכות י"ג ע"א: "למען תזכרו את כל-מצוותי — שלא תהא מחשבתך מפוזרת"; ובסוטה כ"ב ע"א: "מי שיש בו תורה — יש בו יראת-חטא". רמב"ם, הלכות תשובה י,ו: "ואינו עובד אלא מתוך אהבה"; ובהלכות יסודי-התורה ב,ב: "ומה היא האהבה הראויה? — שיאהב את השם באהבה גדולה יתרה". רמב"ם, שמונה-פרקים, פרק ח (נוסח-מקובל): "ואיחוד-המחשבה באלהים — הוא תכלית-העבודה". בהקשר הקלמנטיני: פטרוס רואה באיחוד-המחשבה עם האלהים את תכלית-השקידה — הקבלה לרבנית-רמב"מית. הערה ביקורתית: זוהי הקבלה משמעותית — הרמב"ם רואה באיחוד-המחשבה את מדרגת-הנבואה העליונה, והקלמנטינה רואה בו את מדרגת-השקידה של התלמידים. --- ↩
