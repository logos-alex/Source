פרק טז — בִּכְיַת פֶּטְרוֹס וְדִבְרֵי רַחֲמֵי שָׁמַיִם

(א) וַיְהִי כַּאֲשֶׁר שָׁמַע פֶּטְרוֹס אֶת-הַדְּבָרִים הָאֵלֶּה — וַיֵּבְךְּ וַיֹּאמַר.

(ב) הַאֻמְנָם יִתְפַּלְּאוּ בְּעִקְּשׁוּת רוּחַ-הָרֹב שֶׁל-הָאֱלֹהִים, וּבְרֹחַק-דַּעְתָּם מֵעַל-עוֹלַם בְּנֵי-אָדָם! כִּי-דָּבָר זֶה רַב הוּא, אֲשֶׁר יָכוֹל לְהַחֲזִיק אֶת-סִימוֹן הַמַּכְחִישׁ אֶת-הָאֱלֹהִים, כִּי-אֱמֶת הוּא אֲשֶׁר לָהֶם — עֲוֹנָם וַחֲטָאָתָם סְעָרוּם.

(ג) מַה-זֶּה כְּמוֹ-סוֹבֵר, אֲשֶׁר מִנְּפִישׁוֹ מִשְׁתַּמֵּשׁ בָּהֶם אֲשֶׁר-חָפֵץ בַּשַּׂרְרָה? וְאֵינֶנּוּ נֶפֶשׁ, כִּי אִם-דַּיּוֹ הֵם אֲשֶׁר מִצְטַדִּים בָּהּ; וְכֵן אָמַר: אַף נַפְשׁוֹ אֵינָהּ-מֵתָה, וְנֶאֱבָק עַל-אֵלֶּה אֲשֶׁר-הִסְעִירָם עוֹד-יוֹתֵר, וּבְמַעֲשֵׂה-אֱלִיל רָאָה אִישׁ אֲשֶׁר-אֵין-לוֹ חֵלֶק בַּדָּבָר.

(ד) מַאֲמִין הוּא, כִּי-אַף-מִצְטַדֶּה בָּהּ דַּיּוֹ — אֶפְסוּ-סוֹבֵר כִּי-קְיוּמָהּ שֶׁל-נֶפֶשׁ רָאָה; וּבָזֶה כֻּלּוֹ — רָעָה מִתְרַבָּה, וַאֲמִתִּי כִּי-מוֹדֶה כִּי-רָעָה סְעָרָהּ וְעָשָׂה, אוֹ כִּי-מִתְכַּוֵּן בָּהֶם שֶׁבְּעֵת אַחַת מָכַר נַפְשׁוֹ לָאוֹיֵב (לבעלדבבא — לָאוֹיֵב, לֹא "לַשָּׂטָן") לְאֱלֹהִים, וְעִם-אֵלֶּה הֶרְחִיק אֵת-אֵלֶּה אֲשֶׁר-אַף לֹא-שׁוֹלֵט — כֹּה אָמַר.

(ה) אוֹ-אָחִי, לָאָנָשִׁים אֵין-זֹאת מִפְּנֵי חֶסְרוֹן תְּחִנָּה לָהֶם, כִּי לֹא יִכָּנְעוּ לִדְבַר-יִרְאַת הָאֱלֹהִים, אֶלָּא מִפְּנֵי רֹב-חַטֹּאותֵיהֶם — אֵלֶּה אֲשֶׁר-הֵם אוֹ אֲשֶׁר-קְדָמָם; רָעָה הִיא אֲשֶׁר-אֵלֶּה בְּדַעְתָּם מַסְתִּימִים.

(ו) וּמִפְּנֵי זֹאת גַּם לֹא-יַאֲמִינוּ; כִּי-כְבָר יָכֹל מִצְטַדֶּה בָּהּ, אֶפְסוּ מִפְּנֵי שֶׁ-סָפֵק הוּא, חוֹבָה עִם-בְּרִיאַת-הָעוֹלָם לְהִוָּשַׁע, וּבָא-לָהֶם אֲשֶׁר-אוֹהֲבִים אוֹתָהּ; וְנִתְפָּרְסָה הָרָעָה אֲשֶׁר-עָשׂוּ אוֹתָם לִבְנֵי-אָדָם לָאוֹיֵב לָהּ, וּמִפְּנֵי זֹאת גַּם-שָׂמָם וְגִלָּה לָהֶם עֲצָמוֹת-אֱנוֹשׁ מֵאֲצִירֵיהֶם.

הערות וביאורים

  • פרק טז — בִּכְיַת פֶּטרוֹס וְדִבְרֵי רַחֲמֵי שָׁמַיִם **⁴⁹ וַיֵּבְךְּ** — בכיית-פטרוס. **ההקבלה המקראית**: ראה ירמיהו ט,יז: "וְהִשְׁלִיכוּם עֲלֵיהֶם דְּמָעָה"; וכן תהילים ו,ז: "כִּי-נִבְלְתִּי בְּגָעֲלִי כָּל-הַלַּיְלָה מֵרֹב דִּמְעָתִי מַשְׁכִּיבָה". **חז"ל**: ראה תלמוד-בבלי, שבת ק"ד ע"א: "כל הדור של חורבן — ראוי לבכות עליהם"; ובבבא-מציעא פ"ד ע"ב: "כל הבוכה על צדיק שמת — מוחלין לו כל עוונותיו". **רמב"ם**, הלכות אבל יג,יב: "והבוכה ומצטער על הצדיק — שכרו מרובה מאד". **בהקשר הקלמנטיני**: פטרוס בוכה על קשיות-הלב של סימון וחבריו. זוהי בכייה על חורבן-רוחני, לא חומרי — הקבלה לבכיית-ירמיהו על חורבן-ירושלים. **⁵⁰ עִקְּשׁוּת רוּחַ-הָרֹב שֶׁל-הָאֱלֹהִים** — קשיות-רוח. **ההקבלה המקראית**: ראה דברים י,טז: "וּמַלְתֶּם אֵת עָרְלַת לְבַבְכֶם וְעָרְפְּכֶם לֹא תַקְשׁוּ מוֹד"; וכן ירמיהו ד,ד: "הִמּוֹלוּ לַיהוָה וְהָסִירוּ עָרְלוֹת לְבַבְכֶם". **חז"ל**: ראה תלמוד-בבלי, עבודה-זרה כ"ו ע"ב: "בן-אדם יש לו ערלת-לב"; ובנדרים ל"ב ע"ב: "כל שאפשר למול ולא מל — כאילו ערל". **רמב"ם**, מורה-נבוכים ג,מ"ט (נוסח-מקוצר): "וערלת-הלב — היא קושיות-הרוח". **בהקשר הקלמנטיני**: פטרוס רואה בקשיות-הלב של סימון מחלה-רוחנית — הקבלה ל"ערלת-הלב" המקראית. **⁵¹ מִפְּנֵי רֹב-חַטֹּאותֵיהֶם** — ריבוי-חטאים. **ההקבלה המקראית**: ראה ישעיהו א,ד: "הוֹי גּוֹי חֹטֵא עַם כָּבֵד עָוֹן זֶרַע מְרֵעִים בָּנִים מַשְׁחִיתִם"; וכן ישעיהו נ"ט,יב: "כִּי-רַבּוּ פִּשְׁעֵיכֶם וְנִשְׁנוּ חַטֹּאותֵיכֶם". **חז"ל**: ראה תלמוד-בבלי, יומא פ"ו ע"ב: "רבו חטאות — רבו מכשולות"; ובשבת ל"ב ע"ב: "בשעה שישראל מרבים חטאים — מרובים מכשולות". **רמב"ם**, הלכות תשובה ג,ד: "ואם היו חטאיו מרובים מזכויותיו — נידון". **בהקשר הקלמנטיני**: פטרוס רואה בריבוי-החטאים את הסיבה לקשיות-הלב — הקבלה לתפיסה המקראית-רבנית. --- ↩
