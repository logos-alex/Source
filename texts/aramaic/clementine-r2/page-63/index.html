פרק ס" — פֶּטְרוֹס: הַמַּחֲשָׁבָה בּוֹרֶאתֶת צְלָמִים מִן-הָעָבָר

(א) אֵלֶּה הֵם אֲשֶׁר-עֲתִידִים לָדוּן — שֶׁ-נִהְיֶה — וּמִשְׁעֲרֵיהֶם נִשְׁתַּנֵּינוּ — כֵּן בַּיָּם אֲנַחְנוּ נִסְחָבִים; וְכֵן מֵחֹסֶר הַמְּנוֹת אֲשֶׁר-אֵלֶּה — לֹא מִדַּעְתָּם — חֲסֵרִים.

(ב) וְזֶה אֲשֶׁר-יֵשׁ — שֶׁ-הַנֶּפֶשׁ, אֲשֶׁר-מִתְבַּלְבֶּלֶת בְּרִאשׁוֹנָה, וְלֹא-רוֹאָה שֶׁ-אֵלֶּה יֵשׁ; וְהָאֲחֵרִים אֲשֶׁר-אֵינָם — רָעָב לָהֶם.

(ג) כִּי-אֵלֶּה אֲשֶׁר-שׂוֹנְאִים אוֹתָם — עַצְמָם הֵם הַמִּכְרָה הַמְרֻבָּה, חוֹשְׁבִים שֶׁ-צֹרֶה יִרְאוּ, מִפְּנֵי-נַפְשָׁם אֲשֶׁר-טְרוּדָה בְּמַחֲשָׁבוֹתֶיהָ בְּרֹב אֵלֶּה אֲשֶׁר-עָלֵינוּ.

(ד) וְכֵן-גַּם לְכָל-אֵלֶּה אֲשֶׁר-מוֹכְרִים עַצְמָם מֵרָפֶה — אֵיךְ זֶה נִבְלָע בְּיֶדַעְתֵּנוּ שֶׁ-מֵתִים; וְאֵלֶּה אֲשֶׁר-יֵשׁ וְקְרוֹבִים — לֹא-רוֹאִים; וְאֵלֶּה אֲשֶׁר-אֵינָם — אֵיךְ שֶׁהֵם וְקְרוֹבִים — חוֹשְׁבִים שֶׁרוֹאִים?

(ה) וְכֵן-גַּם לְרַבִּים אֲשֶׁר-בַּחוּץ — יִקְדּוּ בַּאֲשֶׁר-תְּרַדְמַת-הַמַּעְיָן וְהַנְּהָרוֹת — אֵיךְ שֶׁ-נִסְחָבִים יִרְאוּ; וּבְחָשְׁבָם שֶׁ-יִשְׁתּוּ — לֹא יִשְׁתּוּ.

(ו) וְזֹאת-גַּם-לָהֶם — שֶׁ-מִיּוֹבֶשֶׁת הַגּוּף נַפְשָׁם עַצְמָהּ מִתְחַלֶּפֶת בְּחִפָּזוֹן. כֵּן-הַנֶּפֶשׁ וְהַגּוּף נִסְחָבִים — אֵיךְ שֶׁבְּאֵלֶּה אֲנַחְנוּ חוֹשְׁבִים שֶׁ-יֶשׁ-לָּנוּ.

(ז) וּבְאֵלֶּה אֲשֶׁר-יֵשׁ. שְׁמַע-נָא: לִירוּשָׁלַיִם, אֲשֶׁר-אֵלֶּיהָ אַתָּה מְאֹרָס, כַּאֲשֶׁר-שָׁם יָשַׁבְתָּ — רָאִיתָה אוֹתָהּ, אֵיךְ שֶׁ-אֲשֶׁר-בַּשַּׂרְרָה מִקֶּדֶם הָיְתָה.

(ח) כִּי-נִרְאֲתָה לְךָ בַּשַּׂרְרָה. זֹאת-גַּם לִקְסַרְיָה — בְּצוּרָתָהּ, כַּאֲשֶׁר-תֹּאמַר, שֶׁ-חָשַׁבְתָּ שֶׁ-רָאִיתָ.

(ט) וַאֲנִי — בְּשַׁעַר יַלְדוּתִי, אֲשֶׁר-חָשַׁבְתִּי שֶׁ-בַּשַּׂרְרָה כֵּן הִיא — אֵיךְ שֶׁ-בְּמַחֲשַׁבְתִּי הָיִיתִי אֵלֶיהָ, מִתְרַגֵּשׁ-מְאֹד וּמִתְפָּאֵר-מְאֹד; וּבַשַּׂרְרָה כֵּן הִיא — אָמַרְתִּי, כַּאֲשֶׁר-חָשַׁבְתִּי שֶׁ-רָאִיתִיהָ.

(י) בְּבֹאִי אֵלֶיךָ בָּאתִי בִּמְסִרְכֵי-נַפְשִׁי, שֶׁ-אֲפִלּוּ לֹא מִמַּה-שֶּׁרָאִיתִי רָאִיתִי בָּהּ בְּרֹעַ מַחֲשַׁבְתִּי — אֶלָּא-אֵלֶּה אֲשֶׁר-רָאִיתִי מִמַּה-שֶּׁ-יָדַעְתִּי רָאִיתִי.

(יא) מִשַּׁעַר לְשַׁעַר רָאִיתִי בְּמַחֲשַׁבְתִּי בַּשַּׂרְרָה — מֵאֵלֶּה אֲשֶׁר-מִקֶּדֶם רָאִיתִי לִי.

(יב) לֹא-שֶׁ-אִישׁ יוּכַל צֹרֶת חֲדָשָׁה לִצְרוֹר — אֶלָּא-אֲנִי נוֹטֵל תּוֹעֶפֶת מֵאֵלֶּה אֲשֶׁר-מִקֶּדֶם רָאִיתִי לָהּ.
