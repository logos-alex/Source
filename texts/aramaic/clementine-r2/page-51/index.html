(יט) וְכֶנֶגֶד אֵלֶּה אָמַר סִימוֹן: הֲלֹא-הֵעִידֹתִי, כִּי-גַּם-בַּבּוֹרֵא אָמַרְתִּי.

(ב) כִּי-יֵשׁ-לוֹ לְאֱלֹהִים. וּבְלֹא-מַרְגֵּשְׁנוּתְךָ — צָר לְךָ אֹתוֹ לְדֻמּוּת נַפְשָׁם.

(ג) וְאֵינֶנּוּ נֶפֶשׁ — אָמַרְתָּ כִּי-הֵם. אֶלָּא-לֹא-הָיָה עֲדַנִּי — אֲשֶׁר-דְּבַר זֶה.

(ד) כִּי-נִכְסַף לָזֶה כֻּלּוֹ — לֹא-מַרְגֵּשְׁנוּתְךָ. מִתְפָּרֵץ אֲנִי, אוֹמֵר עַל-אוֹר-הַמַּאִיר אֲשֶׁר-לֹא-נִסְתָּכֵל.

(ה) וּשְׁמַע-נָא זֹאת: כִּי-כְבָר יֵשׁ-מַה-שֶּׁמִּמֶּנּוּ בַּסֵּתֶר — אוֹר-הַמַּאִיר, אֲשֶׁר-גְּבוּרָתוֹ לֹא-נִסְתָּכְלָה — לָרֹב.

(ו) אֲשֶׁר-לֹא-לַבּוֹרֵא יְדָעוּהוּ, וְגַם-לֹא-מֹשֶׁה סָאם נְמוּסָא (מְבַזֵּה-הַתּוֹרָה), וְגַם-לֹא-יֵשׁוּעַ מְלַפְנָא.

(ז) וַיֹּאמֶר פֶּטְרוֹס: לֹא-כֵן מִסְתַּבֵּר לְךָ, כִּי-רַב מְחַסֵּר מַחֲשָׁבָה.

(ח) הוּא שֶׁמֵּאֲמִין כִּי-אֱלֹהִים אַחֵר לְמַעְלָה מִמֶּנּוּ — אֱלֹהֵי הַשַּׂרְרָה.

(ט) אֵלֶּה — אֵלֶּה אֲנִי אוֹמֵר בִּמְחַלְקֹתֶיךָ — כְּבָר וְקֹדֶם — לְנַפְשׁוֹ יִשְׁתַּלֵּם.

(י) לְאַחֵר — כִּי-זֶה חֵפֶץ שֶׁנִּשְׁתַּבַּח אִם-אֱלֹהִים אַחֵר אִישׁ וּמִתְפָּרֵץ וְחָצֵף מִמֶּךָּ — לְךָ נִשְׁאַר.

(יא) כַּאֲשֶׁר גַּם-אַתָּה נֶחְלָק אַתָּה, כִּי-אֲנִי אֱלֹהִים אַחֵר — גְּבוּרָה אֲשֶׁר-לֹא-יָדַע אוֹתָהּ גַּם-לֹא-הוּא הַבּוֹרֵא אֲשֶׁר-לָעוֹלָם.

(יב) וְגַם-לֹא-מֹשֶׁה נָבִיא, וְסָאם נְמוּסָא, וְגַם-לֹא-יֵשׁוּעַ מְלַפְנָא. וְאֵיךְ כֵן — כִּי-הִיא הַגְּבוּרָה הַסְּמוּיָה.

(יג) אֲשֶׁר-לְאָנָשִׁים לֹא-תִיָּדַע — אֶלָּא-לְאֶחָד לְבַדּוֹ. זֶה גַּם-לְךָ, אֲשֶׁר-אֶתָּהּ.

(יד) בָּזֶה אֲנִי כֻלִּי — כֵּיצַד אַחֶרֶת הִיא — רֶגֶשׁ אַחֵר — לֹא-נָתַן-לָנוּ דָּבָר בָּהֶם.

(טו) אַחֶרֶת אֲשֶׁר-נִמְסַר-לָהּ, לָהּ לְהַמְצִיא אֱלֹהִים אַחֵר. כְּבָר אָמַרְתָּ כִּי-לֹא-מוֹצֵא לָקַחַת. וְכֵיצַד לְךָ נָתַן — כְּבָר מִפְּנֵי שֶׁחָפַצְתָּ.

(טז) וְגַם-אֲנַחְנוּ — מוֹצְאִים אֲנַחְנוּ לְחַפֵּץ, אֶלָּא-בִּמְסֻרְגָּלִית וּבִמְחִישׁוּל אֲשֶׁר-נִהְיֶה לֹא-חָפֵץ אֲנַחְנוּ.

(יז) אֲנִי מְכַלֵּל — הוּא גָּלָה לְךָ. מִפְּנֵי-מֶה לֹא-גַּם-לָנוּ. וַאֲנִי — אַתָּה מִנַּפְשְׁךָ יָדַעְתָּ שֶׁהַנְּבִיאִים לֹא-יָדְעוּ.

(יח) וְגַם-אֲנִי כֵּיצַד נָבִיא אָמַר לָנוּ — מֶה נֶחְשָׁב אֲנַחְנוּ. וְלֹא-הוּא. הוּא אֶחָד. אֲנִי גַּם-רוּחַ-הַשָּׁוֶה עִם אֵלֶּה שֶׁדּוֹמִים לְךָ.

(יט) יֵשׁ-בְּךָ כְּבָר יָדַעְתָּ גַּם-אֵלֶּה מַחֲשָׁבָה. וַאֲנִי — מַחְשַׁבְתֵּנוּ אֲשֶׁר-קֵרְבָנוּ אֲנַחְנוּ — לוֹמַר — לֹא-מוֹצֵא אַתָּה.

(כ) כֵּיצַד יָדַעְתָּ סֵתֶר — מַחֲשָׁבוֹתָיו — אוֹ-גַּם-זֹּאת, שֶׁבִּלְבַד אֲשֶׁר-יֶשְׁנוֹ אוֹת-הַסָּתֶר מִכָּל-אֵלֶּה.

(כא) מֵאֵלֶּה לֹא-זֶה יָדַעְתָּ. הֲלֹא-מִמֶּנּוּ הוּא אוֹר-הַמַּאִיר — אִם-לֹא-מִמֶּנּוּ לָקַחְתָּ גַּם-זֹּאת שֶׁתִּרְאֶה.

(כב) וְגַם-זֹּאת שֶׁהוּא נִרְאֶה — מִזֶּה שֶׁיָּדַעְתָּ אֵלֶּה — חֲמִשָּׁה מִינִים רֶגֶשׁ אֲשֶׁר-נִמְסְרוּ לְךָ — מִתְפָּרֵס אַתָּה גַּם-עֲלֵיהֶם.

(כג) מֵאֵלֶּה לָקַחְתָּ צְלוּמָא מִדֵּי-מֶה, רַב הוּא אֲשֶׁר-בַּשַּׂרְרָה חֲלוֹם הוּא, וְלֹא-הוּא שַׂרְרָה.

(כד) עֲדַת-רֶגֶשׁ אַחֶרֶת מִבַּלְעֲדֵי חֲמִשָּׁה אֲשֶׁר-יֶשְׁ-לָנוּ — לֹא-מוֹצֵא אַתָּה לְהַטּוֹת אֱלֹהִים אַחֵר — לֹא יֵשׁ.
