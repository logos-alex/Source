חזון אברהם – נוסח א' – פרק א

מקור: סלאבית · ייחוס: אברהם

גודל טקסט

פרק א

סֵפֶר גִּלּוּי אַבְרָהָם בֶּן תֶּרַח, בֶּן נָחוֹר, בֶּן שְׂרוּג, בֶּן רְעוּ, בֶּן פֶּלֶג, בֶּן עֵבֶר, בֶּן שֶׁלַח, בֶּן אַרְפַּכְשַׁד, בֶּן שֵׁם, בֶּן נֹחַ, בֶּן לֶמֶךְ, בֶּן מְתוּשֶׁלַח, בֶּן חֲנוֹךְ, בֶּן יָרֶד.

בַּיּוֹם אֲשֶׁר הִגִּיעַ לִי זְמַן שֵׁרוּת עֲבוֹדַת אָבִי תֶּרַח — וַאֲנִי חוֹקֵר וּמְבַקֵּשׁ לָדַעַת מִי הוּא הָאֵל הֶחָזָק בֶּאֱמֶת — אֲנִי אַבְרָהָם, בְּעֵת גָּמְרִי אֶת שֵׁרוּת מִנְחַת אָבִי תֶּרַח לֵאלֹהָיו: לֵאלֹהֵי הָעֵץ וְהָאֶבֶן וְהַזָּהָב וְהַכֶּסֶף וְהַנְּחֹשֶׁת וְהַבַּרְזֶל — נִכְנַסְתִּי אֶל בֵּית תְּפִלָּתָם לַעֲשׂוֹת אֶת הָעֲבוֹדָה.

וּמָצָאתִי שָׁם אֱלֹהַּ אֶחָד שְׁמוֹ מָרוּמוֹת, מֵאֶבֶן חֲצוּבָה — נָפַל מִמָּקוֹמוֹ לִפְנֵי הָאֱלֹהִים, אֲשֶׁר שְׁמוֹ נָחוֹר, אֱלֹהֵי הַנְּחֹשֶׁת. וַיְהִי כַּאֲשֶׁר רְאִיתִיו נָפוּל, נִבְהַל לִבִּי.

וָאֲחַשְּׁבָה בְּדַעְתִּי כִּי לֹא אוּכַל לַהֲשִׁיבוֹ אֶל מְקוֹמוֹ לְבַדִּי — כִּי כָּבֵד הָיָה מְאֹד מֵאֶבֶן.

וָאֵלֵךְ וְאַגִּיד לְאָבִי, וַיֵּרֶד עִמִּי — וּשְׁנֵינוּ יַחְדָּו לֹא יָכֹלְנוּ לַהֲקִימוֹ, וַיִּפֹּל רֹאשׁוֹ מֵעָלָיו. וַיֹּאמֶר לִי אָבִי: אֱחֹז בְּרֹאשׁוֹ, אַבְרָהָם. וָאֱחֹז בּוֹ. וַיֹּאמֶר לִי: הָבֵא לִי קֶרֶשׁ וּפַטִּישׁ מִן הַבַּיִת. וָאָבִיא לוֹ, וַיַּנַּח אֶת רֹאשׁ מָרוּמוֹת עַל אֶבֶן אַחֶרֶת.

וַיַּעַשׂ אָבִי מָרוּמוֹת אַחֵר מֵאֶבֶן אַחֶרֶת, בְּלֹא רֹאשׁ. וְאֶת הָרֹאשׁ שֶׁנָּפַל מִמָּרוּמוֹת הָרִאשׁוֹן, וְאֶת הַנּוֹתָרִים — חֲמִשָּׁה אֱלֹהִים שֶׁנִּשְׁבְּרוּ — שְׂרָפָם בָּאֵשׁ. וַיְצַוֵּנִי וַיֹּאמֶר לִי: לֵךְ וּמְכֹר אוֹתָם בַּדֶּרֶךְ אֲשֶׁר אֶל הָעִיר.

הערות וביאורים

  • סֵפֶר גִּלּוּי אַבְרָהָם: החיבור נפתח ברשימת יוחסין מלאה מאברהם ועד ירד — בניגוד לספר בראשית המקראי, המתחיל את סיפור אברהם בפסוק 'וַיֹּאמֶר ה׳ אֶל אַבְרָם לֶךְ לְךָ'. רשימת היוחסין מקשרת את אברהם לשרשרת הדורות מבריאת העולם (דרך חנוך ומתושלח), ומעניקה לחיבור תוקף של ספר עתיק ומוסמך. ההקבלה ליוחסין הקדמוני מספר בראשית מדגישה את המשכיות המסורת. ↩
  • מָרוּמוֹת: שם האליל הראשון שנמצא נופל. השם 'מרומות' (מלשון רום/גובה) אירוני — האליל ששמו מורה על רום וגובה, נמצא נופל ארצה ממשכבו. נפילת האליל היא הזרז למסע הרוחני של אברהם: הוא רואה בעליל שאליל מאבן אינו יכול להציל עצמו, ומתחיל לחפש את האל האמיתי. ↩
  • וָאֲחַשְּׁבָה בְּדַעְתִּי כִּי לֹא אוּכַל לַהֲשִׁיבוֹ אֶל מְקוֹמוֹ לְבַדִּי: אברהם מודה בחולשת האליל — אם אל מאבן זקוק לעזרת אדם כדי לחזור למקומו, הרי שהאדם חזק ממנו. זהו הצעד הראשון בהפרכת האמונה האלילית מתוך היגיון פנימי. המחבר מציג את אברהם כחוקר וכמסיק מסקנות, לא כמקבל מסורת סביל. ↩
  • וַיַּעַשׂ אָבִי מָרוּמוֹת אַחֵר מֵאֶבֶן אַחֶרֶת, בְּלֹא רֹאשׁ: תרח 'מתקן' את האליל השבור על ידי יצירת אליל חדש בלא ראש — פתרון טכני גרידא לבעיה רוחנית. האבן האחרת שממנה נעשה הראש החדש היא יצירה כפולה של ידי אדם, דבר המדגיש את היות האליל תוצר מלאכותי לחלוטין. שריפת חמשת האלילים השבורים מסמלת את חוסר התוחלת שבעבודת אלילים — גם האש אינה מכבדת אותם. ↩
  • לֵךְ וּמְכֹר אוֹתָם בַּדֶּרֶךְ אֲשֶׁר אֶל הָעִיר: הפקודה למכור את האלילים השבורים היא שיא האבסורד — תרח מבקש מאברהם למכור את מה שכבר הוכח כחסר כוח. מכירת האלילים תהפוך בפרק הבא לנקודת מפנה, כאשר המציאות עצמה (הגמל והחמור) תוכיח שוב את חולשתם. ↩
