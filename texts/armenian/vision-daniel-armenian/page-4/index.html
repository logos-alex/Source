פסוקים 22–28

מקור: ארמנית · ייחוס: דניאל

גודל טקסט

22. אָז מַלְאַךְ הַמָּוֶת יַכֶּה אוֹתָךְ בְּכָל־עָרִים וּמְדִינוֹת, וּמִיתוֹת אַכְזָרִיּוֹת יְשֻׁלְּחוּ מִן־הַשָּׁמַיִם; פִּתְאֹם יַחֲרֶה אַף עַל־הָאָרֶץ, הָאָרֶץ תִּרְעַד, וְיִפְּלוּ אַרְמְנוֹתֶיהָ, וּבָתֵּיהֶם יִהְיוּ קִבְרוֹתָם, וְהַיָּם יַרְתִּיחַ גַּלָּיו מִמַּעַל וִיכַסֶּה אֶת־בְּנֵי־אָדָם, וְיִהְיוּ אֲשֶׁר יָנוּסוּ וְיִמָּלֵטוּ. אָז הֲמוֹן מַלְאָכִים יִהְיֶה, וְיַעַמְדוּ נֹכַח וִיחַנְּנוּ לִפְנֵי הַכִּסֵּא; אָז, בָּבֶל הַשְּׁבִיעִית, אֶת־מַרְבַדַּיִךְ תִּבְכִּי, אֵפֶר עַל־רֹאשֵׁךְ תִּזְרְקִי, בִּרְאוֹתֵךְ אֶת־בְּנֵי־אָדָם כִּי בַּעֲוֹנוֹת וּבַפְּשָׁעִים יִפֹּלוּ, אֲנָשִׁים עִם־יְלָדִים וְנָשִׁים עִם־עוֹלָלִים יַחְדָּו יֹאבֵדוּ, כִּי חֲרוֹן אַף יְיָ עֲלֵיהֶם. חוֹמוֹתַיִךְ תִּבָּקַעְנָה, וְאַרְמְנוֹתַיִךְ אַרְצָה יֻשְׁלָכוּ; עוֹלְלִים חֲרוּכִים בְּפִיהֶם יִצְעֲקוּ אֶל־אֱלֹהִים, וְכֹהֲנַיִךְ בִּזְרִיקָה יִסְפְּדוּ עַל־מַטּוֹתַיִךְ, גְּדוֹלַיִךְ יִבְכּוּ, וּבְנֵי עִירֵךְ יִתְאַבְּלוּ מְאֹד, וְעוֹבְרֵי דְּרָכַיִךְ בַּמֵּצָר יִהְיוּ, וְאֶשְׁכֹּלוֹתַיִךְ יִתְנַפְּצוּ, וּמַכָּה גְדוֹלָה מְאֹד, וְעַמּוּד אֵשׁ יֵרָאֶה מִן־הַשָּׁמַיִם אָרְצָה.

23. וְאָז תִּשְׁתַּנֶּה מַלְכוּתֵךְ, וְתִהְיִי נוֹתֶרֶת בְּהַשְׁמָדָה, בַּחֳלָאִים וּבָרָעָב. אָז כָּכָה תִּקָּרַע, וְתָבִיא אֶת־הַמַּטֶּה הַשֵּׁנִי אֶל־עִיר אַחֶרֶת, לְאַיִן מֵהַשַּׂר, וַחִצִּים מִתְלַקְּחִים יַשְׁלֵךְ מִשָּׁמַיִם, אוֹתוֹת וּמַעֲשֵׂי תַחְבֻּלוֹת רַבִּים יִהְיוּ. וְאָז יִמְשֹׁל הָאַנְטִיכְרִיסְט, וּבְנֵי־אָדָם מֵעֲבוֹדַת אֱלֹהִים יִסּוֹגוּ אֶל־כּוֹפְרִים, בְּבוֹא אֲשֶׁר לֹא בִקֵּשׁוּ, וַאֲשֶׁר לֹא צִפּוּ לוֹ, אֵלֶּה אֲשֶׁר לֹא הֶאֱמִינוּ בַּמִּתְנַגֵּד לַכֹּל. הוּא יֵעָבֵר וְיוֹלִיד אֶת־אֲשֶׁר לֹא בִקֵּשׁוּ, מִבְּתוּלָה טְמֵאָה, וְיִתְפֹּשׂ מַטֵּה תַרְמִית אֶת־עַם בְּנֵי־אָדָם, לְמוֹעֵד שָׁלוֹשׁ וָחֵצִי יוֹלִיךְ נַפְשׁוֹת רַבִּים אֶל־אֲבַדּוֹן, לִהְיוֹת חֲבֵרִים לַגֵּיהִנֹּם לְעוֹלָמִים. אָז מְבוּכָה תִּהְיֶה לַמַּלְאָכִים, בִּרְאוֹתָם אֶת־הָאוֹת אֲשֶׁר מִקֶּדֶם הֶרְאָה. וְכַאֲשֶׁר יָבִינוּ זֹאת יִרְאֵי אֱלֹהִים, יַכִּירוּ וְיִרְאוּ אֶת־כָּל־הַתַּרְמִית אֲשֶׁר לִבְנֵי־אָדָם, אֲשֶׁר לוֹ סִימָנִים אֵלֶּה: קֵרֵחַ, מְחֻנָּן, גָּדוֹל, חָכָם, מְתוּק־מַרְאֶה, חוֹזֶה, בַּעַל סוֹד, עֵרָנִי, עָנָו, רַךְ, עוֹשֶׂה מוֹפְתִים, בְּבוֹאוֹ יוֹצִיא נַפְשׁוֹת הַנִּדָּחִים, מִסְּלָעִים יוֹצִיא לֶחֶם, עִוְרִים יִפְקַח עֵינַיִם, פִּסְחִים יְהַלֵּכוּ, וְהָרִים מִמָּקוֹם לְמָקוֹם יַעֲתִיק; לְעֵינֵי הַכֹּל יַעֲשֶׂה אֶת־כָּל־אֵלֶּה, וְרַבִּים יַאֲמִינוּ בוֹ.

24. אוֹי לַמַּאֲמִינִים בּוֹ וְלוֹקְחִים אֶת־סִימָנוֹ, יְמִינָם קְשׁוּרָה אֲשֶׁר לֹא תָשׁוּב אֶל־אֲשֶׁר קִוּוּ לְפָנִים. אָז יִהְיֶה רָעָב גָּדוֹל מְאֹד, לֹא יִתְּנוּ שָׁמַיִם מָטָר, וְהָאָרֶץ לֹא תַצְמִיחַ דֶּשֶׁא, יִיבְשׁוּ כָּל־פֵּרוֹת, וְאָז כָּל־עָרִים וּמְדִינוֹת יִתְאַבְּלוּ עַל־עַצְמָן, נָסוֹ יָנוּסוּ, וְלֹא יָנוּסוּ מִמִּזְרָח לְמַעֲרָב וּמִמַּעֲרָב לְמִזְרָח; אַךְ אֲשֶׁר יֹאחֲזוּ בֶּהָרִים וּבַמְּעָרוֹת וּבַסְּדָקִים וּבְחַגְוֵי הָאָרֶץ, הֵמָּה לְבַדָּם יוּכְלוּ לְהִמָּלֵט עַד־הַבִּיאָה הַשֵּׁנִית, אֲשֶׁר נוֹלַד מִן־הַבְּתוּלָה הַקְּדוֹשָׁה.

25. אָז בְּחִירָיו יִגָּלוּ, בִּרְאוֹתָם אֶת־בִּיאַת יְיָ הַנִּצְחִית; יַעֲמֹד לִפְנֵיהֶם, וְרַבִּים יִשָּׁפֵּטוּ, מְהוּמוֹת מִשָּׁמַיִם תִּהְיֶינָה. וְאָז צָרָה אֲיֻמָּה תִּהְיֶה עַל־תֵּבֵל. אוֹי לֶעָרוֹת וְלַמֵּינִיקוֹת בַּיָּמִים הָאַחֲרוֹנִים; אוֹי לַחֲסֵרֵי־לֵב, אוֹי לַמַּאֲמִינִים בַּמִּתְנַגֵּד; אוֹי לַמִּשְׁתַּחֲוִים לוֹ וּבְבוֹאוֹ נִגְלוּ.

26. וְאַחֲרֵי הֱיוֹת כָּל־אֵלֶּה, וְלָשֵׂאת אֶת־יִסּוּרֵי הָאֲנָשִׁים רוּחַ אֱלֹהִים וְהַצַּדִּיקִים אֲשֶׁר מֵאֹנֶס וָעֹשֶׁק נִלְחָצוּ, תָּבוֹא הַתַּכְלִית. וּלְאַנְשֵׁי אָדָם יַרְאוּ אוֹת וַיַּכִּירוּ אֶת־הַשְּׁבִיעִית וְיֹאמְרוּ: הַאִם עִיר הָיְתָה זֹאת? וְאִשָּׁה אַחַת מִתְיַסֶּרֶת אֶל־מִזְרָח, מַעֲרָב, צָפוֹן וָדָרוֹם תְּבַקֵּשׁ פְּרִי וְלֹא תִמְצָא, רַק תִּמְצָא זַיִת אֶחָד, וּבְקִיחָה מִן־הַזַּיִת תּוֹצִיא מֵחֵיקָהּ וְתֹאמַר: אַשְׁרֵי מִי שֶׁנָּטַע אֶת־הַזַּיִת הַזֶּה! וְרוּחָהּ תַּעֲלֶה מִמֶּנָּה שָׁמָּה אֶל־מְקוֹמָהּ.

27. אָז תֵּהָפֵךְ הַשֶּׁמֶשׁ לַחֹשֶׁךְ, וְיָרֵחַ לְדָם, כּוֹכָבִים כֶּעָלֶה יִפֹּלוּ, וְשָׁמַיִם כִּמְגִלָּה יִגָּלְלוּ, וְהַיָּם מִתְּהוֹם יַרְתִּיחַ גַּלָּיו לְכַסּוֹת אֶת־בְּנֵי־אָדָם, וְכָל־אֲשֶׁר בָּאֲוִיר יֵחָרֵךְ וְיִיבָשׁ. מַלְאָכִים בּוֹעֲרִים מִשָּׁמַיִם יֵרְדוּ, וְאֵשׁ תִּתְלַקֵּחַ בַּתֵּבֵל; עַכְבָּרִים אֵשִׁיִּים וּנְחֻשְׁתִּיִּים יִהְיוּ, וְהַדּוֹמֶה לָהֶם, וְחַיּוֹת טוֹרְפוֹת מֵהָרִים יֵצְאוּ, וְלֹא יִירָאוּ אֶת־אֲשֶׁר לֹא בִקֵּשׁוּ; וְאֶרֶץ רְשָׁעִים תֹּאבַד, וְהַצַּדִּיקִים יֵאָסְפוּ אֶל־הָאָב, כִּי פְקֻדָּה יָצְאָה מֵאֵת יְיָ; כִּסְאוֹת יִפֹּלוּ, וּסְפָרִים יִפָּתֵחוּ, וּמִשְׁפָּטִים יַעֲבֹרוּ, מַלְאָכִים יִתְקְעוּ בַּשּׁוֹפָר, וְצַדִּיקִים יָגִילוּ, אֵשׁ תּוֹצִיא בְּרָכוֹת, וּכְאִישׁ מַעֲשָׂיו יִשָּׁפֵטוּ; רַק יְיָ שׁוֹפֵט צֶדֶק, וְכָל־מַעֲשָׂיו אֱמוּנָה.

28. וְכָל־עַמֵּי בְּנֵי־אָדָם יִפְתְּחוּ פִיהֶם לִצְעֹק וְלוֹמַר: יְיָ יְיָ, אַל־תְּבִיאֵנוּ לִידֵי נִסָּיוֹן, כִּי אִם־חַלְּצֵנוּ מִן־הָרָע, כִּי אַתָּה יְיָ יוֹדֵעַ וּמַכִּיר כִּי לְהַאֲרִיךְ לֹא נוּכָל, כִּי בָשָׂר אֲנָחְנוּ; אֲבָל כְּאָב מֵטִיב וְרַחֲמָן חוּסָה עָלֵינוּ, כִּי לְךָ הַתְּהִלָּה מֵעַתָּה וְעַד־עוֹלָם, אָמֵן. --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- נִסְפַּח: הָעֲרוֹת עַל הַמָּקוֹר הָאַרְמֵנִי
