דף ז — מלכים על דרך אותיות שמותיהם

מקור: ערבית · ייחוס: מתלמידי ישוע הנוצרי

גודל טקסט

שְׁמוֹ אָלֶ"ף נוּ"ן. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ אָלֶ"ף. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ מֵ"ם נוּ"ן. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ עַיִ"ן נוּ"ן. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ פֵ"א — וְיִהְיֶה אִישׁ מַאֲמִין, שַׂר. וְאַחֲרָיו יִמְלֹךְ אִישׁ — רִאשׁוֹן שֶׁל שְׁמוֹ פֵ"א — וְהוּא בַּרְבָּרִי, וְהוּא רִאשׁוֹן הַמְּלָכִים אֲשֶׁר יִתְגַּלּוּ לַמַּרְאוֹת נוּ"ן.

וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ דָּלֶ"ת. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ וָ"ו נוּ"ן — וְזֶה הַמֶּלֶךְ יִהְיֶה כּוֹפֵר וְחָנֵף. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ קוֹ"ף נוּ"ן. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ דָּלֶ"ת נוּ"ן. וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ פֵ"א.

יְגַלֶּה הָאֱמֶת וְהַנֶּאֱמָנוּת, וְתִהְיֶינָה דְּגָלָיו עָלָיו צוּרַת צְלָבִי. וְיוּדַע שְׁמוֹ בְּקַצְוֵי הָאָרֶץ, מִזְרָחָהּ וּמַעֲרָבָהּ נוּ"ן. וְיִהְיוּ הַמְּלָכִים כֻּלָּם מִתְבָּרְכִים מִמֶּנּוּ וּבִשְׁמוֹ, כִּי יְגַלֶּה הָאֱמֶת וְהַצֶּדֶק, וִיצַוֶּה בַטּוֹב וְיַזְהִיר מִן הָרַע. וּבְיָמָיו אַגַּלֶּה אוֹתוֹתַי בַּשָּׁמַיִם וּבָאָרֶץ, וְאֶפְרֹשׂ צְלָבִי לְכָל הַבְּרִיאוּת, וְאֶהְיֶה עִמּוֹ, וְאֶתֵּן ⟨בּוֹ⟩ כֹּחִי. וְיִכְבֹּשׁ מֶלֶךְ זֶה אֶת הָאוֹיֵב, וִיכַבֵּד אֶת הַמַּאֲמִינִים בִּי. וְתִהְיֶינָה תִּפְאֶרֶת הַמְּלָכִים בּוֹ, וּבִדְגָלָיו יִגְבְּרוּ אוֹיְבֵיהֶם.

וְאַחֲרָיו יִמְלְכוּ שְׁנֵי מְלָכִים, שֶׁשֵּׁמוֹתֵיהֶם כְּרַאיֵיס. וְאַחַר שְׁנֵיהֶם יִמְלֹךְ מֶלֶךְ מִבֵּית שְׁנֵי הַמְּלָכִים אֲשֶׁר הָיוּ לְפָנָיו, אֲשֶׁר רֹאשׁ שְׁמוֹ לָמֶ"ד — וְיִהְיֶה חָנֵף, מִתְיַגֵּעַ בְּהַשְׁמָדַת הַמַּאֲמִינִים וּבְהַחְרָבָתָם, עַל דַּת יוֹחָנָן הַמַּטְבִּיל קֹדֶם הַחֹרֶף.

וְיִמְלֹךְ אַחַר מֶלֶךְ זֶה מֶלֶךְ הַמַּאֲמִין בִּי, אֲשֶׁר רֹאשׁ שְׁמוֹ יוֹ"ד; וְיִמְלֹךְ אַחֲרָיו מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ וָ"ו נוּ"ן; וּמֶלֶךְ אַחַר מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ תָּ"ו; וְזֶה הַמֶּלֶךְ יִהְיֶה נִבְחָר בָּאֱמוּנָה בִּי, וְאוֹהֲבִי. וּבְיָמָיו אַגַּלֶּה נִפְלָאוֹת רַבּוֹת, וְעֵקֶב מִדִּבְרֵי הַתְּקוּמָה.

וְאַחֲרָיו יִמְלֹךְ מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ מֵ"ם; וְיִהְיֶה גַּם הוּא נִבְחָר, מְבֹרָךְ, מִתְקַבֵּל בֶּאֱמוּנָה שְׁלֵמָה, אֲשֶׁר אֵין בָּהּ סָפֵק וְלֹא וִכּוּחַ; וְיָמוּת הָאֱמֶת אֶת הַשֶּׁקֶר. וְאֶהְיֶה לוֹ סוֹמֵךְ וְעוֹמֵד עִמּוֹ. וְיִמְלֹךְ אַחֲרָיו מֶלֶךְ אֲשֶׁר רֹאשׁ שְׁמוֹ לָמֶ"ד; וְיִהְיֶה גַם הוּא מְבֹרָךְ, טָהוֹר…

[בשולים: מ"א]
